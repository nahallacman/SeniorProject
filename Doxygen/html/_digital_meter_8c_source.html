48 /************************************************************************************
49 * Function: NumberToString(DWORD Value,XCHAR *pText, BYTE NoOfDigits, BYTE DotPos )
50 *
51 * Notes: convert the number to string
52 *
53 *************************************************************************************/
54 static void NumberToString(DWORD Value, XCHAR *pText, BYTE NoOfDigits, BYTE DotPos)
55 {
56  BYTE i;
57  BYTE bIndex;
58 
59  for(bIndex = 0; bIndex < NoOfDigits; bIndex++)
60  {
61  pText[NoOfDigits - bIndex - 1] = '0' + (Value % 10);
62  Value /= 10;
63  }
64 
65  if(DotPos != 0 && DotPos <= NoOfDigits)
66  {
67  for(i = 0; i < DotPos; i++)
68  {
69  pText[NoOfDigits - i] = pText[NoOfDigits - 1 - i];
70  }
71 
72  pText[NoOfDigits - DotPos] = '.';
73  pText[NoOfDigits + 1] = '\0';
74  }
75 
76  //If dot position is 0 or greater than number of digits, then don't put dot in the display
77  else
78  {
79  pText[NoOfDigits] = '\0';
80  }
81 }

202 /*********************************************************************
203 * Function: WORD DmDraw(void *pObj)
204 *
205 * Notes: This is the state machine to display the changing numbers.
206 *
207 **********************************************************************/
208 WORD DmDraw(void *pObj)
209 {
210  typedef enum
211  {
212  DM_STATE_IDLE,
213  DM_STATE_FRAME,
214  DM_STATE_DRAW_FRAME,
215  DM_STATE_INIT,
216  DM_STATE_SETALIGN,
217  DM_STATE_SETTEXT,
218  DM_STATE_ERASETEXT,
219  DM_STATE_DRAWTEXT,
220  DM_STATE_WRAPUP
221  } DM_DRAW_STATES;
222 
223  DIGITALMETER *pDm = NULL;
224  static DM_DRAW_STATES state = DM_STATE_IDLE;
225  static SHORT charCtr = 0, lineCtr = 0;
226  static XCHAR CurValue[DM_WIDTH], PreValue[DM_WIDTH];
227  static SHORT textWidth = 0;
228  static XCHAR ch = 0, pch = 0;
229 
230  pDm = (DIGITALMETER *)pObj;
231 
232  while(1)
233  {
234 
235  if(IsDeviceBusy())
236  return (0);
237 
238  switch(state)
239  {
240  case DM_STATE_DRAW_FRAME:
241  if(Rectangle(pDm->hdr.left, pDm->hdr.top, pDm->hdr.right, pDm->hdr.bottom) == 0)
242  return (0);
243  state = DM_STATE_INIT;
244  break;
245 
246  case DM_STATE_IDLE:
247  SetClip(CLIP_DISABLE);
248 
249 #ifdef USE_BISTABLE_DISPLAY_GOL_AUTO_REFRESH
250  GFX_DRIVER_SetupDrawUpdate( pDm->hdr.left,
251  pDm->hdr.top,
252  pDm->hdr.right,
253  pDm->hdr.bottom);
254 #endif
255  if(GetState(pDm, DM_HIDE) || GetState(pDm, DM_DRAW))
256  {
257  SetColor(pDm->hdr.pGolScheme->CommonBkColor);
258  if(Bar(pDm->hdr.left, pDm->hdr.top, pDm->hdr.right, pDm->hdr.bottom) == 0)
259  return (0);
260  }
261  // if the draw state was to hide then state is still IDLE STATE so no need to change state
262  if (GetState(pDm, DM_HIDE))
263  {
264 #ifdef USE_BISTABLE_DISPLAY_GOL_AUTO_REFRESH
265  GFX_DRIVER_CompleteDrawUpdate( pDm->hdr.left,
266  pDm->hdr.top,
267  pDm->hdr.right,
268  pDm->hdr.bottom);
269 #endif
270  return (1);
271  }
272  state = DM_STATE_FRAME;
273 
274  case DM_STATE_FRAME:
275 
276  if(GetState(pDm, DM_DRAW | DM_FRAME) == (DM_DRAW | DM_FRAME))
277  {
278  // show frame if specified to be shown
279  SetLineType(SOLID_LINE);
280  SetLineThickness(NORMAL_LINE);
281  if(!GetState(pDm, DM_DISABLED))
282  {
283 
284  // show enabled color
285  SetColor(pDm->hdr.pGolScheme->Color1);
286  }
287  else
288  {
289 
290  // show disabled color
291  SetColor(pDm->hdr.pGolScheme->ColorDisabled);
292  }
293  state = DM_STATE_DRAW_FRAME;
294  break;
295  }
296  else
297  {
298  state = DM_STATE_INIT;
299  }
300 
301  case DM_STATE_INIT:
302  if(IsDeviceBusy())
303  return (0);
304 
305  // set clipping area, text will only appear inside the static text area.
306  SetClip(CLIP_ENABLE);
307  SetClipRgn(pDm->hdr.left + DM_INDENT, pDm->hdr.top, pDm->hdr.right - DM_INDENT, pDm->hdr.bottom);
308 
309  // set the text color
310  if(!GetState(pDm, DM_DISABLED))
311  {
312  SetColor(pDm->hdr.pGolScheme->TextColor0);
313  }
314  else
315  {
316  SetColor(pDm->hdr.pGolScheme->TextColorDisabled);
317  }
318 
319  // convert the values to be displayed in string format
320  NumberToString(pDm->Pvalue, PreValue, pDm->NoOfDigits, pDm->DotPos);
321  NumberToString(pDm->Cvalue, CurValue, pDm->NoOfDigits, pDm->DotPos);
322 
323  // use the font specified in the object
324  SetFont(pDm->hdr.pGolScheme->pFont);
325 
326  state = DM_STATE_SETALIGN; // go to drawing of text
327 
328  case DM_STATE_SETALIGN:
329  if(!charCtr)
330  {
331 
332  // set position of the next character (based on alignment and next character)
333  textWidth = GetTextWidth(CurValue, pDm->hdr.pGolScheme->pFont);
334 
335  // Display text with center alignment
336  if(GetState(pDm, (DM_CENTER_ALIGN)))
337  {
338  MoveTo((pDm->hdr.left + pDm->hdr.right - textWidth) >> 1, pDm->hdr.top + (lineCtr * pDm->textHeight));
339  }
340 
341  // Display text with right alignment
342  else if(GetState(pDm, (DM_RIGHT_ALIGN)))
343  {
344  MoveTo((pDm->hdr.right - textWidth - DM_INDENT), pDm->hdr.top + (lineCtr * pDm->textHeight));
345  }
346 
347  // Display text with left alignment
348  else
349  {
350  MoveTo(pDm->hdr.left + DM_INDENT, pDm->hdr.top + (lineCtr * pDm->textHeight));
351  }
352  }
353  pch = *(PreValue + charCtr);
354  ch = *(CurValue + charCtr);
355  state = DM_STATE_SETTEXT;
356 
357  case DM_STATE_SETTEXT:
358 
359  if (0x0000 != ch)
360  {
361  if(GetState(pDm, DM_DRAW))
362  {
363  SetColor(pDm->hdr.pGolScheme->CommonBkColor);
364  }
365  else if(GetState(pDm, DM_UPDATE))
366  {
367  if(pch != ch)
368  {
369  SetColor(pDm->hdr.pGolScheme->CommonBkColor);
370  }
371  else
372  {
373  state = DM_STATE_DRAWTEXT;
374  break;
375  }
376  }
377  state = DM_STATE_ERASETEXT;
378  break;
379  }
380  else
381  {
382  state = DM_STATE_WRAPUP;
383  break;
384  }
385 
386  case DM_STATE_ERASETEXT:
387  if(Bar(GetX(), pDm->hdr.top + 1, GetX() + textWidth, pDm->hdr.bottom - 1) == 0)
388  return (0);
389  state = DM_STATE_DRAWTEXT;
390 
391  case DM_STATE_DRAWTEXT:
392  SetColor(pDm->hdr.pGolScheme->TextColor0);
393 
394  // render the character
395  while(!OutChar(ch));
396  charCtr++; // update to next character
397  ch = *(CurValue + charCtr);
398  pch = *(PreValue + charCtr);
399  state = DM_STATE_SETTEXT;
400  break;
401 
402  case DM_STATE_WRAPUP:
403 
404  // end of text string is reached no more lines to display
405  lineCtr = 0;
406  charCtr = 0;
407  SetClip(CLIP_DISABLE); // remove clipping
408  state = DM_STATE_IDLE;
409 #ifdef USE_BISTABLE_DISPLAY_GOL_AUTO_REFRESH
410  GFX_DRIVER_CompleteDrawUpdate( pDm->hdr.left,
411  pDm->hdr.top,
412  pDm->hdr.right,
413  pDm->hdr.bottom);
414 #endif
415  return (1);
416 
417  } // end of switch()
418  } // end of while(1)
419 }
