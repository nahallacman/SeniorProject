157 // ENC624J600 Interface Configuration
158 // Comment out ENC100_INTERFACE_MODE if you don't have an ENC624J600 or
159 // ENC424J600. Otherwise, choose the correct setting for the interface you
160 // are using. Legal values are:
161 // - Commented out: No ENC424J600/624J600 present or used. All other
162 // ENC100_* macros are ignored.
163 // - 0: SPI mode using CS, SCK, SI, and SO pins
164 // - 1: 8-bit demultiplexed PSP Mode 1 with RD and WR pins
165 // - 2: 8-bit demultiplexed PSP Mode 2 with R/Wbar and EN pins
166 // - 3: 16-bit demultiplexed PSP Mode 3 with RD, WRL, and WRH pins
167 // - 4: 16-bit demultiplexed PSP Mode 4 with R/Wbar, B0SEL, and B1SEL pins
168 // - 5: 8-bit multiplexed PSP Mode 5 with RD and WR pins
169 // - 6: 8-bit multiplexed PSP Mode 6 with R/Wbar and EN pins
170 // - 9: 16-bit multiplexed PSP Mode 9 with AL, RD, WRL, and WRH pins
171 // - 10: 16-bit multiplexed PSP Mode 10 with AL, R/Wbar, B0SEL, and B1SEL
172 // pins
173 #define ENC100_INTERFACE_MODE 0
174 
175 // If using a parallel interface, direct RAM addressing can be used (if all
176 // addresses wires are connected), or a reduced number of pins can be used
177 // for indirect addressing. If using an SPI interface or PSP Mode 9 or 10
178 // (multiplexed 16-bit modes), which require all address lines to always be
179 // connected, then this option is ignored. Comment out or uncomment this
180 // macro to match your hardware connections.
181 #define ENC100_PSP_USE_INDIRECT_RAM_ADDRESSING
182 
183 // ENC424J600/624J600 parallel indirect address remapping macro function.
184 // This section translates SFR and RAM addresses presented to the
185 // ReadMemory() and WriteMemory() APIs in ENCX24J600.c to the actual
186 // addresses that must be presented on the parallel interface. This macro
187 // must be modified to match your hardware if you are using an indirect PSP
188 // addressing mode (ENC100_PSP_USE_INDIRECT_RAM_ADDRESSING is defined) and
189 // have some of your address lines tied off to Vdd. If you are using the
190 // SPI interface, then this section can be ignored or deleted.
191 #if (ENC100_INTERFACE_MODE == 1) || (ENC100_INTERFACE_MODE == 2) || (ENC100_INTERFACE_MODE == 5) || (ENC100_INTERFACE_MODE == 6) // 8-bit PSP
192  #define ENC100_TRANSLATE_TO_PIN_ADDR(a) ((((a)&0x0100)<<6) | ((a)&0x00FF))
193 #elif (ENC100_INTERFACE_MODE == 3) || (ENC100_INTERFACE_MODE == 4) // 16-bit PSP
194  #define ENC100_TRANSLATE_TO_PIN_ADDR(a) (a)
195 #endif
