396 /*****************************************************************************
397  * FUNCTION: RawMove
398  *
399  * RETURNS: Number of bytes that were overlayed (not always applicable)
400  *
401  * PARAMS:
402  * rawId - RAW ID
403  * srcDest - MRF24W object that will either source or destination of move
404  * rawIsDestination - TRUE if RAW engine is the destination, FALSE if its the source
405  * size - number of bytes to overlay (not always applicable)
406  *
407  * NOTES: Performs a RAW move operation between a RAW engine and a MRF24W object
408  *****************************************************************************/
409  static UINT16 RawMove(UINT16 rawId,
410  UINT16 srcDest,
411  BOOL rawIsDestination,
412  UINT16 size)
413 {
414  UINT16 byteCount;
415  UINT8 regId;
416  UINT8 regValue8;
417  UINT16 ctrlVal = 0;
418 
419  if (rawIsDestination)
420  {
421  ctrlVal |= 0x8000;
422  }
423 
424  /* fix later, simply need to ensure that size is 12 bits are less */
425  ctrlVal |= (srcDest << 8); /* defines are already shifted by 4 bits */
426  ctrlVal |= ((size >> 8) & 0x0f) << 8; /* MS 4 bits of size (bits 11:8) */
427  ctrlVal |= (size & 0x00ff); /* LS 8 bits of size (bits 7:0) */
428 
429  /* Clear the interrupt bit in the register */
430  regValue8 = (rawId == RAW_ID_0)?WF_HOST_INT_MASK_RAW_0_INT_0:WF_HOST_INT_MASK_RAW_1_INT_0;
431  Write8BitWFRegister(WF_HOST_INTR_REG, regValue8);
432 
433  /* write update control value to register to control register */
434  regId = (rawId==RAW_ID_0)?RAW_0_CTRL_0_REG:RAW_1_CTRL_0_REG;
435  Write16BitWFRegister(regId, ctrlVal);
436 
437  // Wait for the RAW move operation to complete, and read back the number of bytes, if any, that were overlayed
438  byteCount = WaitForRawMoveComplete(rawId);
439 
440  return byteCount;
441 }

443 /*****************************************************************************
444  * FUNCTION: RawSetIndex
445  *
446  * RETURNS: True is success, false if timed out, which means attempted to set
447  * raw index past end of raw window. Not a problem as long as no read
448  * or write occurs.
449  *
450  * PARAMS:
451  * rawId - RAW ID
452  * index - desired index
453  *
454  * NOTES: Sets the RAW index for the specified RAW engine. If attempt to set RAW
455  * index outside boundaries of RAW window this function will time out.
456  *****************************************************************************/
457 BOOL RawSetIndex(UINT16 rawId, UINT16 index)
458 {
459  UINT8 regId;
460  UINT16 regValue;
461  UINT32 startTickCount;
462  UINT32 maxAllowedTicks;
463 
464  // set the RAW index
465  regId = (rawId==RAW_ID_0)?RAW_0_INDEX_REG:RAW_1_INDEX_REG;
466  Write16BitWFRegister(regId, index);
467 
468  startTickCount = (UINT32)TickGet();
469  maxAllowedTicks = TICKS_PER_SECOND / 200; /* 5ms */
470 
471  regId = (rawId==RAW_ID_0)?RAW_0_STATUS_REG:RAW_1_STATUS_REG;
472 
473  while (1)
474  {
475  regValue = Read16BitWFRegister(regId);
476  if ((regValue & WF_RAW_STATUS_REG_BUSY_MASK) == 0)
477  {
478  return TRUE;
479  }
480 
481  /* if timed out then trying to set index past end of raw window, which is OK so long as the app */
482  /* doesn't try to access it */
483  if (TickGet() - startTickCount >= maxAllowedTicks)
484  {
485  return FALSE; /* timed out waiting for Raw set index to complete */
486  }
487  }
488 }

624 /*****************************************************************************
625  * FUNCTION: WaitForRawMoveComplete
626  *
627  * RETURNS: Number of bytes that were overlayed (not always applicable)
628  *
629  * PARAMS:
630  * rawId - RAW ID
631  *
632  * NOTES: Waits for a RAW move to complete.
633  *****************************************************************************/
634 static UINT16 WaitForRawMoveComplete(UINT8 rawId)
635 
636 {
637  UINT8 rawIntMask;
638  UINT16 byteCount;
639  UINT8 regId;
640  BOOL intDisabled;
641  #if defined(WF_DEBUG)
642  UINT32 startTickCount;
643  UINT32 maxAllowedTicks;
644  #endif
645 
646  /* create mask to check against for Raw Move complete interrupt for either RAW0 or RAW1 */
647  rawIntMask = (rawId == RAW_ID_0)?WF_HOST_INT_MASK_RAW_0_INT_0:WF_HOST_INT_MASK_RAW_1_INT_0;
648 
649  /*
650  These variables are shared with the ISR so need to be careful when setting them.
651  the WFEintHandler() is the isr that will touch these variables but will only touch
652  them if RawMoveState.waitingForRawMoveCompleteInterrupt is set to TRUE.
653  RawMoveState.waitingForRawMoveCompleteInterrupt is only set TRUE here and only here.
654  so as long as we set RawMoveState.rawInterrupt first and then set RawMoveState.waitingForRawMoveCompleteInterrupt
655  to TRUE, we are guranteed that the ISR won't touch RawMoveState.rawInterrupt and
656  RawMoveState.waitingForRawMoveCompleteInterrupt.
657  */
658  RawMoveState.rawInterrupt = 0;
659  RawMoveState.waitingForRawMoveCompleteInterrupt = TRUE;
660 
661  // save state of external interrupt here
662  intDisabled = WF_EintIsDisabled();
663  // if external interrupt is disabled, enable it because we need it for the while(1) loop to exit
664  if(intDisabled)
665  {
666  WF_EintEnable();
667  }
668  else if(WF_EintIsPending())
669  {
670  WF_EintEnable();
671  }
672 
673  #if defined(WF_DEBUG)
674  // Before we enter the while loop, get the tick timer count and save it
675  maxAllowedTicks = TICKS_PER_SECOND / 2; /* 500 ms timeout */
676  startTickCount = (UINT32)TickGet();
677  #endif
678  while (1)
679  {
680  /* if received an external interrupt that signalled the RAW Move */
681  /* completed then break out of this loop */
682  if(RawMoveState.rawInterrupt & rawIntMask)
683  {
684  break;
685  }
686 
687  #if defined(WF_DEBUG)
688  /* If timed out waiting for RAW Move complete than lock up */
689  if (TickGet() - startTickCount >= maxAllowedTicks)
690  {
691  WF_ASSERT(FALSE);
692  }
693  #endif
694 
695  } /* end while */
696 
697  /* if interrupt was enabled by us here, we should disable it now that we're finished */
698  if(intDisabled)
699  {
700  WF_EintDisable();
701  }
702 
703  /* read the byte count and return it */
704  regId = (rawId == RAW_ID_0)?WF_HOST_RAW0_CTRL1_REG:WF_HOST_RAW1_CTRL1_REG;
705  byteCount = Read16BitWFRegister(regId);
706 
707  return ( byteCount );
708 }

716  * RETURNS: TRUE or FALSE
