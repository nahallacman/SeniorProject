210 static BOOL SendNotification(BYTE receiverIndex, SNMP_ID var, SNMP_VAL val,UINT8 targetIndex)
211 {
212  static enum { SM_PREPARE, SM_NOTIFY_WAIT } smState = SM_PREPARE;
213  IP_ADDR IPAddress;
214  static BYTE tempRxIndex;
215  static IP_ADDR tempIpAddress;
216 
217  // Convert local to network order.
218  IPAddress.v[0] = trapInfo.table[receiverIndex].IPAddress.v[3];
219  IPAddress.v[1] = trapInfo.table[receiverIndex].IPAddress.v[2];
220  IPAddress.v[2] = trapInfo.table[receiverIndex].IPAddress.v[1];
221  IPAddress.v[3] = trapInfo.table[receiverIndex].IPAddress.v[0];
222 
223  if(gtrapSMStateUpdate == TRUE)
224  {
225  smState = SM_PREPARE;
226  gtrapSMStateUpdate = FALSE;
227  }
228 
229  switch(smState)
230  {
231  case SM_PREPARE:
232  gSendTrapSMstate = smState;
233  tempRxIndex=receiverIndex;
234  // Convert local to network order.
235  tempIpAddress.v[0] = trapInfo.table[receiverIndex].IPAddress.v[3];
236  tempIpAddress.v[1] = trapInfo.table[receiverIndex].IPAddress.v[2];
237  tempIpAddress.v[2] = trapInfo.table[receiverIndex].IPAddress.v[1];
238  tempIpAddress.v[3] = trapInfo.table[receiverIndex].IPAddress.v[0];
239  SNMPNotifyPrepare(&IPAddress,
240  trapInfo.table[receiverIndex].community,
241  trapInfo.table[receiverIndex].communityLen,
242  MICROCHIP, // Agent ID Var
243  gSpecificTrapNotification, // Specifc Trap notification code
244  SNMPGetTimeStamp());
245  smState = SM_NOTIFY_WAIT;
246 
247  break;
248 
249  case SM_NOTIFY_WAIT:
250  gSendTrapSMstate = smState;
251  if ( SNMPIsNotifyReady(&IPAddress) )
252  {
253 #if defined(SNMP_V1_V2_TRAP_WITH_SNMPV3)
254  if(targetIndex == (SNMPV3_USM_MAX_USER-1))
255  smState = SM_PREPARE;
256  if((gSnmpv3TrapConfigData[targetIndex].messageProcessingModelType == SNMPV3_MSG_PROCESSING_MODEL)
257  && (gSnmpv3TrapConfigData[targetIndex].securityModelType == SNMPV3_USM_SECURITY_MODEL))
258  Snmpv3Notify(var, val, 0,targetIndex);
259  else if((gSnmpv3TrapConfigData[targetIndex].messageProcessingModelType == SNMPV2C_MSG_PROCESSING_MODEL)
260  && (gSnmpv3TrapConfigData[targetIndex].securityModelType == SNMPV2C_SECURITY_MODEL))
261  SNMPNotify(var, val, 0);
262  else if((gSnmpv3TrapConfigData[targetIndex].messageProcessingModelType == SNMPV1_MSG_PROCESSING_MODEL)
263  && (gSnmpv3TrapConfigData[targetIndex].securityModelType == SNMPV1_SECURITY_MODEL))
264  SNMPNotify(var, val, 0);
265  else
266  return TRUE;
267 #else
268  smState = SM_PREPARE;
269  SNMPNotify(var, val, 0);
270 #endif
271  return TRUE;
272  }
273  /* if trapInfo table address for a perticular index is different comparing to the SM_PREPARE IP address
274  then change the state to SM_PREPARE*/
275  if((tempIpAddress.Val != IPAddress.Val) &&
276  (tempRxIndex == receiverIndex))
277  {
278  smState = SM_PREPARE;
279  }
280  /* Change state machine from SM_NOTIFY_WAIT to SM_PREPARE if incoming trap destination
281  index is different from the SM_PREPARE trap destination index*/
282  if(tempRxIndex != receiverIndex)
283  smState=SM_PREPARE;
284 
285  }
286 
287  return FALSE;
288 }
