120 //***********************************************
121 // SPI Setup
122 //***********************************************
123  // When SPI settings are being changed, even if the settings are the same as before,
124  // for some reasons, the SCK output from the MCU will not be driven for a period of time.
125  // The change of this signal from driven low to tri-state may be treated by S1D13522 as 1
126  // extra serial host input clock if at that moment the HCS_L is low. Because of this extra
127  // clock, the subsequent data read will all be shifted. To avoid this, make sure HCS_L is
128  // high when writing to MCU's SPI configurations.
129  DisplayDisable(); // Set HCS_L high before writing to SPI configurations
130  EPD_SPI_STAT(USE_SPI_CHANNEL) = 0;
131 
132  EPD_SPI_CONbits(USE_SPI_CHANNEL).MSTEN = SPI_MSTEN;
133  EPD_SPI_CONbits(USE_SPI_CHANNEL).MODE16 = SPI_MODE;
134  EPD_SPI_CONbits(USE_SPI_CHANNEL).CKP = SPI_CKP;
135  EPD_SPI_CONbits(USE_SPI_CHANNEL).CKE = SPI_CKE;
136  EPD_SPI_CONbits(USE_SPI_CHANNEL).SMP = SPI_SMP;
137  #ifdef __PIC32MX
138  EPD_SPI_CONbits(USE_SPI_CHANNEL).MODE32 = 0;
139  EPD_SPI_BRG(USE_SPI_CHANNEL) = SPI_BAUDRATE;

500 /*********************************************************************
501 * Function: WORD Bar(SHORT left, SHORT top, SHORT right, SHORT bottom)
502 *
503 * PreCondition: none
504 *
505 * Input: left,top - top left corner coordinates,
506 * right,bottom - bottom right corner coordinates
507 *
508 * Output: For NON-Blocking configuration:
509 * - Returns 0 when device is busy and the shape is not yet completely drawn.
510 * - Returns 1 when the shape is completely drawn.
511 * For Blocking configuration:
512 * - Always return 1.
513 *
514 * Side Effects: none
515 *
516 * Overview: draws rectangle filled with current color
517 *
518 * Note: none
519 *
520 ********************************************************************/
521 WORD Bar(SHORT left, SHORT top, SHORT right, SHORT bottom)
522 {
523  if ( g_ActiveLayer == GFX_CURSOR_LAYER )
524  {
525  // not available for cursor layer
526  return 1;
527  }
528  else
529  {
530  WORD pxl = _color;
531  WORD SrcColor = SRC_COLOR;
532  WORD Width = right - left + 1;
533  WORD Height = bottom - top + 1;
534  DWORD Count = (DWORD)Width * (DWORD)Height;
535 
536  if(left > right)
537  return(1); //Added to make sure function call is accurate
538 
539  if(bottom < top)
540  return(1); //Added to make sure function call is accurate
541 
542  #if ( SRC_COLOR == SRC_1BIT )
543  pxl = (_color==WHITE) ? 0xFFFF : 0;
544  Count = (Count%16) ? (1+Count/16) : (Count/16);
545  #else
546  if ( _color == WHITE || _color == BLACK )
547  {
548  pxl = (_color==WHITE) ? 0xFFFF : 0;
549  Count = (Count%16) ? (1+Count/16) : (Count/16);
550  SrcColor = SRC_1BIT;
551  }
552  else
553  {
554  #if ( SRC_COLOR == SRC_8BIT ) //input packed pixel = 1 byte-per-pixel (8 bpp)
555  Count = (Count%2) ? 1+Count/2 : Count/2;
556  pxl &= 0xFF;
557  pxl = (pxl<<8) | (pxl<<0);
558  #elif ( SRC_COLOR == SRC_4BIT ) //packed pixel = 4 bpp
559  Count = (Count%4) ? 1+Count/4 : Count/4;
560  pxl &= 0x0F;
561  pxl = (pxl<<12) | (pxl<<8) | (pxl<<4) | (pxl<<0);
562  #endif
563  }
564  #endif
565 
566  DisplayEnable(); // enable S1D13522
567 
568  seWAIT_HRDY();
569  DisplaySetCommand(); // set RS line to low for command
570  seDeviceWrite( LD_IMG_AREA ); // write command
571  DisplaySetData(); // set RS line to high for data
572  seDeviceWrite( SrcColor | g_HostMemAccessCfg );
573  seDeviceWrite( left );
574  seDeviceWrite( top );
575  seDeviceWrite( Width );
576  seDeviceWrite( Height );
577 
578  seWAIT_HRDY();
579  DisplaySetCommand(); // set RS line to low for command
580  seDeviceWrite( WR_REG ); // write command
581  DisplaySetData(); // set RS line to high for data
582  seDeviceWrite( REG0154_HOST_MEMORY_ACCESS_PORT );
583 
584  while ( Count-- )
585  {
586  seDeviceWrite( pxl );
587  }
588 
589  seWAIT_HRDY();
590  DisplaySetCommand(); // set RS line to low for command
591  seDeviceWrite( LD_IMG_END ); // write command
592 
593  DisplayDisable(); // disable S1D13522
594 
595  g_UpdateRequest = TRUE;
596 
597  return 1;
598  }

1674 // If specified, image data is sent in pData (otherwise pData=NULL).
1675 //---------------------------------------------------------------------------
1676 void GFX_DRIVER_LoadImage( WORD HostOptions, WORD xStart, WORD yStart, WORD xEnd, WORD yEnd, const WORD* pData )
1677 {
1678  if ( g_ActiveLayer == GFX_CURSOR_LAYER )
1679  {
1680  if ((HostOptions & 0x0030) != SRC_4BIT) //packed pixel is not 4 bpp
1681  return;
1682 
1683  DWORD nWordsToWrite = (((DWORD)g_CursorLayerHeight*g_CursorLayerWidth*4)+15)/16;
1684 
1685  DisplayEnable(); // enable S1D13522
1686 
1687  seWAIT_HRDY();
1688  DisplaySetCommand(); // set RS line to low for command
1689  seDeviceWrite( CSR_ADRCFG ); // write command
1690  DisplaySetData(); // set RS line to high for data

1694 
1695  DisplaySetCommand(); // set RS line to low for command

1697

1699  seDeviceWrite( REG03DE_CRS_MEM_LOAD );
1700 
1701  while( nWordsToWrite-- )
1702  {
1703  seDeviceWrite( *pData++ );
1704  }
1705  seWAIT_HRDY();
1706  DisplaySetCommand(); // set RS line to low for command
1707  seDeviceWrite( CSR_ADRCFG ); // write command
1708  DisplaySetData(); // set RS line to high for data

1727  seDeviceWrite( LD_IMG_AREA ); // write command
1728

1731  seDeviceWrite( xStart );
1732  seDeviceWrite( yStart );
1733  seDeviceWrite( width );
1734  seDeviceWrite( height );
1735 
1736  if ( pData ) //if no data specified, skip image load (data write)
1737  {

1740

1751 
1752  DisplaySetData(); // set RS line to high for data
1753  seDeviceWrite( REG0154_HOST_MEMORY_ACCESS_PORT );
1754 
1755  while( Count-- )
1756  {
1757  seDeviceWrite( *pData++ );
1758  }
1759  }
1760 
1761  seWAIT_HRDY();
1762  DisplaySetCommand(); // set RS line to low for command
1763  seDeviceWrite( LD_IMG_END ); // write command
1764 
1765  DisplayDisable(); // disable S1D13522
1766  }
