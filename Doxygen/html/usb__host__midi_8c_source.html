153 BOOL USBHostMIDIInit ( BYTE address, DWORD flags, BYTE clientDriverID )
154 {
155  BYTE *config_descriptor;
156  BYTE *ptr;
157  BYTE bDescriptorType;
158  BYTE bLength;
159  BYTE bNumEndpoints;
160  BYTE bNumInterfaces;
161  BYTE bInterfaceNumber;
162  BYTE bAlternateSetting;
163  BYTE Class;
164  BYTE SubClass;
165  BYTE Protocol;
166  BYTE currentEndpoint;
167  WORD wTotalLength;
168 
169  BYTE index = 0;
170  BOOL error = FALSE;
171 
172  MIDI_DEVICE *device = &devices[0];
173 
174  config_descriptor = USBHostGetCurrentConfigurationDescriptor(address);
175  ptr = config_descriptor;
176 
177  // Load up the values from the Configuration Descriptor
178  bLength = *ptr++;
179  bDescriptorType = *ptr++;
180  wTotalLength = *ptr++; // In case these are not word aligned
181  wTotalLength += (*ptr++) << 8;
182  bNumInterfaces = *ptr++;
183 
184  // Skip over the rest of the Configuration Descriptor
185  index += bLength;
186  ptr = &config_descriptor[index];
187 
188  while (!error && (index < wTotalLength))
189  {
190  // Check the descriptor length and type
191  bLength = *ptr++;
192  bDescriptorType = *ptr++;
193 
194  // Find an interface descriptor
195  if (bDescriptorType != USB_DESCRIPTOR_INTERFACE)
196  {
197  // Skip over the rest of the Descriptor
198  index += bLength;
199  ptr = &config_descriptor[index];
200  }
201  else
202  {
203  // Read some data from the interface descriptor
204  bInterfaceNumber = *ptr++;
205  bAlternateSetting = *ptr++;
206  bNumEndpoints = *ptr++;
207  Class = *ptr++;
208  SubClass = *ptr++;
209  Protocol = *ptr++;
210 
211  // Check to see if this is a MIDI inteface descripter
212  if (Class != AUDIO_CLASS || SubClass != MIDI_SUB_CLASS || Protocol != MIDI_PROTOCOL)
213  {
214  // If we cannot support this interface, skip it.
215  index += bLength;
216  ptr = &config_descriptor[index];
217  continue;
218  }
219 
220  // Initialize the device
221  device->deviceAddress = address;
222  device->clientDriverID = clientDriverID;
223  device->numEndpoints = bNumEndpoints;
224 
225 
226  // Allocate enough memory for each endpoint
227  if ((device->endpoints = (MIDI_ENDPOINT_DATA*)malloc( sizeof(MIDI_ENDPOINT_DATA) * bNumEndpoints)) == NULL)
228  {
229  // Out of memory
230  error = TRUE;
231  }
232 
233  if (!error)
234  {
235  // Skip over the rest of the Interface Descriptor
236  index += bLength;
237  ptr = &config_descriptor[index];
238 
239  // Find the Endpoint Descriptors. There might be Class and Vendor descriptors in here
240  currentEndpoint = 0;
241  while (!error && (index < wTotalLength) && (currentEndpoint < bNumEndpoints))
242  {
243  bLength = *ptr++;
244  bDescriptorType = *ptr++;
245 
246  if (bDescriptorType != USB_DESCRIPTOR_ENDPOINT)
247  {
248  // Skip over the rest of the Descriptor
249  index += bLength;
250  ptr = &config_descriptor[index];
251  }
252  else
253  {
254  device->endpoints[currentEndpoint].endpointAddress = *ptr++;
255  ptr++;
256  device->endpoints[currentEndpoint].endpointSize = *ptr++;
257  device->endpoints[currentEndpoint].endpointSize += (*ptr++) << 8;
258  device->endpoints[currentEndpoint].busy = FALSE;
259 
260  if(device->endpoints[currentEndpoint].endpointSize > 64)
261  {
262  // For full speed bulk endpoints, only 8, 16, 32, and 64 byte packets are supported
263  // But we will accept anything less than or equal to 64.
264  error = TRUE;
265  }
266 
267  // Get ready for the next endpoint.
268  currentEndpoint++;
269  index += bLength;
270  ptr = &config_descriptor[index];
271  }
272  }
273  }
274 
275  // Ensure that we found all the endpoints for this interface.
276  if (currentEndpoint != bNumEndpoints)
277  {
278  error = TRUE;
279  }
280  }
281  }
282 
283  if (error)
284  {
285  // Destroy whatever list of interfaces, settings, and endpoints we created.
286  // The "new" variables point to the current node we are trying to remove.
287  if (device->endpoints != NULL)
288  {
289  free( device->endpoints );
290  device->endpoints = NULL;
291  }
292  return FALSE;
293  }
294 
295  #ifdef DEBUG_MODE
296  UART2PrintString( "USB MIDI Client Initalized: " );
297  UART2PrintString( " address=" );
298  UART2PutDec( address );
299  UART2PrintString( " Number of Endpoings=" );
300  UART2PutHex( bNumEndpoints );
301  UART2PrintString( "\r\n" );
302  #endif
303 
304  // Notify that application that we've been attached to a device.
305  USB_HOST_APP_EVENT_HANDLER(address, EVENT_MIDI_ATTACH, device, sizeof(MIDI_DEVICE) );
306 
307  return TRUE;
308 
309 } // USBHostMIDIInit

344 BOOL USBHostMIDIEventHandler ( BYTE address, USB_EVENT event, void *data, DWORD size )
345 {
346  unsigned char i;
347 
348  // Make sure it was for one of our devices
349  for( i = 0; i < USB_MAX_MIDI_DEVICES; i++)
350  {
351  if ( address == devices[i].deviceAddress)
352  {
353  break;
354  }
355 
356  }
357  if(i == USB_MAX_MIDI_DEVICES)
358  {
359  return FALSE;
360  }
361 
362  // Handle specific events
363  switch (event)
364  {
365  case EVENT_DETACH:
366  // Notify that application that the device has been detached.
367  USB_HOST_APP_EVENT_HANDLER(devices[i].deviceAddress, EVENT_MIDI_DETACH, &devices[i], sizeof(MIDI_DEVICE) );
368  devices[i].deviceAddress = 0;
369  free(devices[i].endpoints);
370  devices[i].endpoints = NULL;
371  #ifdef DEBUG_MODE
372  UART2PrintString( "USB MIDI Client Device Detached: address=" );
373  UART2PutDec( address );
374  UART2PrintString( "\r\n" );
375  #endif
376  return TRUE;
377 
378  #ifdef USB_ENABLE_TRANSFER_EVENT
379  case EVENT_TRANSFER:
380  if ( (data != NULL) && (size == sizeof(HOST_TRANSFER_DATA)) )
381  {
382  unsigned char currentEndpoint;
383  //DWORD dataCount = ((HOST_TRANSFER_DATA *)data)->dataCount;
384 
385  for(currentEndpoint = 0; currentEndpoint < devices[i].numEndpoints; currentEndpoint++)
386  {
387  if ( ((HOST_TRANSFER_DATA *)data)->bEndpointAddress == devices[i].endpoints[currentEndpoint].endpointAddress )
388  {
389  devices[i].endpoints[currentEndpoint].busy = 0;
390  USB_HOST_APP_EVENT_HANDLER(devices[i].deviceAddress, EVENT_MIDI_TRANSFER_DONE, &devices[i].endpoints[currentEndpoint], sizeof(MIDI_ENDPOINT_DATA));
391  return TRUE;
392  }
393  }
394  }
395  return FALSE;
396  #endif
397 
398  case EVENT_SUSPEND:
399  case EVENT_RESUME:
400  case EVENT_BUS_ERROR:
401  default:
402  break;
403  }
404 
405  return FALSE;
406 } // USBHostMIDIEventHandler
