261 * Overview: translates the GOL message for the radio button

263 ********************************************************************/
264 WORD RbTranslateMsg(void *pObj, GOL_MSG *pMsg)
265 {
266 
267  RADIOBUTTON *pRb;
268 
269  pRb = (RADIOBUTTON *)pObj;
270 
271  // Evaluate if the message is for the radio button
272  // Check if disabled first
273  if(GetState(pRb, RB_DISABLED))
274  return (OBJ_MSG_INVALID);
275 
276  #ifdef USE_TOUCHSCREEN
277  if(pMsg->type == TYPE_TOUCHSCREEN)
278  {
279  if(pMsg->uiEvent == EVENT_PRESS)
280  {
281 
282  // Check if it falls in the radio button borders
283  if
284  (
285  (pRb->hdr.left < pMsg->param1) &&
286  (pRb->hdr.right > pMsg->param1) &&
287  (pRb->hdr.top < pMsg->param2) &&
288  (pRb->hdr.bottom > pMsg->param2)
289  )
290  {
291  if(!GetState(pRb, RB_CHECKED))
292  return (RB_MSG_CHECKED);
293  }
294  }
295 
296  return (OBJ_MSG_INVALID);
297  }
298 
299  #endif
300  #ifdef USE_KEYBOARD
301  if(pMsg->type == TYPE_KEYBOARD)
302  {
303  if((WORD)pMsg->param1 == pRb->hdr.ID)
304  {
305  if(pMsg->uiEvent == EVENT_KEYSCAN)
306  {
307  if((pMsg->param2 == SCAN_SPACE_PRESSED) || (pMsg->param2 == SCAN_CR_PRESSED))
308  {
309  if(!GetState(pRb, RB_CHECKED))
310  return (RB_MSG_CHECKED);
311  }
312  }
313  }
314 
315  return (OBJ_MSG_INVALID);
316  }
317 
318  #endif
319  return (OBJ_MSG_INVALID);
320 }

410  }

416  pRb->hdr.right,
417  pRb->hdr.bottom);
418 #endif
419  return (1);
420  }
421 
422  case DRAW_BUTTON0:
423  if(!GetState(pRb, RB_DISABLED))
424  {
425  faceClr = pRb->hdr.pGolScheme->Color0;
426  }
427  else
