92  Remarks:
93  This function is called only one during lifetime of the application.

95 void TickInit(void)
96 {
97 #if defined(__18CXX)
98  // Use Timer0 for 8 bit processors
99  // Initialize the time
100  TMR0H = 0;
101  TMR0L = 0;
102 
103  // Set up the timer interrupt
104  INTCON2bits.TMR0IP = 0; // Low priority
105  INTCONbits.TMR0IF = 0;
106  INTCONbits.TMR0IE = 1; // Enable interrupt
107 
108  // Timer0 on, 16-bit, internal timer, 1:256 prescalar
109  T0CON = 0x87;
110 
111 #else
112  // Use Timer 1 for 16-bit and 32-bit processors
113  // 1:256 prescale
114  T1CONbits.TCKPS = 3;
115  // Base
116  PR1 = 0xFFFF;
117  // Clear counter
118  TMR1 = 0;
119 
120  // Enable timer interrupt
121  #if defined(__C30__)
122  IPC0bits.T1IP = 2; // Interrupt priority 2 (low)
123  IFS0bits.T1IF = 0;
124  IEC0bits.T1IE = 1;
125  #else
126  IPC1bits.T1IP = 2; // Interrupt priority 2 (low)
127  IFS0CLR = _IFS0_T1IF_MASK;
128  IEC0SET = _IEC0_T1IE_MASK;
129  #endif
130 
131  // Start timer
132  T1CONbits.TON = 1;
133 #endif
134 }

139

141  Reads the tick value.
142 
143  Description:
144  This function performs an interrupt-safe and synchronized read of the

146 
147  Precondition:
148  None
149 
150  Parameters:
151  None
152 
153  Returns:
154  None
155  ***************************************************************************/
156 static void GetTickCopy(void)
157 {
158  // Perform an Interrupt safe and synchronized read of the 48-bit
159  // tick value
160 #if defined(__18CXX)
161  do
162  {
163  INTCONbits.TMR0IE = 1; // Enable interrupt
164  Nop();
165  INTCONbits.TMR0IE = 0; // Disable interrupt
166  vTickReading[0] = TMR0L;
167  vTickReading[1] = TMR0H;
168  *((DWORD*)&vTickReading[2]) = dwInternalTicks;
169  } while(INTCONbits.TMR0IF);
170  INTCONbits.TMR0IE = 1; // Enable interrupt
171 #elif defined(__C30__)
172  do
173  {
174  DWORD dwTempTicks;
175  WORD_VAL wTemp;
176 
177  IEC0bits.T1IE = 1; // Enable interrupt
178  Nop();
179  IEC0bits.T1IE = 0; // Disable interrupt
180 
181  // Get low 2 bytes
182  wTemp.Val = TMR1;
183  vTickReading[0] = wTemp.v[0];
184  vTickReading[1] = wTemp.v[1];
185  //((WORD*)vTickReading)[0] = TMR1;
186 
187  // Correct corner case where interrupt increments byte[4+] but
188  // TMR1 hasn't rolled over to 0x0000 yet
189  dwTempTicks = dwInternalTicks;
190  //if(((WORD*)vTickReading)[0] == 0xFFFFu)
191  if(wTemp.Val == 0xFFFFu)
192  dwTempTicks--;
193 
194  // Get high 4 bytes
195  vTickReading[2] = ((BYTE*)&dwTempTicks)[0];
196  vTickReading[3] = ((BYTE*)&dwTempTicks)[1];
197  vTickReading[4] = ((BYTE*)&dwTempTicks)[2];
198  vTickReading[5] = ((BYTE*)&dwTempTicks)[3];
199  } while(IFS0bits.T1IF);
200  IEC0bits.T1IE = 1; // Enable interrupt
201 #else // PIC32
202  do
203  {
204  DWORD dwTempTicks;
205  WORD_VAL wTemp;
206 
207  IEC0SET = _IEC0_T1IE_MASK; // Enable interrupt
208  Nop();
209  IEC0CLR = _IEC0_T1IE_MASK; // Disable interrupt
210 
211  // Get low 2 bytes
212  wTemp.Val = TMR1;
213  vTickReading[0] = wTemp.v[0];
214  vTickReading[1] = wTemp.v[1];
215  //((volatile WORD*)vTickReading)[0] = TMR1;
216 
217  // Correct corner case where interrupt increments byte[4+] but
218  // TMR1 hasn't rolled over to 0x0000 yet
219  dwTempTicks = dwInternalTicks;
220 
221  // PIC32MX3XX/4XX devices trigger the timer interrupt when TMR1 == PR1
222  // (TMR1 prescalar is 0x00), requiring us to undo the ISR's increment
223  // of the upper 32 bits of our 48 bit timer in the special case when
224  // TMR1 == PR1 == 0xFFFF. For other PIC32 families, the ISR is
225  // triggered when TMR1 increments from PR1 to 0x0000, making no special
226  // corner case.
227  #if __PIC32_FEATURE_SET__ <= 460
228  //if(((WORD*)vTickReading)[0] == 0xFFFFu)
229  if(wTemp.Val == 0xFFFFu)
230  dwTempTicks--;
231  #elif !defined(__PIC32_FEATURE_SET__)
232  #error __PIC32_FEATURE_SET__ macro must be defined. You need to download a newer C32 compiler version.
233  #endif
234 
235  // Get high 4 bytes
236  vTickReading[2] = ((BYTE*)&dwTempTicks)[0];
237  vTickReading[3] = ((BYTE*)&dwTempTicks)[1];
238  vTickReading[4] = ((BYTE*)&dwTempTicks)[2];
239  vTickReading[5] = ((BYTE*)&dwTempTicks)[3];
240  } while(IFS0bits.T1IF);
241  IEC0SET = _IEC0_T1IE_MASK; // Enable interrupt
242 #endif
243 }

246 /*****************************************************************************
247  Function:
248  DWORD TickGet(void)
249 
250  Summary:
251  Obtains the current Tick value.
252 
253  Description:
254  This function retrieves the current Tick value, allowing timing and
255  measurement code to be written in a non-blocking fashion. This function
256  retrieves the least significant 32 bits of the internal tick counter,
257  and is useful for measuring time increments ranging from a few
258  microseconds to a few hours. Use TickGetDiv256 or TickGetDiv64K for
259  longer periods of time.
260 
261  Precondition:
262  None
263 
264  Parameters:
265  None
266 
267  Returns:
268  Lower 32 bits of the current Tick value.
269  ***************************************************************************/
270 DWORD TickGet(void)
271 {
272  DWORD dw;
273 
274  GetTickCopy();
275  ((BYTE*)&dw)[0] = vTickReading[0]; // Note: This copy must be done one
276  ((BYTE*)&dw)[1] = vTickReading[1]; // byte at a time to prevent misaligned
277  ((BYTE*)&dw)[2] = vTickReading[2]; // memory reads, which will reset the PIC.
278  ((BYTE*)&dw)[3] = vTickReading[3];
279  return dw;
280 }

360 
361  Summary:
362  Converts a Tick value or difference to milliseconds.
363 
364  Description:
365  This function converts a Tick value or difference to milliseconds. For
366  example, TickConvertToMilliseconds(32768) returns 1000 when a 32.768kHz
367  clock with no prescaler drives the Tick module interrupt.
368 
369  Precondition:
370  None
371 
372  Parameters:
373  dwTickValue - Value to convert to milliseconds
374 
375  Returns:
376  Input value expressed in milliseconds.
377 
378  Remarks:
379  This function performs division on DWORDs, which is slow. Avoid using
380  it unless you absolutely must (such as displaying data to a user). For
381  timeout comparisons, compare the current value to a multiple or fraction
382  of TICK_SECOND, which will be calculated only once at compile time.
383  ***************************************************************************/
384 DWORD TickConvertToMilliseconds(DWORD dwTickValue)
385 {
386  return (dwTickValue+(TICKS_PER_SECOND/2000ul))/((DWORD)(TICKS_PER_SECOND/1000ul));
387 }
