353 void USBDevicePHDCSendData(UINT8 qos, UINT8 *data, UINT16 length,BOOL memtype)
354 {
355  PTR_PHDC_TX_ENDPOINT tx_endpoint;
356  UINT8 index;
357  UINT8 i;
358 
359  for(index=0; index< PHDC_TX_ENDPOINTS;index++)
360  {
361  if((PhdcTXEP[index].qos & qos) != 0)
362  break;
363  }
364  if(index == PHDC_TX_ENDPOINTS)
365  {
366  return; //no endpoint supports the qos
367  }
368  tx_endpoint = &PhdcTXEP[index];
369 
370  if( (tx_endpoint->transfer_size ==0) && (tx_endpoint->offset ==0) ) //new packet
371  {
372  tx_endpoint->transfer_size = length;
373  tx_endpoint->memtype= memtype;
374  if(tx_endpoint->transfer_size > tx_endpoint->size)
375  {
376  tx_endpoint->bytes_to_send = tx_endpoint->size; //multiple send
377  }
378  else
379  {
380  tx_endpoint->bytes_to_send = tx_endpoint->transfer_size; //only packet to send
381  }
382  }
383  else
384  {
385  return; //send in progress
386  }
387  USBMaskInterrupts();
388  tx_endpoint->app_buff = (UINT8*)data;
389  i= tx_endpoint->bytes_to_send;
390  pPHDCDst.bRam = (BYTE*)phdc_data_tx; // Set destination pointer
391  if(tx_endpoint->memtype == MEM_ROM) // Determine type of memory source
392  {
393  pPHDCSrc.bRom = (ROM UINT8 *) data;
394  while(i)
395  {
396  *pPHDCDst.bRam = *pPHDCSrc.bRom;
397  pPHDCDst.bRam++;
398  pPHDCSrc.bRom++;
399  i--;
400  }//end while(byte_to_send)
401  }
402  else // _RAM
403  {
404  pPHDCSrc.bRam = (UINT8*)data;

407  *pPHDCDst.bRam = *pPHDCSrc.bRam;
408  pPHDCDst.bRam++;
409  pPHDCSrc.bRam++;
410  i--;
411  }//end while(byte_to_send._word)
412  }//end if(phdc_mem_type...)
413 
414  tx_endpoint->PHDCDataInHandle = USBTxOnePacket(tx_endpoint->ep_num,
415  (BYTE*)phdc_data_tx,tx_endpoint->bytes_to_send);
416  USBDevicePHDCUpdateStatus(tx_endpoint->ep_num, 1); // update the endpoint status. '1' if endpoint has data. '0' otherwise.
417  USBUnmaskInterrupts();
418 }//end USBDevicePHDCSendData
