480  //
481  // If MTOUCH_INTEGRATION_TYPE is defined as MTOUCH_CONTROLS_ISR, the framework will implement
482  // a dedicated ISR for the mTouch scans' use. If it is not defined, the application may implement
483  // its own ISR.
484  //
485  // A few rules must be followed by custom ISR functions:
486  //
487  // 1. If MTOUCH_INTEGRATION_TYPE is defined as MTOUCH_CALLED_FROM_MAINLOOP, you must set
488  // mTouch_state.isrServiced each time you enter the ISR. This tells the framework that the scan
489  // was interrupted and needs to be repeated.
490  //
491  // 2. If MTOUCH_INTEGRATION_TYPE is defined as MTOUCH_CALLED_FROM_ISR, the example API usage below
492  // is required to service mTouch scanning.
493 
494  #if defined(MCOMM_ENABLED) && defined(MCOMM_TWO_WAY_ENABLED)
495  // If the V2 comms have been enabled, we need to receive incoming requests.
496  #if (MCOMM_TYPE == MCOMM_UART_TWO_WAY)
497  if (MCOMM_UART_RCIE && MCOMM_UART_RCIF) // Check UART for new data
498  #elif (MCOMM_TYPE == MCOMM_I2C_TWO_WAY)
499  if (MCOMM_I2C_SSPIE && MCOMM_I2C_SSPIF) // Check I2C for new data
500  #elif (MCOMM_TYPE == MCOMM_SPI_TWO_WAY)
501  if (MCOMM_SPI_SSPIE && MCOMM_SPI_SSPIF) // Check SPI for new data
