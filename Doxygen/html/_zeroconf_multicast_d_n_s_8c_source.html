967 /***************************************************************
968  Function:
969  size_t mDNSSDFormatServiceInstance(BYTE *string, size_t strSize )
970 
971  Summary:
972  Formats the Service-Instance name according to DNS-SD standard
973  specification.
974 
975  Description:
976  This function is used to format the Service-Instance name, if
977  it contains 'dots' and 'backslashes'
978 
979  As the service-instance name will be merged with service-type &
980  to distinguish the 'dots' seperating the service-type words and
981  'dots' within service-instance name, the 'dots' within service-
982  instance name will be replaced with '\.' in place of '.' Even the
983  '\' are replaced with '\\'.
984 
985  When the resource-record containing service-instance name is
986  pushed out, the formatted dots '\.' are sentout as '.' and the
987  'dots' sperating the service-type & service-instances are replaced
988  with length bytes, as specified in RFC 1035.
989 
990  Precondition:
991  None
992 
993  Parameters:
994  String - Service-Instance name to be formatted
995  strSize - available size for the formatted string, not to be exceeded
996 
997  Returns:
998  size of the formatted string
999  **************************************************************/
1000 MDNS_STATIC size_t mDNSSDFormatServiceInstance(BYTE *string, size_t strSize )
1001 {
1002  BYTE *temp;
1003  BYTE output[MAX_LABEL_SIZE];
1004  BYTE i;
1005  BYTE *right_ptr,*str_token;
1006  BYTE len;
1007 
1008  temp = output;
1009  right_ptr = string;
1010  str_token = string;
1011  while(1)
1012  {
1013  do
1014  {
1015  i = *right_ptr++;
1016  } while((i != 0x00u) && (i != '\\') && (i != '.') );
1017 
1018 
1019  /* Prefix '\' for every occurance of '.' & '\' */
1020  len = (BYTE)(right_ptr-str_token-1);
1021 
1022  memcpy(temp,str_token,len);
1023  temp += len;
1024  str_token += len;
1025  if(i == '.' || i == '\\')
1026  {
1027  *temp = '\\';
1028  temp++;
1029  *temp++ = i;
1030  str_token += 1;
1031 
1032  }
1033  else if(i == 0x00u || i == '/' || i == ',' || i == '>')
1034  break;
1035 
1036  }
1037  *temp++ = '\0';
1038  return strncpy_m((char *)string, strSize, 1, output);
1039 }

1043  void mDNSSDFillResRecords(mdnsd_struct *sd)
1044

1047  sd structure-instance, in which the information is filled from

1050  Description:

1060  resource-reocrds.
1061  1) PTR Resource-Record: This is a shared record, with service-type
1062  as rr-name and fully-qualified name as
1063  rr-data.
1064  2) SRV Resource-Record: This is a unique record, with fully-
1065  qualified name as rr-name and Host-name,
1066  port-num as rr-data.
1067  3) TXT Resource-Record: This is a unique record, with fully-
1068  qualified name as rr-name and additional
1069  information as rr-data like default-page
1070  name (For ex: "/index.htm")
1071 
1072  Precondition:
1073  None
1074 
1075  Parameters:
1076  sd - Service-Discovery structure instance for which Resource-
1077  records to be filled.
1078 
1079  Returns:
1080  None
1081  **************************************************************/

1083 {

1092  memset(&(gResponderCtx.rr_list[QTYPE_TXT_INDEX]),0,(sizeof(mDNSResourceRecord)));
1093 
1094  /* Formatting Service-Instance name.
1095  * And preparing a fully qualified
1096  * Service-instance name. */
1097  strncpy((char *)sd->sd_qualified_name, (char *)sd->srv_name, sizeof(sd->sd_qualified_name));
1098  qual_len= mDNSSDFormatServiceInstance(sd->sd_qualified_name, sizeof(sd->sd_qualified_name));
1099 
1100  // SOFTAP_ZEROCONF_SUPPORT

1599 /***************************************************************
1600  Function:
1601  static WORD mDNSDeCompress(WORD wPos,
1602  BYTE *pcString,
1603  BOOL bFollowPtr,
1604  BYTE cElement,
1605  BYTE cDepth)
1606 
1607  Summary:
1608  Read a string from a resource record, from the Multicast-DNS socket buffer.
1609 
1610  Description:
1611  This function reads a string to the Multicast-DNS socket,
1612  ensuring that it is properly formatted.
1613 
1614  String may be reconstructed through decompression if necessary.
1615  Decompression pointer traversal is done in place, recursively, in UDP's RxBuffer.
1616 
1617  cDepth represents the recursion depth, for debugging purpose.
1618 
1619  cElement represents the number of elements in the string. For example,

1626 
1627  For DNS message compression format, see RFC 1035, section 4.1.4.
1628 
1629  Precondition:
1630  UDP socket is obtained and ready for writing.
1631  wPos correctly reflect the current position in the UDP RxBuffer.
1632 
1633  Parameters:
1634  String - the string to write to the UDP socket.
1635 
1636  Returns:
1637  Number of bytes in THIS resource record field (in RFC 1035's term, NAME or RDATA).
1638  UDP RxBuffer pointer is repositioned to the place right after THIS resource record field.
1639 
1640  **************************************************************/
1641 
1642 MDNS_STATIC WORD mDNSDeCompress(WORD wPos, BYTE *pcString, BOOL bFollowPtr, BYTE cElement, BYTE cDepth)
1643 {
1644  WORD rr_len = 0; // As is in the packet. Could be in compressed format.
1645  WORD startOffset, endOffset, currOffset;
1646  BYTE i, tmp;
1647  WORD offset_in_ptr;
1648  WORD len;
1649  BYTE substr_len;
1650 
1651  currOffset = startOffset = wPos;
1652 
1653  while (1)
1654  {
1655  rr_len++;
1656  if(!UDPGet(&substr_len))
1657  goto mDNSDeCompress_done;
1658 
1659  if(substr_len == 0u)
1660  {
1661  if (pcString)
1662  {
1663  *pcString++ = '\0';
1664  }
1665  goto mDNSDeCompress_done;
1666  }
1667 
1668  if((substr_len & 0xC0) == 0xC0) // b'11 at MSb indicates compression ptr
1669  {
1670  offset_in_ptr = substr_len & 0x3F; // the rest of 6 bits is part of offset_in_ptr.
1671  offset_in_ptr = offset_in_ptr << 8;
1672 
1673  /* Remove label-ptr byte */
1674  rr_len++;
1675  UDPGet(&i);
1676  offset_in_ptr += i;
1677 
1678  if (bFollowPtr)
1679  {
1680  cDepth++;
1681 
1682  DEBUG_MDNS_MESG(zeroconf_dbg_msg, "follow ptr: h'%X, ", offset_in_ptr);
1683  DEBUG_MDNS_PRINT(zeroconf_dbg_msg);
1684  DEBUG_MDNS_MESG(zeroconf_dbg_msg, "%d\r\n", cDepth);
1685  DEBUG_MDNS_PRINT(zeroconf_dbg_msg);
1686 
1687  UDPSetRxBuffer(offset_in_ptr);
1688  len = mDNSDeCompress(offset_in_ptr, pcString, bFollowPtr, cElement, cDepth);
1689 
1690  // compressed ptr is always the last element
1691  goto mDNSDeCompress_done;
1692  }
1693 
1694  goto mDNSDeCompress_done;
1695  }
1696  else
1697  {
1698  if (pcString)
1699  {
1700  if (cElement > 0)
1701  {
1702  // not the first element in name
1703  *pcString++ = '.';
1704  }
1705 
1706  UDPGetArray(pcString, substr_len);
1707  pcString += substr_len;
1708  }
1709  else
1710  {
1711  i = substr_len;
1712  while (i--)
1713  {
1714  UDPGet(&tmp);
1715  }
1716  }
1717 
1718  cElement++;
1719  rr_len += substr_len;
1720  }
1721  }
1722 
1723 mDNSDeCompress_done:
1724 
1725  endOffset = startOffset + rr_len;
1726  UDPSetRxBuffer(endOffset);
1727 
1728  return rr_len;
1729 }
