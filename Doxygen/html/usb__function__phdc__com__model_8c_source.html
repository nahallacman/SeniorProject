345  /******************************************************************************
346  * Function:
347  * void PHDConnect(void)
348  *
349  * Summary:
350  * This function is used to connect to the PHD Manager.
351  *
352  * Description:
353  * This function initiates connection to the PHD Manager by sending an
354  * Association request to manager. The Agent doesn't get connected to
355  * the Manager immediately after calling this function. Upon receiving
356  * the association request from an Agent, the PHD Manager responds with
357  * an association response. The association response tells whether Manager
358  * accepting the request or rejecting it. The Association response from
359  * the Manager is handled by the PHD stack. The PHD stack calls a callback
360  * function (void(* PHDC_APP_CB)(UINT8)) to the application with status of
361  * the connection.
362  * The Manager should respond to the Agent within the specified timeout of
363  * ASSOCIATION_REQUEST_TIMEOUT. The Agent should send the Association request
364  * once more if no response is received from Manager and ASSOCIATION_REQUEST_TIMEOUT
365  * is expired. This function starts a Timer for the Association Timeout request.
366  * The timeout is handled by the PHDTimeoutHandler() function.
367  *
368  * Conditions:
369  * The agent should be in PHD_INITIALIZED state.
370  *
371  * Parameters:
372  * None
373  *
374  * Return:
375  * None
376  *
377  * Side Effects:
378  * None
379  *
380  * Remarks:
381  * None
382  *
383  *****************************************************************************/
384 void PHDConnect(void)
385 {
386  if(PhdComState == PHD_COM_STATE_UNASSOCIATED)
387  {
388  PhdComState= PHD_COM_STATE_ASSOCIATING;
389  USBDevicePHDCSendData(PHDC_BULK_IN_QOS,(UINT8 *) ASSOCIATION_REQUEST,ASSOCIATION_REQUEST_SIZE,MEM_ROM);
390  USBDevicePHDCReceiveData(PHDC_BULK_OUT_QOS,pPhdAppBuffer,0); //get ready to receive
391 
392  PhdAssociationRequestTimeoutStatus = TIMEOUT_ENABLED;
393  PhdAssociationRequestTimeout = ASSOCIATION_REQUEST_TIMEOUT;
394  PhdAssociationRequestRetry = ASSOCIATION_REQUEST_RETRY;

641  }
642  else

647  }

650  ||(data.Val == ROIV_CMIP_SET_CHOSEN)
651  ||(data.Val == ROIV_CMIP_CONFIRMED_SET_CHOSEN)
652  ||(data.Val == ROIV_CMIP_ACTION_CHOSEN))
653  //||(data.Val == ROIV_CMIP_CONFIRMED_ACTION_CHOSEN))
654  {
655  //respond with ROER error.
656  PHDSendRoerMessageToManager( invoke_id.Val, NO_SUCH_OBJECT_INSTANCE);
657  }
658  else
659  {
660  PHDSendAbortRequestToManager(ABORT_REASON_UNDEFINED);
661  }
662  }
663  else if (PhdComState==PHD_COM_STATE_ASSOC_OPERATING)
664  {
665  if(data.Val == ROIV_CMIP_GET_CHOSEN) //Get command from Manager
666  {
667  handle.byte.HB = apdu_val[12];
668  handle.byte.LB = apdu_val[13];
669  if(handle.Val == 0)
670  {
671  PHDMdsAttributesRequestHandler(apdu_val);
672  }
673  else
674  {
675  //Send ROER error.

677  }
