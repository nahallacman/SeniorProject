221  #endif

223  ENC_SPICON1bits.CKE = 1;
224  ENC_SPICON1bits.MSTEN = 1;

235  // Also wait for CLKRDY to become set.
236  // Bit 3 in ESTAT is an unimplemented bit. If it reads out as '1' that

426 /******************************************************************************
427  * Function: void MACDiscardRx(void)
428  *
429  * PreCondition: None
430  *
431  * Input: None
432  *
433  * Output: None
434  *
435  * Side Effects: None
436  *
437  * Overview: Marks the last received packet (obtained using
438  * MACGetHeader())as being processed and frees the buffer
439  * memory associated with it
440  *
441  * Note: Is is safe to call this function multiple times between
442  * MACGetHeader() calls. Extra packets won't be thrown away
443  * until MACGetHeader() makes it available.

445 void MACDiscardRx(void)
446 {
447  WORD_VAL NewRXRDLocation;
448 
449  // Make sure the current packet was not already discarded
450  if(WasDiscarded)
451  return;
452  WasDiscarded = TRUE;
453 
454  // Decrement the next packet pointer before writing it into
455  // the ERXRDPT registers. This is a silicon errata workaround.
456  // RX buffer wrapping must be taken into account if the
457  // NextPacketLocation is precisely RXSTART.
458  NewRXRDLocation.Val = NextPacketLocation.Val - 1;
459  if(NewRXRDLocation.Val > RXSTOP)
460  {
461  NewRXRDLocation.Val = RXSTOP;
462  }
463 
464  // Decrement the RX packet counter register, EPKTCNT
465  BFSReg(ECON2, ECON2_PKTDEC);
466 
467  // Move the receive read pointer to unwrite-protect the memory used by the
468  // last packet. The writing order is important: set the low byte first,
469  // high byte last.
470  WriteReg(ERXRDPTL, NewRXRDLocation.v[0]);
471  WriteReg(ERXRDPTH, NewRXRDLocation.v[1]);
472 }

475 /******************************************************************************
476  * Function: WORD MACGetFreeRxSize(void)
477  *
478  * PreCondition: None
479  *
480  * Input: None
481  *
482  * Output: A WORD estimate of how much RX buffer space is free at
483  * the present time.
484  *
485  * Side Effects: None
486  *
487  * Overview: None
488  *
489  * Note: None
490  *****************************************************************************/
491 WORD MACGetFreeRxSize(void)
492 {
493  WORD_VAL ReadPT, WritePT;
494 
495  // Read the Ethernet hardware buffer write pointer. Because packets can be
496  // received at any time, it can change between reading the low and high
497  // bytes. A loop is necessary to make certain a proper low/high byte pair
498  // is read.
499  BankSel(EPKTCNT);
500  do {
501  // Save EPKTCNT in a temporary location
502  ReadPT.v[0] = ReadETHReg((BYTE)EPKTCNT).Val;
503 
504  BankSel(ERXWRPTL);
505  WritePT.v[0] = ReadETHReg(ERXWRPTL).Val;
506  WritePT.v[1] = ReadETHReg(ERXWRPTH).Val;
507 
508  BankSel(EPKTCNT);
509  } while(ReadETHReg((BYTE)EPKTCNT).Val != ReadPT.v[0]);
510 
511  // Determine where the write protection pointer is
512  BankSel(ERXRDPTL);
513  ReadPT.v[0] = ReadETHReg(ERXRDPTL).Val;
514  ReadPT.v[1] = ReadETHReg(ERXRDPTH).Val;
515 
516  // Calculate the difference between the pointers, taking care to account
517  // for buffer wrapping conditions
518  if(WritePT.Val > ReadPT.Val)
519  {
520  return (RXSTOP - RXSTART) - (WritePT.Val - ReadPT.Val);
521  }
522  else if(WritePT.Val == ReadPT.Val)
523  {
524  return RXSIZE - 1;
525  }
526  else
527  {
528  return ReadPT.Val - WritePT.Val - 1;
529  }
530 }

532 /******************************************************************************
533  * Function: BOOL MACGetHeader(MAC_ADDR *remote, BYTE* type)
534  *
535  * PreCondition: None
536  *
537  * Input: *remote: Location to store the Source MAC address of the
538  * received frame.
539  * *type: Location of a BYTE to store the constant
540  * MAC_UNKNOWN, ETHER_IP, or ETHER_ARP, representing
541  * the contents of the Ethernet type field.
542  *
543  * Output: TRUE: If a packet was waiting in the RX buffer. The
544  * remote, and type values are updated.
545  * FALSE: If a packet was not pending. remote and type are
546  * not changed.
547  *
548  * Side Effects: Last packet is discarded if MACDiscardRx() hasn't already
549  * been called.
550  *
551  * Overview: None
552  *
553  * Note: None
554  *****************************************************************************/
555 BOOL MACGetHeader(MAC_ADDR *remote, BYTE* type)
556 {
557  ENC_PREAMBLE header;
558  BYTE PacketCount;
559 
560  // Test if at least one packet has been received and is waiting
561  BankSel(EPKTCNT);
562  PacketCount = ReadETHReg((BYTE)EPKTCNT).Val;
563  BankSel(ERDPTL);
564  if(PacketCount == 0u)
565  return FALSE;
566 
567  // Make absolutely certain that any previous packet was discarded
568  if(WasDiscarded == FALSE)
569  {
570  MACDiscardRx();
571  return FALSE;
572  }
573 
574  // Set the SPI read pointer to the beginning of the next unprocessed packet
575  CurrentPacketLocation.Val = NextPacketLocation.Val;
576  WriteReg(ERDPTL, CurrentPacketLocation.v[0]);
577  WriteReg(ERDPTH, CurrentPacketLocation.v[1]);
578 
579  // Obtain the MAC header from the Ethernet buffer
580  MACGetArray((BYTE*)&header, sizeof(header));
581 
582  // The EtherType field, like most items transmitted on the Ethernet medium
583  // are in big endian.
584  header.Type.Val = swaps(header.Type.Val);
585 
586  // Validate the data returned from the ENC28J60. Random data corruption,
587  // such as if a single SPI bit error occurs while communicating or a
588  // momentary power glitch could cause this to occur in rare circumstances.
589  if(header.NextPacketPointer > RXSTOP || ((BYTE_VAL*)(&header.NextPacketPointer))->bits.b0 ||
590  header.StatusVector.bits.Zero ||
591  header.StatusVector.bits.CRCError ||
592  header.StatusVector.bits.ByteCount > 1518u ||
593  !header.StatusVector.bits.ReceiveOk)
594  {
595  Reset();
596  }
597 
598  // Save the location where the hardware will write the next packet to
599  NextPacketLocation.Val = header.NextPacketPointer;
600 
601  // Return the Ethernet frame's Source MAC address field to the caller
602  // This parameter is useful for replying to requests without requiring an
603  // ARP cycle.
604  memcpy((void*)remote->v, (void*)header.SourceMACAddr.v, sizeof(*remote));
605 
606  // Return a simplified version of the EtherType field to the caller

669 /******************************************************************************
670  * Function: void MACFlush(void)
671  *
672  * PreCondition: A packet has been created by calling MACPut() and
673  * MACPutHeader().
674  *
675  * Input: None
676  *
677  * Output: None
678  *
679  * Side Effects: None
680  *
681  * Overview: MACFlush causes the current TX packet to be sent out on
682  * the Ethernet medium. The hardware MAC will take control
683  * and handle CRC generation, collision retransmission and
684  * other details.
685  *
686  * Note: After transmission completes (MACIsTxReady() returns TRUE),
687  * the packet can be modified and transmitted again by calling
688  * MACFlush() again. Until MACPutHeader() or MACPut() is
689  * called (in the TX data area), the data in the TX buffer
690  * will not be corrupted.
691  *****************************************************************************/
692 void MACFlush(void)
693 {
694  // Reset transmit logic if a TX Error has previously occured
695  // This is a silicon errata workaround
696  BFSReg(ECON1, ECON1_TXRST);
697  BFCReg(ECON1, ECON1_TXRST);
698  BFCReg(EIR, EIR_TXERIF | EIR_TXIF);
699 
700  // Start the transmission
701  // After transmission completes (MACIsTxReady() returns TRUE), the packet
702  // can be modified and transmitted again by calling MACFlush() again.
703  // Until MACPutHeader() is called, the data in the TX buffer will not be
704  // corrupted.
705  BFSReg(ECON1, ECON1_TXRTS);
706 
707  // Revision B5 and B7 silicon errata workaround
708  if(ENCRevID == 0x05u || ENCRevID == 0x06u)
709  {
710  WORD AttemptCounter = 0x0000;
711  while(!(ReadETHReg(EIR).Val & (EIR_TXERIF | EIR_TXIF)) && (++AttemptCounter < 1000u));
712  if(ReadETHReg(EIR).EIRbits.TXERIF || (AttemptCounter >= 1000u))
713  {
714  WORD_VAL ReadPtrSave;
715  WORD_VAL TXEnd;
716  TXSTATUS TXStatus;
717  BYTE i;
718 
719  // Cancel the previous transmission if it has become stuck set
720  BFCReg(ECON1, ECON1_TXRTS);
721 
722  // Save the current read pointer (controlled by application)
723  ReadPtrSave.v[0] = ReadETHReg(ERDPTL).Val;
724  ReadPtrSave.v[1] = ReadETHReg(ERDPTH).Val;
725 
726  // Get the location of the transmit status vector
727  TXEnd.v[0] = ReadETHReg(ETXNDL).Val;
728  TXEnd.v[1] = ReadETHReg(ETXNDH).Val;
729  TXEnd.Val++;
730 
731  // Read the transmit status vector
732  WriteReg(ERDPTL, TXEnd.v[0]);
733  WriteReg(ERDPTH, TXEnd.v[1]);
734  MACGetArray((BYTE*)&TXStatus, sizeof(TXStatus));
735 
736  // Implement retransmission if a late collision occured (this can
737  // happen on B5 when certain link pulses arrive at the same time
738  // as the transmission)
739  for(i = 0; i < 16u; i++)
740  {
741  if(ReadETHReg(EIR).EIRbits.TXERIF && TXStatus.bits.LateCollision)
742  {
743  // Reset the TX logic
744  BFSReg(ECON1, ECON1_TXRST);
745  BFCReg(ECON1, ECON1_TXRST);
746  BFCReg(EIR, EIR_TXERIF | EIR_TXIF);
747 
748  // Transmit the packet again
749  BFSReg(ECON1, ECON1_TXRTS);
750  while(!(ReadETHReg(EIR).Val & (EIR_TXERIF | EIR_TXIF)));
751 
752  // Cancel the previous transmission if it has become stuck set
753  BFCReg(ECON1, ECON1_TXRTS);
754 
755  // Read transmit status vector
756  WriteReg(ERDPTL, TXEnd.v[0]);
757  WriteReg(ERDPTH, TXEnd.v[1]);
758  MACGetArray((BYTE*)&TXStatus, sizeof(TXStatus));
759  }
760  else
761  {
762  break;
763  }
764  }
765 
766  // Restore the current read pointer
767  WriteReg(ERDPTL, ReadPtrSave.v[0]);
768  WriteReg(ERDPTH, ReadPtrSave.v[1]);
769  }
770  }

921 /******************************************************************************
922  * Function: WORD CalcIPBufferChecksum(WORD len)
923  *
924  * PreCondition: Read buffer pointer set to starting of checksum data
925  *
926  * Input: len: Total number of bytes to calculate the checksum over.
927  * The first byte included in the checksum is the byte
928  * pointed to by ERDPT, which is updated by calls to
929  * MACSetReadPtr(), MACGet(), MACGetArray(),
930  * MACGetHeader(), etc.
931  *
932  * Output: 16-bit checksum as defined by RFC 793
933  *
934  * Side Effects: None
935  *
936  * Overview: This function performs a checksum calculation in the MAC
937  * buffer itself. The ENC28J60 has a hardware DMA module
938  * which can calculate the checksum faster than software, so
939  * this function replaces the CaclIPBufferChecksum() function
940  * defined in the helpers.c file. Through the use of
941  * preprocessor defines, this replacement is automatic.
942  *
943  * Note: This function works either in the RX buffer area or the TX
944  * buffer area. No validation is done on the len parameter.
945  *****************************************************************************/
946 WORD CalcIPBufferChecksum(WORD len)
947 {
948  WORD_VAL Start;
949  DWORD_VAL Checksum = {0x00000000ul};
950  WORD ChunkLen;
951  WORD DataBuffer[10];
952  WORD *DataPtr;
953 
954  // Save the SPI read pointer starting address
955  Start.v[0] = ReadETHReg(ERDPTL).Val;
956  Start.v[1] = ReadETHReg(ERDPTH).Val;
957 
958  while(len)
959  {
960  // Obtain a chunk of data (less SPI overhead compared
961  // to requesting one byte at a time)
962  ChunkLen = len > sizeof(DataBuffer) ? sizeof(DataBuffer) : len;
963  MACGetArray((BYTE*)DataBuffer, ChunkLen);
964 
965  len -= ChunkLen;
966 
967  // Take care of a last odd numbered data byte
968  if(((WORD_VAL*)&ChunkLen)->bits.b0)
969  {
970  ((BYTE*)DataBuffer)[ChunkLen] = 0x00;
971  ChunkLen++;
972  }
973 
974  // Calculate the checksum over this chunk
975  DataPtr = DataBuffer;
976  while(ChunkLen)
977  {
978  Checksum.Val += *DataPtr++;
979  ChunkLen -= 2;
980  }
981  }
982 
983  // Restore old read pointer location
984  WriteReg(ERDPTL, Start.v[0]);
985  WriteReg(ERDPTH, Start.v[1]);
986 
987  // Do an end-around carry (one's complement arrithmatic)
988  Checksum.Val = (DWORD)Checksum.w[0] + (DWORD)Checksum.w[1];
989 
990  // Do another end-around carry in case if the prior add
991  // caused a carry out
992  Checksum.w[0] += Checksum.w[1];
993 
994  // Return the resulting checksum
995  return ~Checksum.w[0];
996 }

1179 /******************************************************************************
1180  * Function: WORD MACGetArray(BYTE *val, WORD len)
1181  *
1182  * PreCondition: SPI bus must be initialized (done in MACInit()).
1183  * ERDPT must point to the place to read from.
1184  *

1186  * len: Number of bytes to read from the data buffer.
1187  *
1188  * Output: Byte(s) of data read from the data buffer.
1189  *
1190  * Side Effects: None
1191  *
1192  * Overview: Burst reads several sequential bytes from the data buffer
1193  * and places them into local memory. With SPI burst support,
1194  * it performs much faster than multiple MACGet() calls.
1195  * ERDPT is incremented after each byte, following the same
1196  * rules as MACGet().
1197  *
1198  * Note: None
1199  *****************************************************************************/
1200 WORD MACGetArray(BYTE *val, WORD len)
1201 {

1203 // with I2C temperature sensor on the same SPI wires
1204 #if defined(__18F8722) || defined(_18F8722) || defined(__18F8723) || defined(_18F8723)
1205  WORD i;
1206  volatile BYTE Dummy;
1207 
1208  i = len;
1209  Dummy = 0xFF;
1210  ClearSPIDoneFlag();
1211  while(i--)
1212  {
1213  if(((BYTE_VAL*)&Dummy)->bits.b0)
1214  {
1215  // End bust operation
1216  ENC_CS_IO = 1;
1217  ((BYTE_VAL*)&Dummy)->bits.b0 = 0;
1218 
1219  // Start the burst operation
1220  ENC_CS_IO = 0;
1221  ENC_SSPBUF = RBM; // Send the Read Buffer Memory opcode.
1222  WaitForDataByte(); // Wait until opcode/address is transmitted.
1223  }
1224  else
1225  Dummy = 0xFF;
1226 
1227  ENC_SSPBUF = 0; // Send a dummy byte to receive a byte
1228  if(val)
1229  {
1230  WaitForDataByte(); // Wait until byte is received.
1231  *val++ = ENC_SSPBUF;
1232  }
1233  else
1234  {
1235  WaitForDataByte(); // Wait until byte is received.
1236  }
1237  }
1238 
1239  ENC_CS_IO = 1;
1240 
1241  return len;
1242 #else
1243  WORD i;
1244  volatile BYTE Dummy;
1245 
1246  // Start the burst operation
1247  ENC_CS_IO = 0;
1248  ClearSPIDoneFlag();
1249  ENC_SSPBUF = RBM; // Send the Read Buffer Memory opcode.
1250  i = 0;
1251  if(val)
1252  val--;
1253  WaitForDataByte(); // Wait until opcode/address is transmitted.
1254  Dummy = ENC_SSPBUF;
1255 
1256  #if defined(__C32__)
1257  {
1258  DWORD_VAL dwv;
1259 
1260  // Read the data, 4 bytes at a time, for as long as possible
1261  if(len >= 4)
1262  {
1263  ENC_SPICON1bits.MODE32 = 1;
1264  while(1)
1265  {
1266  ENC_SSPBUF = 0x00000000; // Send a dummy DWORD to generate 32 clocks

2143  *

2147  * table will be cleared of all entries and the
2148  * filter will be disabled.
2149  *

2153  *

2168  * addresses from the hash table since it is possible to have
2169  * a hash collision and therefore multiple MAC addresses
2170  * relying on the same hash table bit. The stack would have
2171  * to individually store each 6 byte MAC address to support
2172  * this feature, which would waste a lot of RAM and be
2173  * unnecessary in most applications. As a simple compromise,
2174  * you can call SetRXHashTableEntry() using a
2175  * 00-00-00-00-00-00 destination MAC address, which will clear
2176  * the entire hash table and disable the hash table filter.
