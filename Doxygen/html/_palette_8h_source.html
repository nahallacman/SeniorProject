51 /*********************************************************************
52 * Overview: Structure used for the palette entries.
53 * - For TFT: color is defined as 5-6-5 RGB format
54 * (5-bits for RED, 6-bits for GREEN and 5-bits for BLUE.
55 * - For Monochrome: 4 bits are used to represent the luma.
56 *
57 *********************************************************************/
58 typedef union __attribute__ ((packed))
59 {
60  WORD value; // a 16-bit value representing a color or palette entry
61  struct __attribute__ ((packed))
62  {
63  WORD r : 5; // represents the RED component
64  WORD g : 6; // represents the GREEN component
65  WORD b : 5; // represents the BLUE component
66  } color; // color value in 5-6-5 RGB format
67 
68  struct __attribute__ ((packed))
69  {
70  WORD reserved : 12; // reserved, used as a filler
71  WORD luma : 4; // monochrome LUMA value
72  } monochrome; // monochrome LUMA value
73 
74 } PALETTE_ENTRY;
