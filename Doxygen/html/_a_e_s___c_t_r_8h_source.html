74 /* AES_CTR_STATE_DATA, *P_AES_CTR_STATE_DATA
75 
76  Summary:
77  State information that must be maintained between blocks and initialization
78  data.
79 
80  Description:
81  The AES_CTR_STATE_DATA structure contains state information that needs to be
82  retained between blocks of CTR data. This structure is also used to pass
83  some initialization data to the first block of a new stream.
84 */
85 typedef struct
86 {
87  /* Buffer containing the initial NONCE and counter for the CTR mode
88  encryption. This buffer should be initialized before the
89  first call of either the CTR encrypt or decrypt function */
90  UINT8 __attribute__((aligned)) nonce_plus_counter[16];
91 
92  /* Buffer containing the current counter value. This memory
93  should not be touched by the user. */
94  UINT8 __attribute__((aligned)) counter[16];
95 
96  /*
97  must be aligned to the processors required alignment
98  must be a multiple of AES_BLOCK_SIZE number of bytes but smaller
99  than 2^25 (33,554,432) bytes
100  */
101  void* key_stream;
102 
103  /*
104  pointer that points to the current position in the key stream.
105  */
106  void* key_stream_current_position;
107 
108  /*
109  must be a multiple of AES_BLOCK_SIZE number of bytes but smaller
110  than 2^25 (33,554,432) bytes
111  */
112  UINT32 key_stream_size;
113 
114  /* This field indicates the number of bytes left remaining in the key stream. This can be used to
115  determine if you need to generate more key stream before the next call to the encrypt or decrypt
116  functions. This variable should not be checked by the user during the middle of an execution
117  of the encrypt or decrypt functions. This variable can be checked between calls to these
118  functions. */
119  UINT32 bytes_remaining_in_key_stream;
120 
121 } AES_CTR_STATE_DATA, *P_AES_CTR_STATE_DATA;

139  Summary:
140  Encrypts a specified amount of data using counter (CTR) mode.
141  Description:
142  Encrypts a specified amount of data using counter (CTR) mode.
143  Conditions:
144  * p_ctr_state_data must contain initialized data.
145  * num_bytes must be less than or equal to
146  p_ofb_state_data->bytes_remaining_in_key_stream. If not then the
147  AESCTRGenerateKeyStream() will be called in this function
148  to generate additional key stream material as many times that are
149  required to generate the required key stream based on the input.
150  AESCTRGenerateKeyStream() can be called independantly if you wish to
151  pregenerate the key data.
152  * round_keys must be properly generated using AESCreateRoundKeys().
153  Input:
154  cipher_text - [out] pointer to the buffer where the encrypted data sould
155  reside. Must be num_bytes bytes long.
156  plain_text - [in] pointer to the data that needs to be encrypted. Must be
157  num_bytes bytes long.
158  num_bytes - [in] the number of bytes in the plain_text and cipher_text
159  buffers
160  round_keys - [in] pointer to the round key buffer that will be used
161  for this encryption cycle.
162  p_ctr_state_data - [in] pointer to an instance of the AES_CTR_STATE_DATA for this
163  stream.
164  options - [in] Set of options that are ORed together (e.g. \-
165  OPTION_1 | OPTION_2). Valid options are\:
166  * AES_STREAM_START
167  * AES_STREAM_CONTINUE
168  * AES_CTR_32BIT
169  * AES_CTR_64BIT
170  * AES_CTR_128BIT
171 
172  If AES_STREAM_START is not selected then AES_STREAM_CONTINUE
173  is assumed.
174  Return:
175  Returns a member of the AES_ERRORS enumeration.
176  * AES_NONE - No errors creating the requested key stream
177  * AES_KEY_STREAM_GEN_OUT_OF_SPACE - There was not enough room remaining
178  in the p_ctr_state_data->key_stream buffer to fit the key data requested by
179  the num_blocks parameter.
180  Example:
181  <code>
182  //Example variables. The key and initialization vector information
183  // should be customized for the target application.
184  static const UINT8 AESKey128[] = { 0x95, 0xA8, 0xEE, 0x8E,
185  0x89, 0x97, 0x9B, 0x9E,
186  0xFD, 0xCB, 0xC6, 0xEB,
187  0x97, 0x97, 0x52, 0x8D
188  };
189  static const UINT8 initial_data[] = { //Nonce
190  0x00, 0x01, 0x02, 0x03,
191  0x04, 0x05, 0x06, 0x07,
192  //Counter
193  0x00, 0x00, 0x00, 0x00,
194  0x00, 0x00, 0x00, 0x00
195  };
196  AES_ROUND_KEYS_128_BIT round_keys;
197  AES_CTR_STATE_DATA current_stream;
198  //key_stream could also be allocated memory instead of fixed memory
199  UINT8 key_stream[AES_BLOCK_SIZE*4];
200 
201  //Create the round keys. This only needs to be done once for each key.
202  // This example is here for completeness.
203  AESCreateRoundKeys( &round_keys,
204  AESKey128,
205  AES_KEY_TYPE_128_BIT
206  );
207 
208  //Before the first call of AESCTRGenerateKeyStream(), the
209  // p_ctr_state_data->nonce_plus_counter must be loaded with the nonce
210  // and initial count value for that specific stream of data.
211  memcpy(current_stream.nonce_plus_counter,initial_data,16);
212 
213  //Before the first call of AESCTRGenerateKeyStream(), the
214  // key_stream variable must be set.
215  current_stream.key_stream = (void*)&key_stream;
216 
217  //Before the first call of AESCTRGenerateKeyStream(), the
218  // key_stream_size variable must be set, AES_BLOCK_SIZE*4 in this example
219  current_stream.key_stream_size = sizeof(key_stream);
220 
221  //Generate 4 blocks of key stream
222  AESCTRGenerateKeyStream( 4, //num_blocks
223  &round_keys,
224  &current_stream,
225  AES_STREAM_START //options
226  );
227 
228  //Encrypt 23 bytes of plain_text and place it in cipher_text.
229  AESCTREncrypt( cipher_text,
230  plain_text,
231  23, //num_bytes
232  &round_keys,
233  &current_stream, //the current state data
234  AES_STREAM_CONTINUE
235  );
236 
237  //At this point of time there are still 41 bytes of key stream left. If
238  // plain_text is loaded with new data for the same stream, then we only
239  // need to call the AESCTREncrypt() function again pointing to the same
240  // stream variable.
241  AESCTREncrypt( cipher_text_part2,
242  plain_text_part2,
243  41, //num_bytes
244  &round_keys,
245  &current_stream, //the current state data
246  AES_STREAM_CONTINUE //We use continue here just in case there
247  // isn't enough key material, then the code
248  // will generate new material from the same
249  // stream.
250  );
251  </code>

390  If only a particular block of cipher stream needs to be decrypted, this
391  can be done by creating a key stream for the specified block and decrypting
392  that block only. This method can be seein the following example:
393  <code>
394  //Before the first call of AESCTRGenerateKeyStream(), the
395  // p_ctr_state_data->nonce_plus_counter must be loaded with the nonce
396  // and initial count value for that specific stream of data.
397  memcpy(current_stream.nonce_plus_counter,initial_data,16);
398 
399  //Increment the nonce to the fourth block. This will allow us to access
400  // block 4 of the cipher stream. If the nonce used is a concatanation
401  // with the counter, it may be possible to just set the counter value to
402  // the desired block instead of incrementing the entire nonce;
403 
404  //Add 4 to the last block of the nonce+counter variable since in this
405  // example the nonce is just concatanated with the nonce.
406  current_stream.nonce_plus_counter[0] += 4;
407 
408  //Before the first call of AESCTRGenerateKeyStream(), the
409  // key_stream variable must be set. Don't need to do this again if
410  // it was already done.
