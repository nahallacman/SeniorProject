243  public boolean WriteIndices()
244  {
245  // Determine if an update is necessary
246  boolean isChanged = false;
247  for(DynamicVariable dv : vars)
248  {
249  if ((dv.getWasUsed() && dv.getCount() == 0) ||
250  (!dv.getWasUsed() && dv.getCount() != 0))
251  {
252  isChanged = true;
253  break;
254  }
255  }
256  if (!isChanged)
257  return false;
258 
259  // Write out HTTPPrint.idx
260  File httpFileIdx = new File(projectDir +"HTTPPrint.idx");
261  try{
262  FileOutputStream file_output = new FileOutputStream(httpFileIdx);
263  DataOutputStream data_out = new DataOutputStream (file_output);
264  BufferedWriter fout = new BufferedWriter(new OutputStreamWriter(data_out));
265 
266  for(DynamicVariable dv : vars)
267  {
268  if (dv.getCount() > 0)
269  fout.write('+');
270  fout.write(dv.GetName());
271  fout.write('\n');
272  }
273  fout.flush();
274  fout.close();
275 
276  // Begin writing HTTPPrint.h
277  file_output = new FileOutputStream(projectDir + "HTTPPrint.h");
278  data_out = new DataOutputStream (file_output);
279  fout = new BufferedWriter(new OutputStreamWriter(data_out));
280  fout.write(HTTPPRINT_H_HEADER);
281 
282  // Write the prototypes
283  List<String> prototypes = new ArrayList<String>();
284  Pattern rGetParts = Pattern.compile("([A-Za-z0-9_]{0,40})(\\(([A-Za-z0-9_,]*)\\))?");
285  for(DynamicVariable dv : vars)
286  {
287  if (dv.GetName().startsWith("inc:") || dv.getCount() == 0)
288  continue;
289  Matcher m = rGetParts.matcher(dv.GetName());
290  if(m.find()){
291  if (prototypes.contains(m.group(1)))
292  continue;
293 
294  // Add the prototype
295  prototypes.add(m.group(1));
296  String temp_str = "void HTTPPrint_" + m.group(1) + "(";
297  fout.write(temp_str);
298  if(m.group(3) != null)
299  {
300  if(m.groupCount() == 3 && m.group(3).length() > 0)
301  {
302  int numParams = m.group(3).split(",").length;
303  for (int i = 0; i < numParams - 1; i++)
304  fout.write("WORD,");
305  fout.write("WORD");
306  }
307  }
308  else
309  {
310  fout.write("void");
311  }
312  fout.write(");\r\n");
313  }
314  }
315 
316  // Write the function itself
317  fout.write(HTTPPRINT_H_MIDDLE);
318  int index = 0;
319  for(DynamicVariable dv : vars)
320  {
321  if (dv.getCount() == 0)
322  {
323  index++;
324  continue;
325  }
326  String temp_str = " case 0x"+String.format("%08x",index++)+":\r\n";
327  fout.write(temp_str);
328 
329  // Write the actual case statement
330  if(dv.GetName().startsWith("inc:"))
331  {
332  temp_str = "\t\t\tHTTPIncFile((ROM BYTE*)\"" + dv.GetName().substring(4) + "\");\r\n\t\t\tbreak;\r\n";
333  fout.write(temp_str);
334  }
335  else
336  {
337  temp_str = "\t\t\tHTTPPrint_" + dv.GetName();
338  fout.write(temp_str);
339  if(!dv.GetName().endsWith(")"))
340  fout.write("()");
341  fout.write(";\r\n\t\t\tbreak;\r\n");
342  }
343  }
344 
345  // Write the footer part
346  fout.write(HTTPPRINT_H_FOOTER);
347  fout.flush();
348  fout.close();
349  }catch (IOException e){}
350  return true;
351  }

359  private int GetIndex(String name)
360  {
361  int count=0;
362  // Search for the dynamic variable
363  DynamicVariable dv = new DynamicVariable(name);
364 
365  int i = vars.indexOf(dv);
366  // If not found, add a new one
367  if (i == -1)
368  {
369  vars.add(dv);
370  i = vars.size()-1;
371  }
372  // Mark as used and return the index
373  count = vars.get(i).getCount();
374  vars.get(i).setCount(++count);
375 
376  return i;
377  }
