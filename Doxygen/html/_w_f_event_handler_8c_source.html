109 /*****************************************************************************
110  * FUNCTION: WFProcessMgmtIndicateMsg
111  *
112  * RETURNS: error code
113  *
114  * PARAMS: None
115  *
116  * NOTES: Processes a management indicate message
117  *****************************************************************************/
118 void WFProcessMgmtIndicateMsg()
119 {
120  tMgmtIndicateHdr hdr;
121  UINT8 buf[6];
122  UINT8 event = 0xff;
123  UINT16 eventInfo;
124 
125 #if defined(MRF24WG)
126  tMgmtIndicatePassphraseReady passphraseReady;
127  tMgmtIndicateSoftAPEvent softAPEvent;
128 #endif
129  UINT8 *extra = NULL;
130 
131  /* read 2-byte header of management message */
132  RawRead(RAW_MGMT_RX_ID, 0, sizeof(tMgmtIndicateHdr), (UINT8 *)&hdr);
133 
134  /* Determine which event occurred and handle it */
135  switch (hdr.subType)
136  {
137  /*-----------------------------------------------------------------*/
138  case WF_EVENT_CONNECTION_ATTEMPT_STATUS_SUBTYPE:
139  /*-----------------------------------------------------------------*/
140 #if defined(MRF24WG)
141 /* There is one data byte with this message */
142  RawRead(RAW_MGMT_RX_ID, sizeof(tMgmtIndicateHdr),2, buf); /* read first 2 bytes after header */
143  /* if connection attempt successful */
144  if (buf[0] == CONNECTION_ATTEMPT_SUCCESSFUL)
145  {
146  event = WF_EVENT_CONNECTION_SUCCESSFUL;
147  eventInfo = WF_NO_ADDITIONAL_INFO;
148  SignalWiFiConnectionChanged(TRUE);
149  #if defined (STACK_USE_DHCP_CLIENT)
150  #if !defined(CONFIG_WPA_ENTERPRISE)
151  RenewDhcp();
152  #endif
153  #endif
154  SetLogicalConnectionState(TRUE);
155  #if defined(CONFIG_WPA_ENTERPRISE)
156  g_EapConnEvent = EAP_EVT_SUCCESS;
157  #endif
158  }
159  /* else connection attempt failed */
160  else
161  {
162  event = WF_EVENT_CONNECTION_FAILED;
163  eventInfo = (UINT16)(buf[0] << 8 | buf[1]); /* contains connection failure code */
164  SetLogicalConnectionState(FALSE);
165  #if defined(CONFIG_WPA_ENTERPRISE)
166  g_EapConnEvent = EAP_EVT_FAILURE;
167  #endif
168  }
169 
170 #else /* !defined(MRF24WG) */
171  /* There is one data byte with this message */
172  RawRead(RAW_MGMT_RX_ID, sizeof(tMgmtIndicateHdr), 1, buf); /* read first byte after header */
173  /* if connection attempt successful */
174  if (buf[0] == CONNECTION_ATTEMPT_SUCCESSFUL)
175  {
176  event = WF_EVENT_CONNECTION_SUCCESSFUL;
177  eventInfo = WF_NO_ADDITIONAL_INFO;
178  SignalWiFiConnectionChanged(TRUE);
179  #if defined (STACK_USE_DHCP_CLIENT)
180  RenewDhcp();
181  #endif
182  SetLogicalConnectionState(TRUE);
183  }
184  /* else connection attempt failed */
185  else
186  {
187  event = WF_EVENT_CONNECTION_FAILED;
188  eventInfo = (UINT16)buf[0]; /* contains connection failure code */
189  SetLogicalConnectionState(FALSE);
190  }
191 #endif /* defined(MRF24WG) */
192  break;
193 
194  /*-----------------------------------------------------------------*/
195  case WF_EVENT_CONNECTION_LOST_SUBTYPE:
196  /*-----------------------------------------------------------------*/
197  /* read index 2 and 3 from message and store in buf[0] and buf[1]
198  buf[0] -- 1: Connection temporarily lost 2: Connection permanently lost 3: Connection Reestablished
199  buf[1] -- 0: Beacon Timeout 1: Deauth from AP */
200  RawRead(RAW_MGMT_RX_ID, sizeof(tMgmtIndicateHdr), 2, buf);
201 
202  if (buf[0] == CONNECTION_TEMPORARILY_LOST)
203  {
204  event = WF_EVENT_CONNECTION_TEMPORARILY_LOST;
205  eventInfo = (UINT16)buf[1]; /* lost due to beacon timeout or deauth */
206  SignalWiFiConnectionChanged(FALSE);
207  }
208  else if (buf[0] == CONNECTION_PERMANENTLY_LOST)
209  {
210  event = WF_EVENT_CONNECTION_PERMANENTLY_LOST;
211  eventInfo = (UINT16)buf[1]; /* lost due to beacon timeout or deauth */
212  SetLogicalConnectionState(FALSE);
213  SignalWiFiConnectionChanged(FALSE);
214  #if defined(CONFIG_WPA_ENTERPRISE)
215  g_EapConnEvent = EAP_EVT_FAILURE;
216  #endif
217  }
218  else if (buf[0] == CONNECTION_REESTABLISHED)
219  {
220  event = WF_EVENT_CONNECTION_REESTABLISHED;
221  eventInfo = (UINT16)buf[1]; /* originally lost due to beacon timeout or deauth */
222  #if defined(STACK_USE_DHCP_CLIENT)
223  RenewDhcp();
224  #endif
225  SignalWiFiConnectionChanged(TRUE);
226  SetLogicalConnectionState(TRUE);
227  }
228  else
229  {
230  /* invalid parameter in message */
231  WF_ASSERT(FALSE);
232  }
233  break;
234 
235  /*-----------------------------------------------------------------*/
236  case WF_EVENT_SCAN_RESULTS_READY_SUBTYPE:
237  /*-----------------------------------------------------------------*/
238  /* read index 2 of mgmt indicate to get the number of scan results */
239  RawRead(RAW_MGMT_RX_ID, sizeof(tMgmtIndicateHdr), 1, buf);
240  event = WF_EVENT_SCAN_RESULTS_READY;
241  eventInfo = (UINT16)buf[0]; /* number of scan results */
242  break;
243 
244  /*-----------------------------------------------------------------*/
245  case WF_EVENT_SCAN_IE_RESULTS_READY_SUBTYPE:
246  /*-----------------------------------------------------------------*/
247  event = WF_EVENT_IE_RESULTS_READY;
248  /* read indexes 2 and 3 containing the 16-bit value of IE bytes */
249  RawRead(RAW_MGMT_RX_ID, sizeof(tMgmtIndicateHdr), 2, (UINT8 *)&eventInfo);
250  eventInfo = WFSTOHS(eventInfo); /* fix endianess of 16-bit value */
251  break;
252 
253 #if defined(MRF24WG)
254  case WF_EVENT_KEY_CALCULATION_REQUEST_SUBTYPE:
255  event = WF_EVENT_KEY_CALCULATION_REQUEST;
256  RawRead(RAW_MGMT_RX_ID, sizeof(tMgmtIndicateHdr),
257  sizeof(tMgmtIndicatePassphraseReady), (UINT8 *)&passphraseReady);
258  extra = (UINT8 *)&passphraseReady;
259  break;
260  case WF_EVENT_SOFT_AP_EVENT_SUBTYPE: /* Valid only with 3108 or the later module FW version */
261  event = WF_EVENT_SOFT_AP_EVENT;
262  RawRead(RAW_MGMT_RX_ID, sizeof(tMgmtIndicateHdr),
263  sizeof(tMgmtIndicateSoftAPEvent), (UINT8 *)&softAPEvent);
264  extra = (UINT8 *)&softAPEvent;
265  break;
266 #endif
267 
268  /*-----------------------------------------------------------------*/
269  default:
270  /*-----------------------------------------------------------------*/
271  WF_ASSERT(FALSE);
272  break;
273  }
274 
275  /* free mgmt buffer */
276  DeallocateMgmtRxBuffer();
277 
278 #if defined(HOST_CM_TEST)
279  g_event = event;
280 #endif
281 
282  /* if the application wants to be notified of the event */
283  if (isNotifyApp(event))
284  {
285  WF_ProcessEvent(event, eventInfo, extra);
286  }
287 }

290 /*****************************************************************************
291  * FUNCTION: isEventNotifyBitSet
292  *
293  * RETURNS: TRUE if the notify bit is set in the notify mask.
294  *
295  * PARAMS: notifyMask -- the bit mask of events the application wishes to be
296  * notified of
297  * notifyBit -- the specific event that occurred
298  *
299  * NOTES: Determines if the input event it enabled in the notify mask
300  *****************************************************************************/
301 static BOOL isEventNotifyBitSet(UINT8 notifyMask, UINT8 notifyBit)
302 {
303  /* check if the event notify bit is set */
304  return ((notifyMask & notifyBit) > 0);
305 }

307 /*****************************************************************************
308  * FUNCTION: isNotifyApp
309  *
310  * RETURNS: TRUE if application wants to be notified of event, else FALSE
311  *
312  * PARAMS: event -- the event that occurred
313  *
314  * NOTES: Determines if the input event is one which the application should be
315  * notified of.
316  *****************************************************************************/
317 static BOOL isNotifyApp(UINT8 event)
318 {
319  BOOL notify = FALSE;
320  UINT8 notifyMask;
321 
322  notifyMask = GetEventNotificationMask();
323 
324  /* determine if user wants to be notified of event */
325  switch (event)
326  {
327  case WF_EVENT_CONNECTION_SUCCESSFUL:
328  if (isEventNotifyBitSet(notifyMask, WF_NOTIFY_CONNECTION_ATTEMPT_SUCCESSFUL))
329  {
330  notify = TRUE;
331  }
332  break;
333 
334  case WF_EVENT_CONNECTION_FAILED:
335  if (isEventNotifyBitSet(notifyMask, WF_NOTIFY_CONNECTION_ATTEMPT_FAILED))
336  {
337  notify = TRUE;
338  }
339  break;
340 
341  case WF_EVENT_CONNECTION_TEMPORARILY_LOST:
342  if (isEventNotifyBitSet(notifyMask, WF_NOTIFY_CONNECTION_TEMPORARILY_LOST))
343  {
344  notify = TRUE;
345  }
346  break;
347 
348  case WF_EVENT_CONNECTION_PERMANENTLY_LOST:
349  if (isEventNotifyBitSet(notifyMask, WF_NOTIFY_CONNECTION_PERMANENTLY_LOST))
350  {
351  notify = TRUE;
352  }
353  break;
354 
355  case WF_EVENT_CONNECTION_REESTABLISHED:
356  if (isEventNotifyBitSet(notifyMask, WF_NOTIFY_CONNECTION_REESTABLISHED))
357  {
358  notify = TRUE;
359  }
360  break;
361 
362  default:
363  notify = TRUE; /* the app gets notified of all other events */
364  break;
365  }
366 
367  return notify;
368 
369 }
