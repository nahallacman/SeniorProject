331 /*********************************************************************
332 * Function: SLIDER *SldCreate(WORD ID, SHORT left, SHORT top, SHORT right,
333 * SHORT bottom, WORD state, SHORT range,
334 * SHORT page, SHORT pos, GOL_SCHEME *pScheme);
335 *
336 * Overview: This function creates a SLIDER object with the parameters given.
337 * Depending on the SLD_SCROLLBAR state bit slider or scrollbar mode
338 * is set. If SLD_SCROLLBAR is set, mode is scrollbar; if not set
339 * mode is slider. It automatically attaches the new object into a
340 * global linked list of objects and returns the address of the object.
341 *
342 * PreCondition: none
343 *
344 * Input: ID - Unique user defined ID for the object instance.
345 * left - Left most position of the Object.
346 * top - Top most position of the Object.
347 * right - Right most position of the Object.
348 * bottom - Bottom most position of the Object.
349 * state - This defines the initial state of the Object.
350 * range - This specifies the maximum value of the slider when the
351 * thumb is on the rightmost position for a horizontal orientation
352 * and bottom most position for the vertical orientation. Minimum
353 * value is always at zero.
354 * page - This is the incremental change of the slider when user action
355 * requests to move the slider thumb. This value is added or
356 * subtracted to the current position of the thumb.
357 * pos - This defines the initial position of the thumb.
358 * pScheme - The pointer to the style scheme used for the Object.
359 * Set to NULL if default style scheme is used.
360 *
361 * Output: Returns the pointer to the object created.
362 *
363 * Example:
364 * <CODE>
365 * GOL_SCHEME *pScheme;
366 * SLIDER *slider[3];
367 * WORD state;
368 *
369 * pScheme = GOLCreateScheme();
370 *
371 * // create a slider with
372 * // range = [0 – 50000]
373 * // delta = 500
374 * // initial position = 25000
375 * state = SLD_DRAW;
376 * slider[0] = SldCreate( 5,
377 * 150, 145, 285, 181,
378 * state,
379 * 50000, 500, 25000,
380 * pScheme);
381 * if (slider[0] == NULL)
382 * return 0;
383 *
384 * // create a scrollbar with
385 * // range = [0 – 100]
386 * // delta = 20
387 * // initial position = 0
388 *
389 * state = SLD_DRAW|SLD_SCROLLBAR;
390 * slider[1] = SldCreate( 6,
391 * 150, 190, 285, 220,
392 * state,
393 * 100, 20, 0,
394 * pScheme);
395 * if (slider[1] == NULL)
396 * return 0;
397 *
398 * // create a vertical slider with
399 * // range = [0 – 30]
400 * // delta = 2
401 * // initial position = 20
402 *
403 * state = SLD_DRAW|SLD_VERTICAL;
404 * slider[2] = SldCreate( 7,
405 * 120, 145, 140, 220,
406 * state,
407 * 30, 2, 20,
408 * pScheme);
409 * if (slider[2] == NULL)
410 * return 0;
411 *
412 * // draw the sliders

469 *
470 * Overview: This function evaluates the message from a user if the
471 * message will affect the object or not. The table below
472 * enumerates the translated messages for each event of the
473 * touch screen and keyboard inputs.
474 *
475 * <TABLE>
476 * Translated Message Input Source Events Description
477 * ################## ############ ###### ###########

479 * Keyboard EVENT_KEYSCAN If event occurs and parameter1 passed matches the object’s ID and parameter 2 passed matches SCAN_UP_PRESSED or SCAN_LEFT_PRESSED.
480 * SLD_MSG_DEC Touch Screen EVENT_PRESS, EVENT_MOVE If events occurs and the x,y position falls in the area of the slider and the slider position is to the RIGHT of the x,y position for a horizontal slider or ABOVE the x,y position for a vertical slider.
481 * Keyboard EVENT_KEYSCAN If event occurs and parameter1 passed matches the object’s ID and parameter 2 passed matches SCAN_DOWN_PRESSED or SCAN_RIGHT_PRESSED.
482 * OBJ_MSG_PASSIVE Touch Screen EVENT_RELEASE If events occurs and the x,y position falls in the area of the slider.
