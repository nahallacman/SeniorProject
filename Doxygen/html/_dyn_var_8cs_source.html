138  public MPFSFileRecord Parse(MPFSFileRecord file)
139  {
140  byte[] idxData = new byte[0];
141  UInt32 dynVarCntr=0;
142 
143 
144  MatchCollection matches = parser.Matches(ascii.GetString(file.data));
145  foreach(Match m in matches)
146  {
147  int i = GetIndex(m.Value.Replace(" ","").Replace("~",""));
148  Array.Resize(ref idxData, idxData.Length + 8);
149  idxData[idxData.Length - 8] = (byte)m.Index;
150  idxData[idxData.Length - 7] = (byte)(m.Index >> 8);
151  idxData[idxData.Length - 6] = (byte)(m.Index >> 16);
152  idxData[idxData.Length - 5] = (byte)(m.Index >> 24);
153  idxData[idxData.Length - 4] = (byte)i;
154  idxData[idxData.Length - 3] = (byte)(i >> 8);
155  idxData[idxData.Length - 2] = (byte)(i >> 16);
156  idxData[idxData.Length - 1] = (byte)(i >> 24);
157 
158 
159  Array.Resize(ref file.dynVarOffsetAndIndexID, file.dynVarOffsetAndIndexID.Length + 8);
160 
161  file.dynVarOffsetAndIndexID[file.dynVarOffsetAndIndexID.Length - 1] = (byte)(i >> 24);
162  file.dynVarOffsetAndIndexID[file.dynVarOffsetAndIndexID.Length - 2] = (byte)(i >> 16);
163  file.dynVarOffsetAndIndexID[file.dynVarOffsetAndIndexID.Length - 3] = (byte)(i >> 8);
164  file.dynVarOffsetAndIndexID[file.dynVarOffsetAndIndexID.Length - 4] = ((byte)i);
165 
166 
167  file.dynVarOffsetAndIndexID[file.dynVarOffsetAndIndexID.Length - 5] = (byte)(m.Index >> 24);
168  file.dynVarOffsetAndIndexID[file.dynVarOffsetAndIndexID.Length - 6] = (byte)(m.Index >> 16);
169  file.dynVarOffsetAndIndexID[file.dynVarOffsetAndIndexID.Length - 7] = (byte)(m.Index >> 8);
170  file.dynVarOffsetAndIndexID[file.dynVarOffsetAndIndexID.Length - 8] = (byte)m.Index;
171 
172  file.dynVarCntr = ++dynVarCntr;
173  offSetCounter = offSetCounter + 8;
174 
175  }
176 
177  if(parseItrtn == (UInt32)0x0)
178  {
179  file.fileRecordOffset = (UInt32)0x0;
180  offSetCounter = (UInt32)0x0;
181  }
182  else
183  {
184  //file.fileRecordOffset=tempFileRcrdLen+1;
185  file.fileRecordOffset=tempFileRcrdLen;
186  }
187 
188  file.fileRecordLength = 4 /* 4 bytes for file record length itself*/
189  +2 /*To store the hasIndex/isZipped flag*/
190  //+(UInt32)file.FileName.Length
191  + file.dynVarCntr*8;
192 
193  tempFileRcrdLen += file.fileRecordLength;
194 
195  parseItrtn++;
196 
197

199  if (idxData.Length == 0)
200  return null;
201  else
202  {
203  // Set up new file record
204  MPFSFileRecord idxFile = new MPFSFileRecord();
205  idxFile.FileName = "";

207  idxFile.isIndex = true;

218  public bool WriteIndices()
219  {
220  // Determine if an update is necessary
221  bool isChanged = false;
222  foreach (DynamicVariable dv in vars)
223  {
224  if ((dv.WasUsed && dv.Count == 0) ||
225  (!dv.WasUsed && dv.Count != 0))
226  {
227  isChanged = true;
228  break;
229  }
230  }
231  if (!isChanged)
232  return false;
233 
234  // Write out HTTPPrint.idx
235  StreamWriter fout = new StreamWriter(projectDir + "HTTPPrint.idx", false);
236  foreach (DynamicVariable dv in vars)
237  {
238  if (dv.Count > 0)
239  fout.Write('+');
240  fout.WriteLine(dv.Name);
241  }
242  fout.Flush();
243  fout.Close();
244 
245  // Begin writing HTTPPrint.h
246  fout = new StreamWriter(projectDir + "HTTPPrint.h", false);
247  fout.Write(HTTPPRINT_H_HEADER);
248 
249  // Write the prototypes
250  List<string> prototypes = new List<string>();
251  Regex rGetParts = new Regex(@"([A-Za-z0-9_]{0,40})(\(([A-Za-z0-9_,]*)\))?");
252  foreach (DynamicVariable dv in vars)
253  {
254  if (dv.Name.StartsWith("inc:") || dv.Count == 0)
255  continue;
256  Match m = rGetParts.Match(dv.Name);
257  if (prototypes.Contains(m.Groups[1].Value))
258  continue;
259 
260  // Add the prototype
261  prototypes.Add(m.Groups[1].Value);
262  fout.Write("void HTTPPrint_" + m.Groups[1].Value + "(");
263  if(m.Groups.Count == 4 && m.Groups[3].Value.Length > 0)
264  {
265  int numParams = m.Groups[3].Value.Split(',').Length;
266  for (int i = 0; i < numParams - 1; i++)
267  fout.Write("WORD,");
268  fout.Write("WORD");
269  }
270  else
271  {
272  fout.Write("void");
273  }
274  fout.Write(");\r\n");
275  }
276 
277  // Write the function itself
278  fout.Write(HTTPPRINT_H_MIDDLE);
279  int index = 0;
280  foreach (DynamicVariable dv in vars)
281  {
282  if (dv.Count == 0)
283  {
284  index++;
285  continue;
286  }
287 
288  fout.Write(" case 0x" + Convert.ToString(index++, 16).PadLeft(8, '0') + ":\r\n");
289 
290  // Write the actual case statement
291  if(dv.Name.StartsWith("inc:"))
292  {
293  fout.Write("\t\t\tHTTPIncFile((ROM BYTE*)\"" + dv.Name.Substring(4) + "\");\r\n\t\t\tbreak;\r\n");
294  }
295  else
296  {
297  fout.Write("\t\t\tHTTPPrint_" + dv.Name);
298  if(!dv.Name.EndsWith(")"))
299  fout.Write("()");
300  fout.Write(";\r\n\t\t\tbreak;\r\n");
301  }
302  }
303 
304  // Write the footer part
305  fout.Write(HTTPPRINT_H_FOOTER);
306  fout.Flush();
307  fout.Close();
308 
309  return true;
310  }
