317  DWORD Snmpv3TrackAuthEngineTimeTick(void)
318 
319  Summary:
320  Returns the internal tick timer value to be used by 'snmpEngineTime'.
321 
322  Description:
323  This routine reads the internal tick timer 48 bits, converts to seconds.
324 
325  Precondition:
326  None
327 
328  Parameters:
329  None
330 
331  Return Values:
332  timeStamp : DWORD value of the timer ticks
333 
334  Remarks:
335  None
336 ***************************************************************************/
337 DWORD Snmpv3TrackAuthEngineTimeTick(void)
338 {
339 
340  DWORD_VAL dwvHigh, dwvLow;
341  DWORD dw;
342  DWORD timeStamp;
343 
344 
345  //TimeStamp
346  // Get all 48 bits of the internal Tick timer
347  do
348  {
349  dwvHigh.Val = TickGetDiv64K();
350  dwvLow.Val = TickGet();
351  } while(dwvHigh.w[0] != dwvLow.w[1]);
352  dwvHigh.Val = dwvHigh.w[1];
353 
354  // Find total contribution from lower DWORD
355  dw = dwvLow.Val/(DWORD)TICK_SECOND;
356  timeStamp = dw*100ul;
357  dw = (dwvLow.Val - dw*(DWORD)TICK_SECOND)*100ul; // Find fractional seconds and convert to 10ms ticks
358  timeStamp += (dw+((DWORD)TICK_SECOND/2ul))/(DWORD)TICK_SECOND;
359 
360  // Itteratively add in the contribution from upper WORD
361  while(dwvHigh.Val >= 0x1000ul)
362  {
363  timeStamp += (0x100000000000ull*100ull+(TICK_SECOND/2ull))/TICK_SECOND;
364  dwvHigh.Val -= 0x1000;
365  }
366  while(dwvHigh.Val >= 0x100ul)
367  {
368  timeStamp += (0x010000000000ull*100ull+(TICK_SECOND/2ull))/TICK_SECOND;
369  dwvHigh.Val -= 0x100;
370  }
371  while(dwvHigh.Val >= 0x10ul)
372  {
373  timeStamp += (0x001000000000ull*100ull+(TICK_SECOND/2ull))/TICK_SECOND;
374  dwvHigh.Val -= 0x10;
375  }
376  while(dwvHigh.Val)
377  {
378  timeStamp += (0x000100000000ull*100ull+(TICK_SECOND/2ull))/TICK_SECOND;
379  dwvHigh.Val--;
380  }
381 
382  return timeStamp;
383 }
