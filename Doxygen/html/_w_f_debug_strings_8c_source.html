369 static void OutputConnectionFailedMsg(UINT16 eventInfo)
370 {
371 #if defined(MRF24WG)

375  status = (UINT8)(eventInfo >> 8);

378  {
379  putrsUART("Event: Connection Failed : ");
380  putrsUART(connectionFailureStrings[status]);

382  #if !defined(__18CXX)
383  putrsUART(deauthDisssocReasonCodes[reason]);
384  #endif
385 
386  }
387  else if (status == WF_AUTHENTICATION_FAILURE || status == WF_ASSOCIATION_FAILURE)
388  {
389  putrsUART("Event: Connection Failed : ");
390  putrsUART(connectionFailureStrings[status]);
391  putrsUART(" : ");
392  #if !defined(__18CXX)
393  putrsUART(statusCodes[reason]);
394  #endif
395  }
396  else if (status == WF_WPS_FAILURE)
397  {
398  putrsUART("Event: Connection Failed : ");
399  putrsUART(connectionFailureStrings[status]);
400  putrsUART(" : ");
401  #if !defined(__18CXX)
402  putrsUART(wpsState[reason >> 4]);
403  putrsUART(" : ");
404  putrsUART(wpsConfigErr[reason & 0x0f]);
405  #endif
406  }
407  else if (status == WF_P2P_FAILURE)
408  {
409  putrsUART("Event: Connection Failed : ");
410  putrsUART(connectionFailureStrings[status]);
411  putrsUART(" : ");
412  #if !defined(__18CXX)

414  putrsUART(" : ");
415  putrsUART(P2PErr[reason & 0x0f]);
416  #endif
417  }
418  else
419  {
420  putrsUART("Event: Connection Failed : ");
421  putrsUART(connectionFailureStrings[status]);
422  }
423 #else /* !MRF24WG */
424  char buf[8];
425 
426  putrsUART("Event: Connection Failed -- eventInfo = ");
427  sprintf(buf, "%d, ", eventInfo);
428  putsUART(buf);
