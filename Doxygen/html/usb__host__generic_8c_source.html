153 BOOL USBHostGenericInit ( BYTE address, DWORD flags, BYTE clientDriverID )
154 {
155  BYTE *pDesc;
156 
157  // Initialize state
158  gc_DevData.rxLength = 0;
159  gc_DevData.flags.val = 0;
160 
161  // Save device the address, VID, & PID
162  gc_DevData.ID.deviceAddress = address;
163  pDesc = USBHostGetDeviceDescriptor(address);
164  pDesc += 8;
165  gc_DevData.ID.vid = (WORD)*pDesc; pDesc++;
166  gc_DevData.ID.vid |= ((WORD)*pDesc) << 8; pDesc++;
167  gc_DevData.ID.pid = (WORD)*pDesc; pDesc++;
168  gc_DevData.ID.pid |= ((WORD)*pDesc) << 8; pDesc++;
169 
170  // Save the Client Driver ID
171  gc_DevData.clientDriverID = clientDriverID;
172 
173  #ifdef DEBUG_MODE
174  UART2PrintString( "GEN: USB Generic Client Initalized: flags=0x" );
175  UART2PutHex( flags );
176  UART2PrintString( " address=" );
177  UART2PutDec( address );
178  UART2PrintString( " VID=0x" );
179  UART2PutHex( gc_DevData.ID.vid >> 8 );
180  UART2PutHex( gc_DevData.ID.vid & 0xFF );
181  UART2PrintString( " PID=0x" );
182  UART2PutHex( gc_DevData.ID.pid >> 8 );
183  UART2PutHex( gc_DevData.ID.pid & 0xFF );
184  UART2PrintString( "\r\n" );
185  #endif
186 
187  #ifdef USB_GENERIC_SUPPORT_SERIAL_NUMBERS
188  {
189  BYTE *deviceDescriptor;
190  BYTE serialNumberIndex;
191 
192  // MCHP - when multiple devices are implemented, this init will change
193  // to find a free slot
194  gc_DevData.flags.serialNumberValid = 0;
195 
196  gc_DevData.ID.serialNumber = &(serialNumbers[0][0]);
197  gc_DevData.ID.serialNumberLength = 0;
198 
199  deviceDescriptor = USBHostGetDeviceDescriptor( deviceAddress );
200  serialNumberIndex = deviceDescriptor[16];
201 
202  if (serialNumberIndex)
203  {
204  #ifdef DEBUG_MODE
205  UART2PrintString( "GEN: Getting serial number...\r\n" );
206  #endif
207  }
208  else
209  {
210  serialNumberIndex = 1;
211  #ifdef DEBUG_MODE
212  UART2PrintString( "GEN: Getting string descriptor...\r\n" );
213  #endif
214  }
215 
216  if (USBHostGetStringDescriptor( address, serialNumberIndex, (BYTE *)gc_DevData.ID.serialNumber, USB_GENERIC_MAX_SERIAL_NUMBER*2 ))
217  {
218  // We can't get the serial number. Just set the pointer to null
219  // and call it good. We have to call the SN valid so we don't get trapped
220  // in the event handler.
221  gc_DevData.ID.serialNumber = NULL;
222  gc_DevData.flags.initialized = 1;
223  gc_DevData.flags.serialNumberValid = 1;
224 
225  // Tell the application layer that we have a device.
226  USB_HOST_APP_EVENT_HANDLER(address, EVENT_GENERIC_ATTACH, &(gc_DevData.ID), sizeof(GENERIC_DEVICE_ID) );
227 
228  #ifdef DEBUG_MODE
229  UART2PrintString( "GEN: Cannot get string descriptor!\r\n" );
230  #endif
231  }
232  }
233  #else
234  // Generic Client Driver Init Complete.
235  gc_DevData.flags.initialized = 1;
236 
237  // Notify that application that we've been attached to a device.
238  USB_HOST_APP_EVENT_HANDLER(address, EVENT_GENERIC_ATTACH, &(gc_DevData.ID), sizeof(GENERIC_DEVICE_ID) );
239  #endif
240 
241  // TBD
242 
243  return TRUE;
244 
245 } // USBHostGenericInit

280 BOOL USBHostGenericEventHandler ( BYTE address, USB_EVENT event, void *data, DWORD size )
281 {
282  // Make sure it was for our device
283  if ( address != gc_DevData.ID.deviceAddress)
284  {
285  return FALSE;
286  }
287 
288  // Handle specific events.
289  switch (event)
290  {
291  case EVENT_DETACH:
292  // Notify that application that the device has been detached.
293  USB_HOST_APP_EVENT_HANDLER(gc_DevData.ID.deviceAddress, EVENT_GENERIC_DETACH, &gc_DevData.ID.deviceAddress, sizeof(BYTE) );
294  gc_DevData.flags.val = 0;
295  gc_DevData.ID.deviceAddress = 0;
296  #ifdef DEBUG_MODE
297  UART2PrintString( "USB Generic Client Device Detached: address=" );
298  UART2PutDec( address );
299  UART2PrintString( "\r\n" );
300  #endif
301  return TRUE;
302 
303  #ifdef USB_ENABLE_TRANSFER_EVENT
304  case EVENT_TRANSFER:
305  if ( (data != NULL) && (size == sizeof(HOST_TRANSFER_DATA)) )
306  {
307  DWORD dataCount = ((HOST_TRANSFER_DATA *)data)->dataCount;
308 
309  if ( ((HOST_TRANSFER_DATA *)data)->bEndpointAddress == (USB_IN_EP|USB_GENERIC_EP) )
310  {
311  gc_DevData.flags.rxBusy = 0;
312  gc_DevData.rxLength = dataCount;
313  USB_HOST_APP_EVENT_HANDLER(gc_DevData.ID.deviceAddress, EVENT_GENERIC_RX_DONE, &dataCount, sizeof(DWORD) );
314  }
315  else if ( ((HOST_TRANSFER_DATA *)data)->bEndpointAddress == (USB_OUT_EP|USB_GENERIC_EP) )
316  {
317  gc_DevData.flags.txBusy = 0;
318  USB_HOST_APP_EVENT_HANDLER(gc_DevData.ID.deviceAddress, EVENT_GENERIC_TX_DONE, &dataCount, sizeof(DWORD) );
319  }
320  else
321  #ifdef USB_GENERIC_SUPPORT_SERIAL_NUMBERS
322  if (((((HOST_TRANSFER_DATA *)data)->bEndpointAddress & 0x7F) == 0) && !gc_DevData.flags.serialNumberValid)
323  {
324  #ifdef DEBUG_MODE
325  UART2PrintString( "GEN: Got serial number!\r\n" );
326  #endif
327  // Set the serial number information
328  gc_DevData.ID.serialNumberLength = dataCount;
329  gc_DevData.flags.serialNumberValid = 1;
330  gc_DevData.flags.initialized = 1;
331 
332  // Tell the application layer that we have a device.
333  USB_HOST_APP_EVENT_HANDLER(address, EVENT_GENERIC_ATTACH, &(gc_DevData.ID), sizeof(GENERIC_DEVICE_ID) );
334  }
335  else
336  #endif
337  {
338  return FALSE;
339  }
340 
341  return TRUE;
342 
343  }
344  else
345  return FALSE;
346  #endif
347 
348  case EVENT_SUSPEND:
349  case EVENT_RESUME:
350  case EVENT_BUS_ERROR:
351  default:
352  break;
353  }
354 
355  return FALSE;
356 } // USBHostGenericEventHandler

518 BYTE USBHostGenericRead( BYTE deviceAddress, void *buffer, DWORD length )
519 {
520  BYTE RetVal;
521 
522  // Validate the call
523  if (!API_VALID(deviceAddress)) return USB_INVALID_STATE;
524  if (gc_DevData.flags.rxBusy) return USB_BUSY;
525 
526  // Set the busy flag, clear the count and start a new IN transfer.
527  gc_DevData.flags.rxBusy = 1;
528  gc_DevData.rxLength = 0;
529  RetVal = USBHostRead( deviceAddress, USB_IN_EP|USB_GENERIC_EP, (BYTE *)buffer, length );
530  if (RetVal != USB_SUCCESS)
531  {
532  gc_DevData.flags.rxBusy = 0; // Clear flag to allow re-try
533  }
534 
535  return RetVal;
536 
537 } // USBHostGenericRead
