128 BOOL USBHostChargerInitialize( BYTE address, DWORD flags, BYTE clientDriverID )
129 {
130  BYTE *pDesc;
131 
132  // Find a new entry
133  for (currentChargingRecord=0; currentChargingRecord<USB_MAX_CHARGING_DEVICES; currentChargingRecord++)
134  {
135  if (!usbChargingDevices[currentChargingRecord].flags.inUse) break;
136  }
137  if (currentChargingRecord == USB_MAX_CHARGING_DEVICES)
138  {
139  #ifdef DEBUG_MODE
140  UART2PrintString( "CHG: No more space\r\n" );
141  #endif
142  return FALSE; // We have no more room for a new device.
143  }
144 
145  // Initialize state - set the inUse flag.
146  usbChargingDevices[currentChargingRecord].flags.val = 1;
147 
148  // Save device the address, VID, & PID
149  usbChargingDevices[currentChargingRecord].ID.deviceAddress = address;
150  usbChargingDevices[currentChargingRecord].ID.clientDriverID = clientDriverID;
151  pDesc = USBHostGetDeviceDescriptor(address);
152  pDesc += 8;
153  usbChargingDevices[currentChargingRecord].ID.vid = (WORD)*pDesc; pDesc++;
154  usbChargingDevices[currentChargingRecord].ID.vid |= ((WORD)*pDesc) << 8; pDesc++;
155  usbChargingDevices[currentChargingRecord].ID.pid = (WORD)*pDesc; pDesc++;
156  usbChargingDevices[currentChargingRecord].ID.pid |= ((WORD)*pDesc) << 8; pDesc++;
157 
158  #ifdef DEBUG_MODE
159  UART2PrintString( "CHG: USB Charging Client Initalized: flags=0x" );
160  UART2PutHex( flags );
161  UART2PrintString( " address=" );
162  UART2PutDec( address );
163  UART2PrintString( " VID=0x" );
164  UART2PutHex( usbChargingDevices[currentChargingRecord].ID.vid >> 8 );
165  UART2PutHex( usbChargingDevices[currentChargingRecord].ID.vid & 0xFF );
166  UART2PrintString( " PID=0x" );
167  UART2PutHex( usbChargingDevices[currentChargingRecord].ID.pid >> 8 );
168  UART2PutHex( usbChargingDevices[currentChargingRecord].ID.pid & 0xFF );
169  UART2PrintString( "\r\n" );
170  #endif
171 
172  // Notify that application that we've been attached to a device.
173  USB_HOST_APP_EVENT_HANDLER(address, EVENT_CHARGER_ATTACH,
174  &(usbChargingDevices[currentChargingRecord].ID), sizeof(USB_CHARGING_DEVICE_ID) );
175 
176  return TRUE;
177 
178 } // USBHostChargerInit

213 BOOL USBHostChargerEventHandler( BYTE address, USB_EVENT event, void *data, DWORD size )
214 {
215  // Make sure the event is for a connected device
216  if (!_USBHostCharger_FindDevice( address ))
217  {
218  return FALSE;
219  }
220 
221  // Handle specific events.
222  switch (event)
223  {
224  case EVENT_DETACH:
225  // Notify that application that the device has been detached.
226  USB_HOST_APP_EVENT_HANDLER(usbChargingDevices[currentChargingRecord].ID.deviceAddress, EVENT_CHARGER_DETACH, &usbChargingDevices[currentChargingRecord].ID.deviceAddress, sizeof(BYTE) );
227  usbChargingDevices[currentChargingRecord].flags.val = 0;
228  usbChargingDevices[currentChargingRecord].ID.deviceAddress = 0;
229  #ifdef DEBUG_MODE
230  UART2PrintString( "USB Charging Client Device Detached: address=" );
231  UART2PutDec( address );
232  UART2PrintString( "\r\n" );
233  #endif
234  return TRUE;
235  break;
236 
237  case EVENT_SUSPEND:
238  case EVENT_RESUME:
239  case EVENT_BUS_ERROR:
240  default:
241  break;
242  }
243 
244  return FALSE;
245 } // USBHostChargerEventHandler

327 BOOL USBHostChargerGetDeviceAddress(USB_CHARGING_DEVICE_ID *pDevID)
328 {
329  if (pDevID == NULL)
330  {
331  return FALSE;
332  }
333 
334  for (currentChargingRecord=0; currentChargingRecord<USB_MAX_CHARGING_DEVICES; currentChargingRecord++)
335  {
336  if (usbChargingDevices[currentChargingRecord].flags.inUse &&
337  (usbChargingDevices[currentChargingRecord].ID.vid == pDevID->vid) &&
338  (usbChargingDevices[currentChargingRecord].ID.pid == pDevID->pid))
339  {
340  pDevID->deviceAddress = usbChargingDevices[currentChargingRecord].ID.deviceAddress;
341  return TRUE;
342  }
343  }
344 
345  return FALSE; // The device was not found.
346 
347 } // USBHostChargingGetDeviceAddress
