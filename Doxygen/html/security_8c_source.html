86  /*********************************************************************
87  * void encode(INPUT WORD *text, INPUT WORD *key)
88  *
89  * Overview: This function apply XTEA-64 security engine to
90  * the input data buffer with input security key.
91  * The encoded data will replace the input data
92  *
93  * PreCondition: None
94  *
95  * Input:
96  * WORD * data The input buffer to the XTEA engine. The
97  * encoded data will replace the original
98  * content after the function call
99  * WORD * key The security key for the XTEA engine
100  * Output:
101  * None
102  *
103  * Side Effects: None
104  *
105  ********************************************************************/
106  void encode(WORD *text, WORD *key)
107  {
108  WORD part1=text[0], part2=text[1];
109  WORD sum=0, delta=0x9E37;
110  BYTE i;
111 
112  for(i=0; i<XTEA_ROUND; i++)
113  {
114  part1 += (((part2 << 4) ^ (part2 >> 5)) + part2) ^ (sum + key[sum & 3]);
115  sum += delta;
116  part2 += (((part1 << 4) ^ (part1 >> 5)) + part1) ^ (sum + key[(sum>>11) & 3]);
117  }
118  text[0]=part1; text[1]=part2;
119  }

147  void CTR(BYTE *text, BYTE len, BYTE *key, BYTE *nounce)
148  {
149  BYTE block = len/BLOCK_SIZE+1;
150  BYTE i, j;
151  #if defined(__18CXX)
152  BYTE ITStatus = INTCONbits.GIEH;
153

156 
157  for(i = 0; i < block; i++)
158  {
159  for(j = 0; j < BLOCK_SIZE-1; j++)
160  {
161  tmpBlock[j] = nounce[j];
162  }
163  tmpBlock[BLOCK_SIZE-1] = i;
164  encode((BLOCK_UNIT *)tmpBlock, (BLOCK_UNIT *)key);
165  for(j = 0; j < BLOCK_SIZE; j++)
166  {

169  break;
170  }

173  }
