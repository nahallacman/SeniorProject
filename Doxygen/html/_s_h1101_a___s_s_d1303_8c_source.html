282 /*********************************************************************
283 * Function: void PutPixel(SHORT x, SHORT y)
284 *
285 * PreCondition: none
286 *
287 * Input: x,y - pixel coordinates
288 *
289 * Output: none
290 *
291 * Side Effects: none
292 *
293 * Overview: puts pixel
294 *
295 * Note: none
296 *
297 ********************************************************************/
298 void PutPixel(SHORT x, SHORT y)
299 {
300  BYTE page, add, lAddr, hAddr;
301  BYTE mask, display;
302 
303  // check if point is in clipping region
304  if(_clipRgn)
305  {
306  if(x < _clipLeft)
307  return;
308  if(x > _clipRight)
309  return;
310  if(y < _clipTop)
311  return;
312  if(y > _clipBottom)
313  return;
314  }
315 
316  // Assign a page address
317  if(y < 8)
318  page = 0xB0;
319  else if(y < 16)
320  page = 0xB1;
321  else if(y < 24)
322  page = 0xB2;
323  else if(y < 32)
324  page = 0xB3;
325  else if(y < 40)
326  page = 0xB4;
327  else if(y < 48)
328  page = 0xB5;
329  else if(y < 56)
330  page = 0xB6;
331  else
332  page = 0xB7;
333 
334  add = x + OFFSET;
335  lAddr = 0x0F & add; // Low address
336  hAddr = 0x10 | (add >> 4); // High address
337 
338  // Calculate mask from rows basically do a y%8 and remainder is bit position
339  add = y >> 3; // Divide by 8
340  add <<= 3; // Multiply by 8
341  add = y - add; // Calculate bit position
342  mask = 1 << add; // Left shift 1 by bit position
343  DisplayEnable();
344 
345  SetAddress(page, lAddr, hAddr); // Set the address (sets the page,
346 
347 
348  // lower and higher column address pointers)
349  display = SingleDeviceRead(); // Read to initiate Read transaction on PMP and dummy read
350  // (requirement for data synchronization in the controller)
351  display = SingleDeviceRead(); // Read again as a requirement for data synchronization in the display controller
352  display = SingleDeviceRead(); // Read actual data from from display buffer
353 
354  if(_color > 0) // If non-zero for pixel on
355  display |= mask; // or in mask
356  else
357  // If 0 for pixel off
358  display &= ~mask; // and with inverted mask
359  SetAddress(page, lAddr, hAddr); // Set the address (sets the page,
360 
361  // lower and higher column address pointers)
362  DeviceWrite(display); // restore the byte with manipulated bit
363  DisplayDisable();
364 }

366 /*********************************************************************
367 * Function: BYTE GetPixel(SHORT x, SHORT y)
368 *

386 
387  // check if point is in clipping region
388  if(_clipRgn)

390  if(x < _clipLeft)
391  return (0);
392  if(x > _clipRight)
393  return (0);
394  if(y < _clipTop)
395  return (0);
396  if(y > _clipBottom)
397  return (0);
398  }
399 
400  // Assign a page address
401  if(y < 8)
402  page = 0xB0;
403  else if(y < 16)
404  page = 0xB1;
405  else if(y < 24)
406  page = 0xB2;
407  else if(y < 32)
408  page = 0xB3;
409  else if(y < 40)
410  page = 0xB4;
411  else if(y < 48)
412  page = 0xB5;
413  else if(y < 56)
414  page = 0xB6;
415  else
416  page = 0xB7;
417 
418  add = x + OFFSET;
419  lAddr = 0x0F & add; // Low address
420  hAddr = 0x10 | (add >> 4); // High address
421 
422  // Calculate mask from rows basically do a y%8 and remainder is bit position
423  temp = y >> 3; // Divide by 8
424  temp <<= 3; // Multiply by 8
425  temp = y - temp; // Calculate bit position
426  mask = 1 << temp; // Left shift 1 by bit position
427  DisplayEnable();
428 
429  SetAddress(page, lAddr, hAddr); // Set the address (sets the page,
430 
431  // lower and higher column address pointers)
432  display = SingleDeviceRead(); // Read to initiate Read transaction on PMP
433  // Dummy Read (requirement for data synchronization in the controller)
434  display = DeviceRead(); // Read data from display buffer
435  DisplayDisable();
436 
437  return (display & mask); // mask all other bits and return the result
438 }
