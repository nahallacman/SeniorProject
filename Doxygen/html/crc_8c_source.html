101  /*********************************************************************
102  * WORD CRC16( INPUT BYTE * data,
103  * INPUT signed char dataLength,
104  * INPUT WORD initCRC)
105  *
106  * Overview: This function generates 16-bit CRC code for the input
107  * data, with initial CRC value
108  *
109  * PreCondition: None
110  *
111  * Input:
112  * BYTE * data The input buffer to generate CRC
113  * code from.
114  * signed char dataLength The size of input data buffer
115  * WORD initCRC The initial value of CRC. The
116  * initial CRC value enables the
117  * CRC generate module generates
118  * CRC block by block
119  * Output:
120  * WORD Generated CRC code
121  *
122  * Side Effects: None
123  *
124  ********************************************************************/
125  WORD CRC16(BYTE * data, signed char dataLength, WORD initCRC)
126  {
127  WORD crc;
128  BYTE i;
129 
130  crc = initCRC;
131 
132  for(i=0;i<dataLength;i++)
133  {
134  crc = (crc << 8) ^ CRC16Table[(crc>>8) ^ *data++];
135  }
136  return crc;
137  }
