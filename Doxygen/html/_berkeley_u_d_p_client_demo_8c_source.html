142  This function periodically checks a pool of time servers to obtain the

144 
145  Precondition:
146  UDP is initialized.
147 
148  Parameters:
149  None

152  None
153 
154  Remarks:
155  This function requires once available UDP socket while processing, but
156  frees that socket when the SNTP module is idle.
157  ***************************************************************************/
158 void BerkeleyUDPClientDemo(void)
159 {
160  #if defined(STACK_USE_DNS)
161  NTP_PACKET pkt;
162  int i;
163  static DWORD dwServerIP;
164  static DWORD dwTimer;
165  static SOCKET bsdUdpClient;
166  int addrlen = sizeof(struct sockaddr_in);
167  static struct sockaddr_in udpaddr;
168  static enum
169  {
170  SM_HOME = 0,
171  SM_NAME_RESOLVE,
172  SM_CREATE_SOCKET,
173  //SM_BIND, // Not required since we are sending the first packet
174  SM_UDP_SEND,
175  SM_UDP_RECV,
176  SM_SHORT_WAIT,
177  SM_WAIT
178  } SNTPState = SM_HOME;
179 
180  switch(SNTPState)
181  {
182  case SM_HOME:
183  // Obtain ownership of the DNS resolution module
184  if(!DNSBeginUsage())
185  break;
186 
187  // Obtain the IP address associated with the server name
188  DNSResolveROM((ROM BYTE*)NTP_SERVER, DNS_TYPE_A);
189  dwTimer = TickGet();
190  SNTPState = SM_NAME_RESOLVE;
191  break;
192 
193  case SM_NAME_RESOLVE:
194  // Wait for DNS resolution to complete
195  if(!DNSIsResolved((IP_ADDR*)&dwServerIP))
196  {
197  if((TickGet() - dwTimer) > (5 * TICK_SECOND))
198  {
199  DNSEndUsage();
200  dwTimer = TickGetDiv64K();
201  SNTPState = SM_SHORT_WAIT;
202  }
203  break;
204  }
205 
206  // Obtain DNS resolution result
207  if(!DNSEndUsage())
208  {
209  // No valid IP address was returned from the DNS
210  // server. Quit and fail for a while if host is not valid.
211  dwTimer = TickGetDiv64K();
212  SNTPState = SM_SHORT_WAIT;
213  break;
214  }
215 
216  SNTPState = SM_CREATE_SOCKET;
217  // No break needed
218 
219  case SM_CREATE_SOCKET:
220  bsdUdpClient = socket(AF_INET, SOCK_DGRAM, IPPROTO_UDP);
221  if(bsdUdpClient == INVALID_SOCKET)
222  return;
223 
224 // No explicit binding is necessary since we are going to be the first one
225 // sending a UDP packet. sendto() will do implicit binding. Explicit binding
226 // is necessary only when creating UDP servers that will be receiving the first
227 // packet.
228 // SNTPState = SM_BIND;
229 // // No break needed
230 //
231 // case SM_BIND:
232 // udpaddr.sin_port = 0;
233 // udpaddr.sin_addr.S_un.S_addr = IP_ADDR_ANY;
234 // if( bind(bsdUdpClient, (struct sockaddr*)&udpaddr, addrlen) == SOCKET_ERROR )
235 // break;
236 //
237  SNTPState = SM_UDP_SEND;
238  // No break needed
239 
240  case SM_UDP_SEND:
241  // Transmit a time request packet
242  memset(&pkt, 0, sizeof(pkt));
243  pkt.flags.versionNumber = 3; // NTP Version 3
244  pkt.flags.mode = 3; // NTP Client
245  pkt.orig_ts_secs = swapl(NTP_EPOCH);
246  udpaddr.sin_port = NTP_SERVER_PORT;
247  udpaddr.sin_addr.S_un.S_addr = dwServerIP;
248  if(sendto(bsdUdpClient, (const char*)&pkt, sizeof(pkt), 0, (struct sockaddr*)&udpaddr, addrlen)>0)
249  {
250  dwTimer = TickGet();
251  SNTPState = SM_UDP_RECV;
252  }
253  break;
254 
255  case SM_UDP_RECV:
256  // Look for a response time packet
257  i = recvfrom(bsdUdpClient, (char*)&pkt, sizeof(pkt), 0, (struct sockaddr*)&udpaddr, &addrlen);
258  if(i < (int)sizeof(pkt))
259  {
260  if((TickGet()) - dwTimer > NTP_REPLY_TIMEOUT)
261  {
262  // Abort the request and wait until the next timeout period
263  closesocket(bsdUdpClient);
264  dwTimer = TickGetDiv64K();
265  SNTPState = SM_SHORT_WAIT;
266  break;
267  }
268  break;
269  }
270  closesocket(bsdUdpClient);
271  dwTimer = TickGetDiv64K();
272  SNTPState = SM_WAIT;
273 
274  // Set out local time to match the returned time
275  dwLastUpdateTick = TickGet();
276  dwSNTPSeconds = swapl(pkt.tx_ts_secs) - NTP_EPOCH;
277  // Do rounding. If the partial seconds is > 0.5 then add 1 to the seconds count.
278  if(((BYTE*)&pkt.tx_ts_fraq)[0] & 0x80)
279  dwSNTPSeconds++;
280 
281  #ifdef WIFI_NET_TEST
282  wifi_net_test_print("SNTP: current time", dwSNTPSeconds);
283  #endif
284  break;
285 
286  case SM_SHORT_WAIT:
287  // Attempt to requery the NTP server after a specified NTP_FAST_QUERY_INTERVAL time (ex: 8 seconds) has elapsed.
288  if(TickGetDiv64K() - dwTimer > (NTP_FAST_QUERY_INTERVAL/65536ull))
289  SNTPState = SM_HOME;
290  break;
291 
292  case SM_WAIT:
293  // Requery the NTP server after a specified NTP_QUERY_INTERVAL time (ex: 10 minutes) has elapsed.
294  if(TickGetDiv64K() - dwTimer > (NTP_QUERY_INTERVAL/65536ull))
295  SNTPState = SM_HOME;
296 
297  break;
298  }
299 
300  //#if defined(STACK_USE_DNS)
301  #else
302  #warning You must define STACK_USE_DNS for BerkeleyUDPClientDemo to work
303  #endif
304 }

307 /*****************************************************************************
308  Function:
309  DWORD BerkeleySNTPGetUTCSeconds(void)
310 
311  Summary:
312  Obtains the current time from the SNTP module.
313 
314  Description:
315  This function obtains the current time as reported by the SNTP module.
316  Use this value for absolute time stamping. The value returned is (by
317  default) the number of seconds since 01-Jan-1970 00:00:00.
318 
319  Precondition:
320  None
321 
322  Parameters:
323  None
324 
325  Returns:
326  The number of seconds since the Epoch. (Default 01-Jan-1970 00:00:00)
327 
328  Remarks:
329  Do not use this function for time difference measurements. The Tick
330  module is more appropriate for those requirements.
331  ***************************************************************************/
332 DWORD BerkeleySNTPGetUTCSeconds(void)
333 {
334  DWORD dwTickDelta;
335  DWORD dwTick;
336 
337  // Update the dwSNTPSeconds variable with the number of seconds
338  // that has elapsed
339  dwTick = TickGet();
340  dwTickDelta = dwTick - dwLastUpdateTick;
341  while(dwTickDelta > TICK_SECOND)
342  {
343  dwSNTPSeconds++;
344  dwTickDelta -= TICK_SECOND;
345  }
346 
347  // Save the tick and residual fractional seconds for the next call
348  dwLastUpdateTick = dwTick - dwTickDelta;
349 
350  return dwSNTPSeconds;
351 }
