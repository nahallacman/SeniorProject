112 static BOOL iwprivSetCb(void)
113 {
114  tWFCPElements cprof;
115  BOOL cpIdChanged = FALSE;
116 
117  if ( !iwprivCbInitialized ) // first time call of iwprivSetCb
118  {
119  memset(&iwprivCb, 0, sizeof(iwprivCb));
120  iwprivCbInitialized = TRUE;
121  }
122 
123  if (!WF_hibernate.state && !iwconfigSetCb() ) // first set iwconfigCb
124  return FALSE;
125 
126  if ( iwprivCb.cpId != iwconfigCb.cpId)
127  {
128  iwprivCb.cpId = iwconfigCb.cpId;
129  cpIdChanged = TRUE;
130  }
131 
132  WF_CPGetElements(iwprivCb.cpId, &cprof);
133 
134  // set refreshable part of iwprivCb
135  {
136  memcpy((void*)iwprivCb.ssid, (const void*)cprof.ssid, cprof.ssidLength);
137  iwprivCb.ssid[cprof.ssidLength] = '\0';
138 
139  iwprivCb.wepDefaultKeyId = cprof.wepDefaultKeyId;
140  }
141 
142  // set non-refreshable part of iwprivCb only when cpId has changed
143  if (cpIdChanged)
144  {
145  iwprivCb.securityType = cprof.securityType;
146  if (iwprivCb.securityType == WF_SECURITY_WPA_WITH_KEY || iwprivCb.securityType == WF_SECURITY_WPA2_WITH_KEY)
147  {
148  iwprivCb.securityType = WF_SECURITY_WPA_AUTO_WITH_KEY;
149  }

151  {
152  iwprivCb.securityType = WF_SECURITY_WPA_AUTO_WITH_PASS_PHRASE;
153  }
154 
155  iwprivCb.securityKeyLength = 0;
156  }
157 
158  return TRUE;

327 static BOOL iwprivSetKey(void)
328 {
329  UINT8 webKey;
330 
331  if (iwprivCb.securityType != WF_SECURITY_WEP_40 && iwprivCb.securityType != WF_SECURITY_WEP_104)
332  {
333  WFConsolePrintRomStr("WEP encryption mode is not selected", TRUE);
334  return FALSE;
335  }
336 
337  if (ARGC < 3u)
338  {
339  WFConsolePrintRomStr("Missing value for last parameter", TRUE);
340  return FALSE;
341  }
342 
343  if ( (3u <= ARGC) && (strcmppgm2ram((char*)ARGV[2], "[1]") == 0) )
344  {
345  webKey = 0u;
346  }
347  else if ( (3u <= ARGC) && (strcmppgm2ram((char*)ARGV[2], "[2]") == 0) )
348  {
349  webKey = 1u;
350  }
351  else if ( (3u <= ARGC) && (strcmppgm2ram((char*)ARGV[2], "[3]") == 0) )
352  {
353  webKey = 2u;
354 
355  }
356  else if ( (3u <= ARGC) && (strcmppgm2ram((char*)ARGV[2], "[4]") == 0) )
357  {
358  webKey = 3u;
359  }
360  else
361  {
362  WFConsolePrintRomStr("Invalid WEP key index", TRUE);
363  return FALSE;
364  }
365 
366  if (4u <= ARGC)
367  {
368  if ( convertAsciiToHexInPlace(ARGV[3], IWPRIV_WEB_LONG_KEY_LEN) ) // for long web key
369  {
370  iwprivCb.securityType = WF_SECURITY_WEP_104;
371 
372  memcpy((void*)(iwprivCb.securityKey + webKey * IWPRIV_WEB_LONG_KEY_LEN), (const void*)ARGV[3], IWPRIV_WEB_LONG_KEY_LEN);
373  iwprivCb.securityKeyLength = IWPRIV_WEB_KEY_NUM * IWPRIV_WEB_LONG_KEY_LEN;
374 
375  WF_CPSetSecurity(iwprivCb.cpId, iwprivCb.securityType, webKey,
376  iwprivCb.securityKey + webKey * IWPRIV_WEB_LONG_KEY_LEN, IWPRIV_WEB_LONG_KEY_LEN);
377  }
378  else if ( convertAsciiToHexInPlace(ARGV[3], IWPRIV_WEB_SHORT_KEY_LEN) ) // for short web key
379  {
380  iwprivCb.securityType = WF_SECURITY_WEP_40;
381 
382  memcpy((void*)(iwprivCb.securityKey + webKey * IWPRIV_WEB_SHORT_KEY_LEN), (const void*)ARGV[3], IWPRIV_WEB_SHORT_KEY_LEN);
383  iwprivCb.securityKeyLength = IWPRIV_WEB_KEY_NUM * IWPRIV_WEB_SHORT_KEY_LEN;
384 
385  WF_CPSetSecurity(iwprivCb.cpId, iwprivCb.securityType, webKey,
386  iwprivCb.securityKey + webKey * IWPRIV_WEB_SHORT_KEY_LEN, IWPRIV_WEB_SHORT_KEY_LEN);
387  }
388  else
389  {
390  WFConsolePrintRomStr("64/128bit WEP key format not valid", TRUE);
391  return FALSE;
392  }
393  }
394  else // ARGC == 3u
395  {
396  WF_CPSetDefaultWepKeyIndex(iwprivCb.cpId, webKey);
397  }
398 
399  return TRUE;
400 }

432 static BOOL iwprivSetPhrase(void)
433 {
434  UINT8 j;
435  UINT8 securityType;
436  UINT8* phraseStart;
437  UINT8* phraseEnd;
438  UINT8 phraseLen;
439 
440  if ( iwprivCb.securityType == WF_SECURITY_WPA_AUTO_WITH_KEY || iwprivCb.securityType == WF_SECURITY_WPA_AUTO_WITH_PASS_PHRASE )
441  {
442  securityType = WF_SECURITY_WPA_AUTO_WITH_PASS_PHRASE;
443  }
444  else
445  {
446  WFConsolePrintRomStr("WPA-PSK or WPA-PHRASE encryption mode is not selected", TRUE);
447  return FALSE;
448  }
449 
450  if (ARGC < 3u)
451  {
452  WFConsolePrintRomStr("Missing value for last parameter", TRUE);
453  return FALSE;
454  }
455 
456  phraseStart = (UINT8*) ARGV[2];
457  if (*phraseStart == '\"') // cancatenate remaining args into one string
458  {
459  for (j = 2; j < (ARGC-1); j++)
460  {
461  UINT8 argLen = strlen((char*)ARGV[j]);
462  ARGV[j][argLen] = ' '; // replace '\0' with ' '
463  }
464 
465  // searching for an ending quote
466  phraseEnd = phraseStart + strlen((char *)phraseStart) - 1;
467  while (*phraseEnd != '\"')
468  phraseEnd--;
469 
470  // remove the double quotes
471  phraseStart++;
472  phraseEnd--;
473  }
474  else // a single word
475  {
476  phraseEnd = phraseStart + strlen((char *)phraseStart) - 1;
477  }
478 
479  phraseLen = phraseEnd - phraseStart + 1;
480  if (phraseLen < WF_MIN_WPA_PASS_PHRASE_LENGTH || WF_MAX_WPA_PASS_PHRASE_LENGTH < phraseLen)
481  {
482  WFConsolePrintRomStr("Phrase string must be at least 8 chars and no greater than 64", TRUE);
483  return FALSE;
484  }
485 
486  iwprivCb.securityType = securityType;
487 
488  memcpy((void*)iwprivCb.securityKey, (const void*)phraseStart, phraseLen);
489  iwprivCb.securityKey[phraseLen] = '\0'; // just for easy printing on the console
490  iwprivCb.securityKeyLength = phraseLen;
491 
492  WF_CPSetSecurity(iwprivCb.cpId, iwprivCb.securityType, 0,
493  iwprivCb.securityKey, iwprivCb.securityKeyLength);
494 
495  return TRUE;
496 }
