886  the client driver table.
887 
888  Description:

903 
904  Remarks:
905  This is a internal API only. This should not be called by anything other
906  than the USB host stack via the client driver table
907  ***************************************************************************/
908 BOOL AndroidAppInitialize ( BYTE address, DWORD flags, BYTE clientDriverID )
909 {
910  BYTE *config_descriptor = NULL;
911  BYTE *device_descriptor = NULL;
912  WORD tempWord;
913  BYTE *config_desc_end;
914 
915  ANDROID_DEVICE_DATA* device = NULL;
916  BYTE i;
917 
918  device_descriptor = USBHostGetDeviceDescriptor(address);
919 
920  ReadWORD(&tempWord, &device_descriptor[USB_DEV_DESC_VID_OFFSET]);
921 
922  for(i=0;i<NUM_ANDROID_DEVICES_SUPPORTED;i++)
923  {
924  if( (devices[i].state == WAITING_FOR_ACCESSORY_RETURN) || ( (flags & ANDROID_INIT_FLAG_BYPASS_PROTOCOL) == ANDROID_INIT_FLAG_BYPASS_PROTOCOL) )
925  {
926  device = &devices[i];
927  device->state = RETURN_OF_THE_ACCESSORY;
928  break;
929  }
930  }
931 
932  //if this isn't an old accessory, then it must be a new one
933  if(device == NULL)
934  {
935  //Find the first available device.
936  for(i=0;i<NUM_ANDROID_DEVICES_SUPPORTED;i++)
937  {
938  if(devices[i].state == NO_DEVICE)
939  {
940  device = &devices[i];
941  if( (flags & ANDROID_INIT_FLAG_BYPASS_PROTOCOL) == ANDROID_INIT_FLAG_BYPASS_PROTOCOL)
942  {
943  device->state = RETURN_OF_THE_ACCESSORY;
944  }
945  else
946  {
947  device->state = DEVICE_ATTACHED;
948  device->countDown = 1000;
949  }
950  break;
951  }
952  }
953  }
954 
955  if(device == NULL)
956  {
957  return TRUE;
958  }
959 
960  config_descriptor = USBHostGetCurrentConfigurationDescriptor( address );
961 
962  //Save the total length for this configuration descriptor
963  ReadWORD(&tempWord,&config_descriptor[2]);
964 
965  //Record the end of the descriptor so we know when to stop searching through
966  // the descriptor list
967  config_desc_end = config_descriptor + tempWord;
968 
969  //Skip past the configuration part of this descriptor to the next
970  // descriptor in the configuration descriptor list. The size of the config
971  // part of the descriptor is the first byte of the list.
972  config_descriptor += *config_descriptor;
973 
974  //Search the entire configuration descriptor for COMM interfaces
975  while(config_descriptor < config_desc_end)
976  {
977  //We are expecting a interface descriptor
978  if(config_descriptor[USB_DESC_BDESCRIPTORTYPE_OFFSET] != USB_DESCRIPTOR_INTERFACE)
979  {
980  //Jump past this descriptor by adding the current descriptor length
981  // to the current descriptor pointer.
982  config_descriptor += config_descriptor[USB_DESC_BLENGTH_OFFSET];
983 
984  //Jump back to the top of the while loop to continue searching through
985  // this configuration for the next interface
986  continue;
987  }
988 
989  device->address = address;
990  device->clientDriverID = clientDriverID;
991 
992  if( (config_descriptor[USB_INTERFACE_DESC_BINTERFACECLASS_OFFSET] == 0xFF) &&
993  (config_descriptor[USB_INTERFACE_DESC_BINTERFACESUBCLASS_OFFSET] == 0xFF))
994  {
995  //Jump past this descriptor to the next descriptor.
996  config_descriptor += config_descriptor[USB_DESC_BLENGTH_OFFSET];
997 
998  //Parse through the rest of this interface. Stop when we reach the
999  // next interface or the end of the configuration descriptor
1000  while((config_descriptor[USB_DESC_BDESCRIPTORTYPE_OFFSET] != USB_DESCRIPTOR_INTERFACE) && (config_descriptor < config_desc_end))
1001  {
1002  if(config_descriptor[USB_DESC_BDESCRIPTORTYPE_OFFSET] == USB_DESCRIPTOR_ENDPOINT)
1003  {
1004  //If this is an endpoint descriptor in the DATA interface, then
1005  // copy all of the endpoint data to the device information.
1006 
1007  if((config_descriptor[USB_ENDPOINT_DESC_BENDPOINTADDRESS_OFFSET] & 0x80) == 0x80)
1008  {
1009  //If this is an IN endpoint, record the endpoint number
1010  device->INEndpointNum = config_descriptor[USB_ENDPOINT_DESC_BENDPOINTADDRESS_OFFSET];
1011 
1012  //record the endpoint size (2 bytes)
1013  device->INEndpointSize = (config_descriptor[USB_ENDPOINT_DESC_WMAXPACKETSIZE_OFFSET]) + (config_descriptor[USB_ENDPOINT_DESC_WMAXPACKETSIZE_OFFSET+1] << 8);
1014  }
1015  else
1016  {
1017  //Otherwise this is an OUT endpoint, record the endpoint number
1018  device->OUTEndpointNum = config_descriptor[USB_ENDPOINT_DESC_BENDPOINTADDRESS_OFFSET];
1019 
1020  //record the endpoint size (2 bytes)
1021  device->OUTEndpointSize = (config_descriptor[USB_ENDPOINT_DESC_WMAXPACKETSIZE_OFFSET]) + (config_descriptor[USB_ENDPOINT_DESC_WMAXPACKETSIZE_OFFSET+1] << 8);
1022  }
1023  }
1024  config_descriptor += config_descriptor[USB_DESC_BLENGTH_OFFSET];
1025  }
1026  }
1027  else
1028  {
1029  //Jump past this descriptor by adding the current descriptor length
1030  // to the current descriptor pointer.
1031  config_descriptor += config_descriptor[USB_DESC_BLENGTH_OFFSET];
1032  }
1033  }
1034 
1035  return TRUE;
