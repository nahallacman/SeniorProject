3  USB Host Mass Storage Device Driver
4 
5 This is the Mass Storage Class driver file for a USB Embedded Host device.
6 This file should be used in a project with usb_host.c to provided the USB
7 hardware interface.
8 
9 Acronyms/abbreviations used by this class:
10  * LUN - Logical Unit Number
11  * CBW - Command Block Wrapper
12  * CSW - Command Status Wrapper
13 
14 To interface with usb_host.c, the routine USBHostMSDClientInitialize() should be
15 specified as the Initialize() function, and USBHostMSDClientEventHandler() should
16 be specified as the EventHandler() function in the usbClientDrvTable[] array
17 declared in usb_config.h.
18 
19 This driver can be configured to use transfer events from usb_host.c. Transfer
20 events require more RAM and ROM than polling, but it cuts down or even
21 eliminates the required polling of the various USBxxxTasks functions. For this
22 class, USBHostMSDTasks() is compiled out if transfer events from usb_host.c
23 are used. However, USBHostTasks() still must be called to provide attach,
24 enumeration, and detach services. If transfer events from usb_host.c
25 are going to be used, USB_ENABLE_TRANSFER_EVENT should be defined. If transfer
26 status is going to be polled, USB_ENABLE_TRANSFER_EVENT should not be defined.
27 
28 This driver can also be configured to provide mass storage transfer events to
29 the next layer. Generating these events requires a small amount of extra ROM,
30 but no extra RAM. The layer above this driver must be configured to receive
31 and respond to the events. If mass storage transfer events are going to be
32 sent to the next layer, USB_MSD_ENABLE_TRANSFER_EVENT should be defined. If
33 mass storage transfer status is going to be polled,
34 USB_MSD_ENABLE_TRANSFER_EVENT should not be defined.
35 
36 Since mass storage is performed with bulk transfers, USB_SUPPORT_BULK_TRANSFERS
37 must be defined. For maximum throughput, it is recommended that
38 ALLOW_MULTIPLE_BULK_TRANSACTIONS_PER_FRAME be defined. For maximum
39 compatibility with mass storage devices, it is recommended that
40 ALLOW_MULTIPLE_NAKS_PER_FRAME not be defined.

392 BYTE USBHostMSDDeviceStatus( BYTE deviceAddress )
393 {
394  BYTE i;
395  BYTE status;
396 
397  // Make sure a valid device is being requested.
398  if ((deviceAddress == 0) || (deviceAddress > 127))
399  {
400  return USB_MSD_DEVICE_NOT_FOUND;
401  }
402 
403  // Find the correct device.
404  for (i=0; (i<USB_MAX_MASS_STORAGE_DEVICES) && (deviceInfoMSD[i].deviceAddress != deviceAddress); i++);
405  if (i == USB_MAX_MASS_STORAGE_DEVICES)
406  {
407  return USB_MSD_DEVICE_NOT_FOUND;
408  }
409 
410  status = USBHostDeviceStatus( deviceAddress );
411  if (status != USB_DEVICE_ATTACHED)
412  {
413  return status;
414  }
415  else
416  {
417  // The device is attached and done enumerating. We can get more specific now.
418 
419  #ifndef USB_ENABLE_TRANSFER_EVENT
420  switch (deviceInfoMSD[i].state & STATE_MASK)
421  {
422  case STATE_INITIALIZE_DEVICE:
423  return USB_MSD_INITIALIZING;
424  break;
425 
426  case STATE_RUNNING:
427  return USB_MSD_NORMAL_RUNNING;
428  break;
429 
430  case STATE_HOLDING:
431  return USB_MSD_ERROR_STATE;
432  break;
433 
434  case STATE_MSD_RESET_RECOVERY:
435  case STATE_MSD_CLEAR_DATA_IN:
436  case STATE_MSD_CLEAR_DATA_OUT:
437  return USB_MSD_RESETTING_DEVICE;
438  break;
439 
440  default:
441  return USB_MSD_DEVICE_DETACHED;
442  break;
443  }
444  #else
445  switch (deviceInfoMSD[i].state)
446  {
447  case STATE_WAIT_FOR_MAX_LUN:
448  return USB_MSD_INITIALIZING;
449  break;
450 
451  case STATE_RUNNING:
452  case STATE_CBW_WAIT:
453  case STATE_TRANSFER_WAIT:
454  case STATE_CSW_WAIT:
455  case STATE_REQUEST_CSW:
456  return USB_MSD_NORMAL_RUNNING;
457  break;
458 
459  case STATE_HOLDING:
460  return USB_MSD_ERROR_STATE;
461  break;
462 
463  case STATE_WAIT_FOR_RESET:
464  case STATE_WAIT_FOR_CLEAR_IN:
465  case STATE_WAIT_FOR_CLEAR_OUT:
466  return USB_MSD_RESETTING_DEVICE;
467  break;
468 
469  default:
470  return USB_MSD_DEVICE_DETACHED;
471  break;
472  }
473  #endif
474  }

570 void USBHostMSDTasks( void )
571 {
572 
573 #ifndef USB_ENABLE_TRANSFER_EVENT
574 
575  DWORD byteCount;
576  BYTE errorCode;
577  BYTE i;
578 
579  for (i=0; i<USB_MAX_MASS_STORAGE_DEVICES; i++)
580  {
581  if (deviceInfoMSD[i].deviceAddress != 0)
582  {
583  switch (deviceInfoMSD[i].state & STATE_MASK)
584  {
585  case STATE_DETACHED:
586  // No device attached.
587  break;
588 
589  case STATE_INITIALIZE_DEVICE:
590  switch (deviceInfoMSD[i].state & SUBSTATE_MASK)
591  {
592  case SUBSTATE_WAIT_FOR_ENUMERATION:
593  if (USBHostDeviceStatus( deviceInfoMSD[i].deviceAddress ) == USB_DEVICE_ATTACHED)
594  {
595  _USBHostMSD_SetNextSubState();
596  #ifdef DEBUG_MODE
597  UART2PrintString( "MSD: Getting max LUN...\r\n" );
598  #endif
599  }
600  break;
601 
602  case SUBSTATE_SEND_GET_MAX_LUN:
603  // If we are currently sending a token, we cannot do anything.
604  if (U1CONbits.TOKBUSY)
605  break;
606 
607  if (!USBHostIssueDeviceRequest( deviceInfoMSD[i].deviceAddress, USB_SETUP_DEVICE_TO_HOST | USB_SETUP_TYPE_CLASS | USB_SETUP_RECIPIENT_INTERFACE,
608  USB_MSD_GET_MAX_LUN, 0, deviceInfoMSD[i].interface, 1, deviceInfoMSD[i].block.data, USB_DEVICE_REQUEST_GET, deviceInfoMSD[i].clientDriverID ))
609  {
610  _USBHostMSD_SetNextSubState();
611  }
612  else
613  {
614  // We'll try again.
615  }
616  break;
617 
618  case SUBSTATE_WAIT_FOR_MAX_LUN:
619  if (USBHostTransferIsComplete( deviceInfoMSD[i].deviceAddress, 0, &errorCode, &byteCount ))
620  {
621  deviceInfoMSD[i].maxLUN = 0;
622  if (!errorCode)
623  {
624  deviceInfoMSD[i].maxLUN = deviceInfoMSD[i].block.data[0];
625  }
626  else
627  {
628  // Clear the STALL. Since it is EP0, we do not have to clear the stall.
629  USBHostClearEndpointErrors( deviceInfoMSD[i].deviceAddress, 0 );
630  }
631  _USBHostMSD_SetNextSubState();
632  }
633  break;
634 
635  case SUBSTATE_GET_MAX_LUN_COMPLETE:
636  #ifdef DEBUG_MODE
637  UART2PrintString( "MSD: Max LUN is " );
638  UART2PutHex( deviceInfoMSD[i].maxLUN );
639  UART2PrintString( "\r\nMSD: Running...\r\n" );
640  #endif
641 
642  // Tell the media interface layer that we have a MSD attached.
643  if (usbMediaInterfaceTable.Initialize( deviceInfoMSD[i].deviceAddress, usbMediaInterfaceTable.flags, 0 ))
644  {
645  usbMediaInterfaceTable.EventHandler( deviceInfoMSD[i].deviceAddress, EVENT_MSD_MAX_LUN, &(deviceInfoMSD[i].maxLUN), 1 );
646  _USBHostMSD_SetNextState();
647  }
648  else
649  {
650  // The media interface layer cannot support the device.
651  deviceInfoMSD[i].errorCode = USB_MSD_MEDIA_INTERFACE_ERROR;
652  deviceInfoMSD[i].state = STATE_HOLDING;
653  }
654  break;
655  }
656  break;
657 
658  case STATE_RUNNING:
659  switch (deviceInfoMSD[i].state & SUBSTATE_MASK)
660  {
661  case SUBSTATE_HOLDING:
662  break;
663 
664  case SUBSTATE_SEND_CBW:
665  #ifdef DEBUG_MODE
666  UART2PrintString( "MSD: Writing CBW\r\n" );
667  #endif
668  errorCode = USBHostWrite( deviceInfoMSD[i].deviceAddress, deviceInfoMSD[i].endpointOUT, deviceInfoMSD[i].block.data, CBW_SIZE );
669  if (errorCode)
670  {
671  _USBHostMSD_TerminateTransfer( errorCode );
672  }
673  else
674  {
675  _USBHostMSD_SetNextSubState();
676  }
677  break;
678 
679  case SUBSTATE_CBW_WAIT:
680  if (USBHostTransferIsComplete( deviceInfoMSD[i].deviceAddress, deviceInfoMSD[i].endpointOUT, &errorCode, &byteCount ))
681  {
682  if (errorCode)
683  {
684  #ifdef DEBUG_MODE
685  UART2PrintString( "MSD: Error with sending CBW\r\n" );
686  #endif
687  _USBHostMSD_TerminateTransfer( errorCode );
688  }
689  else if (byteCount != CBW_SIZE)
690  {
691  #ifdef DEBUG_MODE
692  UART2PrintString( "MSD: CBW size not correct\r\n" );
693  #endif
694  _USBHostMSD_TerminateTransfer( USB_MSD_CBW_ERROR );
695  }
696  else
697  {
698  if (deviceInfoMSD[i].block.cbw.dCBWDataTransferLength == 0)
699  {
700  #ifdef DEBUG_MODE
701  UART2PrintString( "MSD: Transfer length=0\r\n" );
702  #endif
703  // Skip to get the CSW
704  deviceInfoMSD[i].state = STATE_RUNNING | SUBSTATE_REQUEST_CSW;
705  }
706  else
707  {
708  #ifdef DEBUG_MODE
709  UART2PrintString( "MSD: Going on...\r\n" );
710  #endif
711  _USBHostMSD_SetNextSubState();
712  }
713  }
714  }
715  break;
716 
717  case SUBSTATE_TRANSFER_DATA:
718  #ifdef DEBUG_MODE
719  UART2PrintString( "MSD: Transferring data, length ");
720  UART2PutHexDWord( deviceInfoMSD[i].userDataLength );
721  UART2PrintString( "\r\n" );
722  #endif
723  if (deviceInfoMSD[i].userDataLength == 0)
724  {
725  deviceInfoMSD[i].state = STATE_RUNNING | SUBSTATE_REQUEST_CSW;
726  }
727  else
728  {
729  if (!deviceInfoMSD[i].flags.bfDirection) // OUT
730  {
731  errorCode = USBHostWrite( deviceInfoMSD[i].deviceAddress, deviceInfoMSD[i].endpointDATA, deviceInfoMSD[i].userData, deviceInfoMSD[i].userDataLength );
732  }
733  else
734  {
735  errorCode = USBHostRead( deviceInfoMSD[i].deviceAddress, deviceInfoMSD[i].endpointDATA, deviceInfoMSD[i].userData, deviceInfoMSD[i].userDataLength );
736  }
737 
738  if (errorCode)
739  {
740  _USBHostMSD_TerminateTransfer( errorCode );
741  }
742  else
743  {
744  _USBHostMSD_SetNextSubState();
745  }
746  }
747  break;
748 
749  case SUBSTATE_TRANSFER_WAIT:
750  if (USBHostTransferIsComplete( deviceInfoMSD[i].deviceAddress, deviceInfoMSD[i].endpointDATA, &errorCode, &byteCount ))
751  {
752  if (errorCode)
753  {
754  if (errorCode == USB_ENDPOINT_STALLED)
755  {
756  // Clear the stall, then try to get the CSW.
757  #ifdef DEBUG_MODE
758  UART2PrintString( "MSD: Stall on data\r\n" );
759  #endif
760  USBHostClearEndpointErrors( deviceInfoMSD[i].deviceAddress, deviceInfoMSD[i].endpointDATA );
761  if (!deviceInfoMSD[i].flags.bfDirection) // OUT
762  {
763  deviceInfoMSD[i].flags.bfClearDataOUT = 1;
764  }
765  else
766  {
767  deviceInfoMSD[i].flags.bfClearDataIN = 1;
768  }
769  deviceInfoMSD[i].returnState = STATE_RUNNING | SUBSTATE_REQUEST_CSW;
770  _USBHostMSD_ResetStateJump( i );
771 
772  }
773  else
774  {
775  //Error recovery here is not explicitly covered in the spec. Unfortunately, some
776  // thumb drives generate a turn-around time error here sometimes.
777  //_USBHostMSD_TerminateTransfer( errorCode );
778  deviceInfoMSD[i].flags.val |= MARK_RESET_RECOVERY;
779  deviceInfoMSD[i].returnState = STATE_RUNNING | SUBSTATE_SEND_CBW; // Try the transfer again.
780  _USBHostMSD_ResetStateJump( i );
781  }
782  }
783  else
784  {
785  _USBHostMSD_SetNextSubState();
786  }
787  }
788  break;
789 
790  case SUBSTATE_REQUEST_CSW:
791  #ifdef DEBUG_MODE
792  UART2PrintString( "MSD: Getting CSW\r\n" );
793  #endif
794 
795  errorCode = USBHostRead( deviceInfoMSD[i].deviceAddress, deviceInfoMSD[i].endpointIN, deviceInfoMSD[i].block.data, CSW_SIZE );
796  if (errorCode)
797  {
798  _USBHostMSD_TerminateTransfer( errorCode );
799  }
800  else
801  {
802  _USBHostMSD_SetNextSubState();
803  }
804  break;
805 
806  case SUBSTATE_CSW_WAIT:
807  if (USBHostTransferIsComplete( deviceInfoMSD[i].deviceAddress, deviceInfoMSD[i].endpointIN, &errorCode, &byteCount ))
808  {
809  #ifdef DEBUG_MODE
810  UART2PrintString( "MSD: Got CSW-" );
811  #endif
812  if (errorCode)
813  {
814  deviceInfoMSD[i].attemptsCSW--;
815  if (deviceInfoMSD[i].attemptsCSW)
816  {
817  USBHostClearEndpointErrors( deviceInfoMSD[i].deviceAddress, deviceInfoMSD[i].endpointIN );
818  deviceInfoMSD[i].flags.bfClearDataIN = 1;
819  deviceInfoMSD[i].returnState = STATE_RUNNING | SUBSTATE_REQUEST_CSW;
820  _USBHostMSD_ResetStateJump( i );
821  }
822  else
823  {
824  _USBHostMSD_TerminateTransfer( errorCode );
825  }
826  }
827  else if ((byteCount != CSW_SIZE) |
828  (deviceInfoMSD[i].block.csw.dCSWSignature != USB_MSD_DCSWSIGNATURE) |
829  (deviceInfoMSD[i].block.csw.dCSWTag != deviceInfoMSD[i].dCBWTag) )
830  {
831  _USBHostMSD_TerminateTransfer( USB_MSD_CSW_ERROR );
832  }
833  else
834  {
835  deviceInfoMSD[i].bytesTransferred = deviceInfoMSD[i].userDataLength - deviceInfoMSD[i].block.csw.dCSWDataResidue;
836 
837  if (deviceInfoMSD[i].block.csw.dCSWStatus != 0x00)
838  {
839  _USBHostMSD_TerminateTransfer( deviceInfoMSD[i].block.csw.dCSWStatus | USB_MSD_ERROR );
840  }
841  else
842  {
843  _USBHostMSD_TerminateTransfer( USB_SUCCESS );
844  }
845 
846  // If we have a phase error, we need to perform corrective action instead of
847  // returning to normal running.
848  if (deviceInfoMSD[i].block.csw.dCSWStatus == MSD_PHASE_ERROR)
849  {
850  deviceInfoMSD[i].flags.val |= MARK_RESET_RECOVERY;
851  deviceInfoMSD[i].returnState = STATE_RUNNING | SUBSTATE_HOLDING;
852  _USBHostMSD_ResetStateJump( i );
853  }
854  }
855  }
856  break;
857 
858  case SUBSTATE_TRANSFER_DONE:
859  deviceInfoMSD[i].state = STATE_RUNNING | SUBSTATE_HOLDING;
860  #ifdef USB_MSD_ENABLE_TRANSFER_EVENT
861  usbMediaInterfaceTable.EventHandler( deviceInfoMSD[i].deviceAddress, EVENT_MSD_TRANSFER, NULL, 0 );
862  #endif
863  break;
864  }
865  break;
866 
867  case STATE_MSD_RESET_RECOVERY:
868  switch (deviceInfoMSD[i].state & SUBSTATE_MASK)
869  {
870  case SUBSTATE_SEND_RESET:
871  errorCode = USBHostIssueDeviceRequest( deviceInfoMSD[i].deviceAddress, USB_SETUP_HOST_TO_DEVICE | USB_SETUP_TYPE_CLASS | USB_SETUP_RECIPIENT_INTERFACE,
872  USB_MSD_RESET, 0, deviceInfoMSD[i].interface, 0, NULL, USB_DEVICE_REQUEST_SET, deviceInfoMSD[i].clientDriverID );
873 
874  if (errorCode)
875  {
876  //Error recovery here is not explicitly covered in the spec.
877  _USBHostMSD_TerminateTransfer( USB_MSD_RESET_ERROR );
878  }
879  else
880  {
881  _USBHostMSD_SetNextSubState();
882  }
883  break;
884 
885  case SUBSTATE_WAIT_FOR_RESET:
886  if (USBHostTransferIsComplete( deviceInfoMSD[i].deviceAddress, 0, &errorCode, &byteCount ))
887  {
888  if (errorCode)
889  {
890  //Error recovery here is not explicitly covered in the spec.
891  _USBHostMSD_TerminateTransfer( USB_MSD_RESET_ERROR );
892  }
893  else
894  {
895  deviceInfoMSD[i].flags.bfReset = 0;
896  _USBHostMSD_SetNextSubState();
897  }
898  }
899  break;
900 
901  case SUBSTATE_RESET_COMPLETE:
902  _USBHostMSD_ResetStateJump( i );
903  break;
904  }
905  break;
906 
907  case STATE_MSD_CLEAR_DATA_IN:
908  switch (deviceInfoMSD[i].state & SUBSTATE_MASK)
909  {
910  case SUBSTATE_SEND_CLEAR_IN:
911  errorCode = USBHostIssueDeviceRequest( deviceInfoMSD[i].deviceAddress, USB_SETUP_HOST_TO_DEVICE | USB_SETUP_TYPE_STANDARD | USB_SETUP_RECIPIENT_ENDPOINT,
912  USB_REQUEST_CLEAR_FEATURE, USB_FEATURE_ENDPOINT_HALT, deviceInfoMSD[i].endpointIN, 0, NULL, USB_DEVICE_REQUEST_SET, deviceInfoMSD[i].clientDriverID );
913 
914  if (errorCode)
915  {
916  //Error recovery here is not explicitly covered in the spec.
917  _USBHostMSD_TerminateTransfer( USB_MSD_RESET_ERROR );
918  }
919  else
920  {
921  _USBHostMSD_SetNextSubState();
922  }
923  break;
924 
925  case SUBSTATE_WAIT_FOR_CLEAR_IN:
926  if (USBHostTransferIsComplete( deviceInfoMSD[i].deviceAddress, 0, &errorCode, &byteCount ))
927  {
928  if (errorCode)
929  {
930  //Error recovery here is not explicitly covered in the spec.
931  _USBHostMSD_TerminateTransfer( USB_MSD_RESET_ERROR );
932  }
933  else
934  {
935  deviceInfoMSD[i].flags.bfClearDataIN = 0;
936  _USBHostMSD_SetNextSubState();
937  }
938  }
939  break;
940 
941  case SUBSTATE_CLEAR_IN_COMPLETE:
942  _USBHostMSD_ResetStateJump( i );
943  break;
944  }
945  break;
946 
947  case STATE_MSD_CLEAR_DATA_OUT:
948  switch (deviceInfoMSD[i].state & SUBSTATE_MASK)
949  {
950  case SUBSTATE_SEND_CLEAR_OUT:
951  errorCode = USBHostIssueDeviceRequest( deviceInfoMSD[i].deviceAddress, USB_SETUP_HOST_TO_DEVICE | USB_SETUP_TYPE_STANDARD | USB_SETUP_RECIPIENT_ENDPOINT,
952  USB_REQUEST_CLEAR_FEATURE, USB_FEATURE_ENDPOINT_HALT, deviceInfoMSD[i].endpointOUT, 0, NULL, USB_DEVICE_REQUEST_SET, deviceInfoMSD[i].clientDriverID );
953 
954  if (errorCode)
955  {
956  //Error recovery here is not explicitly covered in the spec.
957  _USBHostMSD_TerminateTransfer( USB_MSD_RESET_ERROR );
958  }
959  else
960  {
961  _USBHostMSD_SetNextSubState();
962  }
963  break;
964 
965  case SUBSTATE_WAIT_FOR_CLEAR_OUT:
966  if (USBHostTransferIsComplete( deviceInfoMSD[i].deviceAddress, 0, &errorCode, &byteCount ))
967  {
968  if (errorCode)
969  {
970  //Error recovery here is not explicitly covered in the spec.
971  _USBHostMSD_TerminateTransfer( USB_MSD_RESET_ERROR );
972  }
973  else
974  {
975  deviceInfoMSD[i].flags.bfClearDataOUT = 0;
976  _USBHostMSD_SetNextSubState();
977  }
978  }
979  break;
980 
981  case SUBSTATE_CLEAR_OUT_COMPLETE:
982  _USBHostMSD_ResetStateJump( i );
983  break;
984  }
985  break;
986 
987  case STATE_HOLDING:
988  break;
989  }
990  }
991  }
992 #endif

1090 
1091 BYTE USBHostMSDTransfer( BYTE deviceAddress, BYTE deviceLUN, BYTE direction, BYTE *commandBlock,
1092  BYTE commandBlockLength, BYTE *data, DWORD dataLength )
1093 {
1094  BYTE i;
1095  BYTE j;
1096 
1097  #ifdef DEBUG_MODE
1098  UART2PrintString( "MSD: Transfer: " );
1099  if (direction)
1100  {
1101  UART2PrintString( "Read, " );
1102  }
1103  else
1104  {
1105  UART2PrintString( "Write, " );
1106  }
1107  #endif
1108 
1109  // Make sure a valid device is being requested.
1110  if ((deviceAddress == 0) || (deviceAddress > 127))
1111  {
1112  return USB_MSD_DEVICE_NOT_FOUND;
1113  }
1114 
1115  // Find the correct device.
1116  for (i=0; (i<USB_MAX_MASS_STORAGE_DEVICES) && (deviceInfoMSD[i].deviceAddress != deviceAddress); i++);
1117  if (i == USB_MAX_MASS_STORAGE_DEVICES)
1118  {
1119  return USB_MSD_DEVICE_NOT_FOUND;
1120  }
1121 
1122  // Make sure the device is in a state ready to read/write.
1123  #ifndef USB_ENABLE_TRANSFER_EVENT
1124  if (deviceInfoMSD[i].state != (STATE_RUNNING | SUBSTATE_HOLDING))
1125  #else
1126  if (deviceInfoMSD[i].state != STATE_RUNNING)
1127  #endif
1128  {
1129  return USB_MSD_DEVICE_BUSY;
1130  }
1131

1133  if (deviceLUN > deviceInfoMSD[i].maxLUN)
1134  {
1135  return USB_MSD_INVALID_LUN;
1136  }
1137 
1138  // Initialize the transfer information.
1139  deviceInfoMSD[i].attemptsCSW = CSW_RECEIVE_ATTEMPTS;
1140  deviceInfoMSD[i].bytesTransferred = 0;
1141  deviceInfoMSD[i].errorCode = USB_SUCCESS;
1142  deviceInfoMSD[i].flags.val = 0;
1143  deviceInfoMSD[i].flags.bfDirection = direction;
1144  deviceInfoMSD[i].userData = data;
1145  deviceInfoMSD[i].userDataLength = dataLength;
1146  deviceInfoMSD[i].dCBWTag = _USBHostMSD_GetNextTag();

1148  if (!direction) // OUT
1149  {
1150  deviceInfoMSD[i].endpointDATA = deviceInfoMSD[i].endpointOUT;
1151  }
1152  #ifdef DEBUG_MODE
1153  UART2PrintString( "Data EP: " );
1154  UART2PutHex( deviceInfoMSD[i].endpointDATA );
1155  UART2PrintString( "\r\n" );
1156  #endif
1157 
1158  // Prepare the CBW so we can give the user back his command block RAM.
1159  deviceInfoMSD[i].block.cbw.dCBWSignature = USB_MSD_DCBWSIGNATURE;
1160  deviceInfoMSD[i].block.cbw.dCBWTag = deviceInfoMSD[i].dCBWTag;
1161  deviceInfoMSD[i].block.cbw.dCBWDataTransferLength = deviceInfoMSD[i].userDataLength;
1162  deviceInfoMSD[i].block.cbw.bmCBWflags.val = 0;
1163  deviceInfoMSD[i].block.cbw.bmCBWflags.bfDirection = direction;
1164  deviceInfoMSD[i].block.cbw.bCBWLUN = deviceLUN;

1166  for (j=0; j<commandBlockLength; j++)
1167  {
1168  deviceInfoMSD[i].block.cbw.CBWCB[j] = commandBlock[j];
1169  }
1170 
1171  #ifndef USB_ENABLE_TRANSFER_EVENT
1172  // Jump to the transfer state.
1173  deviceInfoMSD[i].state = STATE_RUNNING | SUBSTATE_SEND_CBW;
1174  #else
1175  j = USBHostWrite( deviceInfoMSD[i].deviceAddress, deviceInfoMSD[i].endpointOUT, deviceInfoMSD[i].block.data, CBW_SIZE );
1176  if (j)
1177  {
1178  _USBHostMSD_TerminateTransfer( j );
1179  }
1180  else
1181  {
1182  deviceInfoMSD[i].state = STATE_CBW_WAIT;
1183  }
1184  #endif
1185 
1186  return USB_SUCCESS;
1187 }

1332

1334 
1335  #ifdef DEBUG_MODE

1337  UART2PutDec( descriptor[i+5] );

1339  #endif
1340
