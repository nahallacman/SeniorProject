65 * Function: WORD GFXTranslateDirection(WORD direction, WORD orientation)
66 *
67 * Overview: This function returns the actual direction taking the
68 * screen orientation into consideration (local function)
69 *
70 * Input: direction -> The absolute direction required
71 * orientation -> The screen rotation/orientation
72 *
73 * Output: The new direction taking the screen orientation into
74 * consideration
75 *
76 ************************************************************************/
77 WORD GFXTranslateDirection(WORD direction, WORD orientation)
78 {
79  switch(orientation)
80  {
81  case 90:
82  switch(direction)

85  case RIGHT_TO_LEFT: return TOP_TO_BOTTOM;
86  case TOP_TO_BOTTOM: return LEFT_TO_RIGHT;
87  case BOTTOM_TO_TOP: return RIGHT_TO_LEFT;
88  case HORIZONTAL: return VERTICAL;
89  case VERTICAL: return HORIZONTAL;
90  }
91  break;
92 
93  case 180:
94  switch(direction)
95  {
96  case LEFT_TO_RIGHT: return RIGHT_TO_LEFT;
97  case RIGHT_TO_LEFT: return LEFT_TO_RIGHT;
98  case TOP_TO_BOTTOM: return BOTTOM_TO_TOP;
99  case BOTTOM_TO_TOP: return TOP_TO_BOTTOM;
100  }
101  break;
102 
103  case 270:

105  {
106  case LEFT_TO_RIGHT: return TOP_TO_BOTTOM;
107  case RIGHT_TO_LEFT: return BOTTOM_TO_TOP;

109  case BOTTOM_TO_TOP: return LEFT_TO_RIGHT;
110  case HORIZONTAL: return VERTICAL;
111  case VERTICAL: return HORIZONTAL;

217 * Function: void ExpandRectangle(void)
218 *
219 * Overview: Expanding rectangle from the mid to the periphery
220 *
221 * Input: param1 (GFXTransition()) -> Pixel-block size to expand
222 *
223 * Output: none
224 *
225 ************************************************************************/
226 void ExpandRectangle(void)
227 {
228  #define SCALE 100
229 
230  WORD x = Width / 2;
231  WORD y = Height / 2;
232  WORD den = (x > y)? y: x;
233  WORD xpitch = x * SCALE / den;
234  WORD ypitch = y * SCALE / den;
235  WORD i;
236  WORD blocksize = _param1;
237 
238  if(blocksize == 0)
239  {
240  blocksize = 1;
241  }
242  else if(blocksize > GFX_TRANSITION_MAX_BLOCKSIZE)
243  {
244  blocksize = GFX_TRANSITION_MAX_BLOCKSIZE;
245  }
246 
247  while(IsDeviceBusy());
248 
249  for(i = 0; i <= den; i += blocksize)
250  {
251  x = (Width / 2) - ((i * xpitch) / SCALE);
252  y = (Height / 2) - ((i * ypitch) / SCALE);
253  CopyBlock(_srcpageaddr, _destpageaddr, ((DWORD)(Starty + y) * DISP_HOR_RESOLUTION) + Startx + x, ((DWORD)(Starty + y) * DISP_HOR_RESOLUTION) + Startx + x, 2 * xpitch * i / SCALE, 2 * ypitch * i / SCALE);
254  DelayMs(_delay_ms);
255  while(IsDeviceBusy());
256  }

491 * WORD delay_ms, WORD param1, WORD param2)
492 *
493 * Overview: This function saves the parameters to be used and marks the
494 * transition to be pending which is executed when
495 * GFXExecutePendingTransition() is called
496 *
497 * Input: left -> left x coordinate
498 * top -> top y coordinate
499 * right -> right x coordinate
500 * bottom -> bottom y coordinate
501 * type -> Transition type
502 * srcpageaddr -> Source page address for the transition
503 * destpageaddr -> Destination page address for the transition
504 * delay_ms -> Delay in milli seconds between redraws in the
505 * screen while executing the transition
506 * param1 -> Transition-type specific parameter
507 * param2 -> Transition-type specific parameter
508 *
509 * Output: 0 -> Transition executed successfully
510 * -1 -> Transition not executed
511 *
512 ************************************************************************/
513 BYTE GFXTransition(SHORT left, SHORT top, SHORT right, SHORT bottom, GFX_TRANSITION_TYPE type, DWORD srcpageaddr, DWORD destpageaddr, WORD delay_ms, WORD param1, WORD param2)
514 {
515  BYTE retval = 0;
516  SHORT width, height;
517 
518  if(left > right)
519  {
520  return (-1); /* Don't draw */
521  }
522 
523  if(top > bottom)
524  {
525  return (-1); /* Don't draw */
526  }
527 
528  left = (left < 0) ? 0: (left > GetMaxX()) ? GetMaxX(): left;
529  top = (top < 0) ? 0: (top > GetMaxY()) ? GetMaxY(): top;
530  right = (right < 0) ? 0: (right > GetMaxX()) ? GetMaxX(): right;
531  bottom = (bottom < 0) ? 0: (bottom > GetMaxY())? GetMaxY(): bottom;
532 
533  #if (DISP_ORIENTATION == 0) || (DISP_ORIENTATION == 180)
534 
535  width = right - left + 1;
536  height = bottom - top + 1;
537 
538  #elif (DISP_ORIENTATION == 90) || (DISP_ORIENTATION == 270)
539 
540  height = right - left + 1;
541  width = bottom - top + 1;
542 
543  #endif
544 
545  #if (DISP_ORIENTATION == 90)
546 
547  bottom = DISP_VER_RESOLUTION - left;
548  left = top;
549  right = left + width;
550  top = bottom - height;
551 
552  #elif (DISP_ORIENTATION == 180)
553 
554  right = DISP_HOR_RESOLUTION - left;
555  bottom = DISP_VER_RESOLUTION - top;
556  left = right - width;
557  top = bottom - height;
558 
559  #elif (DISP_ORIENTATION == 270)
560 
561  right = DISP_HOR_RESOLUTION - top;
562  top = left;
563  bottom = top + height;
564  left = right - width;
565 
566  #endif
567 
568  Startx = left;
569  Starty = top;
570  Width = width;
571  Height = height;
572 
573  _delay_ms = delay_ms;
574  _srcpageaddr = srcpageaddr;
575  _destpageaddr = destpageaddr;
576  _param1 = param1;
577  _param2 = param2;
578 
579  switch(type)
580  {
581 
582  case EXPANDING_RECTANGLE:
583  ExpandRectangle();
584  break;
585 
586  case CONTRACTING_RECTANGLE:
587  ContractRectangle();
588  break;
589 
590  case SLIDE:
591  SlideRectangle();
592  break;
593 
594  case PUSH:
595  PushRectangle();
596  break;
597 
598  case EXPANDING_LINE:
599  ExpandLine();
600  break;
601 
602  case CONTRACTING_LINE:
603  ContractLine();
604  break;
605 
606  default:
607  PlainCopyRectangle();
608  retval = 0xFF;
609  break;
610  };
611 
612  return (retval);
613 }
