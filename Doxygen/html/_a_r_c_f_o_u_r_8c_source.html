97 void ARCFOURInitialize(ARCFOUR_CTX* ctx, BYTE* key, WORD len)
98 {
99  BYTE temp, i, j, *Sbox;
100 
101  // Initialize the context indicies
102  i = 0;
103  j = 0;
104  Sbox = ctx->Sbox;
105 
106  // Initialize each S-box element with its index
107  do
108  {
109  Sbox[i] = i;
110  i++;
111  } while(i != 0u);
112 
113  // Fill in the S-box
114  do
115  {
116  j = j + Sbox[i] + key[i % len];
117  temp = Sbox[i];
118  Sbox[i] = Sbox[j];
119  Sbox[j] = temp;
120  i++;
121  } while(i != 0u);
122 
123  // Reset the context indicies
124  ctx->i = 0;
125  ctx->j = 0;
126 
127 }

152 void ARCFOURCrypt(ARCFOUR_CTX* ctx, BYTE* data, WORD len)
153 {
154  BYTE temp, temp2, i, j, *Sbox;
155 
156  // Buffer context variables in local RAM for faster access
157  i = ctx->i;
158  j = ctx->j;
159  Sbox = ctx->Sbox;
160 
161  // Loop over each byte. Extract its key and XOR
162  while(len--)
163  {
164  i++;
165  temp = Sbox[i];
166  j += temp;
167  temp2 = Sbox[j];
168  Sbox[i] = temp2;
169  Sbox[j] = temp;
170  temp += temp2;
171  temp2 = Sbox[temp];
172 
173  *data++ ^= temp2;
174  }
175 
176  // Save the new context
177  ctx->i = i;
178  ctx->j = j;
179 
180 }
