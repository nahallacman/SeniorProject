46 // Loops over all pending AJAX events to determine if any action is required
47 function pollAJAX() {
48  var curAjax = new Object();
49  var theTimer = new Date();
50  var elapsed;
51 
52  // Read off the ajaxList objects one by one
53  for(i = ajaxList.length; i > 0; i--)
54  {
55  curAjax = ajaxList.shift();
56  if(!curAjax)
57  continue;
58  elapsed = theTimer.getTime() - curAjax.lastCalled;
59 
60  // If we suceeded
61  if(curAjax.ajaxReq.readyState == 4 && curAjax.ajaxReq.status == 200) {
62  // If it has a container, write the result
63  if(typeof(curAjax.container) == 'function'){
64  curAjax.container(curAjax.ajaxReq.responseXML.documentElement);
65  } else if(typeof(curAjax.container) == 'string') {
66  document.getElementById(curAjax.container).innerHTML = curAjax.ajaxReq.responseText;
67  } // (otherwise do nothing for null values)
68 
69  curAjax.ajaxReq.abort();
70  curAjax.ajaxReq = null;
71 
72  // If it's a repeatable request, then do so
73  if(curAjax.repeat)
74  newAJAXCommand(curAjax.url, curAjax.container, curAjax.repeat);
75  continue;
76  }
77 
78  // If we've waited over 1 second, then we timed out
79  if(elapsed > timeOutMS) {
80  // Invoke the user function with null input
81  if(typeof(curAjax.container) == 'function'){
82  curAjax.container(null);
83  } else {
84  // Alert the user

86  }
87 
88  curAjax.ajaxReq.abort();

94  continue;
95  }
96 
97  // Otherwise, just keep waiting

99  }
100

104 }
