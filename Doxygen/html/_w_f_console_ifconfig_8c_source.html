325 /*****************************************************************************
326  * FUNCTION: isMacAddress
327  *
328  * RETURNS: True if valid MAC address, else False
329  *
330  * PARAMS: p_string -- string to check
331  * p_Address -- Array where MAC values will be written
332  *
333  * NOTES: Determines if the input string is a valid MAC address.
334  * If it is, then returns an array of 6 bytes for each of the values.
335  * MAC address must be in hex in the format xx:xx:xx:xx:xx:xx
336  *****************************************************************************/
337 static BOOL isMacAddress(INT8 *p_string, UINT8 *p_Address)
338 {
339  UINT8 i;
340  UINT16 tmp;
341 
342  if (strlen((char *)p_string) != 17u)
343  {
344  return FALSE;
345  }
346 
347  // ensure the ':' is in the right place, and if so, set them to 0
348  for (i = 2; i < 17u; i += 3)
349  {
350  if (p_string[i] == (INT8)':')
351  {
352  p_string[i] = '\0';
353  }
354  else
355  {
356  return FALSE;
357  }
358  }
359 
360  // now extract each hex number string
361  for (i = 0; i < 6u; ++i)
362  {
363  if (!ConvertASCIIHexToBinary(&p_string[i * 3], &tmp))
364  {
365  return FALSE;
366  }
367 
368  p_Address[i] = (UINT8) (tmp & 0xFF);
369 
370  }
371 
372  return TRUE;
373 }
