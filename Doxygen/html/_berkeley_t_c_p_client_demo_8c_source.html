92  static enum
93  {
94  DNS_START_RESOLUTION = 0,
95  DNS_GET_RESULT,
96  BSD_START,
97  BSD_CONNECT,
98  BSD_SEND,
99  BSD_OPERATION,
100  BSD_CLOSE,
101  BSD_DONE
102  } BSDClientState = BSD_DONE;
103 
104  switch(BSDClientState)
105  {
106  case DNS_START_RESOLUTION:
107  if(DNSBeginUsage())
108  {
109  DNSResolveROM(ServerName, DNS_TYPE_A);
110  BSDClientState = DNS_GET_RESULT;
111  }
112  break;
113 
114  case DNS_GET_RESULT:
115  if(!DNSIsResolved((IP_ADDR*)&addr.sin_addr.S_un.S_addr))
116  break;
117 
118  if(!DNSEndUsage())
119  {
120  #if defined(STACK_USE_UART)
121  putrsUART((ROM char*)"Could not resolve ServerName[] to IP address.\r\n");
122  #endif
123  BSDClientState = BSD_DONE;
124  break;
125  }
126 
127  BSDClientState = BSD_START;
128  // No break; here.
129 
130  case BSD_START:
131  // Create a socket for this client to connect with
132  if((bsdClientSocket = socket(AF_INET, SOCK_STREAM, IPPROTO_TCP)) == INVALID_SOCKET )
133  return;
134 
135  #if defined(STACK_USE_UART)
136  putrsUART((ROM char*)"\r\n\r\nConnecting using Berkeley Sockets TCP API...\r\n");
137  putrsUART((ROM char*)" Note: this demo will do nothing if an underlying TCP_PURPOSE_BERKELEY_CLIENT type \r\n"
138  " socket is unavailable, as declared by the TCPSocketInitializer[] array in \r\n"
139  " TCPIPConfig.h.\r\n\r\n");
140  #endif
141 
142  BSDClientState = BSD_CONNECT;
143  break;
144 
145  case BSD_CONNECT:
146  // addr.sin_addr.S_un.S_addr destination IP address was set earlier in DNS step
147  addr.sin_port = PORTNUM;
148  addrlen = sizeof(struct sockaddr);
149  if(connect( bsdClientSocket, (struct sockaddr*)&addr, addrlen) < 0)
150  return;
151 
152  BSDClientState = BSD_SEND;
153  // No break needed
154 
155  case BSD_SEND:
156  //send TCP data
157  send(bsdClientSocket, (const char*)sendRequest, strlen((char*)sendRequest), 0);
158  BSDClientState = BSD_OPERATION;
159  break;
160 
161  case BSD_OPERATION:
162  // Obtian and print the server reply
163  while(1)
164  {
165  i = recv(bsdClientSocket, recvBuffer, sizeof(recvBuffer)-1, 0); //get the data from the recv queue
166 
167  if(i == 0)
168  break;
169 
170  if(i< 0) //error condition
171  {
172  BSDClientState = BSD_CLOSE;
173  break;
174  }
175 
176  #if defined(STACK_USE_UART)
177  recvBuffer[i] = '\0'; // Null terminate data
178  putsUART((char*)recvBuffer);
179  #endif
180 
181  if(BSDClientState == BSD_OPERATION)
182  break;
183  }
184  break;
185 
186  case BSD_CLOSE:
187  closesocket(bsdClientSocket);
188  BSDClientState = BSD_DONE;
189  // No break needed
190 
191  case BSD_DONE:
192  if(BUTTON2_IO == 0u)
193  BSDClientState = DNS_START_RESOLUTION;
194  break;
195 
196  default:
197  return;
198  }
