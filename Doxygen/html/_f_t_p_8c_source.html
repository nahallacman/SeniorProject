240 /*********************************************************************
241  * Function: void FTPServer(void)
242  *
243  * PreCondition: FTPInit() must already be called.
244  *
245  * Input: None
246  *
247  * Output: Opened FTP connections are served.
248  *
249  * Side Effects: None
250  *
251  * Overview:
252  *
253  * Note: This function acts as a task (similar to one in
254  * RTOS). This function performs its task in
255  * co-operative manner. Main application must call
256  * this function repeatdly to ensure all open
257  * or new connections are served on time.
258  ********************************************************************/
259 BOOL FTPServer(void)
260 {
261  BYTE v;
262  DWORD currentTick;
263 
264  if ( !TCPIsConnected(FTPSocket) )
265  {
266  FTPStringLen = 0;
267  FTPCommand = FTP_CMD_NONE;
268  smFTP = SM_FTP_NOT_CONNECTED;
269  FTPFlags.Val = 0;
270  smFTPCommand = SM_FTP_CMD_IDLE;
271  if(FTPDataSocket != INVALID_SOCKET)
272  {
273  TCPDisconnect(FTPDataSocket);
274  FTPDataSocket = INVALID_SOCKET;
275  }
276 
277  return TRUE;
278  }
279 
280  if ( TCPIsGetReady(FTPSocket) )
281  {
282  lastActivity = TickGet();
283 
284  while( TCPGet(FTPSocket, &v ) )
285  {
286  FTPString[FTPStringLen++] = v;
287  if ( FTPStringLen == MAX_FTP_CMD_STRING_LEN )
288  FTPStringLen = 0;
289  }
290 
291 
292  if ( v == '\n' )
293  {
294  FTPString[FTPStringLen] = '\0';
295  FTPStringLen = 0;
296  ParseFTPString();
297  FTPCommand = ParseFTPCommand(FTP_argv[0]);
298  }
299  }
300  else if ( smFTP != SM_FTP_NOT_CONNECTED )
301  {
302  currentTick = TickGet();
303  currentTick = currentTick - lastActivity;
304  if ( currentTick >= FTP_TIMEOUT )
305  {
306  lastActivity = TickGet();
307  FTPCommand = FTP_CMD_QUIT;
308  smFTP = SM_FTP_CONNECTED;
309  }
310  }
311 
312  switch(smFTP)
313  {
314  case SM_FTP_NOT_CONNECTED:
315  FTPResponse = FTP_RESP_BANNER;
316  lastActivity = TickGet();
317  // No break - Continue...
318 
319  case SM_FTP_RESPOND:
320 SM_FTP_RESPOND_Label:
321  // Make sure there is enough TCP TX FIFO space to put our response
322  if(TCPIsPutReady(FTPSocket) < strlenpgm(FTPResponseString[FTPResponse]))
323  return TRUE;
324 
325  TCPPutROMString(FTPSocket, (ROM BYTE*)FTPResponseString[FTPResponse]);
326  TCPFlush(FTPSocket);
327  FTPResponse = FTP_RESP_NONE;
328  smFTP = SM_FTP_CONNECTED;
329  // No break - this will speed up little bit
330 
331  case SM_FTP_CONNECTED:
332  if ( FTPCommand != FTP_CMD_NONE )
333  {
334  if ( ExecuteFTPCommand(FTPCommand) )
335  {
336  if ( FTPResponse != FTP_RESP_NONE )
337  smFTP = SM_FTP_RESPOND;
338  else if ( FTPCommand == FTP_CMD_QUIT )
339  smFTP = SM_FTP_NOT_CONNECTED;
340 
341  FTPCommand = FTP_CMD_NONE;
342  smFTPCommand = SM_FTP_CMD_IDLE;
343  }
344  else if ( FTPResponse != FTP_RESP_NONE )
345  {
346  smFTP = SM_FTP_RESPOND;
347  goto SM_FTP_RESPOND_Label;
348  }
349  }
350  break;
351  case SM_FTP_USER_NAME:
352  case SM_FTP_USER_PASS:
353  break;
354 
355  }
356 
357  return TRUE;
358 }

367 static BOOL ExecuteFTPCommand(FTP_COMMAND cmd)
368 {
369  switch(cmd)
370  {
371  case FTP_CMD_USER:
372  FTPFlags.Bits.bUserSupplied = TRUE;
373  FTPFlags.Bits.bLoggedIn = FALSE;
374  FTPResponse = FTP_RESP_USER_OK;
375  strncpy((char*)FTPUser, (char*)FTP_argv[1], sizeof(FTPUser));
376  break;
377 
378  case FTP_CMD_PASS:
379  if ( !FTPFlags.Bits.bUserSupplied )
380  FTPResponse = FTP_RESP_LOGIN;
381  else
382  {
383  if ( FTPVerify(FTPUser, FTP_argv[1]) )
384  {
385  FTPFlags.Bits.bLoggedIn = TRUE;
386  FTPResponse = FTP_RESP_PASS_OK;
387  }
388  else
389  FTPResponse = FTP_RESP_LOGIN;
390  }
391  break;
392 
393  case FTP_CMD_QUIT:
394  return Quit();
395 
396  case FTP_CMD_PORT:
397  FTPDataPort.v[1] = (BYTE)atoi((char*)FTP_argv[5]);
398  FTPDataPort.v[0] = (BYTE)atoi((char*)FTP_argv[6]);
399  FTPResponse = FTP_RESP_OK;
400  break;
401 
402  case FTP_CMD_STOR:
403  return PutFile();
404 
405  case FTP_CMD_PWD:
406  FTPResponse = FTP_RESP_PWD;
407  break;
408 
409  case FTP_CMD_CWD:
410  case FTP_CMD_TYPE:
411  FTPResponse = FTP_RESP_OK;
412  break;
413 
414  case FTP_CMD_ABORT:
415  FTPResponse = FTP_RESP_OK;
416  if ( FTPDataSocket != INVALID_SOCKET )
417  {
418  TCPDisconnect(FTPDataSocket);
419  FTPDataSocket = INVALID_SOCKET;
420  }
421  break;
422 
423  default:
424  FTPResponse = FTP_RESP_UNKNOWN;
425  break;
426  }
427  return TRUE;
428 }

476 static BOOL PutFile(void)
477 {
478  BYTE v;
479 
480 
481  switch(smFTPCommand)
482  {
483  case SM_FTP_CMD_IDLE:
484  if ( !FTPFlags.Bits.bLoggedIn )
485  {
486  FTPResponse = FTP_RESP_LOGIN;
487  return TRUE;
488  }
489  else
490  {
491  FTPResponse = FTP_RESP_DATA_OPEN;
492  FTPDataSocket = TCPOpen((PTR_BASE)&TCPGetRemoteInfo(FTPSocket)->remote, TCP_OPEN_NODE_INFO, FTPDataPort.Val, TCP_PURPOSE_FTP_DATA);
493 
494  // Make sure that a valid socket was available and returned
495  // If not, return with an error
496  if(FTPDataSocket == INVALID_SOCKET)
497  {
498  FTPResponse = FTP_RESP_DATA_NO_SOCKET;
499  return TRUE;
500  }
501 
502  smFTPCommand = SM_FTP_CMD_WAIT;
503  }
504  break;
505 
506  case SM_FTP_CMD_WAIT:
507  if ( TCPIsConnected(FTPDataSocket) )
508  {
509 #if defined(FTP_PUT_ENABLED)
510  FTPFileHandle = MPFSFormat();
511 #endif
512 
513  smFTPCommand = SM_FTP_CMD_RECEIVE;
514  }
515  break;
516 
517  case SM_FTP_CMD_RECEIVE:
518  if ( TCPIsGetReady(FTPDataSocket) )
519  {
520  // Reload timeout timer.
521  lastActivity = TickGet();
522  while( TCPGet(FTPDataSocket, &v) )
523  {
524 #if defined(FTP_PUT_ENABLED)
525  MPFSPutArray(FTPFileHandle,&v,1);
526 #endif
527  }
528  MPFSPutEnd(TRUE);
529 
530  }
531  else if ( !TCPIsConnected(FTPDataSocket) )
532  {
533 #if defined(FTP_PUT_ENABLED)
534  MPFSClose(FTPFileHandle);
535 #endif
536  TCPDisconnect(FTPDataSocket);
537  FTPDataSocket = INVALID_SOCKET;
538  FTPResponse = FTP_RESP_DATA_CLOSE;
539  return TRUE;
540  }
541  case SM_FTP_CMD_WAIT_FOR_DISCONNECT:
542  break;
543  }
544  return FALSE;
545 }

562 static void ParseFTPString(void)
563 {
564  BYTE *p;
565  BYTE v;
566  enum { SM_FTP_PARSE_PARAM, SM_FTP_PARSE_SPACE } smParseFTP;
567 
568  smParseFTP = SM_FTP_PARSE_PARAM;
569  p = (BYTE*)&FTPString[0];
570 
571  // Skip white blanks
572  while( *p == ' ' )
573  p++;
574 
575  FTP_argv[0] = (BYTE*)p;
576  FTP_argc = 1;
577 
578  while( (v = *p) )
579  {
580  switch(smParseFTP)
581  {
582  case SM_FTP_PARSE_PARAM:
583  if ( v == ' ' || v == ',' )
584  {
585  *p = '\0';
586  smParseFTP = SM_FTP_PARSE_SPACE;
587  }
588  else if ( v == '\r' || v == '\n' )
589  *p = '\0';
590  break;
591 
592  case SM_FTP_PARSE_SPACE:
593  if ( v != ' ' )
594  {
595  FTP_argv[FTP_argc++] = (BYTE*)p;
596  smParseFTP = SM_FTP_PARSE_PARAM;
597  }
598  break;
599  }
600  p++;
601  if(FTP_argc == MAX_FTP_ARGS)
602  break;
603  }
604 }
