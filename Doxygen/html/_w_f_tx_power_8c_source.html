87  void WF_TxPowerSetMax(INT8 maxTxPower)
88 
89  Summary:
90  Sets the Tx max power on the MRF24WG0MA/B.
91 
92  Description:
93  After initialization the MRF24WG0MA/B max Tx power is determined by a
94  factory-set value. This function can set a different maximum
95  Tx power levels. However, this function can never set a maximum Tx power
96  greater than the factory-set value, which can be read via
97  WF_TxPowerGetFactoryMax().
98 
99  Precondition:
100  MACInit must be called first.
101 
102  Parameters:
103  maxTxPower - valid range (0 to 18 dBm)
104 
105  Returns:
106  None.
107 
108  Remarks:
109  No conversion of units needed, input to MRF24WG0MA/B is in dBm.
110  *****************************************************************************/
111 void WF_TxPowerSetMax(INT8 maxTxPower)
112 {
113  INT8 factoryMaxPower;
114  UINT8 msgData[2];
115  INT16 max = (INT16)maxTxPower;
116 
117  WF_TxPowerGetFactoryMax(&factoryMaxPower);
118  WF_ASSERT(maxTxPower <= factoryMaxPower); /* cannot set max tx power greater than factor-set max tx power */
119 
120  msgData[0] = (INT8)(max >> 8); /* msb of max power */
121  msgData[1] = (INT8)(max & 0xff); /* lsb of max power */
122 
123  SendSetParamMsg(PARAM_TX_POWER, msgData, sizeof(msgData));
124 }

186  maxTxPower - Desired maxTxPower (-10 to 10dB)
187

189  None.
190 
191  Remarks:
192  No conversion of units needed, input to MRF24WB0MA/B is in dB.
193  *****************************************************************************/
194 void WF_TxPowerSetMinMax(INT8 minTxPower, INT8 maxTxPower)
195 {
196  INT8 factoryMaxPower;
197  UINT8 msgData[4]; /* need to input to chip two signed 16-bit values, max power followed by min power */
198  INT16 max;
199  INT16 min;
200 
201  max = (INT16)maxTxPower;
202  min = (INT16)minTxPower;
203 
204  WF_ASSERT(minTxPower <= maxTxPower);
205 
206  WF_TxPowerGetFactoryMax(&factoryMaxPower);
207  WF_ASSERT(maxTxPower <= factoryMaxPower); /* cannot set max tx power greater than factor-set max tx power */
208 
209  msgData[0] = (INT8)(max >> 8); /* msb of max power */
210  msgData[1] = (INT8)(max & 0xff); /* lsb of max power */
211 
212  msgData[2] = (INT8)(min >> 8); /* msb of min power */
213  msgData[3] = (INT8)(min & 0xff); /* lsb of min power */
214 
215  SendSetParamMsg(PARAM_TX_POWER, msgData, sizeof(msgData));
216 }
