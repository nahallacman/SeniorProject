144 /*******************************************************************************
145 Function: UART2Init()
146 
147 Precondition: None.
148 
149 Overview:
150  This routine sets up the UART2 module.
151 
152 Input: None.
153 
154 Output: None.
155 
156 Notes:
157  Allow the peripheral to set the I/O pin directions. If we set the TRIS
158  bits manually, then when we disable the UART, the shape of the stop bit
159  changes, and some terminal programs have problems.
160 *******************************************************************************/
161 void UART2Init()
162 {
163  U2BRG = BAUDRATEREG2;
164  U2MODE = 0;
165  U2MODEbits.BRGH = BRGH2;
166  U2STA = 0;
167  U2MODEbits.UARTEN = 1;
168  U2STAbits.UTXEN = 1;
169  IFS1bits.U2RXIF = 0;
170 
171  #if defined (__PIC32MX__)
172  U2STAbits.URXEN = 1;
173  #endif
174 }

246 /*******************************************************************************
247 Function: UART2PutDec(unsigned char dec)
248 
249 Precondition:
250  UART2Init must be called prior to calling this routine.
251 
252 Overview:
253  This function converts decimal data into a string and outputs it to UART.
254 
255 Input: Binary data.
256 
257 Output: None.
258 
259 *******************************************************************************/
260 void UART2PutDec(unsigned char dec)
261 {
262  unsigned char res;
263  unsigned char printed_already = 0;
264 
265  res = dec;
266 
267  if (res/100)
268  {
269  UART2PutChar( res/100 + '0' );
270  printed_already = 1;
271  }
272  res = res - (res/100)*100;
273 
274  if ((res/10) || (printed_already == 1))
275  {
276  UART2PutChar( res/10 + '0' );
277  }
278  res = res - (res/10)*10;
279 
280  UART2PutChar( res + '0' );
281 }

397 char UART2Char2Hex(char ch){
398  // Wrong char
399  if(ch > 102)
400  return 0;
401 
402  // From a to f
403  if(ch > 96)
404  return (ch-87);
405 
406  // Wrong char
407  if(ch > 70)
408  return 0;
409 
410  // From A to F
411  if(ch > 64)
412  return (ch-55);
413 
414  // Wrong char
415  if(ch > 57)
416  return 0;
417 
418  // From 0 - 9
419  if(ch > 47)
420  return(ch-48);
421  else

423  return 0;
