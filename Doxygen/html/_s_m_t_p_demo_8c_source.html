66  Summary:
67  Demonstrates use of the e-mail (SMTP) client.
68 
69  Description:
70  This function demonstrates the use of the SMTP client. The function is
71  called periodically by the stack, and checks if BUTTON2 and BUTTON3 are
72  pressed simultaneously. If they are, it attempts to send an e-mail
73  message using parameters hard coded in the function below.
74 
75  While the client is executing, LED1 will be used as a busy indicator.
76  LED2 will light when the transmission has been completed successfully.
77  If both LEDs extinguish, an error occurred.
78 
79  For an example of sending a longer message (one that does not exist in
80  RAM all at once), see the commented secondary implementation of this
81  function in this file (SMTPDemo.c) below. For an example of sending
82  a message using parameters gathered at run time, and/or a message with
83  attachments, see the implementation of HTTPPostEmail in CustomHTTPApp.c.
84 
85  Precondition:
86  The SMTP client is initialized.
87 
88  Parameters:
89  None
90 
91  Returns:
92  None
93  ***************************************************************************/
94 void SMTPDemo(void)
95 {
96  // Send an email once if someone pushes BUTTON2 and BUTTON3 at the same time
97  // This is a simple message example, where the message
98  // body must already be in RAM.
99  // LED1 will be used as a busy indicator
100  // LED2 will be used as a mail sent successfully indicator
101  static enum
102  {
103  MAIL_HOME = 0,
104  MAIL_BEGIN,
105  MAIL_SMTP_FINISHING,
106  MAIL_DONE
107  } MailState = MAIL_HOME;
108  static DWORD WaitTime;
109 
110  switch(MailState)
111  {
112  case MAIL_HOME:
113  if((BUTTON2_IO == 0u) && (BUTTON3_IO == 0u))
114  {
115  // Start sending an email
116  LED1_IO = 1;
117  MailState++;
118  LED2_IO = 0;
119  }
120  break;
121 
122  case MAIL_BEGIN:
123  if(SMTPBeginUsage())
124  {
125  // Note that these strings must stay allocated in
126  // memory until SMTPIsBusy() returns FALSE. To
127  // guarantee that the C compiler does not reuse this
128  // memory, you must allocate the strings as static.
129 
130  static BYTE RAMStringTo[] = "joe@picsaregood.com";
131  //static BYTE RAMStringCC[] = "foo@picsaregood.com, \"Jane Smith\" <jane.smith@picsaregood.com>";
132  //static BYTE RAMStringBCC[] = "";
133  static BYTE RAMStringBody[] = "Message generated by stack " TCPIP_STACK_VERSION " \r\n\r\nButtons: 3210";
134  RAMStringBody[sizeof(RAMStringBody)-2] = '0' + BUTTON0_IO;
135  RAMStringBody[sizeof(RAMStringBody)-3] = '0' + BUTTON1_IO;
136  RAMStringBody[sizeof(RAMStringBody)-4] = '0' + BUTTON2_IO;
137  RAMStringBody[sizeof(RAMStringBody)-5] = '0' + BUTTON3_IO;
138 
139  SMTPClient.Server.szROM = (ROM BYTE*)"mail"; // SMTP server address
140  SMTPClient.ROMPointers.Server = 1;
141  //SMTPClient.Username.szROM = (ROM BYTE*)"mchpboard";
142  //SMTPClient.ROMPointers.Username = 1;
143  //SMTPClient.Password.szROM = (ROM BYTE*)"secretpassword";
144  //SMTPClient.ROMPointers.Password = 1;
145  SMTPClient.To.szRAM = RAMStringTo;
146  SMTPClient.From.szROM = (ROM BYTE*)"\"SMTP Service\" <mchpboard@picsaregood.com>";
147  SMTPClient.ROMPointers.From = 1;
148  SMTPClient.Subject.szROM = (ROM BYTE*)"Hello world! SMTP Test.";
149  SMTPClient.ROMPointers.Subject = 1;
150  SMTPClient.Body.szRAM = RAMStringBody;
151  SMTPSendMail();
152  MailState++;
153  }
154  break;
155 
156  case MAIL_SMTP_FINISHING:
157  if(!SMTPIsBusy())
158  {
159  // Finished sending mail
160  LED1_IO = 0;
161  MailState++;
162  WaitTime = TickGet();
163  LED2_IO = (SMTPEndUsage() == SMTP_SUCCESS);
164  }
165  break;
166 
167  case MAIL_DONE:
168  // Wait for the user to release BUTTON2 or BUTTON3 and for at
169  // least 1 second to pass before allowing another
170  // email to be sent. This is merely to prevent
171  // accidental flooding of email boxes while
172  // developing code. Your application may wish to
173  // remove this.
174  if(BUTTON2_IO && BUTTON3_IO)
175  {
176  if(TickGet() - WaitTime > TICK_SECOND)
177  MailState = MAIL_HOME;
178  }
179  break;
180  }
181 }
