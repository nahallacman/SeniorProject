229 /* You return: TRUE if the data is successfully read */

232 
233  // Example code
234  WORD index;
235  BYTE status = TRUE;
236  BYTE dummyVariable;
237 
238  data_bus_TRIS_BITS = 0xFF;
239 
240  // Set the address to read from
