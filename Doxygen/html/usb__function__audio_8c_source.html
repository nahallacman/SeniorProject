152 void USBCheckAudioRequest(void)
153 {
154  /*
155  * If request recipient is not an interface then return
156  */
157  if(SetupPkt.Recipient != USB_SETUP_RECIPIENT_INTERFACE_BITFIELD) return;
158  /*
159  * If request type is not class-specific then return
160  */
161  if(SetupPkt.RequestType != USB_SETUP_TYPE_CLASS_BITFIELD) return;
162  /*
163  * Interface ID must match interface numbers associated with
164  * Audio class, else return
165  */
166  if((SetupPkt.bIntfID != AUDIO_CONTROL_INTERFACE_ID)&&
167  (SetupPkt.bIntfID != AUDIO_STREAMING_INTERFACE_ID)) return;
168  switch(SetupPkt.W_Index.byte.HB)// checking for the Entity ID (Entity ID are defined in the config.h file)
169  {
170  case ID_INPUT_TERMINAL:
171  #if defined USB_AUDIO_INPUT_TERMINAL_CONTROL_REQUESTS_HANDLER
172  USB_AUDIO_INPUT_TERMINAL_CONTROL_REQUESTS_HANDLER();
173  #endif
174  break;
175  case ID_OUTPUT_TERMINAL:
176  #if defined USB_AUDIO_OUTPUT_TERMINAL_CONTROL_REQUESTS_HANDLER
177  USB_AUDIO_OUTPUT_TERMINAL_CONTROL_REQUESTS_HANDLER();
178  #endif
179  break;
180  case ID_MIXER_UNIT:
181  #if defined USB_AUDIO_MIXER_UNIT_CONTROL_REQUESTS_HANDLER
182  USB_AUDIO_MIXER_UNIT_CONTROL_REQUESTS_HANDLER();
183  #endif
184  break;
185  case ID_SELECTOR_UNIT:
186  #if defined USB_AUDIO_SELECTOR_UNIT_CONTROL_REQUESTS_HANDLER
187  USB_AUDIO_SELECTOR_UNIT_CONTROL_REQUESTS_HANDLER();
188  #endif
189  break;
190  case ID_FEATURE_UNIT:
191  #if defined USB_AUDIO_FEATURE_UNIT_CONTROL_REQUESTS_HANDLER
192  USB_AUDIO_FEATURE_UNIT_CONTROL_REQUESTS_HANDLER();
193  #endif
194  break;
195  case ID_PROCESSING_UNIT:
196  #if defined USB_AUDIO_PROCESSING_UNIT_CONTROL_REQUESTS_HANDLER
197  USB_AUDIO_PROCESSING_UNIT_CONTROL_REQUESTS_HANDLER();
198  #endif
199  break;
200  case ID_EXTENSION_UNIT:
201  #if defined USB_AUDIO_EXTENSION_UNIT_CONTROL_REQUESTS_HANDLER
202  USB_AUDIO_EXTENSION_UNIT_CONTROL_REQUESTS_HANDLER();
203  #endif
204  break;
205  default:
206  break;

208  }//end switch(SetupPkt.bRequest
