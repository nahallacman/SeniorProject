90 /*****************************************************************************
91  * FUNCTION: SendMgmtMsg
92  *
93  * RETURNS: error code
94  *
95  * PARAMS: p_header -- pointer to mgmt message header data
96  * headerLength -- number of bytes in the header
97  * will be written
98  * p_data -- pointer to mgmt message data
99  * dataLength -- number of byte of data
100  *
101  * NOTES: Sends a management message
102  *****************************************************************************/
103 void SendMgmtMsg(UINT8 *p_header,
104  UINT8 headerLength,
105  UINT8 *p_data,
106  UINT8 dataLength)
107 {
108  #if defined(__18CXX)
109  static UINT32 startTickCount;
110  static UINT32 maxAllowedTicks;
111  #else
112  UINT32 startTickCount;
113  UINT32 maxAllowedTicks;
114  #endif
115 
116 
117  /* cannot send management messages while in WF_ProcessEvent() */
118  WF_ASSERT(!isInWFProcessEvent());
119 
120  EnsureWFisAwake();
121 
122  /* if a Rx Data packet is mounted that has not yet been processed */
123  if (GetRawWindowState(RAW_RX_ID) == WF_RAW_DATA_MOUNTED)
124  {
125  /* save it, so after mgmt response received it can be restored */
126  PushRawWindow(RAW_RX_ID);
127  RestoreRxData = TRUE;
128  }
129 
130 
131  /* mounts a tx mgmt buffer on the MRF24W when data tx is done */
132  maxAllowedTicks = TICKS_PER_SECOND / 200; /* 5 ms timeout */
133  startTickCount = (UINT32)TickGet();
134  while (!WFisTxMgmtReady() )
135  {
136  MACProcess();
137 
138  /* DEBUG -- REMOVE AFTER FIGURE OUT WHY TIMING OUT (RARELY HAPPENS) */
139  if (TickGet() - startTickCount >= maxAllowedTicks)
140  {
141  /* force flags so WFisTxMgmtReady will return TRUE */
142  SetRawWindowState(RAW_TX_ID, WF_RAW_UNMOUNTED);
143  RawWindowReady[RAW_TX_ID] = FALSE;
144  }
145  }
146 
147  /* write out management header */
148  RawSetByte(RAW_TX_ID, p_header, headerLength);
149 
150  /* write out data (if any) */
151  if (dataLength > 0)
152  {
153  RawSetByte(RAW_TX_ID, p_data, dataLength);
154  }
155 
156  /* send mgmt msg to MRF24W */
157  SendRAWManagementFrame(headerLength + dataLength);
158 }

176 /*****************************************************************************
177  * FUNCTION: WaitForMgmtResponse
178  *
179  * RETURNS: None
180  *
181  * PARAMS: expectedSubtype -- The expected subtype of the mgmt response
182  * freeAction -- FREE_MGMT_BUFFER or DO_NOT_FREE_MGMT_BUFFER
183  *
184  * NOTES: Called after sending a mgmt request. This function waits for a mgmt
185  * response. The caller can optionally request the the management
186  * response be freed immediately (by this function) or not freed. If not
187  * freed the caller is responsible to free the response buffer.
188  *****************************************************************************/
189 void WaitForMgmtResponse(UINT8 expectedSubtype, UINT8 freeAction)
190 {
191  #if defined(__18CXX)
192  static tMgmtMsgRxHdr hdr;
193  #else
194  tMgmtMsgRxHdr hdr;
195  #endif
196 
197  g_WaitingForMgmtResponse = TRUE;
198 
199  /* Wait until mgmt response is received */
200  while (gMgmtConfirmMsgReceived == FALSE)
201  {
202  WFProcess();
203 
204  /* if received a data packet while waiting for mgmt packet */
205  if (g_HostRAWDataPacketReceived)
206  {
207  // We can't let the StackTask processs data messages that come in while waiting for mgmt
208  // response because the application might send another mgmt message, which is illegal until the response
209  // is received for the first mgmt msg. And, we can't prevent the race condition where a data message
210  // comes in before a mgmt response is received. Thus, the only solution is to throw away a data message
211  // that comes in while waiting for a mgmt response. This should happen very infrequently. If using TCP then the
212  // stack takes care of retries. If using UDP, the application has to deal with occasional data messages not being
213  // received. Also, applications typically do not send a lot of management messages after connected.
214 
215  // throw away the data rx
216  RawMountRxBuffer();
217  DeallocateDataRxBuffer();
218  g_HostRAWDataPacketReceived = FALSE;
219 
220  /* ensure interrupts enabled */
221  WF_EintEnable();
222  }
223  }
224 
225  /* set this back to FALSE so the next mgmt send won't think he has a response before one is received */
226  gMgmtConfirmMsgReceived = FALSE;
227 
228 
229  /* if the caller wants to delete the response immediately (doesn't need any data from it */
230  if (freeAction == FREE_MGMT_BUFFER)
231  {
232  /* read and verify result before freeing up buffer to ensure our message send was successful */
233  RawRead(RAW_RX_ID, 0, (UINT16)(sizeof(tMgmtMsgRxHdr)), (UINT8 *)&hdr);
234 
235  /* mgmt response subtype had better match subtype we were expecting */
236  WF_ASSERT(hdr.subtype == expectedSubtype);
237 
238  if (hdr.result == WF_ERROR_DISCONNECT_FAILED
239  || hdr.result == WF_ERROR_NOT_CONNECTED) {
240  #if defined(STACK_USE_UART)
241  putrsUART("Disconnect failed. Disconnect is allowed only when module is in connected state\r\n");
242  #endif
243  } else if (hdr.result == WF_ERROR_NO_STORED_BSS_DESCRIPTOR) {
244  #if defined(STACK_USE_UART)
245  putrsUART("No stored scan results\r\n");
246  #endif
247  } else {
248  if (!(hdr.result == WF_ERROR_CP_INVALID_PROFILE_ID && hdr.subtype == WF_CM_CONNECT_SUBYTPE)) {
249  /* 120c host scan bug workaround */
250  WF_ASSERT(hdr.result == WF_SUCCESS);
251  }
252  }
253 
254  /* free mgmt buffer */
255  DeallocateMgmtRxBuffer();
256 
257  /* if there was a mounted data packet prior to the mgmt tx/rx transaction, then restore it */
258  if (RestoreRxData == TRUE)
259  {
260  RestoreRxData = FALSE;
261  PopRawWindow(RAW_RX_ID);
262  SetRawWindowState(RAW_RX_ID, WF_RAW_DATA_MOUNTED);
263  }
264  }
265 }

268 /*****************************************************************************
269  * FUNCTION: WaitForMgmtRespAndReadData
270  *
271  * RETURNS: None
272  *
273  * PARAMS: expectedSubtype -- management message subtype that we are expecting
274  * p_data -- pointer where any desired management data bytes
275  * will be written
276 
277  * numDataBytes -- Number of data bytes from mgmt response to write to
278  * p_data. Data always starts at index 4 of mgmt response.
279  * skipDataRead -- if TRUE, then no data will be read and the mgmt buffer will not
280  * be freed. If FALSE, the data will be read and the mgmt buffer
281  * will be freed.
282  *
283  * NOTES: Waits for the mgmt response message and validates it by:
284  * 1) checking the result field
285  * 2) verifying that the received subtype matches the execpted subtype
286  *
287  * In addition, this function reads the desired number of data bytes from
288  * the mgmt response, copies them to p_data, and then frees the mgmt buffer.
289  *****************************************************************************/
290 void WaitForMgmtResponseAndReadData(UINT8 expectedSubtype,
291  UINT8 numDataBytes,
292  UINT8 startIndex,
293  UINT8 *p_data)
294 
295 {
296  tMgmtMsgRxHdr hdr; /* management msg header struct */
297 
298  WaitForMgmtResponse(expectedSubtype, DO_NOT_FREE_MGMT_BUFFER);
299 
300  /* if made it here then received a management message */
301  RawRead(RAW_RX_ID, 0, (UINT16)(sizeof(tMgmtMsgRxHdr)), (UINT8 *)&hdr);
302 
303  /* check header result and subtype fields */
304  WF_ASSERT(hdr.result == WF_SUCCESS || hdr.result == WF_ERROR_NO_STORED_BSS_DESCRIPTOR);
305  WF_ASSERT(hdr.subtype == expectedSubtype);
306 
307  /* if caller wants to read data from this mgmt response */
308  if (numDataBytes > 0)
309  {
310  RawRead(RAW_RX_ID, startIndex, numDataBytes, p_data);
311  }
312 
313  /* free the mgmt buffer */
314  DeallocateMgmtRxBuffer();
315 
316  /* if there was a mounted data packet prior to the mgmt tx/rx transaction, then restore it */
317  if (RestoreRxData == TRUE)
318  {
319  RestoreRxData = FALSE;
320  PopRawWindow(RAW_RX_ID);
321  SetRawWindowState(RAW_RX_ID, WF_RAW_DATA_MOUNTED);
322  }
323 }
