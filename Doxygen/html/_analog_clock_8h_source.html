139 /*********************************************************************
140 * Function: WORD AcDraw(ANALOGCLOCK *pAc)
141 *
142 * Overview: This function renders the object on the screen using
143 * the current parameter settings. Location of the object is
144 * determined by the left, top, right and bottom parameters.
145 * The colors used are dependent on the state of the object.
146 * The font used is determined by the style scheme set.
147 *
148 *
149 * PreCondition: Object must be created before this function is called.
150 *
151 * Input: pAc - Pointer to the object to be rendered.
152 *
153 * Output: Returns the status of the drawing
154 * - 1 - If the rendering was completed and
155 * - 0 - If the rendering is not yet finished.
156 * Next call to the function will resume the
157 * rendering on the pending drawing state.
158 *
159 * Example:
160 * <CODE>
161 * void MyGOLDraw(){
162 * static OBJ_HEADER *pCurrentObj = NULL;
163 * int done;
164 *
165 * // There are no objects
166 * if(GOLGetList() == NULL)
167 * return;
168 *
169 * // If it's last object jump to head
170 * if(pCurrentObj == NULL)
171 * pCurrentObj = GOLGetList();
172 *
173 * done = 0;
174 *
175 * // this only process Button and Window
176 * while(pCurrentObj != NULL){
177 * // check if object state indicates redrawing
178 * if(pCurrentObj->state&0xFC00) {
179 * switch(pCurrentObj->type){
180 * case OBJ_ANALOGCLOCK:
181 * done = AcDraw((ANALOGCLOCK*)pCurrentObj);
182 * break;
183 * case OBJ_WINDOW:
184 * done = WndDraw((WINDOW*)pCurrentObj);
185 * break;
186 * default:
187 * done = 1;
188 * break;
189 * }
190 * if(done){
191 * // reset only the state if drawing was finished
192 * pCurrentObj->state = 0;
193 * }else{
194 * // done processing the list
195 * return;
196 * }
197 * }
198 * // go to next object
199 * pCurrentObj = pCurrentObj->pNxtObj;
200 * }
201 * }
202 * </CODE>
203 *
204 * Side Effects: none
205 *
206 ********************************************************************/
