138 void WF_Init(void)
139 {
140  tWFDeviceInfo deviceInfo;
141 
142  WFHardwareInit();
143  RawInit();
144 
145  WFEnableMRF24WB0MMode();
146  WF_GetDeviceInfo(&deviceInfo);
147 
148  // if MRF24WB
149  #if !defined(MRF24WG)
150  WF_ASSERT(deviceInfo.romVersion == 0x12);
151  WF_ASSERT(deviceInfo.patchVersion >= 0x02);
152  if (deviceInfo.romVersion == 0x12 && deviceInfo.patchVersion >= 0x09)
153  {
154  gRFModuleVer1209orLater = TRUE;
155  }
156  #else // must be a MRF24WG
157  WF_ASSERT(deviceInfo.romVersion == 0x30 || deviceInfo.romVersion == 0x31);
158  #endif
159 
160  /* send init messages to MRF24W */
161  WF_LibInitialize();
162 
163  #if defined(WF_CONSOLE)
164  WFConsoleInit();
165  #if defined(WF_CONSOLE_DEMO) || (defined(WF_EASY_CONFIG_DEMO) && defined(__C32__))
166  IperfAppInit();
167  #endif
168  #endif
169 
170  if(AppConfig.Flags.bIsDHCPEnabled)
171  {
172  SetDhcpProgressState();
173  }
174 }

177 /*****************************************************************************
178  * FUNCTION: WF_LibInitialize
179  *
180  * RETURNS: None
181  *
182  * PARAMS: None
183  *
184  * NOTES: Performs initialization which is specific to the Microchip Demo code.
185  *****************************************************************************/
186 static void WF_LibInitialize()
187 {
188  /* Disable Tx Data confirms (from the MRF24W) */
189  WF_SetTxDataConfirm(WF_DISABLED);
190 
191  /* if the user has left the default MAC address in TCPIPConfig.h unchanged then use */
192  /* the unique MRF24W MAC address so prevent multiple devices from having the same */
193  /* MAC address. */
194  if ( memcmp((void *)AppConfig.MyMACAddr.v, (void *)MchpDefaultMacAddress, WF_MAC_ADDRESS_LENGTH) == 0)
195  {
196  /* get the MRF24W MAC address and overwrite the MAC in AppConfig */
197  WF_GetMacAddress((UINT8 *)AppConfig.MyMACAddr.v);
198  }
199  /* else presume the user has a unique MAC address of their own that they wish to use */
200  else
201  {
202  // set MAC address with user-supplied MAC */
203  WF_SetMacAddress((UINT8 *)AppConfig.MyMACAddr.v);
204  }
205 
206  #ifdef WF_CONFIG_DHCP
207  WF_SET_DHCP_STATE(DHCP_ENABLED);
208  #endif
209 
210 }
