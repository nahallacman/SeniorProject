126  This function performs any periodic tasks requied by the DHCP server

138 void DHCPServerTask(void)
139 {
140  BYTE i;
141  BYTE Option, Len;
142  BOOTP_HEADER BOOTPHeader;
143  DWORD dw;
144  BOOL bAccept, bRenew;
145 
146 
147  // Init IP pool
148  static BOOL flag_init = FALSE;
149  if(flag_init == FALSE)
150  {
151  flag_init = TRUE;
152  DHCPServerInit();
153  }
154 
155 #if defined(STACK_USE_DHCP_CLIENT)
156  // Make sure we don't clobber anyone else's DHCP server
157  if(DHCPIsServerDetected(0))
158  return;
159 #endif
160 
161  if(!bDHCPServerEnabled)
162  return;
163 
164  renew_dhcps_Pool();
165  switch(smDHCPServer)
166  {
167  case DHCPS_DISABLE:
168  break;
169  case DHCPS_OPEN_SOCKET:
170  // Obtain a UDP socket to listen/transmit on
171  //MySocket = UDPOpen(DHCP_SERVER_PORT, NULL, DHCP_CLIENT_PORT);
172  MySocket = UDPOpenEx(0,UDP_OPEN_SERVER,DHCP_SERVER_PORT, DHCP_CLIENT_PORT);
173  if(MySocket == INVALID_UDP_SOCKET)
174  break;
175 
176 
177  // Decide which address to lease out
178  // Note that this needs to be changed if we are to
179  // support more than one lease
180  DHCPNextLease.Val = (AppConfig.MyIPAddr.Val & AppConfig.MyMask.Val) + 0x02000000;
181  if(DHCPNextLease.v[3] == 255u)
182  DHCPNextLease.v[3] += 0x03;
183  if(DHCPNextLease.v[3] == 0u)
184  DHCPNextLease.v[3] += 0x02;
185 
186  smDHCPServer++;
187 
188  case DHCPS_LISTEN:
189  // Check to see if a valid DHCP packet has arrived
190  if(UDPIsGetReady(MySocket) < 241u)
191  break;
192 
193  // Retrieve the BOOTP header
194  UDPGetArray((BYTE*)&BOOTPHeader, sizeof(BOOTPHeader));
195 
196  if(TRUE == isIpAddrInPool(BOOTPHeader.ClientIP)){bRenew= TRUE; bAccept = TRUE;}
197  else if(BOOTPHeader.ClientIP.Val == 0x00000000u) {bRenew = FALSE; bAccept = TRUE;}
198  else {bRenew = FALSE; bAccept = FALSE;}
199  //bAccept = (BOOTPHeader.ClientIP.Val == DHCPNextLease.Val) || (BOOTPHeader.ClientIP.Val == 0x00000000u);
200 
201  // Validate first three fields
202  if(BOOTPHeader.MessageType != 1u)
203  break;
204  if(BOOTPHeader.HardwareType != 1u)
205  break;
206  if(BOOTPHeader.HardwareLen != 6u)
207  break;
208 
209  // Throw away 10 unused bytes of hardware address,
210  // server host name, and boot file name -- unsupported/not needed.
211  for(i = 0; i < 64+128+(16-sizeof(MAC_ADDR)); i++)
212  UDPGet(&Option);
213 
214  // Obtain Magic Cookie and verify
215  UDPGetArray((BYTE*)&dw, sizeof(DWORD));
216  if(dw != 0x63538263ul)
217  break;
218 
219  // Obtain options
220  while(1)
221  {
222  // Get option type
223  if(!UDPGet(&Option))
224  break;
225  if(Option == DHCP_END_OPTION)
226  break;
227 
228  // Get option length
229  UDPGet(&Len);
230 
231  // Process option
232  switch(Option)
233  {
234  case DHCP_MESSAGE_TYPE:
235  UDPGet(&i);
236  switch(i)
237  {
238  case DHCP_DISCOVER_MESSAGE:
239  DHCPReplyToDiscovery(&BOOTPHeader);
240  break;
241 
242  case DHCP_REQUEST_MESSAGE:
243  DHCPReplyToRequest(&BOOTPHeader, bAccept, bRenew);
244  break;
245 
246  // Need to handle these if supporting more than one DHCP lease
247  case DHCP_RELEASE_MESSAGE:
248  case DHCP_DECLINE_MESSAGE:
249  break;
250  }
251  break;
252 
253  case DHCP_PARAM_REQUEST_IP_ADDRESS:
254  if(Len == 4u)
255  {
256  IP_ADDR tmp_ip;
257  // Get the requested IP address and see if it is the one we have on offer.
258  UDPGetArray((BYTE*)&dw, 4);
259  Len -= 4;
260  tmp_ip.Val = dw;
261  //bAccept = (dw == DHCPNextLease.Val);
262  if(TRUE == isIpAddrInPool(tmp_ip)){bRenew= TRUE; bAccept = TRUE;}
263  else if(tmp_ip.Val == 0x00000000u) {bRenew = FALSE; bAccept = TRUE;}
264  else {bRenew = FALSE; bAccept = FALSE;}
265 
266  }
267  break;
268 
269  case DHCP_END_OPTION:
270  UDPDiscard();
271  return;
272  }
273 
274  // Remove any unprocessed bytes that we don't care about
275  while(Len--)
276  {
277  UDPGet(&i);
278  }
279  }
280 
281  UDPDiscard();
282  break;
283  }
284 }

307 static void DHCPReplyToDiscovery(BOOTP_HEADER *Header)
308 {
309  BYTE i;
310  INT8 IndexOfPool;
311  IP_ADDR ipAddr;
312  // Set the correct socket to active and ensure that
313  // enough space is available to generate the DHCP response
314  if(UDPIsPutReady(MySocket) < 300u)
315  return;
316 
317  // find in pool
318  IndexOfPool = preAssign_ToDHCPClient_FromPool(Header);
319  if( -1 == IndexOfPool) return;
320 
321  // Begin putting the BOOTP Header and DHCP options
322  UDPPut(BOOT_REPLY); // Message Type: 2 (BOOTP Reply)
323  // Reply with the same Hardware Type, Hardware Address Length, Hops, and Transaction ID fields
324  UDPPutArray((BYTE*)&(Header->HardwareType), 7);
325  UDPPut(0x00); // Seconds Elapsed: 0 (Not used)
326  UDPPut(0x00); // Seconds Elapsed: 0 (Not used)
327  UDPPutArray((BYTE*)&(Header->BootpFlags), sizeof(Header->BootpFlags));
328  UDPPut(0x00); // Your (client) IP Address: 0.0.0.0 (none yet assigned)
329  UDPPut(0x00); // Your (client) IP Address: 0.0.0.0 (none yet assigned)
330  UDPPut(0x00); // Your (client) IP Address: 0.0.0.0 (none yet assigned)
331  UDPPut(0x00); // Your (client) IP Address: 0.0.0.0 (none yet assigned)
332  //UDPPutArray((BYTE*)&DHCPNextLease, sizeof(IP_ADDR)); // Lease IP address to give out
333  ipAddr = GetIPAddrFromIndex_DhcpPool(IndexOfPool);
334  UDPPutArray((UINT8*)&ipAddr, sizeof(IP_ADDR)); // Lease IP address to give out
335  UDPPut(0x00); // Next Server IP Address: 0.0.0.0 (not used)
336  UDPPut(0x00); // Next Server IP Address: 0.0.0.0 (not used)
337  UDPPut(0x00); // Next Server IP Address: 0.0.0.0 (not used)
338  UDPPut(0x00); // Next Server IP Address: 0.0.0.0 (not used)
339  UDPPut(0x00); // Relay Agent IP Address: 0.0.0.0 (not used)
340  UDPPut(0x00); // Relay Agent IP Address: 0.0.0.0 (not used)
341  UDPPut(0x00); // Relay Agent IP Address: 0.0.0.0 (not used)
342  UDPPut(0x00); // Relay Agent IP Address: 0.0.0.0 (not used)
343  UDPPutArray((BYTE*)&(Header->ClientMAC), sizeof(MAC_ADDR)); // Client MAC address: Same as given by client
344  for(i = 0; i < 64+128+(16-sizeof(MAC_ADDR)); i++) // Remaining 10 bytes of client hardware address, server host name: Null string (not used)
345  UDPPut(0x00); // Boot filename: Null string (not used)
346  UDPPut(0x63); // Magic Cookie: 0x63538263
347  UDPPut(0x82); // Magic Cookie: 0x63538263
348  UDPPut(0x53); // Magic Cookie: 0x63538263
349  UDPPut(0x63); // Magic Cookie: 0x63538263
350 
351  // Options: DHCP Offer
352  UDPPut(DHCP_MESSAGE_TYPE);
353  UDPPut(1);
354  UDPPut(DHCP_OFFER_MESSAGE);
355 
356  // Option: Subnet Mask
357  UDPPut(DHCP_SUBNET_MASK);
358  UDPPut(sizeof(IP_ADDR));
359  UDPPutArray((BYTE*)&AppConfig.MyMask, sizeof(IP_ADDR));
360 
361  // Option: Lease duration
362  UDPPut(DHCP_IP_LEASE_TIME);
363  UDPPut(4);
364  UDPPut((DHCP_LEASE_DURATION>>24) & 0xFF);
365  UDPPut((DHCP_LEASE_DURATION>>16) & 0xFF);
366  UDPPut((DHCP_LEASE_DURATION>>8) & 0xFF);
367  UDPPut((DHCP_LEASE_DURATION) & 0xFF);
368 
369  // Option: Server identifier
370  UDPPut(DHCP_SERVER_IDENTIFIER);
371  UDPPut(sizeof(IP_ADDR));
372  UDPPutArray((BYTE*)&AppConfig.MyIPAddr, sizeof(IP_ADDR));
373 
374  // Option: Router/Gateway address
375  UDPPut(DHCP_ROUTER);
376  UDPPut(sizeof(IP_ADDR));
377  UDPPutArray((BYTE*)&AppConfig.MyIPAddr, sizeof(IP_ADDR));
378 
379  // Option: DNS server address
380  UDPPut(DHCP_DNS);
381  UDPPut(sizeof(IP_ADDR));
382  UDPPutArray((BYTE*)&AppConfig.MyIPAddr, sizeof(IP_ADDR));
383 
384  // No more options, mark ending
385  UDPPut(DHCP_END_OPTION);
386 
387  // Add zero padding to ensure compatibility with old BOOTP relays that discard small packets (<300 UDP octets)
388  while(UDPTxCount < 300u)
389  UDPPut(0);
390 
391  // Force remote destination address to be the broadcast address, regardless
392  // of what the node's source IP address was (to ensure we don't try to
393  // unicast to 0.0.0.0).
394  memset((void*)&UDPSocketInfo[MySocket].remote.remoteNode, 0xFF, sizeof(NODE_INFO));
395 
396  // Transmit the packet
397  UDPFlush();
398 }
