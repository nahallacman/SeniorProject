147 /*********************************************************************
148 * Function: LISTBOX *LbCreate(WORD ID, SHORT left, SHORT top, SHORT right,
149 * SHORT bottom, WORD state, XCHAR* pText,
150 * GOL_SCHEME *pScheme)
151 *
152 * Overview: This function creates a LISTBOX object with the parameters given.
153 * It automatically attaches the new object into a global linked list of
154 * objects and returns the address of the object.
155 *
156 * PreCondition: none
157 *
158 * Input: ID - Unique user defined ID for the object instance.
159 * left - Left most position of the Object.
160 * top - Top most position of the Object.
161 * right - Right most position of the Object.
162 * bottom - Bottom most position of the Object.
163 * state - Sets the initial state of the object.
164 * pText - Pointer to the initialization text for the items.
165 * pScheme - Pointer to the style scheme.
166 *
167 * Output: Returns the pointer to the object created.
168 *
169 * Example:
170 * <CODE>
171 * #define LISTBOX_ID 10
172 *
173 * const XCHAR ItemList[] = "Line1\n" "Line2\n";
174 *
175 * GOL_SCHEME *pScheme;
176 * LISTBOX *pLb;
177 * XCHAR *pTemp;
178 * WORD state, counter;
179 *
180 * pScheme = GOLCreateScheme();
181 * state = LB_DRAW;
182 *
183 * // create an empty listbox with default style scheme
184 * pLb = LbCreate( LISTBOX_ID, // ID number
185 * 10,10,150,200, // dimension
186 * state, // initial state
187 * NULL, // set items to be empty
188 * NULL); // use default style scheme
189 * // check if Listbox was created
190 * if (pLb == NULL)
191 * return 0;
192 *
193 * // create the list of items to be placed in the listbox
194 * // Add items (each line will become one item,
195 * // lines must be separated by '\n' character)
196 * pTemp = ItemList;
197 * counter = 0;
198 * while(*pTemp){
199 * // since each item is appended NULL is assigned to
200 * // LISTITEM pointer.
201 * if(NULL == LbAddItem(pLb, NULL, pTemp, NULL, 0, counter))
202 * break;
203 * while((unsigned XCHAR)*pTemp++ > (unsigned XCHAR)31);
204 * if(*(pTemp-1) == 0)
205 * break;
206 * counter++;
207 * }
208 * </CODE>
209 *
210 * Side Effects: none
211 *
212 ********************************************************************/
213 LISTBOX * LbCreate
214  (WORD ID, SHORT left, SHORT top, SHORT right, SHORT bottom, WORD state, XCHAR * pText, GOL_SCHEME * pScheme);

216 /*********************************************************************
217 * Function: LISTITEM* LbAddItem(LISTBOX *pLb, LISTITEM *pPrevItem,
218 * XCHAR *pText, void* pBitmap, WORD status, WORD data)
219 *
220 * Overview: This function allocates memory for the LISTITEM and adds
221 * it to the list box. The newly created LISTITEM will store
222 * the location of pText, pBitmap and other parameters describing
223 * the added item.
224 *
225 * PreCondition: none
226 *
227 * Input: pLb - The pointer to the list box object.
228 * pPrevItem - Pointer to the item after which a new item must
229 * be inserted, if this pointer is NULL, the item
230 * will be appended at the end of the items list.
231 * pText - Pointer to the text that will be inserted. Text must
232 * persist in memory for as long as it is referenced
233 * by an item in the list box.
234 * pBitmap - Pointer to the bitmap for the item. Bitmap must
235 * persist in memory for as long as it is referenced
236 * by the an item in the list box.
237 * status - This parameter specifies if the item being added
238 * will be selected or redrawn
239 * (LB_STS_SELECTED or LB_STS_REDRAW). Refer to
240 * LISTITEM structure for details.
241 * data - User assigned data associated with the item.
242 *
243 * Output: Return a pointer to the item created,
244 * NULL if the operation was not successful.
245 *
246 * Example:
247 * <CODE>
248 * const XCHAR ItemList[] = "Line1\n" "Line2\n" "Line3\n";
249 *
250 * extern BITMAP_FLASH myIcon;
251 * LISTBOX *pLb;
252 * LISTITEM *pItem, *pItemList;
253 * XCHAR *pTemp;
254 *
255 * // Assume that pLb is pointing to an existing list box in memory
256 * // that is empty (no list).
257 *
258 * // Create the list of the list box
259 *
260 * // Initialize this to NULL to indicate that items will be added
261 * // at the end of the list if the list exist on the list box or
262 * // start a new list if the list box is empty.
263 * pItem = NULL;
264 * pTemp = ItemList;
265 * pItem = LbAddItem(pLb, pItem, pTemp, NULL, LB_STS_SELECTED, 1)
266 * if(pItem == NULL)
267 * return 0;
268 * LbSetBitmap(pItem, &myIcon);
269 *
270 * // Adjust pTemp to point to the next line
271 * while((unsigned XCHAR)*pTemp++ > (unsigned XCHAR)31);
272 *
273 * // add the next item
274 * pItem = LbAddItem(pLb, pItem, pTemp, NULL, 0, 2)
275 * if(pItem == NULL)
276 * return 0;
277 * LbSetBitmap(pItem, &myIcon);
278 *
279 * // Adjust pTemp to point to the next line
280 * while((unsigned XCHAR)*pTemp++ > (unsigned XCHAR)31);
281 *
282 * // this time insert the next item after the first item on the list
283 * pItem = LbGetItemList(pLb);
284 * pItem = LbAddItem(pLb, pItem, pTemp, NULL, 0, 3)
285 * if(pItem == NULL)
286 * return 0;
287 * LbSetBitmap(pItem, &myIcon);
288 *
289 * </CODE>
290 *
291 * Side Effects: none
292 *
293 ********************************************************************/
294 LISTITEM *LbAddItem(LISTBOX *pLb, LISTITEM *pPrevItem, XCHAR *pText, void *pBitmap, WORD status, WORD data);
