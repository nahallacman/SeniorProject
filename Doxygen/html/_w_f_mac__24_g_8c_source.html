243 /*****************************************************************************
244  Function:
245  void SyncENCPtrRAWState(UINT8 encPtrId)
246 
247  Summary:
248  Synchronizes the Ethernet RAM index to the WiFi RAW window index.
249 
250  Description:
251  Any time stack code changes the index within the 'logical' Ethernet RAM
252  this function must be called to assure the RAW driver is synced up with
253  where the stack code thinks it is within the Ethernet RAM. This applies
254  to reading/writing tx data, rx data, or tcb data.
255 
256  This function is typically called right after g_encIndex[] is updated.
257 
258  Precondition:
259  None
260 
261  Parameters:
262  encPtrId -- ENC_RD_PTR_ID or ENC_WT_PTR_ID
263 
264  Returns:
265  None
266 
267  Remarks:
268  None
269 *****************************************************************************/
270 static void SyncENCPtrRAWState(UINT8 encPtrId, UINT16 encIndex)
271 {
272  UINT8 rawId;
273  UINT16 rawIndex;
274  UINT32 startTickCount;
275  UINT32 maxAllowedTicks;
276 
277 
278  EnsureWFisAwake();
279 
280  /* Save the the read or write index value. Code below will convert it to a */
281  /* Raw index */
282  g_encIndex[encPtrId] = encIndex;
283 
284  /*-----------------------------------------------------------------------------*/
285  /* if the input read or write pointer is in the Rx section of the Ethernet RAM */
286  /*-----------------------------------------------------------------------------*/
287  if (g_encIndex[encPtrId] < TXSTART)
288  {
289  /* set the rawId */
290  rawId = RAW_DATA_RX_ID;
291 
292  /* Convert encPtr index to Raw Index */
293  rawIndex = g_encIndex[encPtrId] - ENC_RX_BUF_TO_RAW_RX_BUF_ADJUSTMENT;
294 
295  // encPtr[encPtrId] < (RXSTART + ENC_PREAMBLE_SIZE) is an error since we don't have
296  // the same preamble as the ENC chip
297  WF_ASSERT( g_encIndex[encPtrId] >= (RXSTART + ENC_PREAMBLE_SIZE) );
298 
299  }
300  /*----------------------------------------------------------------------------------*/
301  /* else if the input read or write pointer is in the Tx section of the Ethernet RAM */
302  /*----------------------------------------------------------------------------------*/
303  else if (g_encIndex[encPtrId] < BASE_SCRATCH_ADDR/*BASE_TCB_ADDR*/)
304  {
305  /* if the Tx data raw window has not yet been allocated (the stack is creating a new Tx data packet) */
306  if (GetRawWindowState(RAW_DATA_TX_ID) != WF_RAW_DATA_MOUNTED)
307  {
308  /* WiFi chip may be in processs of transmitting a prior tx packet, so it is very possible that */
309  /* we won't be able to allocate a tx data packet instantly. Set up timer and wait a reasonable */
310  /* amount of time. */
311  maxAllowedTicks = TICKS_PER_SECOND * 6; /* 6 second timeout, needed if scan in progress */
312  startTickCount = (UINT32)TickGet();
313  while ( !AllocateDataTxBuffer(MCHP_DATA_PACKET_SIZE) )
314  {
315  /* If timed out than lock up -- something bad happened */
316  if (TickGet() - startTickCount >= maxAllowedTicks)
317  {
318  WF_ASSERT(FALSE); /* timeout occurred */
319  }
320  }
321  }
322 
323  /* set the rawId */
324  rawId = RAW_DATA_TX_ID;
325 
326  /* convert enc Ptr index to raw index */
327  rawIndex = g_encIndex[encPtrId] - ENC_TX_BUF_TO_RAW_TX_BUF_ADJUSTMENT;
328 
329  /* encPtr[encPtrId] < BASE_TX_ADDR is an error since we don't have the same */
330  /* pre-BASE_TX_ADDR or post tx buffer as the ENC chip */
331  WF_ASSERT((g_encIndex[encPtrId] >= BASE_TX_ADDR) && (g_encIndex[encPtrId] <= (BASE_TX_ADDR + MAX_PACKET_SIZE)));
332  }
333  /*----------------------------------------------------------------------------------*/
334  /* else input read or write pointer is in the TCB (Scratch) section of Ethernet RAM */
335  /*----------------------------------------------------------------------------------*/
336  else
337  {
338  /* set the raw Id, scratch is always mounted during intialization */
339  rawId = RAW_SCRATCH_ID;
340 
341  /* convert Enc ptr index to raw index */
342  rawIndex = g_encIndex[encPtrId] - ENC_TCB_BUF_TO_RAW_SCRATCH_BUF_ADJUSTMENT;
343  }
344 
345  /* set the raw index in the specified raw window */
346  RawSetIndex(rawId, rawIndex);
347 
348  /* Update state variable so we know which raw window is being indexed by the read or write index. */
349  /* Will be either RAW_DATA_RX_ID, RAW_DATA_TX_ID, or RAW_SCRATCH_ID. */
350  g_encPtrRAWId[encPtrId] = rawId;
351 
352 }

544 static void ConfigWep(tWFWpsCred *cred, UINT8 *secType, union sec_key *key)

548  struct sec_wep40 *wep_ctx = (struct sec_wep40 *)key;

553  *secType = WF_SECURITY_WEP_40;
554  ConvAsciiKey2Hex(cred->netKey, cred->keyLen, wep_key);
555  key_len = cred->keyLen / 2;
556  } else if (cred->keyLen == WEP_SHORT_KEY_SIZE) {
557  *secType = WF_SECURITY_WEP_40;
558  memcpy(wep_key, cred->netKey, cred->keyLen);
559  key_len = cred->keyLen;
560  } else if (cred->keyLen == WEP_LONG_KEY_SIZE * 2) {
561  *secType = WF_SECURITY_WEP_104;
562  ConvAsciiKey2Hex(cred->netKey, cred->keyLen, wep_key);
563  key_len = cred->keyLen / 2;
564  } else if (cred->keyLen == WEP_LONG_KEY_SIZE) {
565  *secType = WF_SECURITY_WEP_104;
566  memcpy(wep_key, cred->netKey, cred->keyLen);
567  key_len = cred->keyLen;
568  } else {
569  WF_ASSERT(FALSE);
570  }
571 
572  for (i = 0; i < 4; i++)
573  memcpy(keys + i * key_len, wep_key, key_len);
574 
575  wep_ctx->key_idx = cred->keyIdx - 1;
576 }

1188 /******************************************************************************
1189  * Function: BOOL MACGetHeader(MAC_ADDR *remote, BYTE* type)
1190  *
1191  * PreCondition: None
1192  *
1193  * Input: *remote: Location to store the Source MAC address of the
1194  * received frame.
1195  * *type: Location of a BYTE to store the constant
1196  * MAC_UNKNOWN, ETHER_IP, or ETHER_ARP, representing
1197  * the contents of the Ethernet type field.
1198  *
1199  * Output: TRUE: If a packet was waiting in the RX buffer. The
1200  * remote, and type values are updated.
1201  * FALSE: If a packet was not pending. remote and type are
1202  * not changed.
1203  *
1204  * Side Effects: Last packet is discarded if MACDiscardRx() hasn't already
1205  * been called.
1206  *
1207  * Overview: None
1208  *
1209  * Note: None
1210  *****************************************************************************/
1211 BOOL MACGetHeader(MAC_ADDR *remote, BYTE* type)
1212 {
1213  UINT16 len;
1214  tWFRxPreamble header;
1215 
1216  /* stack no longer cares about currently mounted rx packet, so clear flag */
1217  ClearIndexOutOfBoundsFlag(RAW_DATA_RX_ID);
1218 
1219  /* if we currently have a rx buffer mounted then deallocate it as stack no */
1220  /* longer needs it. */
1221  if ( GetRawWindowState(RAW_DATA_RX_ID) == WF_RAW_DATA_MOUNTED )
1222  {
1223  DeallocateDataRxBuffer();
1224  }
1225 
1226  /* check WiFi chip to see if a data frame has been received. If so, the length of the */
1227  /* data frame is returned. If no data packet received then 0 will be returned. */
1228  len = MACIFService();
1229  if ( len == 0 )
1230  {
1231  return FALSE;
1232  }
1233 
1234 
1235 
1236  /* read preamble header. The RAW index was left at index 2 in the call to MACIFService, which */
1237  /* is the logical start of the received data frame LLC Header, also known as the SNAP header. */
1238  RawRead(RAW_DATA_RX_ID, ENC_PREAMBLE_OFFSET, WF_RX_PREAMBLE_SIZE, (UINT8 *)&header);
1239 
1240  /* as a sanity check verify that the expected bytes contain the SNAP header */
1241  if (!(header.snap[0] == SNAP_VAL &&
1242  header.snap[1] == SNAP_VAL &&
1243  header.snap[2] == SNAP_CTRL_VAL &&
1244  header.snap[3] == SNAP_TYPE_VAL &&
1245  header.snap[4] == SNAP_TYPE_VAL &&
1246  header.snap[5] == SNAP_TYPE_VAL) )
1247  {
1248  /* if a vendor proprietary packet, throw away */
1249  DeallocateDataRxBuffer();
1250  return FALSE;
1251  }
1252 
1253 
1254 #if 0
1255 /* this should no longer be needed */
1256  /* we can flush any saved RAW state now by saving and restoring the current rx buffer. */
1257  PushRawWindow(RAW_DATA_RX_ID);
1258  PopRawWindow(RAW_DATA_RX_ID);
1259 #endif
1260 
1261  // set RAW pointer to 802.11 payload
1262  RawSetIndex(RAW_DATA_RX_ID, (ENC_PREAMBLE_OFFSET + WF_RX_PREAMBLE_SIZE));
1263 
1264  g_encPtrRAWId[ENC_RD_PTR_ID] = RAW_DATA_RX_ID;
1265  g_encIndex[ENC_RD_PTR_ID] = RXSTART + sizeof(ENC_PREAMBLE);
1266 
1267  // The EtherType field, like most items transmitted on the Ethernet medium
1268  // are in big endian.
1269  header.Type.Val = swaps(header.Type.Val);
1270 
1271  // Return the Ethernet frame's Source MAC address field to the caller
1272  // This parameter is useful for replying to requests without requiring an
1273  // ARP cycle.
1274  memcpy((void*)remote->v, (void*)header.SourceMACAddr.v, sizeof(*remote));
1275 
1276  // Return a simplified version of the EtherType field to the caller
1277  *type = MAC_UNKNOWN;
1278  if( (header.Type.v[1] == 0x08u) &&
1279  ((header.Type.v[0] == ETHER_IP) || (header.Type.v[0] == ETHER_ARP)) )
1280  {
1281  *type = header.Type.v[0];
1282  }
1283 
1284  return TRUE;
1285 }

1291  *
1292  * Input: *remote: Pointer to memory which contains the destination
1293  * MAC address (6 bytes)
1294  * type: The constant ETHER_ARP or ETHER_IP, defining which
1295  * value to write into the Ethernet header's type field.

1297  *
1298  * Output: None
1299  *
1300  * Side Effects: None
1301  *
1302  * Overview: None
1303  *
1304  * Note: Because of the dataLen parameter, it is probably
1305  * advantagous to call this function immediately before
1306  * transmitting a packet rather than initially when the
1307  * packet is first created. The order in which the packet
1308  * is constructed (header first or data first) is not
1309  * important.

1311 void MACPutHeader(MAC_ADDR *remote, BYTE type, WORD dataLen)

1316  g_txBufferFlushed = FALSE;
1317  #endif

1322  SyncENCPtrRAWState(ENC_WT_PTR_ID, TXSTART + WF_TX_PREAMBLE_SIZE);
1323 
1324  /* write the Ethernet destination address to buffer (6 bytes) */
1325  memcpy(&buf[0], (void *)remote, sizeof(*remote));
1326  /* write snap header to buffer (6 bytes) */
1327  buf[6] = SNAP_VAL;
1328  buf[7] = SNAP_VAL;
1329  buf[8] = SNAP_CTRL_VAL;
1330  buf[9] = SNAP_TYPE_VAL;

1554 /******************************************************************************
1555  * Function: void MACMemCopyAsync(WORD destAddr, WORD sourceAddr, WORD len)
1556  *
1557  * PreCondition: SPI bus must be initialized (done in MACInit()).
1558  *
1559  * Input: destAddr: Destination address in the Ethernet memory to
1560  * copy to. If the MSb is set, the current EWRPT
1561  * value will be used instead.
1562  * sourceAddr: Source address to read from. If the MSb is
1563  * set, the current ERDPT value will be used
1564  * instead.
1565  * len: Number of bytes to copy
1566  *
1567  * Output: Byte read from the MRF24W's RAM
1568  *
1569  * Side Effects: None
1570  *
1571  * Overview: Bytes are asynchronously transfered within the buffer. Call
1572  * MACIsMemCopyDone() to see when the transfer is complete.
1573  *
1574  * Note: If a prior transfer is already in progress prior to
1575  * calling this function, this function will block until it
1576  * can start this transfer.
1577  *
1578  * If a negative value (bit 15 set) is used for the sourceAddr or destAddr
1579  * parameters, then that pointer will get updated with the
1580  * next address after the read or write.
1581  *****************************************************************************/
1582 void MACMemCopyAsync(PTR_BASE destAddr, PTR_BASE sourceAddr, WORD len)
1583 {
1584  UINT16 readSave = 0, writeSave = 0;
1585  BOOL updateWritePointer;
1586  BOOL updateReadPointer;
1587  UINT8 rawScratchId = 0xff; /* garbage value to avoid compiler warning */
1588  UINT8 copyBuf[8];
1589  UINT16 writeIndex, readIndex;
1590  UINT16 bytesLeft;
1591  UINT16 origRawIndex;
1592 
1593  if (len == 0)
1594  {
1595  return; /* nothing to do */
1596  }
1597 
1598  EnsureWFisAwake();
1599 
1600  /* if using current location of write pointer */
1601  if( ((WORD_VAL*)&destAddr)->bits.b15 )
1602  {
1603  updateWritePointer = TRUE;
1604  destAddr = g_encIndex[ENC_WT_PTR_ID]; /* don't need to sync write pointer, should have already been done */
1605  /* when encIndex was set previously */
1606  }
1607  /* else use destAddr as write pointer */
1608  else
1609  {
1610  updateWritePointer = FALSE;
1611  writeSave = g_encIndex[ENC_WT_PTR_ID];
1612  SyncENCPtrRAWState(ENC_WT_PTR_ID, destAddr);
1613  }
1614 
1615  /* if using current location of read pointer */
1616  if( ((WORD_VAL*)&sourceAddr)->bits.b15 )
1617  {
1618  updateReadPointer = TRUE;
1619  sourceAddr = g_encIndex[ENC_RD_PTR_ID]; /* don't need to sync read pointer, should have already been done */
1620  /* when encIndex was set previously */
1621  }
1622  /* else use srcAddr as read pointer */
1623  else
1624  {
1625  updateReadPointer = FALSE;
1626  readSave = g_encIndex[ENC_RD_PTR_ID];
1627  SyncENCPtrRAWState(ENC_RD_PTR_ID, sourceAddr);
1628  }
1629 
1630  /*-------------------------------------------------*/
1631  /* If copying from Scratch to Scratch (TCB to TCB) */
1632  /*-------------------------------------------------*/
1633  /* This is a special case. We cannot perform a RAW Copy operation within the same RAW window, */
1634  /* but we can copy manually data from one section of the scratch to another section of the scratch window. */
1635  if ((destAddr >= BASE_SCRATCH_ADDR/*BASE_TCB_ADDR*/) && (sourceAddr >= BASE_SCRATCH_ADDR/*BASE_TCB_ADDR*/) )
1636  {
1637  bytesLeft = len;
1638  rawScratchId = RAW_SCRATCH_ID;
1639  /* save the current RAW index in this scratch window */
1640  origRawIndex = RawGetIndex(RAW_SCRATCH_ID);
1641 
1642  /*-------------------------------------------------------------------*/
1643  /* If we can safely do a forward copy (from first byte to last byte) */
1644  /*-------------------------------------------------------------------*/
1645  if ( ((sourceAddr + len) <= destAddr) || // end of source before dest (no overlap)
1646  ((destAddr + len) <= sourceAddr) || // end of dest before source (no overlap)
1647  (destAddr < sourceAddr) // dest before source (overlap, but forward copy safe)
1648  )
1649  {
1650  /* map read and write indexs from TCB address to Scratch Index */
1651  readIndex = sourceAddr - ENC_TCB_BUF_TO_RAW_SCRATCH_BUF_ADJUSTMENT;
1652  writeIndex = destAddr - ENC_TCB_BUF_TO_RAW_SCRATCH_BUF_ADJUSTMENT;
1653 
1654  while (bytesLeft > 0)
1655  {
1656  /* if a full copyBuf worth of bytes to copy */
1657  if (bytesLeft >= sizeof(copyBuf))
1658  {
1659  /* read a block of bytes from the source and write to destination */
1660  RawRead( rawScratchId, readIndex, sizeof(copyBuf), copyBuf);
1661  RawWrite(rawScratchId, writeIndex, sizeof(copyBuf), copyBuf);
1662 
1663  // index to next block in source and dest
1664  readIndex += sizeof(copyBuf);
1665  writeIndex += sizeof(copyBuf);
1666  bytesLeft -= sizeof(copyBuf);
1667  }
1668  /* else less than a full copyBuf left to copy */
1669  else
1670  {
1671  if (bytesLeft > 0)
1672  {
1673  RawRead( rawScratchId, readIndex, bytesLeft, copyBuf);
1674  RawWrite(rawScratchId, writeIndex, bytesLeft, copyBuf);
1675  bytesLeft = 0;
1676  }
1677  }
1678  }
1679  } /* end while for foward copy */
1680  /*---------------------------------------------------------------------------*/
1681  /* else we need to do a backward copy (from last byte to first byte) because */
1682  /* start of TCB dest block within TCB src block */
1683  /*---------------------------------------------------------------------------*/
1684  else if (destAddr > sourceAddr)
1685  {
1686  // map read index from TCB address to Scratch Index
1687  readIndex = sourceAddr - ENC_TCB_BUF_TO_RAW_SCRATCH_BUF_ADJUSTMENT + len - 1;
1688  writeIndex = destAddr - ENC_TCB_BUF_TO_RAW_SCRATCH_BUF_ADJUSTMENT + len - 1;
1689 
1690  while (bytesLeft > 0)
1691  {
1692  // if a full copyBuf worth of bytes to copy
1693  if (bytesLeft >= sizeof(copyBuf))
1694  {
1695  RawRead(rawScratchId, readIndex - sizeof(copyBuf) + 1, sizeof(copyBuf), copyBuf);
1696  RawWrite(rawScratchId, writeIndex - sizeof(copyBuf) + 1, sizeof(copyBuf), copyBuf);
1697 
1698  // index to next block in source and dest
1699  readIndex -= sizeof(copyBuf);
1700  writeIndex -= sizeof(copyBuf);
1701  bytesLeft -= sizeof(copyBuf);
1702  }
1703  // else less than a full copyBuf left to copy
1704  else
1705  {
1706  if (bytesLeft > 0)
1707  {
1708  RawRead(rawScratchId, readIndex - bytesLeft + 1, bytesLeft - 1, copyBuf);
1709  RawWrite(rawScratchId, writeIndex - bytesLeft + 1, bytesLeft - 1, copyBuf);
1710  bytesLeft = 0;
1711  }
1712  }
1713  } // end while
1714  }
1715  // restore raw index to where it was when this function was called
1716  RawSetIndex(RAW_SCRATCH_ID, origRawIndex);
1717 
1718  }
1719  /*-------------------------------------------------------------------------------------------*/
1720  /* else copying between RAW Data Rx and RAW Data Tx or between Raw Data Tx/Rx and RAW Scratch */
1721  /*-------------------------------------------------------------------------------------------*/
1722  else
1723  {
1724  #if defined(WF_DEBUG) /* if asserts enabled */
1725  /* if stack trying to copy from RAW Tx Data Window to same RAW Tx Data Window */
1726  if (((g_encIndex[ENC_WT_PTR_ID] >= TXSTART) && (g_encIndex[ENC_RD_PTR_ID] >= TXSTART))
1727  &&
1728  ((g_encIndex[ENC_WT_PTR_ID] < BASE_SCRATCH_ADDR) && (g_encIndex[ENC_RD_PTR_ID] < BASE_SCRATCH_ADDR)))
1729  {
1730  WF_ASSERT(FALSE); /* can't copy from within same raw window */
1731  }
1732 
1733  /* if stack trying to copy from RAW Rx Data Window to RAW Rx Data Window */
1734  if ((g_encIndex[ENC_WT_PTR_ID] < TXSTART) && (g_encIndex[ENC_RD_PTR_ID] < TXSTART))
1735  {
1736  WF_ASSERT(FALSE); /* can't copy from within same raw window */
1737  }
1738  #endif /* WF_DEBUG */
1739 
1740  /* Copy from source RAW window to destination RAW window. The indexes in the respective */
1741  /* RAW windows have been set either in this function or prior to this fucntion call. */
1742  RawToRawCopy(g_encPtrRAWId[ENC_WT_PTR_ID], g_encPtrRAWId[ENC_RD_PTR_ID], len);
1743  }
1744 
1745  if ( !updateReadPointer )
1746  {
1747  SyncENCPtrRAWState(ENC_RD_PTR_ID, readSave);
1748  }
1749 
1750  if ( !updateWritePointer )
1751  {
1752  SyncENCPtrRAWState(ENC_WT_PTR_ID, writeSave);
1753  }
1754 
1755 
1756 } /* end MACMemCopyAsync */
