1 /******************************************************************************
2 
3  USB Host Communication Device Class(CDC) driver
4 
5 This is the Communication device Class driver file for a USB Embedded Host device.
6 This file should be used in a project with usb_host.c to provide the USB
7 hardware interface.
8 
9 Acronyms/abbreviations used by this class:
10  * CDC - Communication Device Class
11  * ACM - Abstract Control Module
12 
13 To interface with usb_host.c, the routine USBHostCDCInitialize() should be
14 specified as the Initialize() function, and USBHostCDCEventHandler() should
15 be specified as the EventHandler() function in the usbClientDrvTable[] array
16 declared in usb_config.h.
17 
18 This driver can be configured to use transfer events from usb_host.c. Transfer
19 events require more RAM and ROM than polling, but it cuts down or even
20 eliminates the required polling of the various USBxxxTasks functions. For this
21 class, USBHostCDCTasks() is compiled out if transfer events from usb_host.c
22 are used. However, USBHostTasks() still must be called to provide attach,
23 enumeration, and detach services. If transfer events from usb_host.c
24 are going to be used, USB_ENABLE_TRANSFER_EVENT should be defined. If transfer
25 status is going to be polled, USB_ENABLE_TRANSFER_EVENT should not be defined.
26 CDC client layer transfer events are also enables once USB_ENABLE_TRANSFER_EVENT
27 is defined. The application must define USB_HOST_APP_EVENT_HANDLER in file usb_config.h.
28 All the CDC client layer event are called back using this function handler. This eliminates
29 the need of polling for transfer status by the application, however handing of these events
30 should be taken care by the application.
31 
32 Transfer of Data Class Interface can be performed with BULK transfers,
33 hence USB_SUPPORT_BULK_TRANSFERS must be defined. Data Class Interface can also use
34 ISOCHRONOUS transfers,however the CDC client is not tested for ISOCHRONOUS transfers.

513  This function performs the maintenance tasks required by CDC class
514 
515  Description:
516  This function performs the maintenance tasks required by the CDC
517  class. If transfer events from the host layer are not being used, then
518  it should be called on a regular basis by the application. If transfer
519  events from the host layer are being used, this function is compiled out,
520  and does not need to be called.
521 
522  Precondition:
523  USBHostCDCInitialize() has been called.
524 
525  Parameters:
526  None - None
527 
528  Returns:
529  None
530 
531  Remarks:
532  None

534 void USBHostCDCTasks( void )
535 {
536 #ifndef USB_ENABLE_TRANSFER_EVENT
537  DWORD byteCount;
538  BYTE errorCode;
539  BYTE i;
540 
541 // BYTE temp;
542 
543  for (i=0; i<USB_MAX_CDC_DEVICES; i++)
544  {
545  if (deviceInfoCDC[i].deviceAddress == 0) /* device address updated by lower layer */
546  {
547  deviceInfoCDC[i].state = STATE_DETACHED;
548  }
549 
550  switch (deviceInfoCDC[i].state & STATE_MASK)
551  {
552  case STATE_DETACHED:
553  // No device attached.
554  break;
555 
556  case STATE_INITIALIZE_DEVICE:
557  switch (deviceInfoCDC[i].state & SUBSTATE_MASK)
558  {
559  case SUBSTATE_WAIT_FOR_ENUMERATION:
560  if (USBHostDeviceStatus( deviceInfoCDC[i].deviceAddress ) == USB_DEVICE_ATTACHED)
561  {
562  _USBHostCDC_SetNextSubState();
563  }
564  break;
565 
566  case SUBSTATE_DEVICE_ENUMERATED:
567  USBHostCDCInitAddress( deviceInfoCDC[i].deviceAddress,0, deviceInfoCDC[i].clientDriverID ); // Initialize device address for application use
568  _USBHostCDC_SetNextSubState(); /* need to add sub states to Set Config, Get LANGID & String Descriptors */
569  break;
570 
571 
572  case SUBSTATE_GET_LINE_CODING:
573 
574  if(!USBHostIssueDeviceRequest( deviceInfoCDC[i].deviceAddress, USB_SETUP_DEVICE_TO_HOST | USB_SETUP_TYPE_CLASS | USB_SETUP_RECIPIENT_INTERFACE,
575  USB_CDC_GET_LINE_CODING, 0 , deviceInfoCDC[i].commInterface.interfaceNum , USB_CDC_LINE_CODING_LENGTH, (BYTE*)&CDC_DEV_LINE_CODING_Buffer,
576  USB_DEVICE_REQUEST_GET , deviceInfoCDC[i].clientDriverID ))
577  {
578  _USBHostCDC_SetNextSubState();
579  }
580  break;
581 
582  case SUBSTATE_WAIT_FOR_GET_LINE_CODING:
583  if (USBHostTransferIsComplete( deviceInfoCDC[i].deviceAddress, 0, &errorCode, &byteCount ))
584  {
585  if (errorCode)
586  {
587  /* Set error code */
588  _USBHostCDC_TerminateTransfer(errorCode);
589  }
590  else
591  {
592  // Clear the STALL. Since it is EP0, we do not have to clear the stall.
593  USBHostClearEndpointErrors( deviceInfoCDC[i].deviceAddress, 0 );
594  // Compare with Line coding expected at Host end with the one received from device
595  if(0 == memcmp(&CDC_DEV_LINE_CODING_Buffer, PTR_HOST_LINE_CODING_BUFFER, USB_CDC_LINE_CODING_LENGTH))
596  {
597  // if fine goto set Control line state
598  // deviceInfoCDC[i].state = STATE_INITIALIZE_DEVICE | SUBSTATE_SET_CONTROL_LINE_STATE;
599  deviceInfoCDC[i].state = STATE_RUNNING;
600  }
601  else
602  {
603  // if data not as expected goto Set Line coding
604  deviceInfoCDC[i].state = STATE_INITIALIZE_DEVICE | SUBSTATE_SET_LINE_CODING;
605  }
606  }
607  }
608  break;
609 
610  case SUBSTATE_SET_LINE_CODING:
611 
612  if(!USBHostIssueDeviceRequest( deviceInfoCDC[i].deviceAddress, USB_SETUP_HOST_TO_DEVICE | USB_SETUP_TYPE_CLASS | USB_SETUP_RECIPIENT_INTERFACE,
613  USB_CDC_SET_LINE_CODING, 1 , deviceInfoCDC[i].commInterface.interfaceNum , USB_CDC_LINE_CODING_LENGTH, (BYTE*)PTR_HOST_LINE_CODING_BUFFER,
614  USB_DEVICE_REQUEST_SET , deviceInfoCDC[i].clientDriverID ))
615  {
616  _USBHostCDC_SetNextSubState();
617  }
618  break;
619 
620  case SUBSTATE_WAIT_FOR_SET_LINE_CODING:
621  if (USBHostTransferIsComplete( deviceInfoCDC[i].deviceAddress, 0, &errorCode, &byteCount ))
622  {
623  if (errorCode)
624  {
625  /* Set error code */
626  _USBHostCDC_TerminateTransfer(errorCode);
627  }
628  else
629  { // check again if line coding is correctly set
630  deviceInfoCDC[i].state = STATE_INITIALIZE_DEVICE | SUBSTATE_GET_LINE_CODING;
631  }
632  }
633  break;
634 
635  case SUBSTATE_SET_CONTROL_LINE_STATE:
636 
637  if(!USBHostIssueDeviceRequest( deviceInfoCDC[i].deviceAddress, USB_SETUP_HOST_TO_DEVICE | USB_SETUP_TYPE_CLASS | USB_SETUP_RECIPIENT_INTERFACE,
638  USB_CDC_SET_CONTROL_LINE_STATE , *(WORD*)PTR_HOST_CONTROL_LINE_BUFFER , deviceInfoCDC[i].commInterface.interfaceNum , USB_CDC_CONTROL_LINE_LENGTH, NULL,
639  USB_DEVICE_REQUEST_SET , deviceInfoCDC[i].clientDriverID ))
640  {
641  _USBHostCDC_SetNextSubState();
642  }
643  break;
644 
645  case SUBSTATE_WAIT_FOR_SET_CONTROL_LINE_STATE:
646  if (USBHostTransferIsComplete( deviceInfoCDC[i].deviceAddress, 0, &errorCode, &byteCount ))
647  {
648  if (errorCode)
649  {
650  /* Set error code */
651  _USBHostCDC_TerminateTransfer(errorCode);
652  }
653  else
654  {
655  _USBHostCDC_SetNextState();
656  }
657  }
658  break;
659 
660  default :
661  break;
662  }
663  break;
664 
665  case STATE_RUNNING:
666  switch (deviceInfoCDC[i].state & SUBSTATE_MASK)
667  {
668  case SUBSTATE_WAITING_FOR_REQ:
669  /* waiting for request from application */
670  break;
671 
672  case SUBSTATE_SEND_READ_REQ:
673  // if endpoint 0 then use control transfer - Communication interface request
674  // otherwise request is for data interface transfer
675  USBHostClearEndpointErrors( deviceInfoCDC[i].deviceAddress, deviceInfoCDC[i].endpointDATA );
676 
677  if(deviceInfoCDC[i].endpointDATA == 0x00)
678  {
679  errorCode = USBHostIssueDeviceRequest( deviceInfoCDC[i].deviceAddress, USB_SETUP_DEVICE_TO_HOST | USB_SETUP_TYPE_CLASS | USB_SETUP_RECIPIENT_INTERFACE,
680  deviceInfoCDC[i].commRequest, 0, deviceInfoCDC[i].interface,deviceInfoCDC[i].reportSize, deviceInfoCDC[i].userData,
681  USB_DEVICE_REQUEST_GET , deviceInfoCDC[i].clientDriverID );
682  }
683  else
684  {
685  errorCode = USBHostRead( deviceInfoCDC[i].deviceAddress, deviceInfoCDC[i].endpointDATA,
686  deviceInfoCDC[i].userData, deviceInfoCDC[i].reportSize );
687  }
688 
689  if (errorCode)
690  {
691  if(USB_ENDPOINT_STALLED == errorCode)
692  {
693  USBHostClearEndpointErrors( deviceInfoCDC[i].deviceAddress, deviceInfoCDC[i].endpointDATA );
694  deviceInfoCDC[i].returnState = STATE_HOLDING;
695  deviceInfoCDC[i].flags.bfReset = 1;
696  _USBHostCDC_ResetStateJump( i );
697  }
698  else
699  {
700  _USBHostCDC_TerminateTransfer(errorCode);
701  }
702  }
703  else
704  {
705  _USBHostCDC_SetNextSubState();
706  }
707  break;
708 
709  case SUBSTATE_READ_REQ_WAIT:
710  if (USBHostTransferIsComplete( deviceInfoCDC[i].deviceAddress, deviceInfoCDC[i].endpointDATA, &errorCode, &byteCount ))
