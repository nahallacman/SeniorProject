121 BOOL USBSleepOnSuspend(void)
122 {
123  unsigned int U1EIE_save, U1IE_save, U1OTGIE_save;
124  unsigned char USB1IE_save;
125 
126  #if defined(USB_POLLING)
127  //If IPL is equal to 0 then there is no way for the USB module to
128  // generate an interrupt to wake up the device.
129  if(_IPL == 0)
130  {
131  return FALSE;
132  }
133 
134  //Set the interrupt priority to a level that will wake up the part (>0)
135  // but will not cause a interrupt vector jump (USB1IP<=IPL)
136  _USB1IP = 1;
137  #endif
138 
139  //Save the old interrupt and CPU settings
140  U1EIE_save = U1EIE;
141  U1IE_save = U1IE;
142  U1OTGIE_save = U1OTGIE;
143  USB1IE_save = IEC5bits.USB1IE;
144 
145  //Disable all USB interrupts
146  U1EIE = 0;
147  U1IE = 0;
148  U1OTGIE = 0;
149 
150  //Enable the interrupt
151  IFS5bits.USB1IF = 0;
152  U1OTGIEbits.ACTVIE = 1;
153  USBClearInterruptFlag(USBActivityIFReg,USBActivityIFBitNum);
154  IEC5bits.USB1IE = 1;
155 
156  Sleep();
157 
158  #if defined(USB_POLLING)
159  //Disable the interrupt
160  _USB1IP = 0;
161  #endif
162 
163  //restore the previous interrupt settings
164  IEC5bits.USB1IE = USB1IE_save;
165  U1EIE = U1EIE_save;
166  U1IE = U1IE_save;
167  U1OTGIE = U1OTGIE_save;
168 
169  return TRUE;
170 }
