175 void USBCheckHIDRequest(void)
176 {
177  if(SetupPkt.Recipient != USB_SETUP_RECIPIENT_INTERFACE_BITFIELD) return;
178  if(SetupPkt.bIntfID != HID_INTF_ID) return;
179 
180  /*
181  * There are two standard requests that hid.c may support.
182  * 1. GET_DSC(DSC_HID,DSC_RPT,DSC_PHY);
183  * 2. SET_DSC(DSC_HID,DSC_RPT,DSC_PHY);
184  */
185  if(SetupPkt.bRequest == USB_REQUEST_GET_DESCRIPTOR)
186  {
187  switch(SetupPkt.bDescriptorType)
188  {
189  case DSC_HID: //HID Descriptor
190  if(USBActiveConfiguration == 1)
191  {
192  USBEP0SendROMPtr(
193  (ROM BYTE*)&configDescriptor1 + 18, //18 is a magic number. It is the offset from start of the configuration descriptor to the start of the HID descriptor.
194  sizeof(USB_HID_DSC)+3,
195  USB_EP0_INCLUDE_ZERO);
196  }
197  break;
198  case DSC_RPT: //Report Descriptor
199  //if(USBActiveConfiguration == 1)
200  {
201  USBEP0SendROMPtr(
202  (ROM BYTE*)&hid_rpt01,
203  HID_RPT01_SIZE, //See usbcfg.h
204  USB_EP0_INCLUDE_ZERO);
205  }
206  break;
207  case DSC_PHY: //Physical Descriptor
208  //Note: The below placeholder code is commented out. HID Physical Descriptors are optional and are not used
209  //in many types of HID applications. If an application does not have a physical descriptor,
210  //then the device should return STALL in response to this request (stack will do this automatically
211  //if no-one claims ownership of the control transfer).
212  //If an application does implement a physical descriptor, then make sure to declare
213  //hid_phy01 (rom structure containing the descriptor data), and hid_phy01 (the size of the descriptors in bytes),
214  //and then uncomment the below code.
215  //if(USBActiveConfiguration == 1)
216  //{
217  // USBEP0SendROMPtr((ROM BYTE*)&hid_phy01, sizeof(hid_phy01), USB_EP0_INCLUDE_ZERO);
218  //}
219  break;
220  }//end switch(SetupPkt.bDescriptorType)
221  }//end if(SetupPkt.bRequest == GET_DSC)
222 
223  if(SetupPkt.RequestType != USB_SETUP_TYPE_CLASS_BITFIELD)
224  {
225  return;
226  }
227 
228  switch(SetupPkt.bRequest)
229  {
230  case GET_REPORT:
231  #if defined USER_GET_REPORT_HANDLER
232  USER_GET_REPORT_HANDLER();
233  #endif
234  break;
235  case SET_REPORT:
236  #if defined USER_SET_REPORT_HANDLER
237  USER_SET_REPORT_HANDLER();
238  #endif
239  break;
240  case GET_IDLE:
241  USBEP0SendRAMPtr(
242  (BYTE*)&idle_rate,
243  1,
244  USB_EP0_INCLUDE_ZERO);
245  break;
246  case SET_IDLE:
247  USBEP0Transmit(USB_EP0_NO_DATA);
248  idle_rate = SetupPkt.W_Value.byte.HB;
249  break;
250  case GET_PROTOCOL:
251  USBEP0SendRAMPtr(
252  (BYTE*)&active_protocol,
253  1,
254  USB_EP0_NO_OPTIONS);
255  break;
256  case SET_PROTOCOL:
257  USBEP0Transmit(USB_EP0_NO_DATA);
258  active_protocol = SetupPkt.W_Value.byte.LB;
259  break;
260  }//end switch(SetupPkt.bRequest)
261 
262 }//end USBCheckHIDRequest
