324 /*****************************************************************************
325  Function:
326  static HTTP_IO_RESULT HTTPPostLCD(void)
327 
328  Summary:
329  Processes the LCD form on forms.htm
330 
331  Description:
332  Locates the 'lcd' parameter and uses it to update the text displayed
333  on the board's LCD display.
334 
335  This function has four states. The first reads a name from the data
336  string returned as part of the POST request. If a name cannot
337  be found, it returns, asking for more data. Otherwise, if the name
338  is expected, it reads the associated value and writes it to the LCD.
339  If the name is not expected, the value is discarded and the next name
340  parameter is read.
341 
342  In the case where the expected string is never found, this function
343  will eventually return HTTP_IO_NEED_DATA when no data is left. In that
344  case, the HTTP2 server will automatically trap the error and issue an
345  Internal Server Error to the browser.
346 
347  Precondition:
348  None
349 
350  Parameters:
351  None
352 
353  Return Values:
354  HTTP_IO_DONE - the parameter has been found and saved
355  HTTP_IO_WAITING - the function is pausing to continue later
356  HTTP_IO_NEED_DATA - data needed by this function has not yet arrived
357  ***************************************************************************/
358 #if defined(USE_LCD)
359 static HTTP_IO_RESULT HTTPPostLCD(void)
360 {
361  BYTE* cDest;
362 
363  #define SM_POST_LCD_READ_NAME (0u)
364  #define SM_POST_LCD_READ_VALUE (1u)
365 
366  switch(curHTTP.smPost)
367  {
368  // Find the name
369  case SM_POST_LCD_READ_NAME:
370 
371  // Read a name
372  if(HTTPReadPostName(curHTTP.data, HTTP_MAX_DATA_LEN) == HTTP_READ_INCOMPLETE)
373  return HTTP_IO_NEED_DATA;
374 
375  curHTTP.smPost = SM_POST_LCD_READ_VALUE;
376  // No break...continue reading value
377 
378  // Found the value, so store the LCD and return
379  case SM_POST_LCD_READ_VALUE:
380 
381  // If value is expected, read it to data buffer,
382  // otherwise ignore it (by reading to NULL)
383  if(!strcmppgm2ram((char*)curHTTP.data, (ROM char*)"lcd"))
384  cDest = curHTTP.data;
385  else
386  cDest = NULL;
387 
388  // Read a value string
389  if(HTTPReadPostValue(cDest, HTTP_MAX_DATA_LEN) == HTTP_READ_INCOMPLETE)
390  return HTTP_IO_NEED_DATA;
391 
392  // If this was an unexpected value, look for a new name
393  if(!cDest)
394  {
395  curHTTP.smPost = SM_POST_LCD_READ_NAME;
396  break;
397  }
398 
399  // Copy up to 32 characters to the LCD
400  if(strlen((char*)cDest) < 32u)
401  {
402  memset(LCDText, ' ', 32);
403  strcpy((char*)LCDText, (char*)cDest);
404  }
405  else
406  {
407  memcpy(LCDText, (void *)cDest, 32);
408  }
409  LCDUpdate();
410 
411  // This is the only expected value, so callback is done
412  strcpypgm2ram((char*)curHTTP.data, "/forms.htm");
413  curHTTP.httpStatus = HTTP_REDIRECT;
414  return HTTP_IO_DONE;
415  }
416 
417  // Default assumes that we're returning for state machine convenience.
418  // Function will be called again later.
419  return HTTP_IO_WAITING;
420 }

423 /*****************************************************************************
424  Function:
425  static HTTP_IO_RESULT HTTPPostConfig(void)
426 
427  Summary:
428  Processes the configuration form on config/index.htm
429 
430  Description:
431  Accepts configuration parameters from the form, saves them to a
432  temporary location in RAM, then eventually saves the data to EEPROM or
433  external Flash.
434 
435  When complete, this function redirects to config/reboot.htm, which will
436  display information on reconnecting to the board.
437 
438  This function creates a shadow copy of the AppConfig structure in
439  RAM and then overwrites incoming data there as it arrives. For each
440  name/value pair, the name is first read to curHTTP.data[0:5]. Next, the
441  value is read to newAppConfig. Once all data has been read, the new
442  AppConfig is saved back to EEPROM and the browser is redirected to
443  reboot.htm. That file includes an AJAX call to reboot.cgi, which
444  performs the actual reboot of the machine.
445 
446  If an IP address cannot be parsed, too much data is POSTed, or any other
447  parsing error occurs, the browser reloads config.htm and displays an error
448  message at the top.
449 
450  Precondition:
451  None
452 
453  Parameters:
454  None
455 
456  Return Values:
457  HTTP_IO_DONE - all parameters have been processed
458  HTTP_IO_NEED_DATA - data needed by this function has not yet arrived
459  ***************************************************************************/
460 #if defined(STACK_USE_HTTP_APP_RECONFIG)
461 static HTTP_IO_RESULT HTTPPostConfig(void)
462 {
463  APP_CONFIG newAppConfig;
464  BYTE *ptr;
465  BYTE i;
466 
467  // Check to see if the browser is attempting to submit more data than we
468  // can parse at once. This function needs to receive all updated
469  // parameters and validate them all before committing them to memory so that
470  // orphaned configuration parameters do not get written (for example, if a
471  // static IP address is given, but the subnet mask fails parsing, we
472  // should not use the static IP address). Everything needs to be processed
473  // in a single transaction. If this is impossible, fail and notify the user.
474  // As a web devloper, if you add parameters to AppConfig and run into this
475  // problem, you could fix this by to splitting your update web page into two
476  // seperate web pages (causing two transactional writes). Alternatively,
477  // you could fix it by storing a static shadow copy of AppConfig someplace
478  // in memory and using it instead of newAppConfig. Lastly, you could
479  // increase the TCP RX FIFO size for the HTTP server. This will allow more
480  // data to be POSTed by the web browser before hitting this limit.
481  if(curHTTP.byteCount > TCPIsGetReady(sktHTTP) + TCPGetRxFIFOFree(sktHTTP))
482  goto ConfigFailure;
483 
484  // Ensure that all data is waiting to be parsed. If not, keep waiting for
485  // all of it to arrive.
486  if(TCPIsGetReady(sktHTTP) < curHTTP.byteCount)
487  return HTTP_IO_NEED_DATA;
488 
489 
490  // Use current config in non-volatile memory as defaults
491  #if defined(EEPROM_CS_TRIS)
492  XEEReadArray(sizeof(NVM_VALIDATION_STRUCT), (BYTE*)&newAppConfig, sizeof(newAppConfig));
493  #elif defined(SPIFLASH_CS_TRIS)
494  SPIFlashReadArray(sizeof(NVM_VALIDATION_STRUCT), (BYTE*)&newAppConfig, sizeof(newAppConfig));
495  #endif
496 
497  // Start out assuming that DHCP is disabled. This is necessary since the
498  // browser doesn't submit this field if it is unchecked (meaning zero).
499  // However, if it is checked, this will be overridden since it will be
500  // submitted.
501  newAppConfig.Flags.bIsDHCPEnabled = 0;
502 
503 
504  // Read all browser POST data
505  while(curHTTP.byteCount)
506  {
507  // Read a form field name
508  if(HTTPReadPostName(curHTTP.data, 6) != HTTP_READ_OK)
509  goto ConfigFailure;
510 
511  // Read a form field value
512  if(HTTPReadPostValue(curHTTP.data + 6, sizeof(curHTTP.data)-6-2) != HTTP_READ_OK)
513  goto ConfigFailure;
514 
515  // Parse the value that was read
516  if(!strcmppgm2ram((char*)curHTTP.data, (ROM char*)"ip"))
517  {// Read new static IP Address
518  if(!StringToIPAddress(curHTTP.data+6, &newAppConfig.MyIPAddr))
519  goto ConfigFailure;
520 
521  newAppConfig.DefaultIPAddr.Val = newAppConfig.MyIPAddr.Val;
522  }
523  else if(!strcmppgm2ram((char*)curHTTP.data, (ROM char*)"gw"))
524  {// Read new gateway address
525  if(!StringToIPAddress(curHTTP.data+6, &newAppConfig.MyGateway))
526  goto ConfigFailure;
527  }
528  else if(!strcmppgm2ram((char*)curHTTP.data, (ROM char*)"sub"))
529  {// Read new static subnet
530  if(!StringToIPAddress(curHTTP.data+6, &newAppConfig.MyMask))
531  goto ConfigFailure;
532 
533  newAppConfig.DefaultMask.Val = newAppConfig.MyMask.Val;
534  }
535  else if(!strcmppgm2ram((char*)curHTTP.data, (ROM char*)"dns1"))
536  {// Read new primary DNS server
537  if(!StringToIPAddress(curHTTP.data+6, &newAppConfig.PrimaryDNSServer))
538  goto ConfigFailure;
539  }
540  else if(!strcmppgm2ram((char*)curHTTP.data, (ROM char*)"dns2"))
541  {// Read new secondary DNS server
542  if(!StringToIPAddress(curHTTP.data+6, &newAppConfig.SecondaryDNSServer))
543  goto ConfigFailure;
544  }
545  else if(!strcmppgm2ram((char*)curHTTP.data, (ROM char*)"mac"))
546  {
547  // Read new MAC address
548  WORD w;
549  BYTE i;
550 
551  ptr = curHTTP.data+6;
552 
553  for(i = 0; i < 12u; i++)
554  {// Read the MAC address
555 
556  // Skip non-hex bytes
557  while( *ptr != 0x00u && !(*ptr >= '0' && *ptr <= '9') && !(*ptr >= 'A' && *ptr <= 'F') && !(*ptr >= 'a' && *ptr <= 'f') )
558  ptr++;
559 
560  // MAC string is over, so zeroize the rest
561  if(*ptr == 0x00u)
562  {
563  for(; i < 12u; i++)
564  curHTTP.data[i] = '0';
565  break;
566  }
567 
568  // Save the MAC byte
569  curHTTP.data[i] = *ptr++;
570  }
571 
572  // Read MAC Address, one byte at a time
573  for(i = 0; i < 6u; i++)
574  {
575  ((BYTE*)&w)[1] = curHTTP.data[i*2];
576  ((BYTE*)&w)[0] = curHTTP.data[i*2+1];
577  newAppConfig.MyMACAddr.v[i] = hexatob(*((WORD_VAL*)&w));
578  }
579  }
580  else if(!strcmppgm2ram((char*)curHTTP.data, (ROM char*)"host"))
581  {// Read new hostname
582  FormatNetBIOSName(&curHTTP.data[6]);
583  memcpy((void*)newAppConfig.NetBIOSName, (void*)curHTTP.data+6, 16);
584  }
585  else if(!strcmppgm2ram((char*)curHTTP.data, (ROM char*)"dhcp"))
586  {// Read new DHCP Enabled flag
587  if(curHTTP.data[6] == '1')
588  newAppConfig.Flags.bIsDHCPEnabled = 1;
589  }
590  }
591 
592 
593  // All parsing complete! Save new settings and force a reboot
594  SaveAppConfig(&newAppConfig);
595 
596  // Set the board to reboot and display reconnecting information
597  strcpypgm2ram((char*)curHTTP.data, "/protect/reboot.htm?");
598  memcpy((void*)(curHTTP.data+20), (void*)newAppConfig.NetBIOSName, 16);
599  curHTTP.data[20+16] = 0x00; // Force null termination
600  for(i = 20; i < 20u+16u; i++)
601  {
602  if(curHTTP.data[i] == ' ')
603  curHTTP.data[i] = 0x00;
604  }
605  curHTTP.httpStatus = HTTP_REDIRECT;
606 
607  return HTTP_IO_DONE;
608 
609 
610 ConfigFailure:
611  lastFailure = TRUE;
612  strcpypgm2ram((char*)curHTTP.data, "/protect/config.htm");
613  curHTTP.httpStatus = HTTP_REDIRECT;
614 
615  return HTTP_IO_DONE;
616 }
