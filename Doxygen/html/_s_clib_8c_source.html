218  Description:
219  This function performs the power on sequence of the SmartCard and

221 
222  Precondition:
223  SC_Initialize() is called, and card is present

228  Return Values:
229  TRUE if Answer to Reset (ATR) was successfuly received and processed

234 BOOL SC_PowerOnATR()
235 {
236  unsigned long atrDelayETUs;
237 
238  //check whether card is present before doing power on reset to the card
239  if( !SCdrv_CardPresent() )
240  {
241  gCardState = UNKNOWN;
242  return FALSE;
243  }
244 
245  //make sure card is in reset
246  SCdrv_SetSwitchCardReset(0);
247  memset( scCardATR, 0xFF, sizeof scCardATR );
248 
249  WaitMilliSec(2);
250 
251  //Turn on power to the card
252  #ifdef ENABLE_SC_POWER_THROUGH_PORT_PIN
253  SCdrv_SetSwitchCardPower(1);
254  #endif
255 
256  scATR_HistoryLength = 0;
257  scATR_HistoryBuffer = NULL;
258  gCardState = UNKNOWN;
259  scLastError = SC_ERR_NONE;
260  scATRLength = 0;
261 
262  // Wait for maximum of 40,000 to 45,000 smard card clock cycles
263  // to get an ATR from card.The number of clock cycles is converted
264  // into etu's for easier usage in the code.
265  atrDelayETUs = (unsigned short int)((45000UL * baudRate)/scReferenceClock);
266 
267  WaitMilliSec(2);
268 
269  // Enable UART
270  SCdrv_EnableUART();
271 
272  WaitMilliSec(2);
273 
274  //Start the clock
275  SCdrv_EnableClock();
276 
277  // Wait for atleast 400 Clock Cycles after applying reference clock to card.
278  WaitMilliSec(2);
279 
280  //Release card reset line. set to high state
281  SCdrv_SetSwitchCardReset(1);
282 
283  // Read Answer to RESET
284  while(1)
285  {
286  //wait for data byte from CARD
287  if( SCdrv_GetRxData( &scCardATR[scATRLength], atrDelayETUs ) )
288  {
289  scATRLength++;
290 
291  if( scATRLength == MAX_ATR_LEN )
292  break;
293  else
294  atrDelayETUs = (unsigned short int)9600;
295  }
296  else
297  break; //no data
298  }
299 
300  //decode the ATR values
301  if( scATRLength >= 3 ) //min TS, T0 and setup byte
302  {
303  BYTE T0 = scCardATR[1];
304  BYTE atrIdx = 2;
305 
306  //Extract Interface bytes TAx TBx TCx and TDx from ATR
307 
308  scTA1 = scTB1 = scTC1 = scTD1 = 0;
309  scTA2 = scTB2 = scTC2 = scTD2 = 0;
310  scTA3 = scTB3 = scTC3 = scTD3 = 0;
311 
312  // Read the global interface bytes
313 
314  if( T0 & 0x10 )
315  scTA1 = scCardATR[atrIdx++];
316 
317  if( T0 & 0x20 )
318  scTB1 = scCardATR[atrIdx++];
319 
320  if( T0 & 0x40 )
321  scTC1 = scCardATR[atrIdx++];
322 
323  if( T0 & 0x80 )
324  scTD1 = scCardATR[atrIdx++];
325 
326  //read the next set of interface bytes if present
327  if( scTD1 & 0xF0 )
328  {
329  if( scTD1 & 0x10 )
330  scTA2 = scCardATR[atrIdx++];
331 
332  if( scTD1 & 0x20 )
333  scTB2 = scCardATR[atrIdx++];
334 
335  if( scTD1 & 0x40 )
336  scTC2 = scCardATR[atrIdx++];
337 
338  if( scTD1 & 0x80 )
339  scTD2 = scCardATR[atrIdx++];
340 
341  if( scTD2 & 0xF0 )
342  {
343  if( scTD2 & 0x10 )
344  {
345  scTA3 = scCardATR[atrIdx++];
346 
347  if ((scTA3 < 0x10) || (scTA3 == 0xFF))
348  {
349  SC_Shutdown();
350  scLastError = SC_ERR_ATR_DATA;
351  return FALSE;
352  }
353 
354  #ifdef SC_PROTO_T1
355  maxSegmentLength = scTA3;
356  #endif
357  }
358 
359  if( scTD2 & 0x20 )
360  scTB3 = scCardATR[atrIdx++];
361 
362  if( scTD2 & 0x40 )
363  {
364  scTC3 = scCardATR[atrIdx++];
365 
366  #ifdef SC_PROTO_T1
367  edcType = (scTC3 & 0x01) ? SC_CRC_TYPE_EDC : SC_LRC_TYPE_EDC;
368  #endif
369  }
370 
371  if( scTD2 & 0x80 )
372  scTD3 = scCardATR[atrIdx++];
373  }
374  }
375 
376  scATR_HistoryLength = T0 & 0x0F;
377  scATR_HistoryBuffer = (scATR_HistoryLength)?(&scCardATR[atrIdx]):NULL;
378  // Calculate the protocol wait times for default values
379  SC_CalculateWaitTime();
380  gCardState = ATR_ON;
381  return TRUE;
382  }
383  else
384  {
385  // Not a Valid ATR Reponse
386  scLastError = SC_ERR_BAR_OR_NO_ATR_RESPONSE;
387  gCardState = UNKNOWN;
388  SC_Shutdown();
389  return FALSE;
390  }

394 /*******************************************************************************
395  Function:
396  BOOL SC_DoPPS(BYTE *ppsPtr)
397 
398  Description:
399  This function does the PPS exchange with the smart card & configures the baud
400  rate of the PIC UART module as per the PPS response from the smart card.
401 
402  Precondition:
403  SC_PowerOnATR was success
404 
405  Parameters:
406  Input is pointer to PPS string
407 
408  Return Values:
409  TRUE if PPS exchange is successful
410 
411  Remarks:
412  This function is called when SC_PowerOnATR() returns TRUE.
413  *****************************************************************************/
414 BOOL SC_DoPPS(BYTE *ppsPtr)
415 {
416  BYTE ppsn[3] = {0,0,0};
417  BYTE index1,index2 = 0x10,index3 = 2,ppsStrLength = 3,pckByte = 0x00;
418  unsigned int ppsCGTinMicroSeconds;
419 
420  // For PPS exchange GT = 12 ETUs
421  ppsCGTinMicroSeconds = (12 * 1000000UL)/baudRate;
422 
423  // Calculate the length of PPS request and store PPS1, PPS2 and PPS3
424  // in local variables for future calculations
425  for(index1 = 0;index1 < 3;index1++)
426  {
427  if(ppsPtr[1] & index2)
428  {
429  ppsStrLength++;
430  ppsn[index1] = ppsPtr[index3];
431  index3++;
432  }
433  index2 = index2 << 1;
434  }
435 
436  // Check for the conditions whether PPS can be done to the card...
437  scPPSresponseLength = 0;
438  if(!SCdrv_CardPresent() || (gCardState != ATR_ON) || (scTD1 & 0x10))
439  return FALSE;
440 
441  // Send PPS request to the card
442  index1 = 0;
443  while(ppsStrLength--)
444  {
445  WaitMicroSec(ppsCGTinMicroSeconds);
446  SCdrv_SendTxData( ppsPtr[index1++] );
447  }
448 
449  // Recieve PPS response from the smart card
450  index1 = 0;
451  while (1)
452  {
453  //wait for data byte from CARD
454  if(SCdrv_GetRxData( &scPPSresponse[index1], 9600 ) )
455  {
456  if(++index1 == 7)
457  return FALSE;
458  }
459  else
460  break;
461  }
462 
463  // If PPS response length is greater or equal to 3 bytes and PPSS byte is 0xFF and lower
464  // nibble of PPS0 byte of PPS response matches with the PPS request then proceed for
465  // further calculation or else exit...
466  if((index1 > 2) && (scPPSresponse[0] == 0xFF) && ((ppsPtr[1] & 0x0F) == (scPPSresponse[1] & 0x0F)))
467  {
468  index2 = 2;
469  if(scPPSresponse[1] & 0x10)
470  {
471  // Check PPS1 Response is identical to PPS1 Request.
472  if(scPPSresponse[index2] != ppsn[0])
473  return FALSE;
474  index2++;
475  }
476 
477  if(scPPSresponse[1] & 0x20)
478  {
479  // Check PPS1 Response is identical to PPS1 Request.
480  if(scPPSresponse[index2] != ppsn[1])
481  return FALSE;
482  index2++;
483  }
484 
485  if(scPPSresponse[1] & 0x40)
486  {
487  // Check PPS1 Response is identical to PPS1 Request.
488  if(scPPSresponse[index2] != ppsn[2])
489  return FALSE;
490  }
491 
492  // Calculate PCK for the PPS response and ex-or it with recieved PCK byte
493  pckByte = 0x00;
494  for(index2 = 0;index2 < index1;index2++)
495  pckByte = pckByte ^ scPPSresponse[index2];
496 
497  // If the final vaue is non-zero then exit...
498  if(pckByte)
499  return FALSE;
500 
501  // If baud rate modification request has been accepted by the smart card,
502  // change the UART baud rate and other wait time constants
503  if(scPPSresponse[1] & 0x10)
504  {
505  SC_FindFDvalues(scPPSresponse[2]);
506  // Calculate the new baud rate
507  baudRate = (unsigned long long)((unsigned long long)((unsigned long long)scReferenceClock * factorDNumerator)/(unsigned long)(factorF * (unsigned long)factorDdenominator));
508  // Configure UART for new baud rate
509  SCdrv_SetBRG();
510  SC_CalculateWaitTime();
511  }
512 
513  // Store the PPS response length in global variable
514  scPPSresponseLength = index1;
515 
516  return TRUE;
517  }
518  else
519  return FALSE;
520 }

763  #endif
764

766  if(scTC1 != 0xFF)
767  {
768  cgtETU = 12 + scTC1;
769  }
770

772  switch(scTD1 & 0x0F)

775  case 1 :
776  // Calculate Character Guard Time in ETU's for T=1 Protocol

789  }

793 
794  tempVariable1 = SC_BWI;
795 
796  }
797 
798  for(index = 0;index < tb2Code;index++)
799  tempVariable2 = tempVariable2 * 2;
