57 /****************************************************************************
58  * Function: EthPhyConfigureCustomisedLED
59  *
60  * PreCondition: - Communication to the PHY should have been established.
61  *
62  * Input: void
63  *
64  * Output: ETH_RES_OK - success,
65  * an error code otherwise
66  *
67  *
68  * Side Effects: None
69  *
70  * Overview: This function is used to Customise LED functionality. If ETH_CFG_LINK
71  * set to 1 to config the Speed and Link to a specific parameter and set to 0 for
72  a deafult Speed and Link configuration.
73  *
74  * Note: None
75  *****************************************************************************/
76 eEthRes EthPhyConfigureCustomisedLED(void)
77 {
78  unsigned short LEDEna=0;
79  int ledFlags ;
80  char ethCnfLink = ETH_CFG_LINK;
81 
82  // Customised LED function selection in page 7,Register 19
83  EthMIIMWriteStart(PHY_REG_PAGESEL, PHY_ADDRESS, PAGENUM_7);
84  EthMIIMReadStart(PHY_REG_INT_WOL_LED_SEL, PHY_ADDRESS);
85  LEDEna=EthMIIMReadResult();
86 
87  LEDEna |= _EN_CUSTOMISED_LED; // enabled Customised LED
88  EthMIIMWriteStart(PHY_REG_INT_WOL_LED_SEL, PHY_ADDRESS, LEDEna);
89  EthMIIMWriteStart(PHY_REG_PAGESEL, PHY_ADDRESS, PAGENUM_0);
90 
91  // LED 0 = 10/100MBPS speed and LED1 = SPEED LINK+ACTIVITY Configure for register 17 page 7
92  EthMIIMWriteStart(PHY_REG_PAGESEL, PHY_ADDRESS, PAGENUM_7);
93  EthMIIMReadStart(PHY_REG_CUSTOMISED_LED_SET, PHY_ADDRESS);
94  LEDEna=EthMIIMReadResult();
95  //For LED1 , Speed and Link Activity details -
96  LEDEna |= _EN_LED1_SEL_ACT|_EN_LED1_SEL_100M|_EN_LED1_SEL_10M;
97 
98  // set LED0 Speed and Link activity details
99  if(ethCnfLink == 1)
100  {
101  ledFlags =ETH_CFG_10?ETH_CFG_10:0;
102  ledFlags|=ETH_CFG_100?ETH_CFG_100:0;
103 
104  // if the speed is configured to 10 MBPS
105  if(ledFlags & ETH_CFG_10)
106  {
107  LEDEna |= _EN_LED0_SEL_10M;
108  }
109  // if the speed is configured to 100 MBPS
110  if(ledFlags & ETH_CFG_100)
111  {
112  LEDEna |= _EN_LED0_SEL_100M;
113  }
114  }
115  else
116  {
117  LEDEna |= _EN_LED0_SEL_100M |_EN_LED0_SEL_10M;
118  }
119 
120  EthMIIMWriteStart(PHY_REG_CUSTOMISED_LED_SET, PHY_ADDRESS, LEDEna);
121  EthMIIMWriteStart(PHY_REG_PAGESEL, PHY_ADDRESS, PAGENUM_0);
122 
123  return ETH_RES_OK;
124 }

128 /****************************************************************************
129  * Function: EthPhyConfigureMII
130  *
131  * PreCondition: Communication to the PHY should have been established.
132  *
133  * Input: cFlags - the requested open flags: ETH_PHY_CFG_RMII/ETH_PHY_CFG_MII
134  *
135  * Output: ETH_RES_OK - success,
136  * an error code otherwise
137  *
138  *
139  * Side Effects: None
140  *
141  * Overview: This function configures the PHY only for RMII operation mode. Both LED0 and LED1
142  * are customised with respect to Speed and Activity. The customised LED function
143  * will be called after RMII mode initialization.
144  * Note: None
145  *****************************************************************************/
146 eEthRes EthPhyConfigureMII(eEthPhyCfgFlags cFlags)
147 {
148  unsigned short phyReg=0;
149 
150  if((cFlags&ETH_PHY_CFG_RMII) == 0)
151  {
152  return ETH_RES_CFG_ERR; // MII configuration not supported for now!
153  }
154 
155  // RMII mode setting is in page 7, Register 16 , 3rd bit is used for MII or RMII mode
156  EthMIIMWriteStart(PHY_REG_PAGESEL, PHY_ADDRESS, PAGENUM_7);
157 
158  EthMIIMReadStart(PHY_REG_RMII_MODE_SEL, PHY_ADDRESS);
159  phyReg=EthMIIMReadResult();
160 
161  phyReg|=_RMII_MODE_SELECT_MASK;
162 
163  EthMIIMWriteStart(PHY_REG_RMII_MODE_SEL, PHY_ADDRESS, phyReg); // update the RMII and Bypass Register
164  EthMIIMWriteStart(PHY_REG_PAGESEL, PHY_ADDRESS, PAGENUM_0);
165 
166  EthPhyConfigureCustomisedLED();
167  return ETH_RES_OK;
168 }

184  * Overview: This function configures the MDIX mode for the PHY.
185  *
186  * Note: None
187  *****************************************************************************/
188 eEthRes EthPhyConfigureMdix(eEthOpenFlags oFlags)
189 {
190  unsigned short phyReg;
191 
192  EthMIIMReadStart(PHY_REG_FIBER_MODE_LOOPBACK_AUTO_MDIX, PHY_ADDRESS);
193  phyReg=EthMIIMReadResult();
194 
195  if(oFlags&ETH_OPEN_MDIX_AUTO)
196  { // enable Auto-MDIX
197  phyReg|=_EN_AUTO_MDIX;
198  }
199  else
200  { // no Auto-MDIX
201  phyReg&=~(_EN_AUTO_MDIX); // disable Auto-MDIX
202  if(oFlags&ETH_OPEN_MDIX_SWAP)
203  {
204  // force MDI
205  phyReg|=_FORCED_MDI; // swap FORCED MDI
206  }
207  else
208  {
209  phyReg&=~(_FORCED_MDI); // normal FORCED MDIX
210  }
211 
212  }
213  return ETH_RES_OK;
214 // return (oFlags&ETH_OPEN_MDIX_AUTO)?ETH_RES_OK:ETH_RES_CFG_ERR;
215 }
