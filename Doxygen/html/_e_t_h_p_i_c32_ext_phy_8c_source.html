103  * Function: EthPhyRestartNegotiation
104  *
105  * PreCondition: - EthPhyInit should have been called.
106  * - The PHY should have been initialized with proper duplex/speed mode!
107  *
108  * Input: None
109  *
110  * Output: ETH_RES_OK for success,
111  * ETH_RES_NEGOTIATION_UNABLE if the auto-negotiation is not supported.
112  *
113  * Side Effects: None

125

168 /****************************************************************************
169  * Function: EthPhyInit
170  *
171  * PreCondition: - EthInit should have been called.
172  *
173  * Input: oFlags - the requested open flags
174  * cFlags - PHY MII/RMII configuration flags
175  * pResFlags - address to store the initialization result
176  *
177  * Output: ETH_RES_OK for success,
178  * an error code otherwise
179  *
180  *
181  * Side Effects: None
182  *
183  * Overview: This function initializes the PHY communication.
184  * It tries to detect the external PHY, to read the capabilties and find a match
185  * with the requested features.
186  * Then it programs the PHY accordingly.
187  *
188  * Note: None
189  *****************************************************************************/
190 eEthRes __attribute__((weak)) EthPhyInit(eEthOpenFlags oFlags, eEthPhyCfgFlags cFlags, eEthOpenFlags* pResFlags)
191 {
192  unsigned short ctrlReg;
193  eEthPhyCfgFlags hwFlags, swFlags;
194  unsigned short phyCpbl, openReqs, matchCpbl;
195  eEthRes res;
196 
197  // the way the hw is configured
198  hwFlags=EthPhyGetHwConfigFlags();
199 
200  if(cFlags&ETH_PHY_CFG_AUTO)
201  {
202  cFlags=hwFlags;
203  }
204  else
205  { // some minimal check against the way the hw is configured
206  swFlags=cFlags&(ETH_PHY_CFG_RMII|ETH_PHY_CFG_ALTERNATE);
207 
208  if((swFlags ^ hwFlags)!=0)
209  { // hw-sw configuration mismatch MII/RMII, ALT/DEF config
210  return ETH_RES_CFG_ERR;
211  }
212  }
213 
214 
215  _PhyAdd=EthPhyMIIMAddress(); // get the PHY address
216 
217 
218  if(oFlags&(ETH_OPEN_PHY_LOOPBACK|ETH_OPEN_MAC_LOOPBACK))
219  {
220  oFlags&=~ETH_OPEN_AUTO; // no negotiation in loopback mode!
221  }
222 
223  if(!(oFlags&ETH_OPEN_AUTO))
224  {
225  oFlags&=~ETH_OPEN_MDIX_AUTO; // Auto-MDIX has to be in auto negotiation only
226  }
227 
228  oFlags|=(cFlags&ETH_PHY_CFG_RMII)?ETH_OPEN_RMII:ETH_OPEN_MII;
229 
230  _PhyInitIo(); // init IO pins
231 
232  EthMIIMConfig(GetSystemClock(), EthPhyMIIMClock());
233 
234 
235  // try to detect the PHY and reset it
236  if(!_PhyDetectReset())
237  { // failed to detect the PHY
238  return ETH_RES_DTCT_ERR;
239  }
240 
241  // provide some defaults
242  if(!(oFlags&(ETH_OPEN_FDUPLEX|ETH_OPEN_HDUPLEX)))
243  {
244  oFlags|=ETH_OPEN_HDUPLEX;
245  }
246  if(!(oFlags&(ETH_OPEN_100|ETH_OPEN_10)))
247  {
248  oFlags|=ETH_OPEN_10;
249  }
250 
251  if(oFlags&ETH_OPEN_AUTO)
252  { // advertise auto negotiation
253  openReqs=_BMSTAT_AN_ABLE_MASK;
254 
255  if(oFlags&ETH_OPEN_100)
256  {
257  if(oFlags&ETH_OPEN_FDUPLEX)
258  {
259  openReqs|=_BMSTAT_BASE100TX_FDX_MASK;
260  }
261  if(oFlags&ETH_OPEN_HDUPLEX)
262  {
263  openReqs|=_BMSTAT_BASE100TX_HDX_MASK;
264  }
265  }
266 
267  if(oFlags&ETH_OPEN_10)
268  {
269  if(oFlags&ETH_OPEN_FDUPLEX)
270  {
271  openReqs|=_BMSTAT_BASE10T_FDX_MASK;
272  }
273  if(oFlags&ETH_OPEN_HDUPLEX)
274  {
275  openReqs|=_BMSTAT_BASE10T_HDX_MASK;
276  }
277  }
278  }
279  else
280  { // no auto negotiation
281  if(oFlags&ETH_OPEN_100)
282  {
283  openReqs=(oFlags&ETH_OPEN_FDUPLEX)?_BMSTAT_BASE100TX_FDX_MASK:_BMSTAT_BASE100TX_HDX_MASK;
284  }
285  else
286  {
287  openReqs=(oFlags&ETH_OPEN_FDUPLEX)?_BMSTAT_BASE10T_FDX_MASK:_BMSTAT_BASE10T_HDX_MASK;
288  }
289  }
290 
291  // try to match the oFlags with the PHY capabilities
292  phyCpbl=_PhyReadReg(PHY_REG_BMSTAT, _PhyAdd);
293  matchCpbl=(openReqs&(MAC_COMM_CPBL_MASK|_BMSTAT_AN_ABLE_MASK))&phyCpbl; // common features
294  if(!(matchCpbl&MAC_COMM_CPBL_MASK))
295  { // no match?
296  return ETH_RES_CPBL_ERR;
297  }
298 
299  // we're ok, we can configure the PHY
300  res=EthPhyConfigureMII(cFlags);
301  if(res!=ETH_RES_OK)
302  {
303  return res;
304  }
305 
306  res=EthPhyConfigureMdix(oFlags);
307  if(res!=ETH_RES_OK)
308  {
309  return res;
310  }
311 
312  if(matchCpbl&_BMSTAT_AN_ABLE_MASK)
313  { // ok, we can perform auto negotiation
314  unsigned short anadReg;
315 
316  anadReg=(((matchCpbl>>_BMSTAT_NEGOTIATION_POS)<<_ANAD_NEGOTIATION_POS)&_ANAD_NEGOTIATION_MASK)|PROT_802_3;
317  if(ETH_MAC_PAUSE_CPBL_MASK & ETH_MAC_PAUSE_TYPE_PAUSE)
318  {
319  anadReg|=_ANAD_PAUSE_MASK;
320  }
321  if(ETH_MAC_PAUSE_CPBL_MASK& ETH_MAC_PAUSE_TYPE_ASM_DIR)
322  {
323  anadReg|=_ANAD_ASM_DIR_MASK;
324  }
325 
326  EthMIIMWriteStart(PHY_REG_ANAD, _PhyAdd, anadReg); // advertise our capabilities
327 
328  EthPhyRestartNegotiation(); // restart negotiation and we'll have to wait
329  }
330  else
331  { // ok, just don't use negotiation
332 
333  ctrlReg=0;
334  if(matchCpbl&(_BMSTAT_BASE100TX_HDX_MASK|_BMSTAT_BASE100TX_FDX_MASK)) // set 100Mbps request/capability
335  {
336  ctrlReg|=_BMCON_SPEED_MASK;
337  }
338 
339  if(matchCpbl&(_BMSTAT_BASE10T_FDX_MASK|_BMSTAT_BASE100TX_FDX_MASK))
340  {
341  ctrlReg|=_BMCON_DUPLEX_MASK;
342  }
343 
344  if(oFlags&ETH_OPEN_PHY_LOOPBACK)
345  {
346  ctrlReg|=_BMCON_LOOPBACK_MASK;
347  }
348 
349  EthMIIMWriteStart(PHY_REG_BMCON, _PhyAdd, ctrlReg); // update the configuration
350  }
351 
352 
353  // now update the open flags
354  // the upper layer needs to know the PHY set-up to further set-up the MAC.
355 
356  // clear the capabilities
357  oFlags&=~(ETH_OPEN_AUTO|ETH_OPEN_FDUPLEX|ETH_OPEN_HDUPLEX|ETH_OPEN_100|ETH_OPEN_10);
358 
359  if(matchCpbl&_BMSTAT_AN_ABLE_MASK)
360  {
361  oFlags|=ETH_OPEN_AUTO;
362  }
363  if(matchCpbl&(_BMSTAT_BASE100TX_HDX_MASK|_BMSTAT_BASE100TX_FDX_MASK)) // set 100Mbps request/capability
364  {
365  oFlags|=ETH_OPEN_100;
366  }
367  if(matchCpbl&(_BMSTAT_BASE10T_HDX_MASK|_BMSTAT_BASE10T_FDX_MASK)) // set 10Mbps request/capability
368  {
369  oFlags|=ETH_OPEN_10;
370  }
371  if(matchCpbl&(_BMSTAT_BASE10T_FDX_MASK|_BMSTAT_BASE100TX_FDX_MASK))
372  {
373  oFlags|=ETH_OPEN_FDUPLEX;
374  }
375  if(matchCpbl&(_BMSTAT_BASE10T_HDX_MASK|_BMSTAT_BASE100TX_HDX_MASK))
376  {
377  oFlags|=ETH_OPEN_HDUPLEX;
378  }
379 
380  *pResFlags=oFlags; // upper layer needs to know the PHY set-up to further set-up the MAC.
381 
382  return ETH_RES_OK;
383 
384 }

387 /****************************************************************************
388  * Function: EthPhyNegotiationComplete
389  *
390  * PreCondition: EthPhyInit (and EthPhyRestartNegotiation) should have been called.
391  *
392  * Input: waitComplete - if wait for completion needed
393  *
394  * Output: ETH_RES_OK if negotiation done,
395  * ETH_RES_NEGOTIATION_INACTIVE if no negotiation in progress
396  * ETH_RES_NEGOTIATION_NOT_STARTED if negotiation not started yet (means tmo if waitComplete was requested)
397  * ETH_RES_NEGOTIATION_ACTIVE if negotiation ongoing (means tmo if waitComplete was requested)
398  *
399  * Side Effects: None
400  *
401  * Overview: This function waits for a previously initiated PHY negotiation to complete.
402  * Subsequently, EthPhyGetNegotiationResult() can be called.
403  *
404  * Note: None
405  *****************************************************************************/
406 eEthRes __attribute__((weak)) EthPhyNegotiationComplete(int waitComplete)
407 {
408  __BMCONbits_t phyBMCon;
409  __BMSTATbits_t phyStat;
410  eEthRes res;
411 
412  phyBMCon.w=_PhyReadReg(PHY_REG_BMCON, _PhyAdd);
413  if(phyBMCon.AN_ENABLE)
414  { // just protect from an accidental call
415  phyBMCon.w=_PhyReadReg(PHY_REG_BMCON, _PhyAdd);
416  phyStat.w=_PhyReadReg(PHY_REG_BMSTAT, _PhyAdd);
417 
418  if(waitComplete)
419  {
420  unsigned int tStart, tWait;
421 
422  if(phyBMCon.AN_RESTART)
423  { // not started yet
424  tWait=(GetSystemClock()/2000)*PHY_NEG_INIT_TMO;
425  tStart=ReadCoreTimer();
426  do
427  {
428  phyBMCon.w=_PhyReadReg(PHY_REG_BMCON, _PhyAdd);
429  }while(phyBMCon.AN_RESTART && (ReadCoreTimer()-tStart)<tWait); // wait auto negotiation start
430  }
431 
432  if(!phyBMCon.AN_RESTART)
433  { // ok, started
434  tWait=(GetSystemClock()/2000)*PHY_NEG_DONE_TMO;
435  tStart=ReadCoreTimer();
436  do
437  {
438  phyStat.w=_PhyReadReg(PHY_REG_BMSTAT, _PhyAdd);
439  }while(phyStat.AN_COMPLETE==0 && (ReadCoreTimer()-tStart)<tWait); // wait auto negotiation done
440 
441  phyStat.w=_PhyReadReg(PHY_REG_BMSTAT, _PhyAdd);
442  }
443  }
444  }
445 
446  if(!phyBMCon.AN_ENABLE)
447  {
448  res=ETH_RES_NEGOTIATION_INACTIVE; // no negotiation is taking place!
449  }
450  else if(phyBMCon.AN_RESTART)
451  {
452  res=ETH_RES_NEGOTIATION_NOT_STARTED; // not started yet/tmo
453  }
454  else
455  {
456  res= (phyStat.AN_COMPLETE==0)?ETH_RES_NEGOTIATION_ACTIVE:ETH_RES_OK; // active/tmo/ok
457  }
458 
459  return res;
460 }

464 /****************************************************************************
465  * Function: EthPhyGetNegotiationResult
466  *
467  * PreCondition: EthPhyInit, EthPhyRestartNegotiation and EthPhyNegotiationComplete should have been called.
468  *
469  * Input: pFlags - address to store the negotiation result
470  * pPauseType - address to store the pause type supported by the LP
471  *
472  * Output: the link status after the (completed) negotiation
473  *
474  * Side Effects: None
475  *
476  * Overview: This function returns the result of a previously initiated negotiation.
477  * The result is based on the PHY status!.
478  *
479  * Note: If no negotiation possible/active/failed, most likely the flags are invalid!
480  *****************************************************************************/
481 eEthLinkStat __attribute__((weak)) EthPhyGetNegotiationResult(eEthOpenFlags* pFlags, eEthMacPauseType* pPauseType)
482 {
483  eEthLinkStat linkStat;
484  eEthOpenFlags oFlags;
485  __BMSTATbits_t phyStat;
486  __ANEXPbits_t phyExp;
487  __ANLPADbits_t lpAD;
488  __ANADbits_t anadReg;
489  eEthMacPauseType pauseType;
490 
491 
492  // should have BMCON.AN_ENABLE==1
493  // wait for it to finish!
494 
495 
496  oFlags=0; // don't know the result yet
497  pauseType=ETH_MAC_PAUSE_TYPE_NONE;
498 
499  phyStat.w=_PhyReadReg(PHY_REG_BMSTAT, _PhyAdd);
500  if(phyStat.AN_COMPLETE==0)
501  {
502  linkStat=(ETH_LINK_ST_DOWN|ETH_LINK_ST_NEG_TMO);
503  }
504  else if(!phyStat.LINK_STAT)
505  {
506  linkStat=ETH_LINK_ST_DOWN;
507  }
508  else
509  { // we're up and running
510  int lcl_Pause, lcl_AsmDir, lp_Pause, lp_AsmDir; // pause capabilities, local and LP
511 
512  linkStat=ETH_LINK_ST_UP;
513 
514  lcl_Pause=(ETH_MAC_PAUSE_CPBL_MASK & ETH_MAC_PAUSE_TYPE_PAUSE)?1:0;
515  lcl_AsmDir=(ETH_MAC_PAUSE_CPBL_MASK & ETH_MAC_PAUSE_TYPE_ASM_DIR)?1:0;
516  lp_Pause=lp_AsmDir=0; // in case negotiation fails
517  lpAD.w=_ANAD_BASE10T_MASK; // lowest priority resolution
518 
519  phyExp.w=_PhyReadReg(PHY_REG_ANEXP, _PhyAdd);
520  if(phyExp.LP_AN_ABLE)
521  { // ok,valid auto negotiation info
522 
523  lpAD.w=_PhyReadReg(PHY_REG_ANLPAD, _PhyAdd);
524  if(lpAD.REM_FAULT)
525  {
526  linkStat|=ETH_LINK_ST_REMOTE_FAULT;
527  }
528 
529  if(lpAD.PAUSE)
530  {
531  linkStat|=ETH_LINK_ST_LP_PAUSE;
532  lp_Pause=1;
533  }
534  if(lpAD.ASM_DIR)
535  {
536  linkStat|=ETH_LINK_ST_LP_ASM_DIR;
537  lp_AsmDir=1;
538  }
539 
540  }
541  else
542  {
543  linkStat|=ETH_LINK_ST_LP_NEG_UNABLE;
544  if(phyExp.PDF)
545  {
546  linkStat|=ETH_LINK_ST_PDF;
547  }
548  }
549 
550  // set the PHY connection params
551 
552  anadReg.w=_PhyReadReg(PHY_REG_ANAD, _PhyAdd); // get our advertised capabilities
553  anadReg.w&=lpAD.w; // get the matching ones
554  // get the settings, according to IEEE 802.3 Annex 28B.3 Priority Resolution
555  // Note: we don't support 100BaseT4 !
556 
557  if(anadReg.w&_ANAD_BASE100TX_FDX_MASK)
558  {
559  oFlags=(ETH_OPEN_100|ETH_OPEN_FDUPLEX);
560  }
561  else if(anadReg.w&_ANAD_BASE100TX_MASK)
562  {
563  oFlags=(ETH_OPEN_100|ETH_OPEN_HDUPLEX);
564  }
565  else if(anadReg.w&_ANAD_BASE10T_FDX_MASK)
566  {
567  oFlags=(ETH_OPEN_10|ETH_OPEN_FDUPLEX);
568  }
569  else if(anadReg.w&_ANAD_BASE10T_MASK)
570  {
571  oFlags=(ETH_OPEN_10|ETH_OPEN_HDUPLEX);
572  }
573  else
574  { // this should NOT happen!
575  linkStat|=ETH_LINK_ST_NEG_FATAL_ERR;
576  linkStat&=~ETH_LINK_ST_UP; // make sure we stop...!
577  }
578 
579 
580  // set the pause type for the MAC
581  // according to IEEE Std 802.3-2002 Tables 28B-2, 28B-3
582  if(oFlags&ETH_OPEN_FDUPLEX)
583  { // pause type relevant for full duplex only
584  if(lp_Pause & (lcl_Pause|(lcl_AsmDir&lp_AsmDir)))
585  {
586  pauseType=ETH_MAC_PAUSE_TYPE_EN_TX;
587  }
588  if(lcl_Pause & (lp_Pause | (lcl_AsmDir&lp_AsmDir)))
589  {
590  pauseType|=ETH_MAC_PAUSE_TYPE_EN_RX;
591  }
592  }
593  }
594 
595  if(pFlags)
596  {
597  *pFlags=oFlags;
598  }
599 
600  if(pPauseType)
601  {
602  *pPauseType=pauseType;
603  }
604  return linkStat;
605 }

639 /****************************************************************************
640  * Function: EthPhyReset
641  *
642  * PreCondition: EthPhyInit() should have been called
643  * Communication with the PHY already established
644  *
645  * Input: waitComplete - if TRUE the procedure will wait for reset to complete
646  *
647  * Output: TRUE if the reset procedure completed (or completion not required)
648  * FALSE otherwise
649  *
650  * Side Effects: None
651  *
652  * Overview: This function immediately resets the PHY.
653  * It does not wait for the reset procedure to complete
654  *
655  * Note: None
656  *****************************************************************************/
657 int __attribute__((weak)) EthPhyReset(int waitComplete)
658 {
659 
660  EthMIIMWriteStart(PHY_REG_BMCON, _PhyAdd, _BMCON_RESET_MASK); // Soft Reset the PHY
661 
662  if(waitComplete)
663  { // wait reset self clear
664  __BMCONbits_t bmcon;
665  unsigned int tStart, tWaitReset;
666 
667  tWaitReset=(GetSystemClock()/2000)*PHY_RESET_CLR_TMO;
668  tStart=ReadCoreTimer();
669  do
670  {
671  bmcon.w=_PhyReadReg(PHY_REG_BMCON, _PhyAdd);
672  }while(bmcon.RESET && (ReadCoreTimer()-tStart)<tWaitReset);
673

856 /****************************************************************************
857  * Function: _PhyDetectReset
858  *
859  * PreCondition: EthMIIMConfig() should have been called
860  *
861  * Input: None
862  *
863  * Output: TRUE if the detection and the reset of the PHY succeeded,
864  * FALSE if no PHY detected
865  *
866  * Side Effects: None
867  *
868  * Overview: This function detects and resets the PHY.
869  *
870  * Note: Needs the system running frequency to for the PHY detection
871  *****************************************************************************/
872 static int _PhyDetectReset(void)
873 {
874  __BMCONbits_t bmcon;
875  unsigned int tStart, tWaitReset;
876 
877  bmcon.w=_PhyReadReg(PHY_REG_BMCON, _PhyAdd); // read the BMCON register
878 
879  if(bmcon.RESET)
880  { // that is already suspicios...but give it a chance to clear itself
881  tWaitReset=(GetSystemClock()/2000)*PHY_RESET_CLR_TMO;
882  tStart=ReadCoreTimer();
883  do
884  {
885  bmcon.w=_PhyReadReg(PHY_REG_BMCON, _PhyAdd);
886  }while(bmcon.RESET && (ReadCoreTimer()-tStart)<tWaitReset); // wait reset self clear
887 
888  bmcon.w=_PhyReadReg(PHY_REG_BMCON, _PhyAdd);
889  if(bmcon.RESET)
890  { // tmo clearing the reset
891  return 0;
892  }
893  }
894 
895  // ok, reset bit is low
896  // try to see if we can write smth to the PHY
897  // we use Loopback and Isolate bits
898  EthMIIMWriteStart(PHY_REG_BMCON, _PhyAdd, _BMCON_LOOPBACK_MASK|_BMCON_ISOLATE_MASK); // write control bits
899  bmcon.w=_PhyReadReg(PHY_REG_BMCON, _PhyAdd); // read back
900  if(bmcon.LOOPBACK==0 || bmcon.ISOLATE==0)
901  { // failed to set
902  return 0;
903  }
904  bmcon.w^=_BMCON_LOOPBACK_MASK|_BMCON_ISOLATE_MASK;
905  EthMIIMWriteStart(PHY_REG_BMCON, _PhyAdd, bmcon.w); // clear bits and write
906  bmcon.w=_PhyReadReg(PHY_REG_BMCON, _PhyAdd); // read back
907  if(bmcon.LOOPBACK || bmcon.ISOLATE)
908  { // failed to clear
909  return 0;
910  }
911 
912  // everything seems to be fine
913  //
914  return EthPhyReset(1);
915 }
