86 void WaitMicroSec( unsigned long int microSec )
87 {
88  // Set the Timer Count as per the delay needed
89  SCdrv_SetDelayTimerCnt(0xFFFF - microSec/TIMER1_SINGLE_COUNT_MICRO_SECONDS);
90 
91  // Enable the delay timer
92  SCdrv_EnableDelayTimer();
93 
94  // Wait until the delay is elapsed
95  while(!delayLapsedFlag);

150 void SCdrv_SendTxData( BYTE data )
151 {
152  BYTE txRetryCounter;
153 
154  // Try to trasmit the byte for 5 times before giving up...
155  for(txRetryCounter = 0;txRetryCounter < 5;txRetryCounter++)
156  {
157  U1STAbits.UTXEN = 1;
158  U1TXREG = data;
159 
160  // check for transmit empty flag
161  while( !U1STAbits.TRMT )
162  Nop();
163 
164  U1STAbits.UTXEN = 0;
165 
166  // Disable UART Module
167  U1MODEbits.UARTEN = 0;
168 
169 // WaitMicroSec( oneETUtimeinMicroSeconds/2 ); // 0.5 Bit time delay
170 
171  // Receiver pulls Tx line low, to indicate PE or FR errors. If Such a
172  // scenario occurs then try to retransmit the byte(maximum of 5 trys allowed).
173  if(SCdrv_GetRxPinData())
174  break;
175  else
176  {
177  WaitMicroSec(oneETUtimeinMicroSeconds); //wait one etu before repeating
178 
179  U1MODEbits.UARTEN = 1;
180  }
181  }
182 
183  U1MODEbits.UARTEN = 1; // Enable UART Module
184 
185  U1STAbits.OERR = 0; //clear any overflow error that we caused
186 
187  while(1) // remove rx data recvd from our Tx line
188  {
189  WORD temp;
190  if( U1STAbits.URXDA )
191  temp = U1RXREG;
192  else
193  break;
194  }
195 }

219 BOOL SCdrv_GetRxData( BYTE* pDat,unsigned short int timeOutETUs )
220 {
221  unsigned long int tmrCntDifference = timeOutETUs * rxETUtimeDivisorFactor;
222  unsigned short int timerCountValue;
223  BYTE rxRetryCounter;
224 
225  delayLapsedFlag = TRUE;
226 
227  // If any error occured while recieving the byte, give chance to smart card
228  // to send the corrected byte within 5 attempts
229  for(rxRetryCounter = 0;rxRetryCounter < 5;rxRetryCounter++)
230  {
231  //wait for data byte
232  while(!U1STAbits.URXDA)
233  {
234  if( delayLapsedFlag )
235  {
236  delayLapsedFlag = FALSE;
237 
238  if(tmrCntDifference)
239  {
240  if(tmrCntDifference > 0xFFFF)
241  {
242  timerCountValue = 0;
243  tmrCntDifference = tmrCntDifference - 0xFFFF;
244  }
245  else
246  {
247  timerCountValue = (unsigned short int)tmrCntDifference;
248  tmrCntDifference = 0;
249  }
250 
251  // Set the Timer Count as per the delay needed
252  SCdrv_SetDelayTimerCnt(0xFFFF - timerCountValue);
253 
254  // Enable the delay timer
255  SCdrv_EnableDelayTimer();
256  }
257  else
258  return FALSE;
259  }
260  }
261 
262  // Disable Timer if it is still running
263  SCdrv_DisableDelayTimer();
264 
265  delayLapsedFlag = FALSE;
266 
267  if( U1STAbits.PERR ) //Parity Error detected
268  {
269  SCdrv_TxPin_Direction(0); //pull it low to tell the card that there was error receiving data
270 
271  #if defined(__PIC24F__)
272  U1MODEbits.RXINV = 1; //do not recognize this low state as a valid start bit
273  #elif defined(__PIC24H__)
274  U1MODEbits.URXINV = 1; //do not recognize this low state as a valid start bit
275  #endif
276 
277  //Read the data from UART to clear the error flag
278  *pDat = U1RXREG;
279 
280  WaitMicroSec( oneETUtimeinMicroSeconds); // 1 Bit time delay
281 
282  SCdrv_TxPin_Direction(1); //release RD10. Card should retransmit now.
283 
284  #if defined(__PIC24F__)
285  U1MODEbits.RXINV = 0;
286  #elif defined(__PIC24H__)
287  U1MODEbits.URXINV = 0;
288  #endif
289  }
290  else
291  {
292  //Read the data from UART
293  *pDat = U1RXREG;
294  return TRUE;
295  }
296  }
297 
298  return FALSE;
299 }

380 void SCdrv_InitUART(void)
381 {
382  unsigned int power2Value = 1;
383  BYTE power2temp;
384 
385  #ifdef ENABLE_SC_POWER_THROUGH_PORT_PIN
386  SCdrv_PowerPin_Direction(0); //set RG8 as output to power the Smart Card
387  #endif
388 
389  SCdrv_ResetPin_Direction(0); //set RD0 as output for Smart Card Reset Pin
390  SCdrv_CardPresent_Direction(1); //RD3 Input Card Present - SmartCard Conn Active Hi
391  SCdrv_SimPresent_Direction(1); //RG12 Input Card Present - SimCard Conn Active Low
392 
393  SCdrv_SetTxPinData(0);
394  SCdrv_TxPin_Direction(1); // use as gpio to pull the line low
395 
396  //Turn on the pull-up on both RX and TX line for faster transitions.
397  SCdrv_EnableTxPinPullUp();
398  SCdrv_EnableRxPinPullUp();
399 
400  // Enable Pull-ups for card present/sim present pins
401  SCdrv_EnableCardPresentPinPullUp();
402  SCdrv_EnableSimPresentPinPullUp();
403 
404  #ifdef ENABLE_SC_POWER_THROUGH_PORT_PIN
405  SCdrv_SetSwitchCardPower(1); //Turn on power to smart card
406  #endif
407 
408  SCdrv_SetSwitchCardReset(0); //keep card in reset state
409 
410  MapUART1RxPin(); // Map UART1 Rx pin
411  MapUART1TxPin(); // Map UART1 Tx pin
412 
413  Scdrv_ClockSet();
414 
415  // Initial Baud Rate of Smart Card for external Clock
416  power2temp = REF_CLOCK_POWER2_VALUE;
417 
418  while(power2temp--)
419  {
420  power2Value = power2Value * (BYTE)2;
421  }
422 
423  scReferenceClock = REF_CLOCK_CIRCUIT_INPUT_CLK/(power2Value + REF_CLOCK_DIVISOR_VALUE);
424 
425  baudRate = scReferenceClock/372;
426 
427  // Calculate the recieve ETU divisor which will be used to calculate the timeouts
428  // while recieving the data bytes. This calculation is done here because baudrate
429  // of ISO 7816 communication is calculated here.
430  oneETUtimeinMicroSeconds = (unsigned short int)(1000000UL/baudRate);
431  rxETUtimeDivisorFactor = oneETUtimeinMicroSeconds/TIMER1_SINGLE_COUNT_MICRO_SECONDS;
432 
433  U1BRG = (unsigned int)((unsigned long)((unsigned long)(FCY/4)/baudRate) - 1);
434 
435  U1MODEbits.PDSEL = 1; //8bits + even parity
436  U1MODEbits.STSEL = 0; //1 stop bit
437  U1MODEbits.BRGH = 1;
438 
439  // Enable Delay Timer Interrupts for future use
440  SCdrv_EnableDelayTimerIntr();
441 }
