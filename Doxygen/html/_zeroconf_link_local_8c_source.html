340 /***************************************************************
341  Function:
342  ARP_PKT_TYPE FindARPPktType (DWORD SrcIPAddr, DWORD DestIPAddr,
343  BYTE op_req)
344 
345  Summary:
346  Finds the Type of ARP-Packet based on the Source IP-address,
347  Destination IP-address and operation-request.
348 
349  Description:
350  This function is used to find out the ARP-packet type. When ARP
351  module passes up a ARP-packet to Zeroconf Link-Local module, it
352  parses contents and finds the packet-type (like ARP-Probe, ARP-
353  Claim, ARP-Defend, or generic ARP-request/response)
354 
355  Parameters:
356  SrcIPAddr - Source IP-Address
357  DestIPAddr - Destination IP-Address
358  op_req - Operation-Request (ARP-Request/Response)
359 
360  Returns:
361  ARP_PKT_TYPE - Type of ARP-Packet (Probe, Claim, Defend or
362  generic ARP-request/response)
363  ***************************************************************/
364 ARP_PKT_TYPE FindARPPktType (DWORD SrcIPAddr, DWORD DestIPAddr,BYTE op_req)
365 {
366  if(op_req == ARP_REQ)
367  {
368  if(SrcIPAddr == 0x0)
369  return ARP_PROBE_TYPE;
370  else if (SrcIPAddr == DestIPAddr)
371  return ARP_CLAIM_TYPE;
372  else
373  return ARP_REQUEST_TYPE;
374 
375  }
376 
377  else if(op_req == ARP_RESP)
378  {
379  if(SrcIPAddr == DestIPAddr)
380  return ARP_DEFEND_TYPE;
381  else
382  return ARP_RESPONSE_TYPE;
383  }
384 
385  else
386  return UNKNOWN_TYPE;
387 }

389 /***************************************************************
390  Function:
391  void ZeroconfARPPktNotify (DWORD SenderIPAddr, DWORD TargetIPAddr,
392  MAC_ADDR* SenderMACAddr,
393  MAC_ADDR* TargetMACAddr, BYTE op_req)
394 
395  Summary:
396  Callback registered with ARP-Module. This gets invoked from ARP-
397  module and runs in the same context.
398 
399  Description:
400  This function is registered as a callback with ARP-module to get
401  notified about incoming Packet-events. Based on the type of packet
402  received and Link-Local current state, appropriate action will be
403  taken. To find the type of ARP-Packet this function makes use of
404  FindARPPktType routine.
405 
406  Primary purpose of this function is to decipher the ARP-Packet rxed
407  and check whether its leading to a conflict with the selected IP-
408  address.
409 
410  Two types of conflicts are defined: Probe-Conflict and Late-Conflict
411  If the current state of Link-Local is Probe/Claim and a conflict is
412  detected its called "Probe-Conflict"
413  If the current state of Link-Local is Defend-state and a conflict is
414  detected its called "Late-Conflict"
415 
416  Parameters:
417  SenderIPAddr - Sender IP-Address
418  TargetIPAddr - Target IP-Address
419  SenderMACAddr - Sender MAC-Address
420  TargetMACAddr - Target MAC-Address
421  op_req - Operation-Request (ARP-Request/Response)
422 
423  Returns:
424  None
425  ***************************************************************/
426 void ZeroconfARPPktNotify (DWORD SenderIPAddr, DWORD TargetIPAddr,
427  MAC_ADDR* SenderMACAddr,
428  MAC_ADDR* TargetMACAddr, BYTE op_req)
429 {
430  ARP_PKT_TYPE pkt_type;
431 
432  pkt_type = FindARPPktType (SenderIPAddr, TargetIPAddr, op_req);
433 
434  if(pkt_type == UNKNOWN_TYPE)
435  return; // Can't hit this
436 
437  switch (zcll_state)
438  {
439  case SM_ADDR_PROBE:
440  case SM_ADDR_CLAIM:
441  {
442  switch(pkt_type)
443  {
444  case ARP_PROBE_TYPE:
445  case ARP_CLAIM_TYPE:
446  case ARP_DEFEND_TYPE:
447  if(temp_IP_addr.Val == TargetIPAddr ) // Probe-Conflict
448  {
449  if(memcmp(SenderMACAddr, &AppConfig.MyMACAddr, 6))
450  {
451  DEBUG_ZCLL_PRINT("ARPPktNotify: Somebody has initiated " \
452  "the procedure to use same Addr.\r\n");
453 
454  zcll_flags.bits.probe_conflict =1;
455  }
456  }
457  break;
458 
459  case ARP_RESPONSE_TYPE:
460  /* Some-body has been using probed addr
461  * We need to choose different Address */
462  if(temp_IP_addr.Val == SenderIPAddr)
463  {
464  DEBUG_ZCLL_PRINT("ARPPktNotify: Some-body has responded for " \
465  "probed IP-addr \r\n");
466 
467  zcll_flags.bits.probe_conflict =1;
468  }
469  break;
470  default:
471  break;
472  }
473  }
474  break;
475 
476  case SM_ADDR_DEFEND:
477  {
478  if(AppConfig.MyIPAddr.Val == SenderIPAddr)
479  {
480  if(memcmp(SenderMACAddr, &AppConfig.MyMACAddr, 6))
481  {
482  DEBUG_ZCLL_PRINT("ARPPktNotify: Some-one has started using" \
483  " our IP-address \r\n");
484 
485  zcll_flags.bits.late_conflict = 1;
486  }
487  }
488  }
489  break;
490 
491  default:
492  break; // Nothing to do in other states
493  }
494 
495 }
