83 /*******************************************************************************
84  Function:
85  UINT16 WF_Scan(UINT8 CpId)
86 
87  Summary:
88  Commands the MRF24W to start a scan operation. This will generate the
89  WF_EVENT_SCAN_RESULTS_READY event.
90 
91  Description:
92  Directs the MRF24W to initiate a scan operation utilizing the input
93  Connection Profile ID. The Host Application will be notified that the scan
94  results are ready when it receives the WF_EVENT_SCAN_RESULTS_READY event.
95  The eventInfo field for this event will contain the number of scan results.
96  Once the scan results are ready they can be retrieved with
97  WF_ScanGetResult().
98 
99  Scan results are retained on the MRF24W until:
100  1. Calling WF_Scan() again (after scan results returned from previous
101  call).
102  2. MRF24W reset.
103 
104  MRF24WB0M & MRF24WG0M support up to max of 60 scan results (SSIDs).
105 
106  Precondition:
107  MACInit must be called first.
108 
109  Parameters:
110  CpId - Connection Profile to use.
111  If the CpId is valid then the values from that Connection Profile
112  will be used for filtering scan results. If the CpId is set to
113  WF_SCAN_ALL (0xFF) then a default filter will be used.
114 
115  Valid CpId
116  * If CP has a defined SSID only scan results with that SSID are
117  retained.
118  * If CP does not have a defined SSID then all scanned SSID’s will be
119  retained
120  * Only scan results from Infrastructure or AdHoc networks are
121  retained, depending on the value of networkType in the Connection Profile
122  * The channel list that is scanned will be determined from
123  channelList in the Connection Algorithm (which must be defined
124  before calling this function).
125 
126  CpId is equal to WF_SCAN_ALL
127  * All scan results are retained (both Infrastructure and Ad Hoc
128  networks).
129  * All channels within the MRF24W’s regional domain will be
130  scanned.
131  * No Connection Profiles need to be defined before calling this
132  function.
133  * The Connection Algorithm does not need to be defined before
134  calling this function.
135 
136  Returns:
137  Operation results. Success or Failure
138 
139  Remarks:
140  Host scan is allowed only in idle or connected state.
141  If MRF24W FW is in the midst of connection ( or reconnection) process, then
142  host scan can hammer connection process, and furthermore it may cause
143  fatal failure in MRF24W FW operation. To use host scan, we strongly
144  recommend the user to disable MRF24W FW connection manager by enabling
145  #define DISABLE_MODULE_FW_CONNECT_MANAGER_IN_INFRASTRUCTURE
146  in WF_Config.h
147  *****************************************************************************/
148 UINT16 WF_Scan(UINT8 CpId)
149 {
150  UINT8 hdr[4];
151 #ifndef MRF24WG
152  UINT8 connectionState;
153  UINT8 dummy;
154 #endif
155 
156  /* WARNING !!! :
157  * Host scan is allowed only in idle or connected state.
158  * If module FW is in the midst of connection ( or reconenction) process, then
159  * host scan can hammer connection process, and furthermore it may cause
160  * fatal failure in module FW operation. To be safte to use host scan, we strongly
161  * recommend you to disable module FW connection manager by uncommenting
162  * #define DISABLE_MODULE_FW_CONNECT_MANAGER_IN_INFRASTRUCTURE
163  * in WF_Config.h
164  */
165  if (!WF_CMIsHostScanAllowed())
166  return WF_ERROR_OPERATION_CANCELLED;
167 
168 #ifndef MRF24WG
169  WF_CMGetConnectionState(&connectionState, &dummy);
170  if (connectionState == WF_CSTATE_NOT_CONNECTED)
171  WF_CMConnect(0xff); /* 120c host scan bug workaround */
172 #endif
173 
174  hdr[0] = WF_MGMT_REQUEST_TYPE;
175  hdr[1] = WF_SCAN_START_SUBTYPE;
176  hdr[2] = CpId; /* Connection Profile ID */
177  hdr[3] = 0; /* not used */
178 
179  SendMgmtMsg(hdr, /* header */
180  sizeof(hdr), /* size of header */
181  NULL, /* no data */
182  0); /* no data */
183 
184  /* wait for mgmt response, free it after it comes in (no data needed) */
185  WaitForMgmtResponse(WF_SCAN_START_SUBTYPE, FREE_MGMT_BUFFER);
186 
187  return WF_SUCCESS;
188 }

190 /*******************************************************************************
191  Function:
192  void WF_ScanGetResult(UINT8 listIndex, tWFScanResult *p_scanResult)
193 
194  Summary:
195  Read scan results back from MRF24W.
196 
197  Description:
198  After a scan has completed this function is used to read one or more of the
199  scan results from the MRF24W. The scan results will be written
200  contiguously starting at p_scanResult (see tWFScanResult structure for
201  format of scan result).
202  MRF24WB0M & MRF24WG0M support up to max of 60 scan results (SSIDs).
203 
204  Precondition:
205  MACInit must be called first. WF_EVENT_SCAN_RESULTS_READY event must have
206  already occurrerd.
207 
208  Parameters:
209  listIndex - Index (0-based list) of the scan entry to retrieve.
210  p_scanResult - Pointer to location to store the scan result structure
211 
212  Returns:
213  None.
214 
215  Remarks:
216  RSSI can only be obtained from the scan results p_scanResult->rssi.
217  MRF24W checks out the signal strength from the preamble of the incoming packets.
218  The higher the values, the stronger is the received signal strength.
219 
220  p_scanResult->rssi contains received signal strength indicator (RSSI).
221  * MRF24WB : RSSI_MAX (200) , RSSI_MIN (106).
222  * MRF24WG : RSSI_MAX (128) , RSSI_MIN (43).
223 
224  The RSSI value is not directly translated to dbm because this is not calibrated number.
225  However, as a guideline, MAX(200) corresponds to 0 dbm, MIN (106) corresponds to -94 dbm.
226 
227  *****************************************************************************/
