139 /*********************************************************************
140 * Function: WORD PictDraw(void *pObj)
141 *
142 * Output: returns the status of the drawing
143 * 0 - not completed
144 * 1 - done
145 *
146 * Overview: draws picture
147 *
148 ********************************************************************/
149 WORD PictDraw(void *pObj)
150 {
151  typedef enum
152  {
153  REMOVE,
154  DRAW_IMAGE,
155  DRAW_BACKGROUND1,
156  DRAW_BACKGROUND2,
157  DRAW_BACKGROUND3,
158  DRAW_BACKGROUND4,
159  DRAW_FRAME
160  } PICT_DRAW_STATES;
161 
162  static PICT_DRAW_STATES state = REMOVE;
163  static SHORT posleft;
164  static SHORT postop;
165  static SHORT posright;
166  static SHORT posbottom;
167  PICTURE *pPict;
168 
169  pPict = (PICTURE *)pObj;
170 
171  if(IsDeviceBusy())
172  return (0);
173 
174  switch(state)
175  {
176  case REMOVE:
177  if(GetState(pPict, PICT_HIDE))
178  {
179  SetColor(pPict->hdr.pGolScheme->CommonBkColor);
180  if(!Bar(pPict->hdr.left, pPict->hdr.top, pPict->hdr.right, pPict->hdr.bottom))
181  return (0);
182  return (1);
183  }
184 
185  posleft = (pPict->hdr.left + 1 + pPict->hdr.right - pPict->scale * GetImageWidth(pPict->pBitmap)) >> 1;
186  postop = (pPict->hdr.top + 1 + pPict->hdr.bottom - pPict->scale * GetImageHeight(pPict->pBitmap)) >> 1;
187  posright = (pPict->hdr.right + pPict->hdr.left + pPict->scale * GetImageWidth(pPict->pBitmap)) >> 1;
188  posbottom = (pPict->hdr.bottom + pPict->hdr.top + pPict->scale * GetImageHeight(pPict->pBitmap)) >> 1;
189  state = DRAW_IMAGE;
190 
191  case DRAW_IMAGE:
192  if(pPict->pBitmap != NULL)
193  {
194  if(IsDeviceBusy())
195  return (0);
196  if(!PutImagePartial(posleft, postop, pPict->pBitmap, pPict->scale, pPict->partial.xoffset,pPict->partial.yoffset,pPict->partial.width,pPict->partial.height))
197  return (0);
198  }
199 
200  SetColor(pPict->hdr.pGolScheme->CommonBkColor);
201  state = DRAW_BACKGROUND1;
202 
203  case DRAW_BACKGROUND1:
204  if(!Bar(pPict->hdr.left + 1, pPict->hdr.top + 1, pPict->hdr.right - 1, postop - 1))
205  return (0);
206  state = DRAW_BACKGROUND2;
207 
208  case DRAW_BACKGROUND2:
209  if(!Bar(pPict->hdr.left + 1, posbottom, pPict->hdr.right - 1, pPict->hdr.bottom - 1))
210  return (0);
211  state = DRAW_BACKGROUND3;
212 
213  case DRAW_BACKGROUND3:
214  if(!Bar(pPict->hdr.left + 1, postop, posleft - 1, posbottom))
215  return (0);
216  state = DRAW_BACKGROUND4;
217 
218  case DRAW_BACKGROUND4:
219  if(!Bar(posright, postop, pPict->hdr.right - 1, posbottom))
220  return (0);
221  state = DRAW_FRAME;
222 
223  case DRAW_FRAME:
224  if(GetState(pPict, PICT_FRAME))
225  {
226  SetLineType(SOLID_LINE);
227  SetColor(pPict->hdr.pGolScheme->TextColor0);
228  if(!Rectangle(pPict->hdr.left, pPict->hdr.top, pPict->hdr.right, pPict->hdr.bottom))
229  return (0);
230  }
231 
232  state = REMOVE;
233  return (1);
234  }
235 
236  return (1);
237 }
