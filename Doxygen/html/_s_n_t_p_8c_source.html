370 /*****************************************************************************
371  Function:
372  DWORD SNTPGetUTCSeconds(void)
373 
374  Summary:
375  Obtains the current time from the SNTP module.
376 
377  Description:
378  This function obtains the current time as reported by the SNTP module.
379  Use this value for absolute time stamping. The value returned is (by
380  default) the number of seconds since 01-Jan-1970 00:00:00.
381 
382  Precondition:
383  None
384 
385  Parameters:
386  None
387 
388  Returns:
389  The number of seconds since the Epoch. (Default 01-Jan-1970 00:00:00)
390 
391  Remarks:
392  Do not use this function for time difference measurements. The Tick
393  module is more appropriate for those requirements.
394  ***************************************************************************/
395 DWORD SNTPGetUTCSeconds(void)
396 {
397  DWORD dwTickDelta;
398  DWORD dwTick;
399 
400  // Update the dwSNTPSeconds variable with the number of seconds
401  // that has elapsed
402  dwTick = TickGet();
403  dwTickDelta = dwTick - dwLastUpdateTick;
404  while(dwTickDelta > TICK_SECOND)
405  {
406  dwSNTPSeconds++;
407  dwTickDelta -= TICK_SECOND;
408  }
409 
410  // Save the tick and residual fractional seconds for the next call
411  dwLastUpdateTick = dwTick - dwTickDelta;
412 
413  return dwSNTPSeconds;
414 }
