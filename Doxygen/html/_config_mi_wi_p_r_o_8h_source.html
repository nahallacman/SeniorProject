63  // ENABLE_MIWI_PRO_ACKNOWLEDGEMENT enables the MiWi PRO stack to
64  // send back an acknowledgement packet when a MiWi PRO data packet
65  // for application is received. This process will be handled
66  // automatically in the stack without application layer involvement.
67  // MiWi PRO acknowledgement is ack in network layer. It may be sent
68  // across multiple hops and should be differiented with MAC ack.
69  // When message is sent to a sleeping device, the MiWi PRO
70  // acknowledgemnt will be received later and notified by call
71  // back function MiApp_CB_RFDAcknowledgement. Otherwise, whether
72  // receive MiWi PRO acknowledgement will be used to notify application
73  // layer if transmission is successful.
74  /*********************************************************************/
75  //#define ENABLE_MIWI_PRO_ACKNOWLEDGEMENT

107  /*********************************************************************/
108  // For a sleeping device, when establishing an indirect connection
109  // (socket), it may not be desirable to poll the data at the normal
110  // interval, which can be longer than OPEN_SOCKET_TIMEOUT, the

133  /*********************************************************************/
134  // When broadcasting to a sleeping device is enabled, it is hard for
135  // a parent node to track which end device has received the broadcast
136  // message. However, if no tracking is provided, the end device may
137  // receive the same broadcast multiple times. MiWi PRO solves this problem
138  // by tracking the broadcast message on sleeping device side.
139  // INDIRECT_MESSAGE_TIMEOUT_CYCLE defines the total number of messages
140  // receives before the broadcast record times out. It is hard for a
141  // sleeping node to track timing, so tracking the number of message
142  // received is a simpler way.
143  /*********************************************************************/
144  #define INDIRECT_MESSAGE_TIMEOUT_CYCLE 2

219  // device. Once Time Synchronization feature is enabled, following

226  /*********************************************************************/
227  // TIME_SYNC_SLOTS defines the total number of time slot available
228  // within one duty cycle. As a rule, the number of time slot must be
229  // equal or higher than the total number of sleeping devices that are
230  // connected to the FFD, so that each sleeping device can be assigned
231  // to a time slot. The time slot period is calcualted by following
232  // formula:
233  // Time Slot Period = RFD_WAKEUP_INTERVAL / TIME_SYNC_SLOTS
234  // The length of time slot period depends on the primary oscillator
235  // accuracy on the FFD as well as the 32KHz crystal accuracy on sleeping
236  // devices.
237  // The definition of TIME_SYNC_SLOTS is only valid if ENABLE_TIME_SYNC
238  // is defined
239  /*********************************************************************/
240  #define TIME_SYNC_SLOTS 10
241 
242  /*********************************************************************/

245  // when MCU is in sleep.

273  /*********************************************************************/
274  // COMM_RSSI_THRESHOLD defines the minimum signal strength that is
275  // acceptable for a Coordinator to route messages through. A neighbor
276  // Coordinator that has signal strength that is higher than this
277  // threshold can be list as neighbor to be able to route message.
278  // This setting usually is related to the signal strength
279  // interpretation for the specific RF transceiver. This value is
280  // represented by BYTE with 8-bit data width, thus should be less
281  // than 256.
282  /*********************************************************************/

285  /*********************************************************************/
286  // RANDOM_DELAY_RANGE defines random delay range in milliseconds.
287  // When rebroadcast a message, it is recommended that a Coordinator
288  // should introduce a random delay to avoid multiple Coordinators
289  // rebroadcast and collide the messages at the same time. The actual
290  // random delay will be randomly selected between 0 and
291  // RANDOM_DELAY_RANGE. This setting is recommended to be set
292  // higher than 20. The value is represented by BYTE with 8-bit data
293  // width, thus should be less than 256.
294  /*********************************************************************/
295  #define RANDOM_DELAY_RANGE 200
