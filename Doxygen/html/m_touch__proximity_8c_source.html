163  while (MTOUCH_PROX_time[i] != MTOUCH_PROX_currentTime) // As long as we have not hit the time index that
164  { // will be replaced in the array, continue.
165  if (MTOUCH_PROX_value[i] > newValue) // If the value in the sorted array is greater
166  { // than the new value, drop the new value here
167  uint16_t tempValue = MTOUCH_PROX_value[i]; // and pick up the value in the array.
168  uint8_t tempTime = MTOUCH_PROX_time[i];

171  MTOUCH_PROX_time [i] = newTime;
172 
173  newValue = tempValue; // Now we start shifting these values up through
174  newTime = tempTime; // the sorted list until we hit the time index
175  } // that is to be replaced.
176  i++;
177  }

180  MTOUCH_PROX_time[i] = newTime; // and place the held value there.
181 
182  for (i++; i < MTOUCH_PROX_BUFFER_SIZE; i++) // Now, loop through the values larger than
183  { // the one we just removed.
184  if (newValue > MTOUCH_PROX_value[i]) // If the value in the sorted array is less
185  { // than the new value, push the value in the
186  MTOUCH_PROX_value[i-1] = MTOUCH_PROX_value[i]; // array down one location, and move the new
187  MTOUCH_PROX_time [i-1] = MTOUCH_PROX_time[i]; // value up.
188 
189  MTOUCH_PROX_value[i] = newValue;
190  MTOUCH_PROX_time[i] = newTime;
191  }
192  else break; // If the value in the sorted array is now
193  } // greater than the new value, stop.

406  //
