344  public String GetProtocol()
345  {
346  // For images being built, base on the output version
347  if (radWebDir.isSelected())
348  {
349  if ((outPutVersion == MPFS_OUTPUT_VERSION.MPFS2)&&
350  (uploadProtocol == UPLOAD_PROTOCOL_OPTION.HTTP))
351  return "http";
352  else
353  return "ftp";
354  }
355  // For images being uploaded from disk, try to read the file header
356  // If file does not exist, default to http
357  else
358  {
359  try
360  {
361  FileInputStream bin = new FileInputStream(projectBinFilePath);
362  if (bin.read() == (byte)'M' && bin.read() == (byte)'P' &&
363  bin.read() == (byte)'F' && bin.read() == (byte)'S' &&
364  bin.read() == (byte)0x02)
365  {
366  // Upload an MPFS2 image
367  bin.close();
368  if ((outPutVersion == MPFS_OUTPUT_VERSION.MPFS2)&&
369  (uploadProtocol == UPLOAD_PROTOCOL_OPTION.HTTP))
370  return "http";
371  else
372  return "ftp";
373  //return "http";
374  }
375  else
376  {
377  // Upload an MPFS Classic image
378  bin.close();
379  generateLog.add("Error : Invalid Source file.");
380  //return "ftp";
381  return null;
382  }
383  }
384  catch(IOException e)
385  {
386  // Will trap if the file did not exist or was unreadable
387  return "http";
388  }
389  }
