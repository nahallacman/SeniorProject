131 WORD GridDraw(void *pObj)
132 {
133  SHORT i;
134  SHORT j;
135  SHORT xText, yText;
136  GRID *pGrid;
137 
138  pGrid = (GRID *)pObj;
139 
140  if
141  (
142  (pGrid->hdr.state & GRID_DRAW_ITEMS) ||
143  (pGrid->hdr.state & GRID_DRAW_ALL) ||
144  (pGrid->hdr.state & GRID_SHOW_FOCUS)
145  )
146  {
147  if(pGrid->hdr.state & GRID_DRAW_ALL)
148  {
149 
150  // Clear the entire region.
151  SetColor(pGrid->hdr.pGolScheme->CommonBkColor);
152  while(!Bar(pGrid->hdr.left, pGrid->hdr.top, pGrid->hdr.right, pGrid->hdr.bottom));
153 
154  // initialize the global cursor positions to the Grid left top position.
155  _cursorX = pGrid->hdr.left;
156  _cursorY = pGrid->hdr.top;
157 
158  // Draw the grid lines
159  if(pGrid->hdr.state & (GRID_SHOW_LINES | GRID_SHOW_BORDER_ONLY | GRID_SHOW_SEPARATORS_ONLY))
160  {
161  SetLineType(SOLID_LINE);
162  SetColor(pGrid->hdr.pGolScheme->EmbossLtColor);
163 
164  // Draw the outside of the box
165  if(pGrid->hdr.state & (GRID_SHOW_LINES | GRID_SHOW_BORDER_ONLY))
166  {
167  while(!Line(pGrid->hdr.left, pGrid->hdr.top, pGrid->hdr.right, pGrid->hdr.top));
168  LineTo(_cursorX, pGrid->hdr.bottom);
169  LineTo(pGrid->hdr.left, _cursorY);
170  LineTo(pGrid->hdr.left, pGrid->hdr.top);
171  }
172 
173  // Draw the lines between each cell
174  if(pGrid->hdr.state & (GRID_SHOW_LINES | GRID_SHOW_SEPARATORS_ONLY))
175  {
176  for(i = 1; i < pGrid->numColumns; i++)
177  {
178  while
179  (
180  !Line
181  (
182  pGrid->hdr.left + i * (pGrid->cellWidth + 1),
183  pGrid->hdr.top,
184  pGrid->hdr.left + i * (pGrid->cellWidth + 1),
185  pGrid->hdr.top + pGrid->numRows * (pGrid->cellHeight + 1)
186  )
187  );
188  }
189 
190  for(i = 1; i < pGrid->numRows; i++)
191  {
192  while
193  (
194  !Line
195  (
196  pGrid->hdr.left,
197  pGrid->hdr.top + i * (pGrid->cellHeight + 1),
198  pGrid->hdr.right,
199  pGrid->hdr.top + i * (pGrid->cellHeight + 1)
200  )
201  );
202  }
203  }
204  }
205  }
206 
207  for(i = 0; i < pGrid->numColumns; i++)
208  {
209  for(j = 0; j < pGrid->numRows; j++)
210  {
211  if
212  (
213  (pGrid->hdr.state & GRID_DRAW_ALL) ||
214  (
215  (pGrid->hdr.state & GRID_DRAW_ITEMS) &&
216  (pGrid->gridObjects[CELL_AT(i, j)].status & GRIDITEM_DRAW)
217  ) ||
218  ((pGrid->hdr.state & GRID_SHOW_FOCUS) && (i == pGrid->focusX) && (j == pGrid->focusY))
219  )
220  {
221 
222  // Clear the cell
223  SetColor(pGrid->hdr.pGolScheme->CommonBkColor);
224  while(!Bar(CELL_LEFT, CELL_TOP, CELL_RIGHT, CELL_BOTTOM));
225 
226  // Draw the cell
227  if((pGrid->gridObjects[CELL_AT(i, j)].status & GRID_TYPE_MASK) == GRIDITEM_IS_BITMAP)
228  {
229 
230  // Draw the bitmap
231  if(pGrid->gridObjects[CELL_AT(i, j)].data)
232  {
233  while(!PutImage(CELL_LEFT, CELL_TOP, pGrid->gridObjects[CELL_AT(i, j)].data, BITMAP_SCALE));
234  }
235  }
236  else
237  {
238 
239  // Draw the text
240  if(pGrid->gridObjects[CELL_AT(i, j)].data)
241  {
242  SetFont(pGrid->hdr.pGolScheme->pFont);
243  SetColor(pGrid->hdr.pGolScheme->TextColor0);
244 
245  if(GetState(pGrid, GRIDITEM_TEXTRIGHT))
246  {
247  xText = CELL_LEFT+pGrid->cellWidth-GetTextWidth(pGrid->gridObjects[CELL_AT(i, j)].data,pGrid->hdr.pGolScheme->pFont);
248  }
249  else if(GetState(pGrid, GRIDITEM_TEXTLEFT))
250  {
251  xText = CELL_LEFT;
252  }
253  else
254  {
255  xText = CELL_LEFT+(pGrid->cellWidth>>1)-(GetTextWidth(pGrid->gridObjects[CELL_AT(i, j)].data,pGrid->hdr.pGolScheme->pFont)>>1);
256  }
257  if(GetState(pGrid, GRIDITEM_TEXTBOTTOM))
258  {
259  yText = CELL_TOP+pGrid->cellHeight-GetTextHeight(pGrid->hdr.pGolScheme->pFont);
260  }
261  else if(GetState(pGrid, GRIDITEM_TEXTTOP))
262  {
263  yText = CELL_TOP;
264  }
265  else
266  {
267  yText = CELL_TOP+(pGrid->cellHeight>>1)-(GetTextHeight(pGrid->hdr.pGolScheme->pFont)>>1);
268  }
269 
270  while(!OutTextXY( xText, yText, pGrid->gridObjects[CELL_AT(i, j)].data));
271  }
272  }
273 
274  // Draw the focus if applicable.
275  if((pGrid->hdr.state & GRID_SHOW_FOCUS) && (i == pGrid->focusX) && (j == pGrid->focusY))
276  {
277  SetColor(pGrid->hdr.pGolScheme->EmbossLtColor);
278  SetLineType(DOTTED_LINE);
279  SetLineThickness(NORMAL_LINE);
280  while(!Rectangle(CELL_LEFT, CELL_TOP, CELL_RIGHT, CELL_BOTTOM));
281  }
282 
283  // If the cell is selected, indicate it.
284  if(pGrid->gridObjects[CELL_AT(pGrid->focusX, pGrid->focusY)].status & GRIDITEM_SELECTED)
285  {
286  SetColor(pGrid->hdr.pGolScheme->EmbossLtColor);
287  SetLineType(SOLID_LINE);
288  if(pGrid->hdr.state & GRID_SHOW_LINES)
289  {
290  SetLineThickness(THICK_LINE);
291  }
292  else
293  {
294  SetLineThickness(NORMAL_LINE);
295  }
296 
297  while(!Rectangle(CELL_LEFT - 1, CELL_TOP - 1, CELL_RIGHT + 1, CELL_BOTTOM + 1));
298  }
299 
300  GridClearCellState(pGrid, i, j, GRIDITEM_DRAW);
301  }
302  }
303  }
304 
305  //pGrid->state &= ~(GRID_DRAW_ITEMS || GRID_DRAW_ALL || GRID_SHOW_FOCUS);
306  pGrid->hdr.state &= ~(GRID_DRAW_ITEMS || GRID_DRAW_ALL);
307 
308  // Set line state
309  SetLineType(SOLID_LINE);
310  }
311 
312  return (1);
313 }

465 
466  GRID *pGrid;

468  pGrid = (GRID *)pObj;

476

485  (pGrid->hdr.right >= pMsg->param1) &&
486  (pGrid->hdr.top <= pMsg->param2) &&
487  (pGrid->hdr.bottom >= pMsg->param2)
488  )
489  {

492
