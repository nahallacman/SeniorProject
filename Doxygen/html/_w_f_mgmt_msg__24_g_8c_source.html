90  void SendMgmtMsg(UINT8 *p_header, UINT8 headerLength, UINT8 *p_data, UINT8 dataLength)
91 
92  Summary:
93  Sends a management message to MRF24W.
94 
95  Description:
96  Allocates management buffer on MRF24W. Then sends the management message
97  (header + data) to MRF24W.
98 
99  Precondition:
100  MACInit must be called.
101 
102  Parameters:
103  p_header -- pointer to mgmt message header data
104  headerLength -- number of bytes in the header will be written
105  p_data -- pointer to mgmt message data
106  dataLength -- number of bytes of data
107 
108  Returns:
109  None
110 
111  Remarks:
112  *****************************************************************************/
113 void SendMgmtMsg(UINT8 *p_header,
114  UINT8 headerLength,
115  UINT8 *p_data,
116  UINT8 dataLength)
117 {
118  UINT32 timeoutPeriod;
119  UINT32 startTickCount;
120 
121  /* cannot send management messages while in WF_ProcessEvent() */
122  WF_ASSERT(!isInWFProcessEvent());
123 
124  EnsureWFisAwake();
125 
126  /* Allocate a management buffer on the WiFi device. May have to wait a bit while the previous Data Tx packet is being sent */
127  /* which will free up the memory for this mgmt packet. */
128  timeoutPeriod = TICKS_PER_SECOND / 2; /* 500 ms */
129  startTickCount = (UINT32)TickGet();
130  while (AllocateMgmtTxBuffer(WF_MAX_TX_MGMT_MSG_SIZE) == FALSE)
131  {
132  if (TickGet() - startTickCount >= timeoutPeriod)
133  {
134  WF_ASSERT(FALSE);
135  }
136  }
137 
138  /* write out management header */
139  RawSetByte(RAW_MGMT_TX_ID, p_header, headerLength);
140 
141  /* write out data (if any) */
142  if (dataLength > 0)
143  {
144  RawSetByte(RAW_MGMT_TX_ID, p_data, dataLength);
145  }
146 
147  /* send mgmt msg to MRF24W */
148  SendRAWManagementFrame(headerLength + dataLength);
149 
150 
151 }

181 /*******************************************************************************
182  Function:
183  void WaitForMgmtResponse(UINT8 expectedSubtype, UINT8 freeAction)
184 
185  Summary:
186  Waits for a management response
187 
188  Description:
189  Called after sending a mgmt request. This function waits for a mgmt
190  response. The caller can optionally request the the management
191  response be freed immediately (by this function) or not freed. If not
192  freed, the caller is responsible to free the response buffer.
193 
194  Precondition:
195  MACInit must be called.
196 
197  Parameters:
198  expectedSubtype -- Expected subtype of the mgmt response
199  freeAction -- FREE_MGMT_BUFFER or DO_NOT_FREE_MGMT_BUFFER
200 
201  Returns:
202  None
203 
204  Remarks:
205  *****************************************************************************/
206 void WaitForMgmtResponse(UINT8 expectedSubtype, UINT8 freeAction)
207 {
208  #if defined(__18CXX)
209  static tMgmtMsgRxHdr hdr;
210  #else
211  tMgmtMsgRxHdr hdr;
212  #endif
213 
214  g_WaitingForMgmtResponse = TRUE;
215 
216  /* Wait until mgmt response is received */
217  while (gMgmtConfirmMsgReceived == FALSE)
218  {
219  WFProcess();
220 
221  /* if received a data packet while waiting for mgmt packet */
222  if (g_HostRAWDataPacketReceived)
223  {
224  // We can't let the StackTask processs data messages that come in while waiting for mgmt
225  // response because the application might send another mgmt message, which is illegal until the response
226  // is received for the first mgmt msg. And, we can't prevent the race condition where a data message
227  // comes in before a mgmt response is received. Thus, the only solution is to throw away a data message
228  // that comes in while waiting for a mgmt response. This should happen very infrequently. If using TCP then the
229  // stack takes care of retries. If using UDP, the application has to deal with occasional data messages not being
230  // received. Also, applications typically do not send a lot of management messages after connected.
231 
232  // throw away the data rx
233  RawMountRxBuffer(RAW_DATA_RX_ID);
234  DeallocateDataRxBuffer();
235  g_HostRAWDataPacketReceived = FALSE;
236 
237  /* ensure interrupts enabled */
238  WF_EintEnable();
239  }
240  }
241  g_WaitingForMgmtResponse = FALSE;
242 
243  /* set this back to FALSE so the next mgmt send won't think he has a response before one is received */
244  gMgmtConfirmMsgReceived = FALSE;
245 
246 
247  /* if the caller wants to delete the response immediately (doesn't need any data from it */
248  if (freeAction == FREE_MGMT_BUFFER)
249  {
250  /* read and verify result before freeing up buffer to ensure our message send was successful */
251  RawRead(RAW_MGMT_RX_ID, 0, (UINT16)(sizeof(tMgmtMsgRxHdr)), (UINT8 *)&hdr);
252 
253  /* mgmt response subtype had better match subtype we were expecting */
254  WF_ASSERT(hdr.subtype == expectedSubtype);
255 
256  /* Mgmt response 'result' field should always indicate success. If this assert is hit the error codes are located */
257  /* WFApi.h. Search for WF_SUCCESS for the list of error codes. */
258  if (hdr.result == WF_ERROR_HOST_SCAN_NOT_ALLOWED)
259  {
260  #if defined(STACK_USE_UART)
261  putrsUART("Host Scan Failed. Host scan is allowed only in idle or connected state\r\n");
262  #endif
263  } else if (hdr.result == WF_ERROR_INVALID_WPS_PIN)
264  {
265  #if defined(STACK_USE_UART)
266  putrsUART("WPS failed : Invalid WPS PIN data\r\n");
267  #endif
268  } else if (hdr.result == WF_ERROR_DISCONNECT_FAILED)
269  {
270  #if defined(STACK_USE_UART)
271  putrsUART("Disconnect failed. Disconnect is allowed only when module is in connected state\r\n");
272  #endif
273  }
274  else if (hdr.result == WF_ERROR_NO_STORED_BSS_DESCRIPTOR)
275  {
276  #if defined(STACK_USE_UART)
277  putrsUART("No stored scan results\r\n");
278  #endif
279  }
280  else if (hdr.result != WF_SUCCESS)
281  {
282  #if defined(STACK_USE_UART)
283  UINT8 buf[8];
284  putrsUART("WaitForMgmtResponse result error: ");
285  sprintf((char *)buf, "%d",hdr.result);
286  putrsUART((char *)buf);
287  putrsUART("\r\n");
288  #endif
289  //WF_ASSERT(FALSE);
290  }
291 
292  /* free mgmt buffer */
293  DeallocateMgmtRxBuffer();
294  }

330 void WaitForMgmtResponseAndReadData(UINT8 expectedSubtype,
331  UINT8 numDataBytes,
332  UINT8 startIndex,
333  UINT8 *p_data)
334 
335 {
336  tMgmtMsgRxHdr hdr; /* management msg header struct */
337 
338  WaitForMgmtResponse(expectedSubtype, DO_NOT_FREE_MGMT_BUFFER);
339 
340  /* if made it here then received a management message */
341  RawRead(RAW_MGMT_RX_ID, 0, (UINT16)(sizeof(tMgmtMsgRxHdr)), (UINT8 *)&hdr);
