278 BYTE USBHostAudioV1ReceiveAudioData( BYTE deviceAddress,
279  ISOCHRONOUS_DATA *pIsochronousData )
280 {
281  DWORD byteCount;
282  BYTE errorCode;
283  BYTE i;
284 
285  #ifdef DEBUG_MODE
286  UART2PrintString( "AUD: Start audio stream\r\n" );
287  #endif
288 
289  // Find the correct device.
290  for (i=0; (i<USB_MAX_AUDIO_DEVICES) && (deviceInfoAudioV1[i].ID.deviceAddress != deviceAddress); i++);
291  if (i == USB_MAX_AUDIO_DEVICES)
292  {
293  return USB_AUDIO_DEVICE_NOT_FOUND;
294  }
295 
296  // Make sure the device is not already receiving data or setting the interface.
297  if (!USBHostTransferIsComplete( deviceInfoAudioV1[i].ID.deviceAddress,
298  deviceInfoAudioV1[i].endpointAudioStream, &errorCode, &byteCount ) ||
299  deviceInfoAudioV1[i].flags.bfSettingInterface)
300  {
301  return USB_AUDIO_DEVICE_BUSY;
302  }
303 
304  // Start receiving data
305  errorCode = USBHostReadIsochronous( deviceInfoAudioV1[i].ID.deviceAddress,
306  deviceInfoAudioV1[i].endpointAudioStream, pIsochronousData );
307  if (errorCode)
308  {
309  #ifdef DEBUG_MODE
310  UART2PrintString( "AUD: Cannot start audio stream.\r\n" );
311  #endif
312  }
313 
314  return errorCode;
315 }

460 /****************************************************************************
461  Function:
462  BYTE USBHostAudioV1SetSamplingFrequency( BYTE deviceAddress, BYTE *frequency )
463 
464  Summary:
465  This function sets the sampling frequency for the device.
466 
467  Description:
468  This function sets the sampling frequency for the device. If the exact
469  frequency is not supported by the device, the device will round it to the
470  closest supported value.
471 
472  IMPORTANT: If the request is initiated successfully, the frequency value
473  must remain valid until the EVENT_AUDIO_FREQUENCY_SET event is received.
474  Therefore, this value cannot be a local (stack) variable. The application
475  can either use a global variable for this value, or it can use the
476  function USBHostAudioV1SupportedFrequencies() to obtain a pointer to the
477  number and list of supported frequencies, and pass a pointer to the desired
478  frequency in this list.
479 
480  Precondition:
481  None
482 
483  Parameters:
484  BYTE deviceAddress - Device address
485  BYTE *frequency - Pointer to three bytes that specify the desired
486  sampling frequency. NOTE: If the request is
487  initiated successfully, this location must
488  remain valid until the EVENT_AUDIO_FREQUENCY_SET
489  event is received.
490 
491  Return Values:
492  USB_SUCCESS - Request started successfully
493  Others - See USBHostIssueDeviceRequest() errors.
494 
495  Example:
496  <code>
497  BYTE numFrequencies;
498  BYTE *ptr;
499 
500  ptr = USBHostAudioV1SupportedFrequencies( deviceAddress );
501  if (ptr)
502  {
503  numFrequencies = *ptr;
504  ptr++;
505  if (numFrequencies == 0)
506  {
507  // Continuous sampling, minimum and maximum are specified.
508  DWORD minFrequency;
509  DWORD maxFrequency;
510 
511  minFrequency = *ptr + (*(ptr+1) << 8) + (*(ptr+2) << 16);
512  ptr += 3;
513  maxFrequency = *ptr + (*(ptr+1) << 8) + (*(ptr+2) << 16);
514  if ((minFrequency <= desiredFrequency) && (desiredFrequency <= maxFrequency))
515  {
516  rc = USBHostAudioV1SetSamplingFrequency( deviceAddress, &desiredFrequency );
517  }
518  else
519  {
520  // Desired frequency out of range
521  }
522  }
523  else
524  {
525  // Discrete sampling frequencies are specified.
526  DWORD frequency;
527 
528  while (numFrequencies)
529  {
530  frequency = *ptr + (*(ptr+1) << 8) + (*(ptr+2) << 16);
531  if (frequency == desiredFrequency)
532  {
533  rc = USBHostAudioV1SetSamplingFrequency( deviceAddress, ptr );
534  continue;
535  }
536  numFrequencies--;
537  ptr += 3;
538  }
539  if (numFrequencies == 0)
540  {
541  // Desired frequency not found.
542  }
543  }
544  }
545  </code>
546 
547  Remarks:
548  If a global variable is used to old the frequency, it can be declared as
549  a DWORD. Since PIC Microcontrollers are little endian machines, a
550  pointer to the DWORD can be used as the frequency parameter:
551  <code>
552  DWORD desiredFrequency = 44100; // Hertz
553 
554  rc = USBHostAudioV1SetSamplingFrequency( deviceAddress, (BYTE *)(&desiredFrequency) );
555  </code>
556 
557  ***************************************************************************/

563 
564  // Find the correct device.

574  deviceInfoAudioV1[i].ID.clientDriverID );

584  deviceInfoAudioV1[i].flags.bfSettingFrequency = 1;
585  }

808  FALSE - We cannot support the device.
809 
810  Remarks:
811  None
812  ***************************************************************************/
813 
814 BOOL USBHostAudioV1Initialize( BYTE address, DWORD flags, BYTE clientDriverID )
815 {
816  BYTE *descriptor;
817  BYTE device;
818  WORD i;
819  WORD saveIndex;
820 
821 
822  // See if the device has already been initialized, and therefore is
823  // already in the audio device table. If we change the configuration, the
824  // host driver will call the Initialize function again.
825  for (device = 0; (device < USB_MAX_AUDIO_DEVICES) && (deviceInfoAudioV1[device].ID.deviceAddress != address); device++);
826  if (device < USB_MAX_AUDIO_DEVICES)
827  {
828  return TRUE;
829  }
830 
831  // Find the free slot in the table. If we cannot find one, kick off the device.
832  for (device = 0; (device < USB_MAX_AUDIO_DEVICES) && (deviceInfoAudioV1[device].ID.deviceAddress != 0); device++);
833  if (device == USB_MAX_AUDIO_DEVICES)
834  {
835  #ifdef DEBUG_MODE
836  UART2PrintString( "AUD: No free slots available for audio.\r\n" );
837  #endif
838  // Kick off the device
839  return FALSE;
840  }
841 
842  #ifdef DEBUG_MODE
843  UART2PrintString( "AUD: USBHostAudioV1Initialize(0x" );
844  UART2PutHex( flags );
845  UART2PrintString( ")\r\n" );
846  #endif
847 
848  // Fill in the VID and PID before we lose the device descriptor pointer.
849  // It's not valid unless deviceAddress is non-zero.
850  descriptor = USBHostGetDeviceDescriptor( address );
851  deviceInfoAudioV1[device].ID.vid = ((USB_DEVICE_DESCRIPTOR *)descriptor)->idVendor;
852  deviceInfoAudioV1[device].ID.pid = ((USB_DEVICE_DESCRIPTOR *)descriptor)->idProduct;
853 
854  // Get ready to parse the configuration descriptor.
855  descriptor = USBHostGetCurrentConfigurationDescriptor( address );
856 
857  i = 0;
858 
859  #ifdef DEBUG_MODE
860  UART2PrintString("AUD: Checking descriptor " );
861  UART2PutDec( descriptor[i+5] );
862  UART2PrintString(" ...\r\n" );
863  #endif
864 
865  // Find the next interface descriptor.
866  while (i < ((USB_CONFIGURATION_DESCRIPTOR *)descriptor)->wTotalLength)
867  {
868  #ifdef DEBUG_MODE
869  UART2PrintString("AUD: Checking interface...\r\n" );
870  #endif
871  // See if we are pointing to an interface descriptor.
872  if (descriptor[i+1] == USB_DESCRIPTOR_INTERFACE)
873  {
874  // See if the interface is a streaming audio interface. Protocol is not used
875  if ((descriptor[i+5] == USB_CLASS_AUDIO) && (descriptor[i+6] == USB_SUBCLASS_AUDIO_STREAMING))
876  {
877  // Record the interface number.
878  deviceInfoAudioV1[device].interface = descriptor[i+2];
879 
880  if (descriptor[i+4] == 0)
881  {
882  // If the number of endpoints is 0, then this is the zero bandwidth configuration.
883  #ifdef DEBUG_MODE
884  UART2PrintString( "AUD: Zero bandwidth configuration\r\n" );
885  #endif
886  deviceInfoAudioV1[device].settingZeroBandwidth = descriptor[i+3];
887  deviceInfoAudioV1[device].flags.bfFoundZeroBandwidthSetting = 1;
888  }
889  else
890  {
891  // If the number of endpoints is not 0, then this is the full bandwidth configuration.
892  #ifdef DEBUG_MODE
893  UART2PrintString( "AUD: Full bandwidth configuration\r\n" );
894  #endif
895  deviceInfoAudioV1[device].settingFullBandwidth = descriptor[i+3];
896  deviceInfoAudioV1[device].flags.bfFoundFullBandwidthSetting = 1;
897 
898  // Initialize variables prior to scanning.
899  i += descriptor[i];
900  saveIndex = i;
901 
902  // Look for an audio stream endpoint.
903  while ((deviceInfoAudioV1[device].endpointAudioStream == 0) &&
904  (i < ((USB_CONFIGURATION_DESCRIPTOR *)descriptor)->wTotalLength))
905  {
906  if (descriptor[i+1] == USB_DESCRIPTOR_ENDPOINT)
907  {
908  if ((descriptor[i+3] & 0x03) == 0x01) // Isochronous
909  {
910  deviceInfoAudioV1[device].endpointAudioStream = descriptor[i+2];
911  deviceInfoAudioV1[device].ID.audioDataPacketSize = (descriptor[i+5] << 8) + descriptor[i+4];
912  #ifdef DEBUG_MODE
913  UART2PrintString( "AUD: Audio stream endpoint: " );
914  UART2PutHex( deviceInfoAudioV1[device].endpointAudioStream );
915  UART2PrintString( ", packet size " );
916  UART2PutHexWord( deviceInfoAudioV1[device].ID.audioDataPacketSize );
917  UART2PrintString( "\r\n" );
918  #endif
919  }
920  }
921  i += descriptor[i];
922  }
923 
924  // Look for the CS_INTERFACE, FORMAT_TYPE descriptor
925  i = saveIndex;
926  while ((deviceInfoAudioV1[device].pFormatTypeDescriptor == NULL) &&
927  (i < ((USB_CONFIGURATION_DESCRIPTOR *)descriptor)->wTotalLength))
928  {
929  if ((descriptor[i+1] == USB_AUDIO_DESCRIPTOR_INTERFACE) &&
930  (descriptor[i+2] == USB_AUDIO_AS_IF_DESCRIPTOR_FORMAT_TYPE))
931  {
932  if (descriptor[i+3] == USB_AUDIO_FORMAT_TYPE_I) // iPods should be only this format.
933  {
934  #ifdef DEBUG_MODE
935  UART2PrintString( "AUD: Found supported frequencies\r\n" );
936  #endif
937  deviceInfoAudioV1[device].pFormatTypeDescriptor = &(descriptor[i]);
938  }
939  }
940  i += descriptor[i];
941  }
942 
943  // Other audio devices may need to check more parameters.
944  // Since we support only iPods, we do not have to check
945  // data format, etc.
946 
947  }
948 
949  // Make sure we've found everything we need.
950  if ((deviceInfoAudioV1[device].flags.bfFoundZeroBandwidthSetting == 1) &&
951  (deviceInfoAudioV1[device].flags.bfFoundFullBandwidthSetting == 1) &&
952  (deviceInfoAudioV1[device].pFormatTypeDescriptor != NULL) &&
953  (deviceInfoAudioV1[device].endpointAudioStream != 0))
954  {
955  // Initialize the device information.
956  deviceInfoAudioV1[device].ID.deviceAddress = address;
957  deviceInfoAudioV1[device].ID.clientDriverID = clientDriverID;
958 
959  #ifdef DEBUG_MODE
960  UART2PrintString( "AUD: AUD attached.\r\n" );
961  #endif
962 
963  // Tell the application layer that we have a device.
964  USB_HOST_APP_EVENT_HANDLER( deviceInfoAudioV1[device].ID.deviceAddress, EVENT_AUDIO_ATTACH, &(deviceInfoAudioV1[device].ID), sizeof(USB_AUDIO_V1_DEVICE_ID) );
965 
966  return TRUE;
967  }
968  }
969  }
970 
971  // Jump to the next descriptor in this configuration.
972  i += descriptor[i];
973  }
974 
975  // This configuration is not valid for an audio device.
976  return FALSE;
977 }

1003  FALSE - Event was not handled
1004 
1005  Remarks:
1006  None
1007  ***************************************************************************/
1008 
1009 BOOL USBHostAudioV1EventHandler( BYTE address, USB_EVENT event, void *data, DWORD size )
1010 {
1011  BYTE i;
1012 
1013  switch (event)
1014  {
1015  case EVENT_NONE: // No event occured (NULL event)
1016  return TRUE;
1017  break;
1018 
1019  case EVENT_DETACH: // USB cable has been detached (data: BYTE, address of device)
1020  #ifdef DEBUG_MODE
1021  UART2PrintString( "AUD: Detach\r\n" );
1022  #endif
1023 
1024  // Find the device in the table. If found, clear the important fields.
1025  for (i=0; (i<USB_MAX_AUDIO_DEVICES) && (deviceInfoAudioV1[i].ID.deviceAddress != address); i++);
1026  if (i < USB_MAX_AUDIO_DEVICES)
1027  {
1028  // Notify that application that the device has been detached.
1029  USB_HOST_APP_EVENT_HANDLER( address, EVENT_AUDIO_DETACH,
1030  &deviceInfoAudioV1[i].ID.deviceAddress, sizeof(BYTE) );
1031 
1032  deviceInfoAudioV1[i].ID.deviceAddress = 0;
1033  deviceInfoAudioV1[i].flags.val = 0;
1034  deviceInfoAudioV1[i].endpointAudioStream = 0;
1035  deviceInfoAudioV1[i].pFormatTypeDescriptor = NULL;
1036  }
1037  return TRUE;
1038  break;
1039 
1040  case EVENT_TRANSFER: // A USB transfer has completed
1041  #ifdef DEBUG_MODE
1042 // UART2PrintString( "AUD: transfer event: " );
1043 // UART2PutHex( address );
1044 // UART2PrintString( "\r\n" );
1045  #endif
1046 
1047  for (i=0; (i<USB_MAX_AUDIO_DEVICES) && (deviceInfoAudioV1[i].ID.deviceAddress != address); i++) {}
1048  if (i == USB_MAX_AUDIO_DEVICES)
1049  {
1050  #ifdef DEBUG_MODE
1051  UART2PrintString( "AUD: Unknown device\r\n" );
1052  #endif
1053  return FALSE;
1054  }
1055 
1056  // The transfer event comes with data of the type HOST_TRANSFER_DATA.
1057 
1058  if (((HOST_TRANSFER_DATA *)data)->bEndpointAddress == 0)
1059  {
1060  if (deviceInfoAudioV1[i].flags.bfSettingInterface)
1061  {
1062  // Either the full or the zero bandwidth interface has been set.
1063  deviceInfoAudioV1[i].flags.bfSettingInterface = 0;
1064  USB_HOST_APP_EVENT_HANDLER( i, EVENT_AUDIO_INTERFACE_SET, NULL, ((HOST_TRANSFER_DATA *)data)->bErrorCode );
1065  }
1066  if (deviceInfoAudioV1[i].flags.bfSettingFrequency)
1067  {
1068  // The sampling frequency has been set.
1069  deviceInfoAudioV1[i].flags.bfSettingFrequency = 0;
1070  USB_HOST_APP_EVENT_HANDLER( i, EVENT_AUDIO_FREQUENCY_SET, data, size );
1071  }
1072  }
1073  #ifdef USB_ENABLE_ISOC_TRANSFER_EVENT
1074  else if (((HOST_TRANSFER_DATA *)data)->bEndpointAddress == deviceInfoAudioV1[i].endpointAudioStream)
1075  {
1076  // If we received streaming audio data, pass the event up to the application.
1077  // It's only one more byte of information more than they need (bmAttributes).
1078  USB_HOST_APP_EVENT_HANDLER( i, EVENT_AUDIO_RECEIVE_STREAM, data, size );
1079  }
1080  #endif
1081  break;
1082 
1083  case EVENT_BUS_ERROR:
1084  #ifdef DEBUG_MODE
1085 // UART2PrintString( "AUD: bus error: " );
1086 // UART2PutHex( address );
1087 // UART2PrintString( "\r\n" );
1088  #endif
1089 
1090  for (i=0; (i<USB_MAX_AUDIO_DEVICES) && (deviceInfoAudioV1[i].ID.deviceAddress != address); i++) {}
1091  if (i == USB_MAX_AUDIO_DEVICES)
1092  {
1093  #ifdef DEBUG_MODE
1094  UART2PrintString( "AUD: Unknown device\r\n" );
1095  #endif
1096  return FALSE;
1097  }
1098 
1099  // The bus error event comes with data of the type HOST_TRANSFER_DATA.
1100 
1101  if (((HOST_TRANSFER_DATA *)data)->bEndpointAddress == 0)
1102  {
1103  ((HOST_TRANSFER_DATA *)data)->bErrorCode = USB_AUDIO_COMMAND_FAILED;
1104  if (deviceInfoAudioV1[i].flags.bfSettingInterface)
1105  {
1106  // The interface could not be set.
1107  deviceInfoAudioV1[i].flags.bfSettingInterface = 0;
1108  USB_HOST_APP_EVENT_HANDLER( i, EVENT_AUDIO_INTERFACE_SET, NULL, ((HOST_TRANSFER_DATA *)data)->bErrorCode );
1109  }
1110  if (deviceInfoAudioV1[i].flags.bfSettingFrequency)
1111  {
1112  // The sampling frequency could not be set.
1113  deviceInfoAudioV1[i].flags.bfSettingFrequency = 0;
1114  USB_HOST_APP_EVENT_HANDLER( i, EVENT_AUDIO_FREQUENCY_SET, data, size );
1115  }
1116  }
1117  else if (((HOST_TRANSFER_DATA *)data)->bEndpointAddress == deviceInfoAudioV1[i].endpointAudioStream)
1118  {
1119  // Bus error on data stream.
1120  USB_HOST_APP_EVENT_HANDLER( i, EVENT_AUDIO_STREAM_RECEIVED, NULL, 0 );
1121  }
1122  break;
1123 
1124  case EVENT_SOF: // Start of frame - NOT NEEDED
1125  case EVENT_RESUME: // Device-mode resume received
1126  case EVENT_SUSPEND: // Device-mode suspend/idle event received
1127  case EVENT_RESET: // Device-mode bus reset received
1128  case EVENT_STALL: // A stall has occured
1129  return TRUE;
1130  break;
1131 
1132  default:
1133  return FALSE;
1134  break;
1135  }
1136  return FALSE;
1137 }
