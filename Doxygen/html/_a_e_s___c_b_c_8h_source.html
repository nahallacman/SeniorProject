73 /* AES_CBC_STATE_DATA, *P_AES_CBC_STATE_DATA
74 
75  Summary:
76  State information that must be maintained between blocks and initialization
77  data.
78 
79  Description:
80  The AES_CBC_STATE_DATA structure contains state information that needs to be
81  retained between blocks of CBC data. This structure is also used to pass
82  some initialization data to the first block of a new stream.
83 */
84 typedef struct
85 {
86  /* The initialization vector must be loaded before the first call to the
87  encrypt or decrypt functions. After that point this variable should
88  not be touched. This buffer is used to hold intermediate data. */
89  UINT32 __attribute__((aligned)) initial_vector[4];
90 
91  /* Buffer for any data bytes that may have been remaining from the last
92  call to the encrypt or decrypt function */
93  /* DOM-IGNORE-BEGIN */
94  /* This buffer is 16 bytes in length instead of 15 bytes in length
95  because this provides a simple, word aligned buffer for the PIC32
96  implementation to use */
97  /* DOM-IGNORE-END */
98  UINT8 __attribute__((aligned)) remaining_data[16];
99 
100  /* The number of valid bytes in the remaining_data buffer */
101  UINT8 bytes_remaining;
102 } AES_CBC_STATE_DATA, *P_AES_CBC_STATE_DATA;

121  Summary:
122  Encrypts a specified amount of data using cipher feedback (CFB) mode.
123  Description:
124  Encrypts a specified amount of data using cipher feedback (CFB) mode.
125  Conditions:
126  * Round keys must be create by AESCreateRoundKeys() before calling
127  this routine.
128  * p_cbc_state_data->intial_vector must be populated with the initialization
129  vector before the first call of this function on a new data stream.
130  * p_cbc_state_data->round_keys must be set to point to a buffer
131  containing the round keys for this session. This buffer must
132  maintain the round keys until this session is complete. The round
133  keys can be created using the AESCreateRoundKeys() function.
134  Input:
135  cipher_text - [out] pointer to the buffer where the encrypted data
136  sould reside. This buffer must be a multiple of 16 bytes
137  even if the plain text passed in is not. This buffer should
138  also always be larger than the plain_text buffer.
139  num_cipher_bytes - [out] the number of bytes written to the cipher_text buffer.
140  Note that this could be more or less than the num_plain bytes due
141  to buffered data or padding.
142  plain_text - [in] pointer to the data that needs to be encrypted.
143  num_plain_bytes - [in] the number of bytes in the plain_text buffer
144  round_keys - [in] pointer to the round key buffer that will be used
145  for this encryption cycle.
146  p_cbc_state_data - [in] pointer to an instance of the AES_CBC_STATE_DATA for this
147  stream. The first call of this function should have the
148  p_cbc_state_data->initial_vector set to the initialization
149  vector. The same AES_CBC_STATE_DATA object should be used for
150  every call in the same data stream. The contents of the
151  AES_CBC_STATE_DATA object shouldn't be changed by the user
152  once an encryption stream has been started.
153  options - [in] Set of options that are ORed together (e.g. \-
154  AES_PAD_NONE | AES_STREAM_START). Valid options are\:
155  * AES_PAD_NONE
156  * AES_PAD_NULLS
157  * AES_PAD_8000
158  * AES_PAD_NUMBER
159  * AES_STREAM_START
160  * AES_STREAM_CONTINUE
161  * AES_STREAM_COMPLETE
162  * AES_CIPHER_TEXT_POINTER_ALIGNED
163  * AES_PLAIN_TEXT_POINTER_ALIGNED
164  Return:
165  None
166  Example:
167  <code>
168  //Example variables. The key and initialization vector information
169  // should be customized for the target application.
170  static const UINT8 AESKey128[] = { 0x95, 0xA8, 0xEE, 0x8E,
171  0x89, 0x97, 0x9B, 0x9E,
172  0xFD, 0xCB, 0xC6, 0xEB,
173  0x97, 0x97, 0x52, 0x8D
174  };
175  static const UINT8 initialization_vector[] = { 0x00, 0x01, 0x02, 0x03,
176  0x04, 0x05, 0x06, 0x07,
177  0x08, 0x09, 0x0A, 0x0B,
178  0x0C, 0x0D, 0x0E, 0x0F
179  };
180  AES_ROUND_KEYS_128_BIT round_keys;
181  AES_CBC_STATE_DATA current_stream;
182  UINT32 num_bytes_encrypted;
183 
184  //Create the round keys. This only needs to be done once for each key.
185  // This example is here for completeness.
186  AESCreateRoundKeys ( &round_keys,
187  AESKey128,
188  AES_KEY_SIZE_128_BIT
189  );
190 
191  //Before the first call of AESCBCEncrypt() for a stream of data, the
192  // p_cbc_state_data->initial_vector must be loaded with the initialization
193  // vector information for that specific stream of data.
194  memcpy(current_stream.initial_vector,initialization_vector,16);
195 
196  //Encrypt 128 bytes of of the current stream. The num_bytes field must
197  // be a multiple of 16 unless the options field specifies that it is the
198  // end of the stream.
199  AESCBCEncrypt( cipher_text,
200  &num_bytes_encrypted,
201  plain_text,
202  128, //num_bytes
203  &round_keys,
204  &current_stream, //stream state data
205  AES_STREAM_START //options
206  );
207 
208  //At some point later we need to encrypt the rest of the stream. The
209  // entire stream could have been encrypted in the first call if it was all
210  // available at the time. Otherwise it can be done in sections, like this
211  // example.
212 
213  //Encrypt the remaining data of this stream.
214  AESCBCEncrypt( cipher_text_part2,
215  &num_bytes_encrypted,
216  plain_text_part2,
217  53, //num_bytes
218  &round_keys,
219  &current_stream, //stream state data
220  AES_PAD_NULLS | AES_STREAM_COMPLETE //options
221  );
222  </code>
