285 #if defined(BI_USE_MOD)
286 void BigIntMod(BIGINT *n, BIGINT* m)
287 {
288  BIGINT_DATA_TYPE *ptrMSBn, MSBm;
289  BIGINT_DATA_TYPE_2 qHatInt;
290  union
291  {
292  BIGINT_DATA_TYPE v[2];
293  BIGINT_DATA_TYPE_2 Val;
294  } topTwoWords;
295 
296  // Find the starting MSBs
297  ptrMSBn = BigIntMSB(n);
298  MSBm = *BigIntMSB(m);
299 
300  // Set up assembly pointers for m
301  // _iB and _xB are limiters in the _mas function
302  _iB = m->ptrLSB;
303  _xB = BigIntMSB(m);
304 
305  // Find out how many bytes we need to shift and move the LSB up
306  _iR = n->ptrLSB + (BigIntMagnitudeDifference(n, m) - 1);
307 
308  // This loops while the order of magnitude (in words) of n > m
309  // Each iteration modulos off one word of magnitude from n
310  while(_iR >= n->ptrLSB)
311  {
312  // Find qHat = MSBn:MSBn-1/MSBm
313  topTwoWords.Val = *((BIGINT_DATA_TYPE_2*)(ptrMSBn - 1));
314  qHatInt = topTwoWords.Val / MSBm;
315  if(qHatInt > BIGINT_DATA_MAX)
316  qHatInt = BIGINT_DATA_MAX;
317 
318 #if BIGINT_DEBUG
319  putrsUART("\r\n\r\n n = ");
320  BigIntPrint(n);
321  putrsUART("\r\n m = ");
322  BigIntPrint(m);
323  putrsUART("\r\n qHat (");
324  putulhexUART(qHatInt);
325  putrsUART(") = topTwo(");
326  putulhexUART(topTwoWords.Val);
327  putrsUART(") / (");
328  putulhexUART(MSBm);
329  putrsUART(") ");
330 #endif
331 
332  // Once qHat is determined, we multiply M by qHat, shift it up
333  // as many bytes as possible, and subtract the result.
334  // In essence, qHat is a rough estimation of the quotient, divided
335  // by a power of 2^8 (PIC18) or 2^16 (PIC24/dsPIC) or 2^32 (PIC32)
336 
337  // This implementation multiplies and subtracts in the same step
338  // using a _mas function which saves about 30% of clock cycles.
339 
340  // Save the old MSB and set up the ASM pointers
341  _wC = (BIGINT_DATA_TYPE)qHatInt;
342 
343  // Do the multiply and subtract
344  // Occassionally this results in underflow...this is solved below.
345  masBI();
346 
347  // qHat may have been 1 or 2 greater than possible. If so,
348  // the new MSB will be greater than the old one, so we *add*
349  // M back to N in the shifted position until overflow occurs
350  // and this case corrects itself.
351  while(topTwoWords.v[1] < *BigIntMSB(n))
352 // while(((BIGINT_DATA_TYPE*)&topTwoWords)[1] < *BigIntMSB(n))
353  {
354  _iA = _iR;
355  _xA = BigIntMSB(n);
356  addBI();
357  }
358 
359  // We should have modulated off a word (or two if we were lucky),
360  // so move our MSB and LSB pointers as applicable
361  while(*ptrMSBn == 0x0u)
362  {
363  _iR--;
364  n->ptrMSB--;
365  ptrMSBn--;
366  }
367  }
368 
369  // Iteration of the _mas function can only handle full-byte orders
370  // of magnitude. The result may still be a little larger, so this
371  // cleans up the last few multiples with simple subtraction.
372  while(BigIntCompare(n, m) >= 0)
373  {
374  _iA = n->ptrLSB;
375  _xA = n->ptrMSB;
376  subBI();
377 
378  // Invalidate MSB pointer
379  n->bMSBValid = 0;
380  }
381 }
382 #endif

506 
507  #if BIGINT_DEBUG_COMPARE

518  #if BIGINT_DEBUG_COMPARE

520  #endif
521

531  }
