68  SM_RSA_ENCRYPT, // RSA encryption is proceeding
69  SM_RSA_DECRYPT_START, // Initial state for decryption processes; decryption is ready to begin
70  SM_RSA_DECRYPT_FIND_M1, // First stage in the CRT decryption algorithm
71  SM_RSA_DECRYPT_FIND_M2, // Second stage in the CRT decryption algorithm
