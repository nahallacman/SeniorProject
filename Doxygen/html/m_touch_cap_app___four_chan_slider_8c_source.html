195 WORD mTouchCapApp_FourChannelSlider_Logic(SLIDER4CH *Slider)
196 {
197  WORD i;
198  static float delta;
199  float FourChDelta[4];
200  static float avgdeltaCh[4];
201  static WORD PercentLevel=0;
202  static WORD prevpercent=0;
203  static WORD calcPercent=0;
204  static WORD AvrgCh0Data;
205  static WORD AvrgCh1Data;
206  static WORD AvrgCh2Data;
207  static WORD AvrgCh3Data;
208  int calc = 0,min,max;
209 
210 
211  FourChDelta[0] = (float)averageData[Slider->Channel_Num[0]] - (float)CurFiltdData[Slider->Channel_Num[0]];
212  FourChDelta[1] = (float)averageData[Slider->Channel_Num[1]] - (float)CurFiltdData[Slider->Channel_Num[1]];
213  FourChDelta[2] = (float)averageData[Slider->Channel_Num[2]] - (float)CurFiltdData[Slider->Channel_Num[2]];
214  FourChDelta[3] = (float)averageData[Slider->Channel_Num[3]] - (float)CurFiltdData[Slider->Channel_Num[3]];
215 
216 
217  delta=(FourChDelta[0] + FourChDelta[1] + FourChDelta[2] + FourChDelta[3]);
218 
219  AvrgCh0Data = averageData[Slider->Channel_Num[0]];
220  AvrgCh1Data = averageData[Slider->Channel_Num[1]];
221  AvrgCh2Data = averageData[Slider->Channel_Num[2]];
222  AvrgCh3Data = averageData[Slider->Channel_Num[3]];
223 
224  avgdeltaCh[0] = (FourChDelta[0]*100)/AvrgCh0Data;
225  avgdeltaCh[1] = (FourChDelta[1]*100)/AvrgCh1Data;
226  avgdeltaCh[2] = (FourChDelta[2]*100)/AvrgCh2Data;
227  avgdeltaCh[3] = (FourChDelta[3]*100)/AvrgCh3Data;
228 
229 
230  delta=avgdeltaCh[0];
231  max=0;
232  for (i=1;i<4;i++)
233  {
234  if(delta < avgdeltaCh[i])
235  {
236  delta=avgdeltaCh[i];
237  max=i;
238  }
239  }
240 
241  if((max!=0)&&(max!=3))
242  {
243  min = (avgdeltaCh[max-1]>avgdeltaCh[max+1]) ? max-1 : max+1 ;
244  }
245  else
246  {
247  min = (max == 0) ? 1 : 2;
248  }
249 
250  delta=avgdeltaCh[max]+avgdeltaCh[min];
251  avgdeltaCh[max]=(avgdeltaCh[max]*100)/delta;
252  avgdeltaCh[min]=(avgdeltaCh[min]*100)/delta;
253 
254 
255  if((avgdeltaCh[max]>80)&&(avgdeltaCh[min]<10))
256  {
257  calc=max*5+1;
258  }
259  else
260  {
261  if((avgdeltaCh[max]>80)&&(avgdeltaCh[min]>10))
262  {
263  calc=max*5+2;
264  }
265  else
266  {
267  if((avgdeltaCh[max]>60)&&(max<min))
268  {
269  calc=max*5+3;
270  }
271  else
272  {
273  if((avgdeltaCh[max]>30)&&(avgdeltaCh[min]<60)&&(max<min))
274  {
275  calc=max*5+4;
276  }
277  else
278  {
279  if((avgdeltaCh[max]>30)&&(avgdeltaCh[min]<60)&&(max>min))
280  {
281  calc=(max-1)*5+5;
282  }
283  }
284  }
285  }
286  }
287 
288  calcPercent=calc*6+2;
289  PercentLevel=(PercentLevel+calcPercent)/2;
290 
291  if(PercentLevel<=0)
292  PercentLevel=0;
293  if(PercentLevel>100)
294  PercentLevel=100;
295 
296  Nop();
297  Nop();
298  Nop();
299 
300  if (PercentLevel>prevpercent)
301  PercentLevel=prevpercent+MIN_PERCENT_DISPLAY;
302  else
303  PercentLevel=prevpercent-MIN_PERCENT_DISPLAY;
304 
305  prevpercent=PercentLevel;
306 
307  return PercentLevel;
