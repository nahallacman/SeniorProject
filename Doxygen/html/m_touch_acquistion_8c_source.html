81  MTouchCTMURearm();
82 
83  // Switch channel to the current sensor
84  MTouchADCSetChannel(channelNumber);
85 
86  // Set current sensor to GND (discharge external)
87  *lat &= ~ioMask; *tris &= ~ioMask;
88 
89  // Wait to discharge
90  MTouchDischargeDelay();

92  // Disconnect current sensor from GND (HI-Z)
93  *tris |= ioMask;
94 
95  // Start charging
96  MTouchCTMUStart();
97

99  MTouchChargeDelay(chargeDelay);
100

102  MTouchCTMUStop();
103 
104  // Start conversion
105  MTouchADCStart();
106 
107  // Ground sensor

109 
110  // Get result
