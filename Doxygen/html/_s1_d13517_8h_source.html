138 extern inline BYTE __attribute__ ((always_inline)) Percentage2Alpha(BYTE alphaPercentage)
139 {
140  DWORD percent = (DWORD)(alphaPercentage);
141  percent *= ALPHA_DELTA;
142  percent >>= 5;

144  return (BYTE)percent;
145 }
