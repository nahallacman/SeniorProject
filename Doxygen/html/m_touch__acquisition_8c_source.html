372 void mTouch_DecimationFilter(void)
373 {
374  // Get the current decimated filter value for the sensor.
375  #if MTOUCH_NUMBER_SENSORS > 1
376  uint16_t result = (uint16_t)((*mTouch_currentAcqData).result.v >> 4);
377  #else
378  uint16_t result = (uint16_t)(mTouch_acqData[0].result.v >> 4);
379  #endif
380 
381  #if defined(MTOUCH_EEPROM_ENABLED)
382  #if MTOUCH_DECIMATION_MAX_STEP > 255
383  uint16_t decMaxStep = MTOUCH_DECIMATION_MAX_VALUE;
384  #else
385  uint8_t decMaxStep = MTOUCH_DECIMATION_MAX_VALUE;
386  #endif
387 
388  #undef MTOUCH_DECIMATION_MAX_VALUE
389  #define MTOUCH_DECIMATION_MAX_VALUE decMaxStep // Replace the EEPROM access code with our temporary variable
390  #endif
391 
392  #if defined(MTOUCH_DECIMATION_MAX_STEP) || defined(MTOUCH_EEPROM_ENABLED)
393  #if (MTOUCH_DECIMATION_MAX_STEP != 0) || defined(MTOUCH_EEPROM_ENABLED)
394  if ((uint16_t)(mTouch_lastResult) > (uint16_t)(result + MTOUCH_DECIMATION_MAX_VALUE))
395  {
396  result += MTOUCH_DECIMATION_MAX_VALUE; // Latest conversion is very large. Increment by max step size only.
397  }
398  else if (result > MTOUCH_DECIMATION_MAX_VALUE)
399  {
400  if ((uint16_t)(mTouch_lastResult) < result - MTOUCH_DECIMATION_MAX_VALUE)
401  {
402  result -= MTOUCH_DECIMATION_MAX_VALUE; // Latest conversion is very small. Decrement by max step size only.
403  } else {
404  result = mTouch_lastResult; // Latest conversion is similar in value to decimated value.
405  }
406  }
407  else // Latest conversion is less than (decimated value + STEP_SIZE)
408  { // AND the decimated value is in danger of underflowing.
409  result = mTouch_lastResult;
410  }
411  #else
412  result = mTouch_lastResult; // Decimation disabled.
413  #endif
414  #else
415  result = mTouch_lastResult; // Decimation disabled.
416  #endif
417 
418  #if MTOUCH_NUMBER_SENSORS > 1
419  (*mTouch_currentAcqData).result.v &= 0x000F; // Store the new decimated value into memory for later use.
420  (*mTouch_currentAcqData).result.v |= (uint16_t)(result << 4);
421  #else
422  mTouch_acqData[0].result.v &= 0x000F; // Store the new decimated value into memory for later use.
423  mTouch_acqData[0].result.v |= (uint16_t)(result << 4);
424  #endif
425 
426  #if MTOUCH_NUMBER_SENSORS > 1
427  (*mTouch_currentAcqData).accumulator.v += (uint16_t)((*mTouch_currentAcqData).result.v >> 4);
428  #else
429  mTouch_acqData[0].accumulator.v += (uint16_t)(mTouch_acqData[0].result.v >> 4);
430  #endif
431 }

439 void mTouch_StoreScanA(void)
440 {
441  #if (PIC_ADC_BITS == 8)
442  mTouch_lastResult = ADRES;

444  mTouch_lastResult = (uint16_t) (ADRESH << 8) + ADRESL;

452 void mTouch_StoreScanB(void)
453 {
454  #if (PIC_ADC_BITS == 8)
455  mTouch_lastResult = (uint16_t) (ADRES | 0x0100) - mTouch_lastResult;
456  #elif (PIC_ADC_BITS == 10)
457  mTouch_lastResult = (uint16_t)((ADRESH | 0x04) << 8) + ADRESL - mTouch_lastResult;
458  #elif (PIC_ADC_BITS == 12)
459  mTouch_lastResult = (uint16_t)((ADRESH | 0x10) << 8) + ADRESL - mTouch_lastResult;
460  #else
461  #error The PIC_ADC_BITS value in your device's hardware profile is invalid. It must be 8, 10, or 12.
462  #endif
463 }
