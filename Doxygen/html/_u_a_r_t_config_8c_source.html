226 // NOTE: The following XMODEM code has been upgarded to MPFS2 from MPFS Classic.
228 // Upgrading to HTTP2 and MPFS2 is *strongly* recommended for all new designs.
229 // MPFS2 images can be uploaded directly using the MPFS2.exe tool.
231 static BOOL DownloadMPFS(void)
232 {
233  enum SM_MPFS
234  {
235  SM_MPFS_SOH,
236  SM_MPFS_BLOCK,
237  SM_MPFS_BLOCK_CMP,
238  SM_MPFS_DATA,
239  } state;
240 
241  BYTE c;
242  MPFS_HANDLE handle;
243  BOOL lbDone;
244  BYTE blockLen=0;
245  BYTE lResult;
246  BYTE tempData[XMODEM_BLOCK_LEN];
247  DWORD lastTick;
248  DWORD currentTick;
249 
250  state = SM_MPFS_SOH;
251  lbDone = FALSE;
252 
253  handle = MPFSFormat();
254 
255  // Notify the host that we are ready to receive...
256  lastTick = TickGet();
257  do
258  {
259  currentTick = TickGet();
260  if ( currentTick - lastTick >= (TICK_SECOND/2) )
261  {
262  lastTick = TickGet();
263  while(BusyUART());
264  WriteUART(XMODEM_NAK);
265 
266  /*
267  * Blink LED to indicate that we are waiting for
268  * host to send the file.
269  */
270  LED6_IO ^= 1;
271  }
272 
273  } while(!DataRdyUART());
274 
275 
276  while(!lbDone)
277  {
278  if(DataRdyUART())
279  {
280  // Toggle LED as we receive the data from host.
281  LED6_IO ^= 1;
282  c = ReadUART();
283  }
284  else
285  {
286  // Real application should put some timeout to make sure
287  // that we do not wait forever.
288  continue;
289  }
290 
291  switch(state)
292  {
293  default:
294  if ( c == XMODEM_SOH )
295  {
296  state = SM_MPFS_BLOCK;
297  }
298  else if ( c == XMODEM_EOT )
299  {
300  // Turn off LED when we are done.
301  LED6_IO = 1;
302 
303  MPFSClose(handle);
304  while(BusyUART());
305  WriteUART(XMODEM_ACK);
306  lbDone = TRUE;
307  }
308  else
309  {
310  while(BusyUART());
311  WriteUART(XMODEM_NAK);
312  }
313 
314  break;
315 
316  case SM_MPFS_BLOCK:
317 
318  // We do not use block information.
319  lResult = XMODEM_ACK;
320  blockLen = 0;
321  state = SM_MPFS_BLOCK_CMP;
322  break;
323 
324  case SM_MPFS_BLOCK_CMP:
325 
326  // We do not use 1's comp. block value.
327  state = SM_MPFS_DATA;
328  break;
329 
330  case SM_MPFS_DATA:
331 
332  // Buffer block data until it is over.
333  tempData[blockLen++] = c;
334  if ( blockLen > XMODEM_BLOCK_LEN )
335  {
336 
337  lResult = XMODEM_ACK;
338  for ( c = 0; c < XMODEM_BLOCK_LEN; c++ )
339  MPFSPutArray(handle,&tempData[c],1);
340 
341  MPFSPutEnd(handle);
342 
343  while(BusyUART());
344  WriteUART(lResult);
345  state = SM_MPFS_SOH;
346  }
347  break;
348 
349  }
350 
351  }
352 
353  return TRUE;
354 }
