234 BYTE MDD_IntFlash_SectorRead(DWORD sector_addr, BYTE* buffer)
235 {
236  WORD i;
237  DWORD flashAddress;
238  BYTE TBLPAGSave;
239  WORD_VAL temp;
240 
241  //Error check. Make sure the host is trying to read from a legitimate
242  //address, which corresponds to the MSD volume (and not some other program
243  //memory region beyond the end of the MSD volume).
244  if(sector_addr >= MDD_INTERNAL_FLASH_TOTAL_DISK_SIZE)
245  {
246  return FALSE;
247  }
248 
249  //Save TBLPAG register
250  TBLPAGSave = TBLPAG;
251 
252  //Compute the 24 bit starting address. Note: this is a word address, but we
253  //only store data in and read from the lower word (even LSB).
254  //Starting address will always be even, since MasterBootRecord[] uses aligned attribute in declaration.
255  flashAddress = (DWORD)FILES_ADDRESS + (DWORD)(sector_addr*(WORD)MEDIA_SECTOR_SIZE);
256 
257  //Read a sector worth of data from the flash, and copy to the user specified "buffer".
258  for(i = 0; i < (MEDIA_SECTOR_SIZE / 2u); i++)
259  {
260  TBLPAG = (BYTE)(flashAddress >> 16); //Load TBLPAG pointer (upper 8 bits of total address. A sector could get split at
261  //a 16-bit address boundary, and therefore could exist on two TBLPAG pages.
262  //Therefore, need to reload TBLPAG every iteration of the for() loop
263  temp.Val = __builtin_tblrdl((WORD)flashAddress);
264  *buffer++ = temp.v[0];
265  *buffer++ = temp.v[1];
266  flashAddress += 2u; //Increment address by 2. No MSD data stored in the upper WORD (which only has one implemented byte anyway).
267 
268  }
269 
270  //Restore TBLPAG register to original value
271  TBLPAG = TBLPAGSave;
272 
273  return TRUE;
274 }
