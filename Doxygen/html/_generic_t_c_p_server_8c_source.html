113  to uppercase, and echos it back.
114

126 void GenericTCPServer(void)
127 {
128  BYTE i;
129  WORD w, w2;
130  BYTE AppBuffer[32];
131  WORD wMaxGet, wMaxPut, wCurrentChunk;
132  static TCP_SOCKET MySocket;
133  static enum _TCPServerState
134  {
135  SM_HOME = 0,
136  SM_LISTENING,
137  SM_CLOSING,
138  } TCPServerState = SM_HOME;
139 
140  switch(TCPServerState)
141  {
142  case SM_HOME:
143  // Allocate a socket for this server to listen and accept connections on
144  MySocket = TCPOpen(0, TCP_OPEN_SERVER, SERVER_PORT, TCP_PURPOSE_GENERIC_TCP_SERVER);
145  if(MySocket == INVALID_SOCKET)
146  return;
147 
148  TCPServerState = SM_LISTENING;
149  break;
150 
151  case SM_LISTENING:
152  // See if anyone is connected to us
153  if(!TCPIsConnected(MySocket))
154  return;
155 
156 
157  // Figure out how many bytes have been received and how many we can transmit.
158  wMaxGet = TCPIsGetReady(MySocket); // Get TCP RX FIFO byte count
159  wMaxPut = TCPIsPutReady(MySocket); // Get TCP TX FIFO free space
160 
161  // Make sure we don't take more bytes out of the RX FIFO than we can put into the TX FIFO
162  if(wMaxPut < wMaxGet)
163  wMaxGet = wMaxPut;
164 
165  // Process all bytes that we can
166  // This is implemented as a loop, processing up to sizeof(AppBuffer) bytes at a time.
167  // This limits memory usage while maximizing performance. Single byte Gets and Puts are a lot slower than multibyte GetArrays and PutArrays.
168  wCurrentChunk = sizeof(AppBuffer);
169  for(w = 0; w < wMaxGet; w += sizeof(AppBuffer))
170  {
171  // Make sure the last chunk, which will likely be smaller than sizeof(AppBuffer), is treated correctly.
172  if(w + sizeof(AppBuffer) > wMaxGet)
173  wCurrentChunk = wMaxGet - w;
174 
175  // Transfer the data out of the TCP RX FIFO and into our local processing buffer.
176  TCPGetArray(MySocket, AppBuffer, wCurrentChunk);
177 
178  // Perform the "ToUpper" operation on each data byte
179  for(w2 = 0; w2 < wCurrentChunk; w2++)
180  {
181  i = AppBuffer[w2];
182  if(i >= 'a' && i <= 'z')
183  {
184  i -= ('a' - 'A');
185  AppBuffer[w2] = i;
186  }
187  else if(i == 0x1B) //escape
188  {
189  TCPServerState = SM_CLOSING;
190  }
191  }
192 
193  // Transfer the data out of our local processing buffer and into the TCP TX FIFO.
194  TCPPutArray(MySocket, AppBuffer, wCurrentChunk);
195  }
196 
197  // No need to perform any flush. TCP data in TX FIFO will automatically transmit itself after it accumulates for a while. If you want to decrease latency (at the expense of wasting network bandwidth on TCP overhead), perform and explicit flush via the TCPFlush() API.
198 
199  break;
200 
201  case SM_CLOSING:
202  // Close the socket connection.
203  TCPClose(MySocket);
204 
205  TCPServerState = SM_HOME;
206  break;
207  }
208 }
