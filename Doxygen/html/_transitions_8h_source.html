48 /*********************************************************************
49 * Overview: Transition type enumeration to determine the type of
50 * the transition. Each type will require specific parameters when
51 * setting up the transition operation (GFXSetupTransition() or
52 * GFXTransition()).
53 *
54 * <TABLE>
55 * GFX_TRANSITION_TYPE param1 param2 (sets the direction of transition)
56 * ######################## ################ ###################################################
57 * PLAIN pixel block size not used
58 * EXPANDING_RECTANGLE pixel block size not used
59 * CONTRACTING_RECTANGLE pixel block size not used
60 * SLIDE pixel block size LEFT_TO_RIGHT, RIGHT_TO_LEFT, TOP_TO_BOTTOM, BOTTOM_TO_TOP
61 * PUSH pixel block size LEFT_TO_RIGHT, RIGHT_TO_LEFT, TOP_TO_BOTTOM, BOTTOM_TO_TOP
62 * EXPANDING_LINE pixel block size HORIZONTAL, VERTICAL
63 * CONTRACTING_LINE pixel block size HORIZONTAL, VERTICAL
64 * </TABLE>
65 
66 *********************************************************************/
67 typedef enum
68 {
69  PLAIN = 0, // no transition effect
70  SLIDE, // param1 -> Pixel-block size, param2 -> Sliding direction LEFT_TO_RIGHT/RIGHT_TO_LEFT/TOP_TO_BOTTOM/BOTTOM_TO_TOP
71  PUSH, // param1 -> Pixel-block size, param2 -> Sliding direction LEFT_TO_RIGHT/RIGHT_TO_LEFT/TOP_TO_BOTTOM/BOTTOM_TO_TOP
72  EXPANDING_RECTANGLE, // param1 -> Pixel-block size
73  CONTRACTING_RECTANGLE, // param1 -> Pixel-block size
74  EXPANDING_LINE, // param1 -> Pixel-block size, param2 -> direction HORIZONTAL/VERTICAL
75  CONTRACTING_LINE // param1 -> Pixel-block size, param2 -> direction HORIZONTAL/VERTICAL
76 } GFX_TRANSITION_TYPE;

113 /************************************************************************

115 * DWORD destpageaddr)
116 *
117 * Overview: This function executes the transition that was set up by
118 * GFXSetupTransition(). Status of the transition is returned to
119 * indicate if the transition was executed or not. This function

132 ************************************************************************/

135 /************************************************************************
136 * Function: BYTE GFXSetupTransition(SHORT left, SHORT top, SHORT right,
137 * SHORT bottom, GFX_TRANSITION_TYPE type,
138 * WORD delay_ms, WORD param1, WORD param2)
139 *
140 * Overview: This sets up the transition effect using the GFX_TRANSITION_TYPE
141 * and the given parameters. The actual transition execution will
142 * occur when GFXExecutePendingTransition() is called. When
143 * DOUBLE_BUFFERING is enabled, GFXExecutePendingTransition() is

145 *
146 * Input: left - left x coordinate
147 * top - top y coordinate
148 * right - right x coordinate
149 * bottom - bottom y coordinate
150 * type - Transition type
151 * delay_ms - Delay in milli seconds between redraws in the
152 * screen while executing the transition
153 * param1 - Transition-type specific parameter
154 * param2 - Transition-type specific parameter
155 *
156 * Output: Returns success of the setup
157 * - 0 : Parameters successfully saved for the new transition
158 * - -1 : Parameters not saved, there is a pending transition
159 *
160 ************************************************************************/
