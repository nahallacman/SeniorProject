98 //Note: If only 1 FAT sector is used, assuming 12-bit (1.5 byte) FAT entry size
99 //(ex: FAT12 filesystem), then the total FAT entries that can fit in a single 512
100 //byte FAT sector is (512 bytes) / (1.5 bytes/entry) = 341 entries. This allows
101 //the FAT table to reference up to 341*512 = ~174kB of space. Therfore, more
102 //FAT sectors are needed if creating an MSD volume bigger than this.
103 #define MDD_INTERNAL_FLASH_NUM_RESERVED_SECTORS 1
104 #define MDD_INTERNAL_FLASH_NUM_VBR_SECTORS 1
105 #define MDD_INTERNAL_FLASH_NUM_FAT_SECTORS 1
106 #define MDD_INTERNAL_FLASH_NUM_ROOT_DIRECTORY_SECTORS ((MDD_INTERNAL_FLASH_MAX_NUM_FILES_IN_ROOT+15)/16) //+15 because the compiler truncates
107 #define MDD_INTERNAL_FLASH_OVERHEAD_SECTORS (\
108  MDD_INTERNAL_FLASH_NUM_RESERVED_SECTORS + \
109  MDD_INTERNAL_FLASH_NUM_VBR_SECTORS + \
110  MDD_INTERNAL_FLASH_NUM_ROOT_DIRECTORY_SECTORS + \
111  MDD_INTERNAL_FLASH_NUM_FAT_SECTORS)
112 #define MDD_INTERNAL_FLASH_TOTAL_DISK_SIZE (\
