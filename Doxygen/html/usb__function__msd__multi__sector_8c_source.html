308 void USBCheckMSDRequest(void)
309 {
310  if(SetupPkt.Recipient != USB_SETUP_RECIPIENT_INTERFACE_BITFIELD) return;
311  if(SetupPkt.bIntfID != MSD_INTF_ID) return;
312 
313  switch(SetupPkt.bRequest)
314  {
315  case MSD_RESET:
316  //First make sure all request parameters are correct:
317  //MSD BOT specs require wValue to be == 0x0000 and wLength == 0x0000
318  if((SetupPkt.wValue != 0) || (SetupPkt.wLength != 0))
319  {
320  return; //Return without handling the request (results in STALL)
321  }
322 
323  //Host would typically issue this after a STALL event on an MSD
324  //bulk endpoint. The MSD reset should re-initialize status
325  //so as to prepare for a new CBW. Any currently ongoing command
326  //block should be aborted, but the STALL and DTS states need to be
327  //maintained (host will re-initialize these seperately using
328  //CLEAR_FEATURE, endpoint halt).
329  MSD_State = MSD_WAIT;
330  MSDCommandState = MSD_COMMAND_WAIT;
331  MSDReadState = MSD_READ10_WAIT;
332  MSDWriteState = MSD_WRITE10_WAIT;
333  MSDCBWValid = TRUE;
334  //Need to re-arm MSD bulk OUT endpoint, if it isn't currently armed,
335  //to be able to receive next CBW. If it is already armed, don't need
336  //to do anything, since we can already receive the next CBW (or we are
337  //STALLed, and the host will issue clear halt first).
338  if(!USBHandleBusy(USBGetNextHandle(MSD_DATA_OUT_EP, OUT_FROM_HOST)))
339  {
340  USBMSDOutHandle = USBRxOnePacket(MSD_DATA_OUT_EP,(BYTE*)&msd_cbw,MSD_OUT_EP_SIZE);
341  }
342 
343  //Let USB stack know we took care of handling the EP0 SETUP request.
344  //Allow zero byte status stage to proceed normally now.
345  USBEP0Transmit(USB_EP0_NO_DATA);
346  break;
347  case GET_MAX_LUN:
348  //First make sure all request parameters are correct:
349  //MSD BOT specs require wValue to be == 0x0000, and wLengh == 1
350  if((SetupPkt.wValue != 0) || (SetupPkt.wLength != 1))
351  {
352  break; //Return without handling the request (results in STALL)
353  }
354 
355  //If the host asks for the maximum number of logical units
356  // then send out a packet with that information
357  CtrlTrfData[0] = MAX_LUN;
358  USBEP0SendRAMPtr((BYTE*)&CtrlTrfData[0],1,USB_EP0_INCLUDE_ZERO);
359  break;

366 
367  Summary:
368  This function runs the MSD class state machines and all of its

372  Description:
373  This function runs the MSD class state machines and all of its
374  sub-systems. This function should be called periodically once the
375  device is in the configured state in order to keep the MSD state
376  machine going.
377 
378  Typical Usage:
379  <code>
380  void main(void)
381  {
382  USBDeviceInit();
383  while(1)
384  {
385  USBDeviceTasks();
386  if((USBGetDeviceState() < CONFIGURED_STATE) ||
387  (USBIsDeviceSuspended() == TRUE))
388  {
389  //Either the device is not configured or we are suspended
390  // so we don't want to do execute any application code
391  continue; //go back to the top of the while loop
392  }
393  else
394  {
395  //Keep the MSD state machine going
396  MSDTasks();
397 
398  //Run application code.
399  UserApplication();
400  }
401  }
402  }
403  </code>
404  Conditions:
405  None
406  Return Values:
407  BYTE - the current state of the MSD state machine the valid values are
408  defined in MSD.h under the MSDTasks state machine declaration section.
409  The possible values are the following\:
410  * MSD_WAIT
411  * MSD_DATA_IN
412  * MSD_DATA_OUT
413  * MSD_SEND_CSW
414  Remarks:
415  None
416  *********************************************************************************/
417 BYTE MSDTasks(void)
418 {
419  BYTE i;
420 
421  //Error check to make sure we have are in the CONFIGURED_STATE, prior to
422  //performing MSDTasks(). Some of the MSDTasks require that the device be
423  //configured first.
424  if(USBGetDeviceState() != CONFIGURED_STATE)
425  {
426  return MSD_WAIT;
427  }
428 
429  //Note: Both the USB stack code (usb_device.c) and this MSD handler code
430  //have the ability to modify the BDT values for the MSD bulk endpoints. If the
431  //USB stack operates in USB_INTERRUPT mode (user option in usb_config.h), we
432  //should temporily disable USB interrupts, to avoid any possibility of both
433  //the USB stack and this MSD handler from modifying the same BDT entry, or
434  //MSD state machine variables (ex: in the case of MSD_RESET) at the same time.
435  USBMaskInterrupts();
436 
437  //Main MSD task dispatcher. Receives MSD Command Block Wrappers (CBW) and
438  //dispatches appropriate lower level handlers to service the requests.
439  switch(MSD_State)
440  {
441  case MSD_WAIT: //idle state, when we are waiting for a command from the host
442  {
443  //Check if we have received a new command block wrapper (CBW)
444  if(!USBHandleBusy(USBMSDOutHandle))
445  {
446  //If we are in the MSD_WAIT state, and we received an OUT transaction
447  //on the MSD OUT endpoint, then we must have just received an MSD
448  //Command Block Wrapper (CBW).
449  //First copy the the received data to to the gblCBW structure, so
450  //that we keep track of the command, but free up the MSD OUT endpoint
451  //buffer for fulfilling whatever request may have been received.
452  //gblCBW = msd_cbw; //we are doing this, but below method can yeild smaller code size
453  for(i = 0; i < MSD_CBW_SIZE; i++)
454  {
455  *((BYTE*)&gblCBW.dCBWSignature + i) = *((BYTE*)&msd_cbw.dCBWSignature + i);
456  }
457 
458  //If this CBW is valid?
459  if((USBHandleGetLength(USBMSDOutHandle) == MSD_CBW_SIZE) && (gblCBW.dCBWSignature == MSD_VALID_CBW_SIGNATURE))
460  {
461  //The CBW was valid, set flag meaning any stalls after this point
462  //should not be "persistent" (as in the case of non-valid CBWs).
463  MSDCBWValid = TRUE;
464 
465  //Is this CBW meaningful?
466  if((gblCBW.bCBWLUN <= MAX_LUN) //Verify the command is addressed to a supported LUN
467  &&(gblCBW.bCBWCBLength <= MSD_MAX_CB_SIZE) //Verify the claimed CB length is reasonable/valid
468  &&(gblCBW.bCBWCBLength >= 0x01) //Verify the claimed CB length is reasonable/valid
469  &&((gblCBW.bCBWFlags & MSD_CBWFLAGS_RESERVED_BITS_MASK) == 0x00)) //Verify reserved bits are clear
470  {
471 
472  //The CBW was both valid and meaningful.
473  //Begin preparing a valid Command Status Wrapper (CSW),
474  //in anticipation of completing the request successfully.
475  //If an error detected is later, we will change the status
476  //before sending the CSW.
477  msd_csw.dCSWSignature = MSD_VALID_CSW_SIGNATURE;
478  msd_csw.dCSWTag = gblCBW.dCBWTag;
479  msd_csw.dCSWDataResidue = 0x0;
480  msd_csw.bCSWStatus = MSD_CSW_COMMAND_PASSED;
481 
482  //Since a new CBW just arrived, we should re-init the
483  //lower level state machines to their default states.
484  //Even if the prior operation didn't fully complete
485  //normally, we should abandon the prior operation, when
486  //a new CBW arrives.
487  MSDCommandState = MSD_COMMAND_WAIT;
488  MSDReadState = MSD_READ10_WAIT;
489  MSDWriteState = MSD_WRITE10_WAIT;
490 
491  //Keep track of retry attempts, in case of temporary
492  //failures during read or write of the media.
493  MSDRetryAttempt = 0;
494 
495  //Check the command. With the exception of the REQUEST_SENSE
496  //command, we should reset the sense key info for each new command block.
497  //Assume the command will get processed successfully (and hence "NO SENSE"
498  //response, which is used for success cases), unless handler code
499  //later on detects some kind of error. If it does, it should
500  //update the sense keys to reflect the type of error detected,
501  //prior to sending the CSW.
502  if(gblCBW.CBWCB[0] != MSD_REQUEST_SENSE)
503  {
504  gblSenseData[LUN_INDEX].SenseKey=S_NO_SENSE;
505  gblSenseData[LUN_INDEX].ASC=ASC_NO_ADDITIONAL_SENSE_INFORMATION;
506  gblSenseData[LUN_INDEX].ASCQ=ASCQ_NO_ADDITIONAL_SENSE_INFORMATION;
507  }
508 
509  //Isolate the data direction bit. The direction bit is bit 7 of the bCBWFlags byte.
510  //Then, based on the direction of the data transfer, prepare the MSD state machine
511  //so it knows how to proceed with processing the request.
512  //If bit7 = 0, then direction is OUT from host. If bit7 = 1, direction is IN to host
513  if (gblCBW.bCBWFlags & MSD_CBW_DIRECTION_BITMASK)
514  {
515  MSD_State = MSD_DATA_IN;
516  }
517  else //else direction must be OUT from host
518  {
519  MSD_State = MSD_DATA_OUT;
520  }
521 
522  //Determine if the host is expecting there to be data transfer or not.
523  //Doing this now will make for quicker error checking later.
524  if(gblCBW.dCBWDataTransferLength != 0)
525  {
526  MSDHostNoData = FALSE;
527  }
528  else
529  {
530  MSDHostNoData = TRUE;
531  }
532 
533  //Copy the received command to the lower level command
534  //state machine, so it knows what to do.
535  MSDCommandState = gblCBW.CBWCB[0];
536  }
537  else
538  {
539  //else the CBW wasn't meaningful. Section 6.4 of BOT specs v1.0 says,
540  //"The response of a device to a CBW that is not meaningful is not specified."
541  //Lets STALL the bulk endpoints, so as to promote the possibility of recovery.
542  USBStallEndpoint(MSD_DATA_IN_EP, IN_TO_HOST);
543  USBStallEndpoint(MSD_DATA_OUT_EP, OUT_FROM_HOST);
544  }
545  }//end of: if((USBHandleGetLength(USBMSDOutHandle) == MSD_CBW_SIZE) && (gblCBW.dCBWSignature == MSD_VALID_CBW_SIGNATURE))
546  else //The CBW was not valid.
547  {
548  //Section 6.6.1 of the BOT specifications rev. 1.0 says the device shall STALL bulk IN and OUT
549  //endpoints (or should discard OUT data if not stalled), and should stay in this state until a
550  //"Reset Recovery" (MSD Reset + clear endpoint halt commands on EP0, see section 5.3.4)
551  USBStallEndpoint(MSD_DATA_IN_EP, IN_TO_HOST);
552  USBStallEndpoint(MSD_DATA_OUT_EP, OUT_FROM_HOST);
553  MSDCBWValid = FALSE; //Flag so as to enable a "persistent"
554  //stall (cannot be cleared by clear endpoint halt, unless preceded
555  //by an MSD reset).
556  }
557  }//if(!USBHandleBusy(USBMSDOutHandle))
558  break;
559  }//end of: case MSD_WAIT:
560  case MSD_DATA_IN:
561  if(MSDProcessCommand() == MSD_COMMAND_WAIT)
562  {
563  // Done processing the command, send the status
564  MSD_State = MSD_SEND_CSW;
565  }
566  break;
567  case MSD_DATA_OUT:
568  if(MSDProcessCommand() == MSD_COMMAND_WAIT)
569  {
570  /* Finished receiving the data prepare and send the status */
571  if ((msd_csw.bCSWStatus == MSD_CSW_COMMAND_PASSED)&&(msd_csw.dCSWDataResidue!=0))
572  {
573  msd_csw.bCSWStatus = MSD_CSW_PHASE_ERROR;
574  }
575  MSD_State = MSD_SEND_CSW;
576  }
577  break;
578  case MSD_SEND_CSW:
579  //Check to make sure the bulk IN endpoint is available before sending CSW.
580  //The endpoint might still be busy sending the last packet on the IN endpoint.
581  if(USBHandleBusy(USBGetNextHandle(MSD_DATA_IN_EP, IN_TO_HOST)) == TRUE)
582  {
583  break; //Not available yet. Just stay in this state and try again later.
584  }
585 
586  //Send the Command Status Wrapper (CSW) packet
587  USBMSDInHandle = USBTxOnePacket(MSD_DATA_IN_EP,(BYTE*)&msd_csw,MSD_CSW_SIZE);
588  //If the bulk OUT endpoint isn't already armed, make sure to do so
589  //now so we can receive the next CBW packet from the host.
590  if(!USBHandleBusy(USBMSDOutHandle))
591  {
592  USBMSDOutHandle = USBRxOnePacket(MSD_DATA_OUT_EP,(BYTE*)&msd_cbw,MSD_OUT_EP_SIZE);
593  }
594  MSD_State=MSD_WAIT;
595  break;
596  default:
597  //Illegal condition that should not happen, but might occur if the
598  //device firmware incorrectly calls MSDTasks() prior to calling
599  //USBMSDInit() during the set-configuration portion of enumeration.
600  MSD_State=MSD_WAIT;
601  msd_csw.bCSWStatus = MSD_CSW_PHASE_ERROR;
602  USBStallEndpoint(MSD_DATA_OUT_EP, OUT_FROM_HOST);
603  }//switch(MSD_State)
604 
605  //Safe to re-enable USB interrupts now.
606  USBUnmaskInterrupts();
607 
608  return MSD_State;
609 }

634  *****************************************************************************/
635 BYTE MSDProcessCommand(void)
636 {
637  //Check if the media is either not present, or has been flagged by firmware
638  //to pretend to be non-present (ex: SoftDetached).
639  if((LUNMediaDetect() == FALSE) || (SoftDetach[gblCBW.bCBWLUN] == TRUE))
640  {
641  //Clear flag so we know the media need initialization, if it becomes
642  //present in the future.
643  gblMediaPresent &= ~((WORD)1<<gblCBW.bCBWLUN);
644  MSDProcessCommandMediaAbsent();
645  }
646  else
647  {
648  //Check if the media is present and hasn't been already flagged as initialized.
649  if((gblMediaPresent & ((WORD)1<<gblCBW.bCBWLUN)) == 0)
650  {
651  //Try to initialize the media
652  if(LUNMediaInitialize())
653  {
654  //The media initialized successfully. Set flag letting software
655  //know that it doesn't need re-initialization again (unless the
656  //media is removable and is subsequently removed and re-inserted).
657  gblMediaPresent |= ((WORD)1<<gblCBW.bCBWLUN);
658 
659  //The media is present and has initialized successfully. However,
660  //we should still notify the host that the media may have changed,
661  //from the host's perspective, since we just initialized it for
662  //the first time.
663  gblSenseData[LUN_INDEX].SenseKey = S_UNIT_ATTENTION;
664  gblSenseData[LUN_INDEX].ASC = ASC_NOT_READY_TO_READY_CHANGE;
665  gblSenseData[LUN_INDEX].ASCQ = ASCQ_MEDIUM_MAY_HAVE_CHANGED;
666  //Signify a soft error to the host, so it knows to check the
667  //sense keys to learn that the media just changed.
668  msd_csw.bCSWStatus = MSD_CSW_COMMAND_FAILED; //No real "error" per se has occurred
669  //Process the command now.
670  MSDProcessCommandMediaPresent();
671  }
672  else
673  {
674  //The media failed to initialize for some reason.
675  MSDProcessCommandMediaAbsent();
676  }
677  }
678  else
679  {
680  //The media was present and was already initialized/ready to process
681  //the host's command.
682  MSDProcessCommandMediaPresent();
683  }
684  }
685 
686  return MSDCommandState;
687 }

768  Function:
769  void MSDProcessCommandMediaPresent(void)
770 
771  Description:
772  This funtion processes a command received through the MSD
773  class driver
774 
775  PreCondition:
776  None
777 
778  Paramters:
779  None
780 
781  Return Values:
782  BYTE - the current state of the MSDProcessCommand state
783  machine. The valid values are defined in MSD.h under the
784  MSDProcessCommand state machine declaration section
785 
786  Remarks:
787  None
788 
789  *****************************************************************************/
790 void MSDProcessCommandMediaPresent(void)
791 {
792  BYTE i;
793  BYTE NumBytesInPacket;
794 
795  //Check what command we are currently processing, to decide how to handle it.
796  switch(MSDCommandState)
797  {
798  case MSD_READ_10:
799  //The host issues a "Read 10" request when it wants to read some number
800  //of 10-bit length blocks (512 byte blocks) of data from the media.
801  //Since this is a common request and is part of the "critical path"
802  //performance wise, we put this at the top of the state machine checks.
803  if(MSDReadHandler() == MSD_READ10_WAIT)
804  {
805  MSDCommandState = MSD_COMMAND_WAIT;
806  }
807  break;
808  case MSD_WRITE_10:
809  //The host issues a "Write 10" request when it wants to write some number
810  //of 10-bit length blocks (512 byte blocks) of data to the media.
811  //Since this is a common request and is part of the "critical path"
812  //performance wise, we put this near the top of the state machine checks.
813  if(MSDWriteHandler() == MSD_WRITE10_WAIT)
814  {
815  MSDCommandState = MSD_COMMAND_WAIT;
816  }
817  break;
818  case MSD_INQUIRY:
819  {
820  //The host wants to learn more about our MSD device (spec version,
821  //supported abilities, etc.)
822 
823  //Error check: If host doesn't want any data, then just advance to CSW phase.
824  if(MSDHostNoData == TRUE)
825  {
826  MSDCommandState = MSD_COMMAND_WAIT;
827  break;
828  }
829 
830  //Get the 16-bit "Allocation Length" (Number of bytes to respond
831  //with. Note: Value provided in CBWCB is in big endian format)
832  TransferLength.byte.HB = gblCBW.CBWCB[3]; //MSB
833  TransferLength.byte.LB = gblCBW.CBWCB[4]; //LSB
834  //Check for possible errors.
835  if(MSDCheckForErrorCases(TransferLength.Val) != MSD_ERROR_CASE_NO_ERROR)
836  {
837  break;
838  }
839 
840  //Compute and load proper csw residue and device in number of byte.
841  MSDComputeDeviceInAndResidue(sizeof(InquiryResponse));
842 
843  //If we get to here, this implies no errors were found and the command is legit.
844 
845  //copy the inquiry results from the defined ROM buffer
846  // into the USB buffer so that it can be transmitted
847  memcpypgm2ram((void *)&msd_buffer[0], (ROM void*)&inq_resp, sizeof(InquiryResponse)); //Inquiry response is 36 bytes total
848  MSDCommandState = MSD_COMMAND_RESPONSE;
849  break;
850  }
851  case MSD_READ_CAPACITY:
852  {
853  //The host asked for the total capacity of the device. The response
854  //packet is 8-bytes (32-bits for last LBA implemented, 32-bits for block size).
855  DWORD_VAL sectorSize;
856  DWORD_VAL capacity;
857 
858  //get the information from the physical media
859  capacity.Val = LUNReadCapacity();
860  sectorSize.Val = LUNReadSectorSize();
861 
862  //Copy the data to the buffer. Host expects the response in big endian format.
863  msd_buffer[0]=capacity.v[3];
864  msd_buffer[1]=capacity.v[2];
865  msd_buffer[2]=capacity.v[1];
866  msd_buffer[3]=capacity.v[0];
867 
868  msd_buffer[4]=sectorSize.v[3];
869  msd_buffer[5]=sectorSize.v[2];
870  msd_buffer[6]=sectorSize.v[1];
871  msd_buffer[7]=sectorSize.v[0];
872 
873  //Compute and load proper csw residue and device in number of byte.
874  TransferLength.Val = 0x08; //READ_CAPACITY always has an 8-byte response.
875  MSDComputeDeviceInAndResidue(0x08);
876 
877  MSDCommandState = MSD_COMMAND_RESPONSE;
878  break;
879  }
880  case MSD_REQUEST_SENSE:
881  //The host normally sends this request after a CSW completed, where
882  //the device indicated some kind of error on the previous transfer.
883  //In this case, the host will typically issue this request, so it can
884  //learn more details about the cause/source of the error condition.
885 
886  //Error check: if the host doesn't want any data, just advance to CSW phase.
887  if(MSDHostNoData == TRUE)
888  {
889  MSDCommandState = MSD_COMMAND_WAIT;
890  break;
891  }
892 
893  //Compute and load proper csw residue and device in number of byte.
894  TransferLength.Val = sizeof(RequestSenseResponse); //REQUEST_SENSE has an 18-byte response.
895  MSDComputeDeviceInAndResidue(sizeof(RequestSenseResponse));
896 
897  //Copy the requested response data from flash to the USB ram buffer.
898  for(i=0;i<sizeof(RequestSenseResponse);i++)
899  {
900  msd_buffer[i]=gblSenseData[LUN_INDEX]._byte[i];
901  }
902  MSDCommandState = MSD_COMMAND_RESPONSE;
903  break;
904  case MSD_MODE_SENSE:
905  msd_buffer[0]=0x03;
906  msd_buffer[1]=0x00;
907  msd_buffer[2]=(LUNWriteProtectState()) ? 0x80 : 0x00;
908  msd_buffer[3]= 0x00;
909 
910  //Compute and load proper csw residue and device in number of byte.
911  TransferLength.Val = 0x04;
912  MSDComputeDeviceInAndResidue(0x04);
913  MSDCommandState = MSD_COMMAND_RESPONSE;
914  break;
915  case MSD_PREVENT_ALLOW_MEDIUM_REMOVAL:
916  gblSenseData[LUN_INDEX].SenseKey=S_ILLEGAL_REQUEST;
917  gblSenseData[LUN_INDEX].ASC=ASC_INVALID_COMMAND_OPCODE;
918  gblSenseData[LUN_INDEX].ASCQ=ASCQ_INVALID_COMMAND_OPCODE;
919  msd_csw.bCSWStatus = MSD_CSW_COMMAND_FAILED;
920  msd_csw.dCSWDataResidue = 0x00;
921  MSDCommandState = MSD_COMMAND_WAIT;
922  break;
923  case MSD_TEST_UNIT_READY:
924  //The host will typically send this command periodically to check if
925  //it is ready to be used and to obtain polled notification of changes
926  //in status (ex: user removed media from a removable media MSD volume).
927  //There is no data stage for this request. The information we send to
928  //the host in response to this request is entirely contained in the CSW.
929 
930  //First check for possible errors.
931  if(MSDCheckForErrorCases(0) != MSD_ERROR_CASE_NO_ERROR)
932  {
933  break;
934  }
935  //The stack sets this condition when the status of the removable media
936  //has just changed (ex: the user just plugged in the removable media,
937  //in which case we want to notify the host of the changed status, by
938  //sending a deliberate "error" notification). This doesn't mean any
939  //real error has occurred.
940  if((gblSenseData[LUN_INDEX].SenseKey==S_UNIT_ATTENTION) && (msd_csw.bCSWStatus==MSD_CSW_COMMAND_FAILED))
941  {
942  MSDCommandState = MSD_COMMAND_WAIT;
943  }
944  else
945  {
946  ResetSenseData();
947  msd_csw.dCSWDataResidue=0x00;
948  MSDCommandState = MSD_COMMAND_WAIT;
949  }
950  break;
951  case MSD_VERIFY:
952  //Fall through to STOP_START
953  case MSD_STOP_START:
954  msd_csw.dCSWDataResidue=0x00;
955  MSDCommandState = MSD_COMMAND_WAIT;
956  break;
957  case MSD_COMMAND_RESPONSE:
958  //This command state didn't originate from the host. This state was
959  //set by the firmware (for one of the other handlers) when it was
960  //finished preparing the data to send to the host, and it is now time
961  //to transmit the data over the bulk IN endpoint.
962  if(USBHandleBusy(USBMSDInHandle) == FALSE)
963  {
964  //We still have more bytes needing to be sent. Compute how many
965  //bytes should be in the next IN packet that we send.
966  if(gblCBW.dCBWDataTransferLength >= MSD_IN_EP_SIZE)
967  {
968  NumBytesInPacket = MSD_IN_EP_SIZE;
969  gblCBW.dCBWDataTransferLength -= MSD_IN_EP_SIZE;
970  }
971  else
972  {
973  //This is a short packet and will be our last IN packet sent
974  //in the transfer.
975  NumBytesInPacket = gblCBW.dCBWDataTransferLength;
976  gblCBW.dCBWDataTransferLength = 0;
977  }
978 
979  //We still have more bytes needing to be sent. Check if we have
980  //already fulfilled the device input expected quantity of bytes.
981  //If so, we need to keep sending IN packets, but pad the extra
982  //bytes with value = 0x00 (see error case 5 MSD device BOT v1.0
983  //spec handling).
984  if(TransferLength.Val >= NumBytesInPacket)
985  {
986  //No problem, just send the requested data and keep track of remaining count.
987  TransferLength.Val -= NumBytesInPacket;
988  }
989  else
990  {
991  //The host is reading more bytes than the device has to send.
992  //In this case, we still need to send the quantity of bytes requested,
993  //but we have to fill the pad bytes with 0x00. The below for loop
994  //is execution speed inefficient, but performance isn't important
995  //since this code only executes in the case of a host error
996  //anyway (Hi > Di).
997  for(i = 0; i < NumBytesInPacket; i++)
998  {
999  if(TransferLength.Val != 0)
1000  {
1001  TransferLength.Val--;
1002  }
1003  else
1004  {
1005  msd_buffer[i] = 0x00;
1006  }
1007  }
1008  }
1009 
1010  //We are now ready to send the packet to the host.
1011  USBMSDInHandle = USBTxOnePacket(MSD_DATA_IN_EP,(BYTE*)&msd_buffer[0],NumBytesInPacket);
1012 
1013  //Check to see if we are done sending all requested bytes of data
1014  if(gblCBW.dCBWDataTransferLength == 0)
1015  {
1016  //We have sent all the requested bytes. Go ahead and
1017  //advance state so as to send the CSW.
1018  MSDCommandState = MSD_COMMAND_WAIT;
1019  break;
1020  }
1021  }
1022  break;
1023  case MSD_COMMAND_ERROR:
1024  default:
1025  //An unsupported command was received. Since we are uncertain how many
1026  //bytes we should send/or receive, we should set sense key data and then
1027  //STALL, to force the host to perform error recovery.
1028  MSDErrorHandler(MSD_ERROR_UNSUPPORTED_COMMAND);
1029  break;
1030  } // end switch
1031 }//void MSDProcessCommandMediaPresent(void)

1034 /******************************************************************************
1035  Function:
1036  static void MSDComputeDeviceInAndResidue(WORD DiExpected)
1037 
1038  Description:
1039  This is a private function that performs Hi > Di data size checking
1040  and handling. This function also computes the proper CSW data residue
1041  and updates the global variable.
1042 
1043  PreCondition:
1044  Should only be called in the context of the
1045  MSDProcessCommandMediaPresent() handler function, after receiving a new
1046  command that needs processing. Before calling this function, make sure
1047  the gblCBW.dCBWDataTransferLength and TransferLength.Val variables have
1048  been pre-loaded with the expected host and device data size values.
1049 
1050  Parameters:
1051  WORD DiExpected - Input: Firmware can specify an addional value that
1052  might be smaller than the TransferLength.Val value. The function will
1053  update TransferLength.Val with the smaller of the original value, or
1054  DiExpected.
1055 
1056  Return Values:
1057  None
1058 
1059  Remarks:
1060  None
1061 
1062  *****************************************************************************/
1063 static void MSDComputeDeviceInAndResidue(WORD DiExpected)
1064 {
1065  //Error check number of bytes to send. Check for Hi < Di
1066  if(gblCBW.dCBWDataTransferLength < DiExpected)
1067  {
1068  //The host has requested less data than the entire reponse. We
1069  //send only the host requested quantity of bytes.
1070  msd_csw.dCSWDataResidue = 0;
1071  TransferLength.Val = gblCBW.dCBWDataTransferLength;
1072  }
1073  else
1074  {
1075  //The host requested greater than or equal to the number of bytes expected.
1076  if(DiExpected < TransferLength.Val)
1077  {
1078  TransferLength.Val = DiExpected;
1079  }
1080  msd_csw.dCSWDataResidue = gblCBW.dCBWDataTransferLength - TransferLength.Val;
1081  }
1082 }

1085 /******************************************************************************
1086  Function:
1087  BYTE MSDReadHandler(void)
1088 
1089  Description:
1090  This funtion processes a read command received through
1091  the MSD class driver
1092 
1093  PreCondition:
1094  None
1095 
1096  Parameters:
1097  None
1098 
1099  Return Values:
1100  BYTE - the current state of the MSDReadHandler state
1101  machine. The valid values are defined in MSD.h under the
1102  MSDReadHandler state machine declaration section
1103 
1104  Remarks:
1105  None
1106 
1107  *****************************************************************************/
1108 #pragma idata
1109 BYTE MSDReadHandler(void)
1110 {
1111  //static BYTE fetchStatus = PRE_FETCH_COMPLETE;
1112  BYTE* ptrSave;
1113  static BYTE* ptrLastData = (BYTE*)&msd_buffer[0];
1114  static BYTE* ptrNextData = (BYTE*)&msd_buffer[MSD_IN_EP_SIZE];
1115  static BOOL NewDataAlreadyAvailable;
1116 
1117  //Call our data fetching poller, unless we already have unprocessed data
1118  //ready for our retrieval. In this case, we need to stop calling our
1119  //MDD_SDSPI_AsyncReadTasks(), until we are ready to consume the new buffer
1120  //worth of data and update the AsyncReadWriteInfo structure with a new pointer.
1121  if((fetchStatus != ASYNC_READ_NEW_PACKET_READY) && (fetchStatus != ASYNC_READ_COMPLETE))
1122  {
1123  fetchStatus = MDD_SDSPI_AsyncReadTasks(&AsyncReadWriteInfo);
1124  }
1125 
1126  switch(MSDReadState)
1127  {
1128  case MSD_READ10_WAIT:
1129  //Extract the LBA from the CBW. Note: Also need to perform endian
1130  //swap, since the multi-byte CBW fields are stored big endian, but
1131  //the Microchip C compilers are little endian.
1132  LBA.v[3]=gblCBW.CBWCB[2];
1133  LBA.v[2]=gblCBW.CBWCB[3];
1134  LBA.v[1]=gblCBW.CBWCB[4];
1135  LBA.v[0]=gblCBW.CBWCB[5];
1136 
1137  TransferLength.byte.HB = gblCBW.CBWCB[7]; //MSB of Transfer Length (in number of blocks, not bytes)
1138  TransferLength.byte.LB = gblCBW.CBWCB[8]; //LSB of Transfer Length (in number of blocks, not bytes)
1139 
1140  //Check for possible error cases before proceeding
1141  if(MSDCheckForErrorCases(TransferLength.Val * (DWORD)MEDIA_SECTOR_SIZE) != MSD_ERROR_CASE_NO_ERROR)
1142  {
1143  break;
1144  }
1145 
1146  //Assume success initially, msd_csw.bCSWStatus will get set to 0x01
1147  //or 0x02 later if an error is detected during the actual read sequence.
1148  msd_csw.bCSWStatus=0x0;
1149  msd_csw.dCSWDataResidue=gblCBW.dCBWDataTransferLength;
1150 
1151  //Set up our fetch info data structure.
1152  AsyncReadWriteInfo.wNumBytes = MSD_IN_EP_SIZE;
1153  AsyncReadWriteInfo.pBuffer = (BYTE*)ptrNextData;
1154  AsyncReadWriteInfo.dwAddress = LBA.Val;
1155  AsyncReadWriteInfo.dwBytesRemaining = gblCBW.dCBWDataTransferLength;
1156  AsyncReadWriteInfo.bStateVariable = ASYNC_READ_QUEUED;
1157  //Advance state machine to begin fetching data
1158  MSDReadState = MSD_READ10_XMITING_DATA;
1159  fetchStatus = ASYNC_READ_BUSY;
1160  NewDataAlreadyAvailable = FALSE;
1161  //Fall through...
1162  case MSD_READ10_XMITING_DATA:
1163  //Check if the MDD_SDSPI_AsyncReadTasks() function has finished sending us
1164  //a new packet of data yet. If not, keep calling it until we get some data.
1165  if(NewDataAlreadyAvailable == FALSE)
1166  {
1167  //Try to fetch a new packet of data
1168  if(fetchStatus == ASYNC_READ_NEW_PACKET_READY) //Check if the MDD_SDSPI_AsyncReadTasks() function is about to return on data on the next call
1169  {
1170  fetchStatus = MDD_SDSPI_AsyncReadTasks(&AsyncReadWriteInfo);
1171  NewDataAlreadyAvailable = TRUE;
1172  }
1173  else
1174  {
1175  fetchStatus = MDD_SDSPI_AsyncReadTasks(&AsyncReadWriteInfo);
1176  }
1177  }
1178 
1179  //Check if we are done handling the whole READ10 request.
1180  if((fetchStatus == ASYNC_READ_COMPLETE) && (msd_csw.dCSWDataResidue == 0))
1181  {
1182  MSDReadState = MSD_READ10_AWAITING_COMPLETION;
1183  break;
1184  }
1185  else if(fetchStatus == ASYNC_READ_ERROR)
1186  {
1187  MSDReadState = MSD_READ10_ERROR;
1188  }
1189 
1190  //Check if we currently have any new data ready and we are ready to
1191  //send the data over USB to the host.
1192  if((!USBHandleBusy(USBMSDInHandle)) && (NewDataAlreadyAvailable == TRUE))
1193  {
1194  //Swap the data pointers.
1195  ptrSave = ptrLastData;
1196  ptrLastData = ptrNextData;
1197  ptrNextData = ptrSave;
1198  //Prepare the USB module to send an IN transaction worth of data to the host.
1199  USBMSDInHandle = USBTxOnePacket(MSD_DATA_IN_EP,ptrLastData,MSD_IN_EP_SIZE);
1200  //Advance state machine now that the fetch operation is returning data.
1201  AsyncReadWriteInfo.pBuffer = (BYTE*)ptrNextData; //Swap pointer, so we use the other buffer (the one not currently in use by USB).
1202  //We are sending a USB packet worth of data. Decrement counter.
1203  msd_csw.dCSWDataResidue-=MSD_IN_EP_SIZE;
1204  //Get the next operation started, so it can occur concurrently
1205  //with the above USB transfer, for maximum transfer speeds.
1206  if(fetchStatus == ASYNC_READ_NEW_PACKET_READY)
1207  {
1208  fetchStatus = MDD_SDSPI_AsyncReadTasks(&AsyncReadWriteInfo);
1209  NewDataAlreadyAvailable = TRUE;
1210  }
1211  else
1212  {
1213  fetchStatus = MDD_SDSPI_AsyncReadTasks(&AsyncReadWriteInfo);
1214  NewDataAlreadyAvailable = FALSE;
1215  }
1216  }
1217  break;
1218  case MSD_READ10_AWAITING_COMPLETION:
1219  //If the old data isn't completely sent over USB yet, need to stay
1220  //in this state and return, until the endpoint becomes available agin.
1221  if(USBHandleBusy(USBMSDInHandle))
1222  {
1223  break;
1224  }
1225  else
1226  {
1227  MSDReadState = MSD_READ10_WAIT;
1228  }
1229  break;
1230  case MSD_READ10_ERROR:
1231  default:
1232  //A read error occurred. Notify the host.
1233  msd_csw.bCSWStatus=0x01; //indicate error
1234  //Set error status sense keys, so the host can check them later

1787  }
