136 void MTouchSensorDecode ( SENSOR* pSensor )
137 {
138 
139 INT16 delta;
140 
141  delta = pSensor->oversampleFilterData - pSensor->average;
142 
143  if(delta < 0)
144  {
145  delta = 0;
146  }
147 
148  pSensor->delta = delta;
149 
150  switch(pSensor->state)
151  {
152  case SENSOR_INITIALIZING:
153  {
154  if (pSensor->debounceCount-- <= 0)
155  {
156  pSensor->state = SENSOR_RELEASED;
157  pSensor->debounceCount = DEBOUNCE_COUNT;
158  }
159  pSensor->average = pSensor->oversampleFilterData;
160  break;
161  }
162 
163 
164  case SENSOR_RELEASED:
165  {
166  if(delta > pSensor->pressThreshold)
167  {
168  if (pSensor->debounceCount-- <= 0)
169  {
170  pSensor->state = SENSOR_PRESSED;
171  pSensor->debounceCount = DEBOUNCE_COUNT;
172  }
173  }
174  else
175  {
176  pSensor->debounceCount = DEBOUNCE_COUNT;
177  }
178 
179  MTouchUpdateSensorAverage(pSensor, AVG_RATE_RELEASED);
180 
181  break;
182 
183  }
184 
185  case SENSOR_PRESSED:
186  {
187  if(delta < pSensor->releaseThreshold)
188  {
189  if (pSensor->debounceCount-- <= 0)
190  {
191  pSensor->state = SENSOR_RELEASED;
192  pSensor->debounceCount = DEBOUNCE_COUNT;
193  }
194  }
195  else
196  {
197  pSensor->debounceCount = DEBOUNCE_COUNT;
198  }
199 
200  MTouchUpdateSensorAverage(pSensor, AVG_RATE_PRESSED);
201 
202  break;
203  }
204 
205  default: break;
206 
207  }
208 
209  pSensor->oversampleFilterData = 0;
210 
211 }

214 void MTouchUpdateSensorAverage(SENSOR* pSensor, UINT16 rate)
215 {
216  if(pSensor->averageCount++ >= rate)
217  {
218  pSensor->averageCount = 0;
219 
220  #if (AVG_UPDATE == 2)
221 
222  // avg = (avg + reading) / 2
223  pSensor->average += pSensor->oversampleFilterData;
224  pSensor->average >>= 1;
225 
226  #elif (AVG_UPDATE == 4)
227 
228  // avg = avg - (25% * avg) + (25% * reading)
229  pSensor->average -= pSensor->average >> 2;
230  pSensor->average += pSensor->oversampleFilterData >> 2;
231 
232  #elif (AVG_UPDATE == 8)
233 
234  // avg = avg - (12.5% * avg) + (12.5% * reading)
235  pSensor->average -= pSensor->average >> 3;
236  pSensor->average += pSensor->oversampleFilterData >> 3;
237 
238  #elif (AVG_UPDATE == 16)
239 
240  // avg = avg - (6.25% * avg) + (6.25% * reading)
241  pSensor->average -= pSensor->average >> 4;
242  pSensor->average += pSensor->oversampleFilterData >> 4;
243 
244  #endif
245  }

249 void MTouchAcquisition(void)
250 {
251 INT16 data;
252 UINT8 i;
253 
254 static UINT8 startSensorNumber = 0;
255 SENSOR* pCurrentSensor;
256 SENSOR* pLastSensor;
257 
258 #if defined(MTOUCH_USE_CVD)
259 SENSOR* pPreviousSensor;
260 #endif
261 
262  // Start scan from different sensor each time to create a jitter.
263  // The jitter improves a noise immunity.
264  pCurrentSensor = (SENSOR*)sensors + startSensorNumber;
265  pLastSensor = (SENSOR*)sensors + (MTOUCH_SENSORS_NUMBER-1);
266 
267  #if defined(MTOUCH_USE_CVD)
268  if(startSensorNumber == (MTOUCH_SENSORS_NUMBER-1))
269  {
270  pPreviousSensor = (SENSOR*)sensors;
271  }else{
272  pPreviousSensor = pCurrentSensor + 1;
273  }
274  #endif
275 
276  for(i=0; i<MTOUCH_SENSORS_NUMBER; i++)
277  {
278  if((pCurrentSensor->state&SENSOR_DISCONNECTED) == 0)
279  {
280 
281  #if defined(MTOUCH_USE_CTMU)
282 
283  data = MTouchAcquisitionCTMU(pCurrentSensor);
284 
285  #elif defined(MTOUCH_USE_CVD)
286 
287  data = MTouchAcquisitionCVD(pCurrentSensor,pPreviousSensor);
288 
289  #endif
290 
291  // Add acquisition result to decimate filter
292  if(data > pCurrentSensor->decimateFilterData)
293  {
294  pCurrentSensor->decimateFilterData++;
295  } else {
296  pCurrentSensor->decimateFilterData--;
297  }
298 
299  // Add decimate filter to the oversampling filter.
300  pCurrentSensor->oversampleFilterData += pCurrentSensor->decimateFilterData;
301 
302  if((pCurrentSensor->sampleCounter)-- <= 0)
303  {
304  MTouchSensorDecode(pCurrentSensor);
305  pCurrentSensor->sampleCounter = pCurrentSensor->samplesNumber;
306  }
307 
308  }
309 
310  // Move pointers to the next sensor.
311  #if defined(MTOUCH_USE_CVD)
312  pPreviousSensor = pCurrentSensor;
313  #endif
314 
315  if(pCurrentSensor == pLastSensor)
316  {
317  pCurrentSensor = (SENSOR*)sensors;
318  continue;
319  }
320 
321  pCurrentSensor++;
322  }
323 
324  startSensorNumber++;
325 
326  if(startSensorNumber > (MTOUCH_SENSORS_NUMBER-1))
327  {
328  startSensorNumber = 0;
329  }
330 
331 }
