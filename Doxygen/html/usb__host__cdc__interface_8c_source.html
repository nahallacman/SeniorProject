1 /******************************************************************************

3  USB Host Communication Device Class(CDC) Interface file
4 
5 This file provides all the interface functions to the CDC host client driver.
6 This file should be used in a project with usb_host_cdc.c to provide access
7 to CDC host.
8 
9 Acronyms/abbreviations used by this class:
10  * CDC - Communication Device Class
11  * ACM - Abstract Control Module
12 
13 This file provides interface functions for USB CDC host to the application.
14 These interface function can be used to set up transfers with the device attached
15 on the bus. Interface function USBHostCDC_Api_Get_IN_Data can be used to set up
16 transfer request to receive data from the device.Interface function
17 USBHostCDC_Api_Send_OUT_Data can be used to set up transfer request to send
18 data to the device. If USB_ENABLE_TRANSFER_EVENT is defined application will receive
19 an event on completion of transfer request otherwise application can poll the status
20 of transfer using function USBHostCDC_ApiTransferIsComplete. This file also provides
21 interface routine for ACM class specific requests in function USBHostCDC_Api_ACM_Request.

261 /*******************************************************************************
262  Function:
263  BYTE USBHostCDC_Api_ACM_Request(BYTE requestType, BYTE size, BYTE* data)
264 
265  Description:
266  This function can be used by application code to dynamically access ACM specific
267  requests. This function should be used only if apllication intends to modify for
268  example the Baudrate from previouly configured rate. Data transmitted/received
269  to/from device is a array of bytes. Application must take extra care of understanding
270  the data format before using this function.
271 
272  Precondition:
273  Device must be enumerated and attached successfully.
274 
275  Parameters:
276  BYTE requestType - USB_CDC_SEND_ENCAPSULATED_COMMAND
277  - USB_CDC_GET_ENCAPSULATED_REQUEST
278  - USB_CDC_SET_LINE_CODING
279  - USB_CDC_SET_CONTROL_LINE_STATE
280  - USB_CDC_SET_CONTROL_LINE_STATE
281  These are the mandatory CDC request supported by the
282  CDC host stack.
283  BYTE size - No of bytes to be transferred.
284  BYTE *data - Pointer to data being transferred.
285 
286  Return Values:
287  USB_SUCCESS - Request started successfully
288  USB_CDC_DEVICE_NOT_FOUND - No device with specified address
289  USB_CDC_DEVICE_BUSY - Device not in proper state for
290  performing a transfer
291  USB_CDC_COMMAND_FAILED - Request is not supported.
292  USB_CDC_ILLEGAL_REQUEST - Requested ID is invalid.
293 
294  Remarks:
295  None
296  *******************************************************************************/
297 BYTE USBHostCDC_Api_ACM_Request(BYTE requestType, BYTE size, BYTE* data)
298 {
299  BYTE i;
300  BYTE return_val = USB_CDC_COMMAND_FAILED;
301 
302  // Find the correct device.
303  for (i=0; (i<USB_MAX_CDC_DEVICES) && (deviceInfoCDC[i].deviceAddress != CDCdeviceAddress); i++);
304  if (i == USB_MAX_CDC_DEVICES)
305  {
306  return USB_CDC_DEVICE_NOT_FOUND;
307  }
308 
309  switch(requestType)
310  {
311  case USB_CDC_SEND_ENCAPSULATED_COMMAND :
312  return_val = USBHostCDCTransfer(CDCdeviceAddress, USB_CDC_SEND_ENCAPSULATED_COMMAND,0,
313  deviceInfoCDC[i].commInterface.interfaceNum,size,data,
314  0);
315  break;
316  case USB_CDC_GET_ENCAPSULATED_REQUEST :
317  return_val = USBHostCDCTransfer(CDCdeviceAddress, USB_CDC_GET_ENCAPSULATED_REQUEST,1,
318  deviceInfoCDC[i].commInterface.interfaceNum,size,data,
319  0);
320  break;
321 
322  case USB_CDC_SET_COMM_FEATURE :
323  break;
324 
325  case USB_CDC_GET_COMM_FEATURE :
326  break;
327 
328  case USB_CDC_SET_LINE_CODING :
329  return_val = USBHostCDCTransfer(CDCdeviceAddress, USB_CDC_SET_LINE_CODING,0,
330  deviceInfoCDC[i].commInterface.interfaceNum,size,data,
331  0);
332  break;
333 
334  case USB_CDC_GET_LINE_CODING :
335  return_val = USBHostCDCTransfer(CDCdeviceAddress, USB_CDC_GET_LINE_CODING,1,
336  deviceInfoCDC[i].commInterface.interfaceNum,size,data,
337  0);
338  break;
339 
340  case USB_CDC_SET_CONTROL_LINE_STATE :
341  return_val = USBHostCDCTransfer(CDCdeviceAddress, USB_CDC_SET_CONTROL_LINE_STATE,0,
342  deviceInfoCDC[i].commInterface.interfaceNum,size,data,
343  0);
344  break;
345 
346  case USB_CDC_SEND_BREAK :
347  break;
348  default:
349  return USB_CDC_ILLEGAL_REQUEST;
350  break;
351  }
352  return return_val;
353 }
