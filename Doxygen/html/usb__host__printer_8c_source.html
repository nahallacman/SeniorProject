5  Summary:
6  This is the Printer client driver file for a USB Embedded Host device.
7 
8  Description:
9  This is the Printer client driver file for a USB Embedded Host device.
10  It allows an embedded application to utilize a USB printer to provide
11  printed output.
12 
13  USB printers utilize the USB Printer Class to communicate with a USB
14  Host. This class defines the USB transfer type, the endpoint structure,
15  a device requests that can be performed. The actual commands sent to
16  the printer, however, are dictated by the printer language used by the
17  particular printer.
18 
19  Many different printer languages are utilized by the wide variety of
20  printers on the market. Typically, low end printers receive printer-specific
21  binary data, utilizing the processing power of the USB Host to perform
22  all of the complex calculations required to translate text and graphics to
23  a simple binary representation. This works well when a PC is the USB Host,
24  but it is not conducive to an embedded application with limited resources.
25 
26  Many printers on the market use a command based printer language, relying
27  on the printer itself to interpret commands to produce the desired output.
28  Some languages are standardized across printers from a particular
29  manufacturer, and some are used across multiple manufacturer. This method
30  lends itself better to embedded applications by allowing the printer to
31  take on some of the computational overhead. Microchip provides support for
32  some printer languages, including PostScript and PCL 5. Additional printer
33  language can be implemented. Refer to the USB Embedded Host Printer Class
34  application notes for more details on implementing printer language support.
35 
36 
37  Remarks:
38  This driver should be used in a project with usb_host.c to provided the USB
39  Embedded Host and hardware interfaces, plus one or more language support
40  files.
41 
42  To interface with USB Embedded Host layer, the routine USBHostPrinterInitialize()
43  should be specified as the Initialize() function, and
44  USBHostPrinterEventHandler() should be specified as the EventHandler()
45  function in the usbClientDrvTable[] array declared in usb_config.c.
46 
47  This driver requires transfer events from usb_host.c, so
48  USB_ENABLE_TRANSFER_EVENT must be defined.
49 
50  Since the printer class is performed with bulk transfers,
51  USB_SUPPORT_BULK_TRANSFERS must be defined.

569 BOOL USBHostPrinterEventHandler ( BYTE address, USB_EVENT event, void *data, DWORD size )
570 {
571  USB_PRINTER_QUEUE_ITEM *transfer;
572 
573  #ifdef DEBUG_MODE
574  //UART2PrintString( "PRN: Receiving event\r\n" );
575  #endif
576 
577  // Make sure it was for our device
578  if (!_USBHostPrinter_FindDevice( address ))
579  {
580  return FALSE; // The device was not found.
581  }
582 
583  // Handle specific events.
584  switch (event)
585  {
586  case EVENT_DETACH:
587  // Purge the IN and OUT transfer queues.
588  while (StructQueueIsNotEmpty( &(usbPrinters[currentPrinterRecord].transferQueueIN), USB_PRINTER_TRANSFER_QUEUE_SIZE ))
589  {
590  transfer = StructQueueRemove( &(usbPrinters[currentPrinterRecord].transferQueueIN), USB_PRINTER_TRANSFER_QUEUE_SIZE );
591  }
592  while (StructQueueIsNotEmpty( &(usbPrinters[currentPrinterRecord].transferQueueOUT), USB_PRINTER_TRANSFER_QUEUE_SIZE ))
593  {
594  transfer = StructQueueRemove( &(usbPrinters[currentPrinterRecord].transferQueueOUT), USB_PRINTER_TRANSFER_QUEUE_SIZE );
595  if (transfer->flags & USB_PRINTER_TRANSFER_COPY_DATA)
596  {
597  USB_FREE( transfer->data );
598  }
599  }
600 
601  // Tell the printer language support that the device has been detached.
602  USBHostPrinterCommand( usbPrinters[currentPrinterRecord].ID.deviceAddress, USB_PRINTER_DETACHED, USB_NULL, 0, 0 );
603  // Notify that application that the device has been detached.
604  USB_HOST_APP_EVENT_HANDLER(usbPrinters[currentPrinterRecord].ID.deviceAddress, EVENT_PRINTER_DETACH,
605  &usbPrinters[currentPrinterRecord].ID.deviceAddress, sizeof(BYTE) );
606  #ifdef USB_PRINTER_ALLOW_DYNAMIC_LANGUAGE_DETERMINATION
607  USB_FREE( usbPrinters[currentPrinterRecord].deviceIDString );
608  #endif
609  usbPrinters[currentPrinterRecord].flags.value = 0;
610  #ifdef DEBUG_MODE
611  UART2PrintString( "PRN: USB Printer Client Device Detached: address=" );
612  UART2PutDec( address );
613  UART2PrintString( "\r\n" );
614  #endif
615  return TRUE;
616 
617  case EVENT_TRANSFER:
618  #ifdef DEBUG_MODE
619  //UART2PrintString( "PRN: TRANSFER event occurred.\r\n" );
620  #endif
621 
622  if ( (data != NULL) && (size == sizeof(HOST_TRANSFER_DATA)) )
623  {
624  DWORD dataCount = ((HOST_TRANSFER_DATA *)data)->dataCount;
625 
626  if ( ((HOST_TRANSFER_DATA *)data)->bEndpointAddress == (USB_IN_EP | usbPrinters[currentPrinterRecord].endpointIN) )
627  {
628  usbPrinters[currentPrinterRecord].flags.rxBusy = 0;
629  usbPrinters[currentPrinterRecord].rxLength = dataCount;
630 
631  transfer = StructQueueRemove( &(usbPrinters[currentPrinterRecord].transferQueueIN), USB_PRINTER_TRANSFER_QUEUE_SIZE );
632  if (transfer->flags & USB_PRINTER_TRANSFER_COPY_DATA)
633  {
634  USB_FREE( transfer->data );
635  }
636 
637  if (StructQueueIsNotEmpty( &(usbPrinters[currentPrinterRecord].transferQueueIN), USB_PRINTER_TRANSFER_QUEUE_SIZE ))
638  {
639  _USBHostPrinter_ReadFromQueue( usbPrinters[currentPrinterRecord].ID.deviceAddress );
640  }
641 
642  if (transfer->flags & USB_PRINTER_TRANSFER_NOTIFY)
643  {
644  USB_HOST_APP_EVENT_HANDLER( usbPrinters[currentPrinterRecord].ID.deviceAddress, EVENT_PRINTER_RX_DONE, &dataCount, sizeof(DWORD) );
645  }
646  }
647  else if ( ((HOST_TRANSFER_DATA *)data)->bEndpointAddress == (USB_OUT_EP | usbPrinters[currentPrinterRecord].endpointOUT) )
648  {
649  usbPrinters[currentPrinterRecord].flags.txBusy = 0;
